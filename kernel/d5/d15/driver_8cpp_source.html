37  void Manager::PreloadDrivers()
38  {
39  debug("Initializing driver manager");
40  const char *DriverDirectory = Config.DriverDirectory;
41  FileNode *drvDirNode = fs->GetByPath(DriverDirectory, nullptr);
42  if (!drvDirNode)
43  {
44  error("Failed to open driver directory %s", DriverDirectory);
45  KPrint("Failed to open driver directory %s", DriverDirectory);
46  return;
47  }
48 
49  foreach (const auto &drvNode in drvDirNode->Children)
50  {
51  debug("Checking driver %s", drvNode->Path.c_str());
52  if (!drvNode->IsRegularFile())
53  continue;
54 
55  if (Execute::GetBinaryType(drvNode->Path) != Execute::BinTypeELF)
56  {
57  error("Driver %s is not an ELF binary", drvNode->Path.c_str());
58  continue;
59  }
60 
61  DriverObject drvObj = {.BaseAddress = 0,
62  .EntryPoint = 0,
63  .vma = new Memory::VirtualMemoryArea(thisProcess->PageTable),
64  .Path = drvNode->Path,
65  .InterruptHandlers = new std::unordered_map<uint8_t, void *>(),
66  .DeviceOperations = new std::unordered_map<dev_t, DriverHandlers>(),
67  .ID = DriverIDCounter};
68 
69  int err = this->LoadDriverFile(drvObj, drvNode);
70  debug("err = %d (%s)", err, strerror(err));
71  if (err != 0)
72  {
73  error("Failed to load driver %s: %s",
74  drvNode->Path.c_str(), strerror(err));
75 
76  delete drvObj.vma;
77  delete drvObj.InterruptHandlers;
78  delete drvObj.DeviceOperations;
79  continue;
80  }
81 
82  debug("gdb: \"0x%lX\" %s", drvObj.BaseAddress, drvObj.Name);
83 
84  Drivers.insert({DriverIDCounter++, drvObj});
85  }
86  }

88  void Manager::LoadAllDrivers()
89  {
90  if (Drivers.empty())
91  {
92  KPrint("\x1b[1;31;41mNo drivers to load");
93  return;
94  }
95 
96  foreach (auto &var in Drivers)
97  {
98  DriverObject &Drv = var.second;
99 
100  debug("Calling driver %s at %#lx", Drv.Path.c_str(), Drv.EntryPoint);
101  int (*DrvInit)(dev_t) = (int (*)(dev_t))Drv.EntryPoint;
102  Drv.ErrorCode = DrvInit(Drv.ID);
103  if (Drv.ErrorCode < 0)
104  {
105  KPrint("FATAL: _start() failed for %s: %s",
106  Drv.Name, strerror(Drv.ErrorCode));
107  error("Failed to load driver %s: %s",
108  Drv.Path.c_str(), strerror(Drv.ErrorCode));
109 
110  Drv.vma->FreeAllPages();
111  continue;
112  }
113 
114  KPrint("Loading driver %s", Drv.Name);
115 
116  debug("Calling Probe()=%#lx on driver %s",
117  Drv.Probe, Drv.Path.c_str());
118  Drv.ErrorCode = Drv.Probe();
119  if (Drv.ErrorCode < 0)
120  {
121  KPrint("Probe() failed for %s: %s",
122  Drv.Name, strerror(Drv.ErrorCode));
123  error("Failed to probe driver %s: %s",
124  Drv.Path.c_str(), strerror(Drv.ErrorCode));
125 
126  Drv.vma->FreeAllPages();
127  continue;
128  }
129 
130  debug("Calling driver Entry()=%#lx function on driver %s",
131  Drv.Entry, Drv.Path.c_str());
132  Drv.ErrorCode = Drv.Entry();
133  if (Drv.ErrorCode < 0)
134  {
135  KPrint("Entry() failed for %s: %s",
136  Drv.Name, strerror(Drv.ErrorCode));
137  error("Failed to initialize driver %s: %s",
138  Drv.Path.c_str(), strerror(Drv.ErrorCode));
139 
140  Drv.vma->FreeAllPages();
141  continue;
142  }
143 
144  debug("Loaded driver %s", Drv.Path.c_str());
145  Drv.Initialized = true;
146  }
147  }

149  void Manager::UnloadAllDrivers()
150  {
151  foreach (auto &var in Drivers)
152  {
153  DriverObject *Drv = &var.second;
154  if (!Drv->Initialized)
155  continue;
156 
157  debug("Unloading driver %s", Drv->Name);
158  int err = Drv->Final();
159  if (err < 0)
160  {
161  warn("Failed to unload driver %s: %s",
162  Drv->Name, strerror(err));
163  }
164 
165  if (!Drv->InterruptHandlers->empty())
166  {
167  foreach (auto &rInt in * Drv->InterruptHandlers)
168  {
169  Interrupts::RemoveHandler((void (*)(CPU::TrapFrame *))rInt.second);
170  }
171  Drv->InterruptHandlers->clear();
172  }
173  }
174  Drivers.clear();
175  }

177  void Manager::Panic()
178  {
179  Memory::Virtual vmm;
180  if (Drivers.size() == 0)
181  return;
182 
183  foreach (auto Driver in Drivers)
184  {
185  if (!Driver.second.Initialized)
186  continue;
187 
188  trace("Panic on driver %s", Driver.second.Name);
189  debug("%#lx", Driver.second.Panic);
190 
191  /* Crash while probing? */
192  if (Driver.second.Panic && vmm.Check((void *)Driver.second.Panic))
193  Driver.second.Panic();
194  else
195  error("No panic function for driver %s",
196  Driver.second.Name);
197  }

200  int Manager::LoadDriverFile(DriverObject &Drv, FileNode *File)
201  {
202  trace("Loading driver %s in memory", File->Name.c_str());
203 
204  Elf_Ehdr ELFHeader{};
205  File->Read(&ELFHeader, sizeof(Elf_Ehdr), 0);
206 
207  AssertReturnError(ELFHeader.e_ident[EI_CLASS] == ELFCLASS64, -ENOEXEC);
208  AssertReturnError(ELFHeader.e_ident[EI_DATA] == ELFDATA2LSB, -ENOEXEC);
209  AssertReturnError(ELFHeader.e_ident[EI_OSABI] == ELFOSABI_SYSV, -ENOEXEC);
210  AssertReturnError(ELFHeader.e_ident[EI_ABIVERSION] == 0, -ENOEXEC);
211  AssertReturnError(ELFHeader.e_type == ET_DYN, -ENOEXEC);
212  AssertReturnError(ELFHeader.e_machine == EM_X86_64, -ENOEXEC);
213  AssertReturnError(ELFHeader.e_version == EV_CURRENT, -ENOEXEC);
214  AssertReturnError(ELFHeader.e_entry != 0x0, -ENOEXEC);
215  AssertReturnError(ELFHeader.e_shstrndx != SHN_UNDEF, -ENOEXEC);
216  Drv.EntryPoint = ELFHeader.e_entry;
217 
218  size_t segSize = 0;
219  Elf_Phdr phdr{};
220  for (Elf_Half i = 0; i < ELFHeader.e_phnum; i++)
221  {
222  File->Read(&phdr, sizeof(Elf_Phdr), ELFHeader.e_phoff + (i * sizeof(Elf_Phdr)));
223  if (phdr.p_type == PT_LOAD || phdr.p_type == PT_DYNAMIC)
224  {
225  if (segSize < phdr.p_vaddr + phdr.p_memsz)
226  segSize = phdr.p_vaddr + phdr.p_memsz;
227  continue;
228  }
229 
230  if (phdr.p_type == PT_INTERP)
231  {
232  char interp[17];
233  File->Read(interp, sizeof(interp), phdr.p_offset);
234  if (strncmp(interp, "/boot/fennix.elf", sizeof(interp)) != 0)
235  {
236  error("Interpreter is not /boot/fennix.elf");
237  return -ENOEXEC;
238  }
239  }
240  }
241  debug("segSize: %ld", segSize);
242 
243  Drv.BaseAddress = (uintptr_t)Drv.vma->RequestPages(TO_PAGES(segSize) + 1);
244  Drv.EntryPoint += Drv.BaseAddress;
245  debug("Driver %s has entry point %#lx and base %#lx",
246  File->Name.c_str(), Drv.EntryPoint, Drv.BaseAddress);
247 
248  Elf64_Shdr sht_strtab{};
249  Elf64_Shdr sht_symtab{};
250  Elf_Shdr shstrtab{};
251  Elf_Shdr shdr{};
252  __DriverInfo driverInfo{};
253  File->Read(&shstrtab, sizeof(Elf_Shdr), ELFHeader.e_shoff + (ELFHeader.e_shstrndx * ELFHeader.e_shentsize));
254  for (Elf_Half i = 0; i < ELFHeader.e_shnum; i++)
255  {
256  if (i == ELFHeader.e_shstrndx)
257  continue;
258 
259  File->Read(&shdr, ELFHeader.e_shentsize, ELFHeader.e_shoff + (i * ELFHeader.e_shentsize));
260 
261  switch (shdr.sh_type)
262  {
263  case SHT_PROGBITS:
264  break;
265  case SHT_SYMTAB:
266  sht_symtab = shdr;
267  continue;
268  case SHT_STRTAB:
269  sht_strtab = shdr;
270  continue;
271  case SHT_NULL:
272  default:
273  continue;
274  }
275 
276  char symName[16];
277  File->Read(symName, sizeof(symName), shstrtab.sh_offset + shdr.sh_name);
278  if (strcmp(symName, ".driver.info") != 0)
279  continue;
280 
281  File->Read(&driverInfo, sizeof(__DriverInfo), shdr.sh_offset);
282 
283  /* Perform relocations */
284  driverInfo.Name = (const char *)(Drv.BaseAddress + (uintptr_t)driverInfo.Name);
285  driverInfo.Description = (const char *)(Drv.BaseAddress + (uintptr_t)driverInfo.Description);
286  driverInfo.Author = (const char *)(Drv.BaseAddress + (uintptr_t)driverInfo.Author);
287  driverInfo.License = (const char *)(Drv.BaseAddress + (uintptr_t)driverInfo.License);
288  }
289 
290  for (size_t h = 0; h < (sht_symtab.sh_size / sizeof(Elf64_Sym)); h++)
291  {
292  Elf64_Sym symEntry{};
293  uintptr_t symOffset = sht_symtab.sh_offset + (h * sizeof(Elf64_Sym));
294  File->Read(&symEntry, sizeof(Elf64_Sym), symOffset);
295 
296  if (symEntry.st_name == 0)
297  continue;
298 
299  char symName[16];
300  File->Read(symName, sizeof(symName), sht_strtab.sh_offset + symEntry.st_name);
301 
302  switch (symEntry.st_shndx)
303  {
304  case SHN_UNDEF:
305  case SHN_ABS:
306  case SHN_LOPROC /* , SHN_LORESERVE and SHN_BEFORE */:
307  case SHN_AFTER:
308  case SHN_HIPROC:
309  case SHN_COMMON:
310  case SHN_HIRESERVE:
311  break;
312  default:
313  {
314  debug("shndx: %d", symEntry.st_shndx);
315  if (strcmp(symName, "DriverEntry") == 0)
316  Drv.Entry = (int (*)())(Drv.BaseAddress + symEntry.st_value);
317  else if (strcmp(symName, "DriverFinal") == 0)
318  Drv.Final = (int (*)())(Drv.BaseAddress + symEntry.st_value);
319  else if (strcmp(symName, "DriverPanic") == 0)
320  Drv.Panic = (int (*)())(Drv.BaseAddress + symEntry.st_value);
321  else if (strcmp(symName, "DriverProbe") == 0)
322  Drv.Probe = (int (*)())(Drv.BaseAddress + symEntry.st_value);
323 
324  debug("Found %s at %#lx", symName, symEntry.st_value);
325  break;
326  }
327  }
328  }
329 
330  for (Elf_Half i = 0; i < ELFHeader.e_phnum; i++)
331  {
332  File->Read(&phdr, sizeof(Elf_Phdr), ELFHeader.e_phoff + (i * sizeof(Elf_Phdr)));
333 
334  switch (phdr.p_type)
335  {
336  case PT_LOAD:
337  case PT_DYNAMIC:
338  {
339  if (phdr.p_memsz == 0)
340  continue;
341 
342  uintptr_t dest = Drv.BaseAddress + phdr.p_vaddr;
343  debug("Copying PHDR %#lx to %#lx-%#lx (%ld file bytes, %ld mem bytes)",
344  phdr.p_type, dest, dest + phdr.p_memsz,
345  phdr.p_filesz, phdr.p_memsz);
346 
347  if (phdr.p_filesz > 0)
348  File->Read(dest, phdr.p_filesz, phdr.p_offset);
349 
350  if (phdr.p_memsz - phdr.p_filesz > 0)
351  {
352  void *zero = (void *)(dest + phdr.p_filesz);
353  memset(zero, 0, phdr.p_memsz - phdr.p_filesz);
354  }
355 
356  if (phdr.p_type != PT_DYNAMIC)
357  break;
358 
359  Elf64_Dyn *dyn = (Elf64_Dyn *)(Drv.BaseAddress + phdr.p_vaddr);
360  Elf64_Dyn *relaSize = nullptr;
361  Elf64_Dyn *pltrelSize = nullptr;
362 
363  while (dyn->d_tag != DT_NULL)
364  {
365  switch (dyn->d_tag)
366  {
367  case DT_PLTRELSZ:
368  {
369  pltrelSize = dyn;
370  break;
371  }
372  case DT_PLTGOT:
373  {
374  Elf_Addr *got = (Elf_Addr *)(Drv.BaseAddress + dyn->d_un.d_ptr);
375  got[1] = 0;
376  got[2] = 0;
377  break;
378  }
379  case DT_RELASZ:
380  {
381  relaSize = dyn;
382  break;
383  }
384  case DT_PLTREL:
385  {
386  AssertReturnError(dyn->d_un.d_val == DT_RELA, -ENOEXEC);
387  break;
388  }
389  default:
390  break;
391  }
392  dyn++;
393  }
394 
395  dyn = (Elf64_Dyn *)(Drv.BaseAddress + phdr.p_vaddr);
396  while (dyn->d_tag != DT_NULL)
397  {
398  switch (dyn->d_tag)
399  {
400  case DT_RELA: /* .rela.dyn */
401  {
402  AssertReturnError(relaSize != nullptr, -ENOEXEC);
403 
404  Elf64_Rela *rela = (Elf64_Rela *)(Drv.BaseAddress + dyn->d_un.d_ptr);
405  for (size_t i = 0; i < (relaSize->d_un.d_val / sizeof(Elf64_Rela)); i++)
406  {
407  Elf64_Rela *r = &rela[i];
408  uintptr_t *reloc = (uintptr_t *)(Drv.BaseAddress + r->r_offset);
409  uintptr_t relocTarget = 0;
410 
411  switch (ELF64_R_TYPE(r->r_info))
412  {
413  case R_X86_64_GLOB_DAT:
414  case R_X86_64_JUMP_SLOT:
415  {
416  relocTarget = Drv.BaseAddress;
417  break;
418  }
419  case R_X86_64_RELATIVE:
420  case R_X86_64_64:
421  {
422  relocTarget = Drv.BaseAddress + r->r_addend;
423  break;
424  }
425  default:
426  {
427  fixme("Unhandled relocation type: %#lx",
428  ELF64_R_TYPE(r->r_info));
429  break;
430  }
431  }
432 
433  *reloc = relocTarget;
434 
435  debug("Relocated %#lx to %#lx",
436  r->r_offset, *reloc);
437  }
438  break;
439  }
440  case DT_JMPREL: /* .rela.plt */
441  {
442  AssertReturnError(pltrelSize != nullptr, -ENOEXEC);
443 
444  std::vector<Elf64_Dyn> symtab = Execute::ELFGetDynamicTag_x86_64(File, DT_SYMTAB);
445  Elf64_Sym *symbols = (Elf64_Sym *)((uintptr_t)Drv.BaseAddress + symtab[0].d_un.d_ptr);
446 
447  std::vector<Elf64_Dyn> StrTab = Execute::ELFGetDynamicTag_x86_64(File, DT_STRTAB);
448  char *DynStr = (char *)((uintptr_t)Drv.BaseAddress + StrTab[0].d_un.d_ptr);
449 
450  Elf64_Rela *rela = (Elf64_Rela *)(Drv.BaseAddress + dyn->d_un.d_ptr);
451  for (size_t i = 0; i < (pltrelSize->d_un.d_val / sizeof(Elf64_Rela)); i++)
452  {
453  Elf64_Rela *r = &rela[i];
454  uintptr_t *reloc = (uintptr_t *)(Drv.BaseAddress + r->r_offset);
455 
456  switch (ELF64_R_TYPE(r->r_info))
457  {
458  case R_X86_64_JUMP_SLOT:
459  {
460  Elf64_Xword symIndex = ELF64_R_SYM(r->r_info);
461  Elf64_Sym *sym = symbols + symIndex;
462 
463  const char *symName = DynStr + sym->st_name;
464  debug("Resolving symbol %s", symName);
465 
466  *reloc = (uintptr_t)GetSymbolByName(symName, driverInfo.Version.APIVersion);
467  break;
468  }
469  default:
470  {
471  fixme("Unhandled relocation type: %#lx",
472  ELF64_R_TYPE(r->r_info));
473  break;
474  }
475  }
476  }
477  break;
478  }
479  case DT_PLTGOT:
480  case DT_PLTRELSZ:
481  case DT_RELASZ:
482  case DT_PLTREL:
483  break;
484  default:
485  {
486  fixme("Unhandled dynamic tag: %#lx", dyn->d_tag);
487  break;
488  }
489  }
490  dyn++;
491  }
492  break;
493  }
494  case PT_PHDR:
495  case PT_INTERP:
496  break;
497  default:
498  {
499  fixme("Unhandled program header type: %#lx", phdr.p_type);
500  break;
501  }
502  }
503  }
504 
505  AssertReturnError(driverInfo.Name != nullptr, -EFAULT);
506  strncpy(Drv.Name, driverInfo.Name, sizeof(Drv.Name));
507  strncpy(Drv.Description, driverInfo.Description, sizeof(Drv.Description));
508  strncpy(Drv.Author, driverInfo.Author, sizeof(Drv.Author));
509  Drv.Version.Major = driverInfo.Version.Major;
510  Drv.Version.Minor = driverInfo.Version.Minor;
511  Drv.Version.Patch = driverInfo.Version.Patch;
512  strncpy(Drv.License, driverInfo.License, sizeof(Drv.License));
513 
514  return 0;
515  }
