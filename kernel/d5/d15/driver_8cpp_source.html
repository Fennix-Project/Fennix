35  void Manager::LoadAllDrivers()
36  {
37  foreach (auto &var in Drivers)
38  {
39  DriverObject *Drv = &var.second;
40  size_t dapiPgs = TO_PAGES(sizeof(__driverAPI));
41  __driverAPI *dApi = (__driverAPI *)Drv->vma->RequestPages(dapiPgs);
42  debug("Driver API at %#lx-%#lx", dApi, dApi + sizeof(__driverAPI));
43 
44  fixme("api version");
45  dApi->APIVersion.Major = 0;
46  dApi->APIVersion.Minor = 0;
47  dApi->APIVersion.Patch = 0;
48 
49  dApi->MajorID = var.first;
50  dApi->Base = Drv->BaseAddress;
51  PopulateDriverAPI(dApi);
52 
53  debug("Calling driver %s at %#lx", Drv->Path, Drv->EntryPoint);
54  int (*DrvInit)(__driverAPI *) = (int (*)(__driverAPI *))Drv->EntryPoint;
55  Drv->ErrorCode = DrvInit(dApi);
56  if (Drv->ErrorCode < 0)
57  {
58  KPrint("FATAL: _start() failed for %s: %s",
59  Drv->Name, strerror(Drv->ErrorCode));
60  error("Failed to load driver %s: %s",
61  Drv->Path, strerror(Drv->ErrorCode));
62 
63  Drv->vma->FreeAllPages();
64  continue;
65  }
66 
67  KPrint("Loading driver %s", Drv->Name);
68 
69  debug("Calling Probe()=%#lx on driver %s",
70  Drv->Probe, Drv->Path);
71  Drv->ErrorCode = Drv->Probe();
72  if (Drv->ErrorCode < 0)
73  {
74  KPrint("Probe() failed for %s: %s",
75  Drv->Name, strerror(Drv->ErrorCode));
76  error("Failed to probe driver %s: %s",
77  Drv->Path, strerror(Drv->ErrorCode));
78 
79  Drv->vma->FreeAllPages();
80  continue;
81  }
82 
83  debug("Calling driver Entry()=%#lx function on driver %s",
84  Drv->Entry, Drv->Path);
85  Drv->ErrorCode = Drv->Entry();
86  if (Drv->ErrorCode < 0)
87  {
88  KPrint("Entry() failed for %s: %s",
89  Drv->Name, strerror(Drv->ErrorCode));
90  error("Failed to initialize driver %s: %s",
91  Drv->Path, strerror(Drv->ErrorCode));
92 
93  Drv->vma->FreeAllPages();
94  continue;
95  }
96 
97  debug("Loaded driver %s", Drv->Path);
98  Drv->Initialized = true;
99  }

114  void Manager::UnloadAllDrivers()
115  {
116  foreach (auto &var in Drivers)
117  {
118  DriverObject *Drv = &var.second;
119  if (!Drv->Initialized)
120  continue;
121 
122  debug("Unloading driver %s", Drv->Name);
123  int err = Drv->Final();
124  if (err < 0)
125  {
126  warn("Failed to unload driver %s: %s",
127  Drv->Name, strerror(err));
128  }
129 
130  if (!Drv->InterruptHandlers->empty())
131  {
132  foreach (auto &rInt in * Drv->InterruptHandlers)
133  {
134  Interrupts::RemoveHandler((void (*)(CPU::TrapFrame *))rInt.second);
135  }
136  Drv->InterruptHandlers->clear();
137  }
138 
139  delete Drv->vma, Drv->vma = nullptr;
140  delete Drv->InterruptHandlers, Drv->InterruptHandlers = nullptr;
141  }
142  Drivers.clear();
143  }

165  int Manager::LoadDriverFile(uintptr_t &EntryPoint,
166  uintptr_t &BaseAddress,
167  Memory::VirtualMemoryArea *dVma,
168  RefNode *rDrv)
169  {
170  Elf64_Ehdr ELFHeader;
171  rDrv->seek(0, SEEK_SET);
172  rDrv->read((uint8_t *)&ELFHeader, sizeof(Elf64_Ehdr));
173  if (ELFHeader.e_type != ET_DYN)
174  {
175  error("Driver %s is not a shared object", rDrv->node->FullPath);
176  return -ENOEXEC;
177  }
178 
179  trace("Loading driver %s in memory", rDrv->node->Name);
180 
181  BaseAddress = 0;
182  {
183  Elf64_Phdr ProgramBreakHeader{};
184  Elf64_Phdr ProgramHeader;
185 
186  size_t SegmentsSize = 0;
187  for (Elf64_Half i = 0; i < ELFHeader.e_phnum; i++)
188  {
189  rDrv->seek(ELFHeader.e_phoff + (i * sizeof(Elf64_Phdr)), SEEK_SET);
190  rDrv->read((uint8_t *)&ProgramHeader, sizeof(Elf64_Phdr));
191 
192  if (ProgramHeader.p_type == PT_LOAD ||
193  ProgramHeader.p_type == PT_DYNAMIC)
194  {
195  if (SegmentsSize < ProgramHeader.p_vaddr + ProgramHeader.p_memsz)
196  {
197  SegmentsSize = ProgramHeader.p_vaddr + ProgramHeader.p_memsz;
198  ProgramBreakHeader = ProgramHeader;
199  }
200  }
201  }
202  debug("SegmentsSize: %#lx", SegmentsSize);
203 
204  /* TODO: Check if this is correct and/or it needs more
205  complex calculations & allocations */
206  void *SegmentsAddress = dVma->RequestPages(TO_PAGES(SegmentsSize) + 1, true);
207  BaseAddress = (uintptr_t)SegmentsAddress;
208  debug("BaseAddress: %#lx, End: %#lx (%#lx)", BaseAddress,
209  BaseAddress + FROM_PAGES(TO_PAGES(SegmentsSize)),
210  SegmentsSize);
211 
212  for (Elf64_Half i = 0; i < ELFHeader.e_phnum; i++)
213  {
214  rDrv->seek(ELFHeader.e_phoff + (i * sizeof(Elf64_Phdr)), SEEK_SET);
215  rDrv->read((uint8_t *)&ProgramHeader, sizeof(Elf64_Phdr));
216 
217  switch (ProgramHeader.p_type)
218  {
219  case PT_LOAD:
220  {
221  /* Because this is ET_DYN, we can load the segments
222  anywhere we want. */
223  uintptr_t SegmentDestination = BaseAddress + ProgramHeader.p_vaddr;
224 
225  if (ProgramHeader.p_memsz == 0)
226  continue;
227 
228  debug("Copying PT_LOAD to %#lx-%#lx (%ld file bytes, %ld mem bytes)",
229  SegmentDestination, SegmentDestination + ProgramHeader.p_memsz,
230  ProgramHeader.p_filesz, ProgramHeader.p_memsz);
231 
232  if (ProgramHeader.p_filesz > 0)
233  {
234  rDrv->seek(ProgramHeader.p_offset, SEEK_SET);
235  rDrv->read((uint8_t *)SegmentDestination, ProgramHeader.p_filesz);
236  }
237 
238  if (ProgramHeader.p_memsz - ProgramHeader.p_filesz > 0)
239  {
240  void *zAddr = (void *)(SegmentDestination + ProgramHeader.p_filesz);
241  memset(zAddr, 0, ProgramHeader.p_memsz - ProgramHeader.p_filesz);
242  }
243  break;
244  }
245  case PT_DYNAMIC:
246  {
247  /* PT_DYNAMIC contains the dynamic linking information for the
248  executable or shared library. */
249 
250  uintptr_t DynamicSegmentDestination = BaseAddress + ProgramHeader.p_vaddr;
251 
252  if (ProgramHeader.p_memsz == 0)
253  continue;
254 
255  debug("Copying PT_DYNAMIC to %#lx-%#lx (%ld file bytes, %ld mem bytes)",
256  DynamicSegmentDestination, DynamicSegmentDestination + ProgramHeader.p_memsz,
257  ProgramHeader.p_filesz, ProgramHeader.p_memsz);
258 
259  if (ProgramHeader.p_filesz > 0)
260  {
261  rDrv->seek(ProgramHeader.p_offset, SEEK_SET);
262  rDrv->read((uint8_t *)DynamicSegmentDestination, ProgramHeader.p_filesz);
263  }
264 
265  if (ProgramHeader.p_memsz - ProgramHeader.p_filesz > 0)
266  {
267  void *zAddr = (void *)(DynamicSegmentDestination + ProgramHeader.p_filesz);
268  memset(zAddr, 0, ProgramHeader.p_memsz - ProgramHeader.p_filesz);
269  }
270  break;
271  }
272  default:
273  {
274  fixme("Unhandled program header type: %#lx",
275  ProgramHeader.p_type);
276  break;
277  }
278  }
279  }
280  }
281 
282  Elf64_Phdr ProgramHeader;
283  for (Elf64_Half i = 0; i < ELFHeader.e_phnum; i++)
284  {
285  rDrv->seek(ELFHeader.e_phoff + (i * sizeof(Elf64_Phdr)), SEEK_SET);
286  rDrv->read((uint8_t *)&ProgramHeader, sizeof(Elf64_Phdr));
287 
288  if (ProgramHeader.p_type == PT_DYNAMIC)
289  {
290  Elf64_Dyn *Dynamic = (Elf64_Dyn *)(BaseAddress + ProgramHeader.p_vaddr);
291  Elf64_Dyn *RelaSize = nullptr;
292  Elf64_Dyn *PltRelSize = nullptr;
293 
294  while (Dynamic->d_tag != DT_NULL)
295  {
296  switch (Dynamic->d_tag)
297  {
298  case DT_RELASZ:
299  RelaSize = Dynamic;
300  debug("RELA Size: %d", RelaSize->d_un.d_val / sizeof(Elf64_Rela));
301 
302  break;
303  case DT_PLTRELSZ:
304  PltRelSize = Dynamic;
305  debug("PLTRELSZ: %d", PltRelSize->d_un.d_val / sizeof(Elf64_Rela));
306 
307  break;
308  default:
309  break;
310  }
311 
312  Dynamic++;
313  }
314  Dynamic = (Elf64_Dyn *)(BaseAddress + ProgramHeader.p_vaddr);
315 
316  while (Dynamic->d_tag != DT_NULL)
317  {
318  switch (Dynamic->d_tag)
319  {
320  case DT_RELA: /* .rela.dyn */
321  {
322  if (!RelaSize)
323  {
324  error("DT_RELASZ is not set");
325  break;
326  }
327 
328  Elf64_Rela *Rela = (Elf64_Rela *)(BaseAddress + Dynamic->d_un.d_ptr);
329  for (size_t i = 0; i < (RelaSize->d_un.d_val / sizeof(Elf64_Rela)); i++)
330  {
331  Elf64_Rela *r = &Rela[i];
332  uintptr_t *RelocationAddress = (uintptr_t *)(BaseAddress + r->r_offset);
333  uintptr_t RelocationTarget = 0;
334 
335  switch (ELF64_R_TYPE(r->r_info))
336  {
337  case R_X86_64_GLOB_DAT:
338  case R_X86_64_JUMP_SLOT:
339  {
340  RelocationTarget = BaseAddress;
341  break;
342  }
343  case R_X86_64_RELATIVE:
344  case R_X86_64_64:
345  {
346  RelocationTarget = BaseAddress + r->r_addend;
347  break;
348  }
349  default:
350  {
351  fixme("Unhandled relocation type: %#lx",
352  ELF64_R_TYPE(r->r_info));
353  break;
354  }
355  }
356 
357  *RelocationAddress = RelocationTarget;
358 
359  debug("Relocated %#lx to %#lx",
360  r->r_offset, *RelocationAddress);
361  }
362  break;
363  }
364  case DT_PLTREL:
365  {
366  if (Dynamic->d_un.d_val != DT_RELA)
367  error("DT_PLTREL is not DT_RELA");
368  break;
369  }
370  case DT_JMPREL: /* .rela.plt */
371  {
372  if (!PltRelSize)
373  {
374  error("DT_PLTRELSZ is not set");
375  break;
376  }
377 
378  int fd = fopen(rDrv->node->FullPath, "r");
379 
380  std::vector<Elf64_Dyn> SymTab = Execute::ELFGetDynamicTag_x86_64(fd, DT_SYMTAB);
381  std::vector<Elf64_Dyn> StrTab = Execute::ELFGetDynamicTag_x86_64(fd, DT_STRTAB);
382  Elf64_Sym *_SymTab = (Elf64_Sym *)((uintptr_t)BaseAddress + SymTab[0].d_un.d_ptr);
383  char *DynStr = (char *)((uintptr_t)BaseAddress + StrTab[0].d_un.d_ptr);
384  UNUSED(DynStr);
385  fclose(fd);
386 
387  Elf64_Rela *Rela = (Elf64_Rela *)(BaseAddress + Dynamic->d_un.d_ptr);
388  for (size_t i = 0; i < (PltRelSize->d_un.d_val / sizeof(Elf64_Rela)); i++)
389  {
390  Elf64_Rela *r = &Rela[i];
391  uintptr_t *RelocationAddress = (uintptr_t *)(BaseAddress + r->r_offset);
392  uintptr_t RelocationTarget = 0;
393 
394  switch (ELF64_R_TYPE(r->r_info))
395  {
396  case R_X86_64_JUMP_SLOT:
397  {
398  Elf64_Xword SymIndex = ELF64_R_SYM(r->r_info);
399  Elf64_Sym *Sym = _SymTab + SymIndex;
400 
401 #ifdef DEBUG
402  const char *SymbolName = DynStr + Sym->st_name;
403  debug("Symbol %s at %#lx", SymbolName, Sym->st_value);
404 #endif
405 
406  RelocationTarget = BaseAddress + Sym->st_value;
407  break;
408  }
409  default:
410  {
411  fixme("Unhandled relocation type: %#lx",
412  ELF64_R_TYPE(r->r_info));
413  break;
414  }
415  }
416 
417  *RelocationAddress = RelocationTarget;
418 
419  debug("Relocated %#lx to %#lx",
420  r->r_offset, *RelocationAddress);
421  }
422  break;
423  }
424  case DT_SYMTAB:
425  {
426  fixme("DT_SYMTAB");
427  break;
428  int fd = fopen(rDrv->node->FullPath, "r");
429 
430  std::vector<Elf64_Dyn> SymTab = Execute::ELFGetDynamicTag_x86_64(fd, DT_SYMTAB);
431  std::vector<Elf64_Dyn> StrTab = Execute::ELFGetDynamicTag_x86_64(fd, DT_STRTAB);
432  Elf64_Sym *_SymTab = (Elf64_Sym *)((uintptr_t)BaseAddress + SymTab[0].d_un.d_ptr);
433  char *DynStr = (char *)((uintptr_t)BaseAddress + StrTab[0].d_un.d_ptr);
434  UNUSED(DynStr);
435  fclose(fd);
436 
437  size_t symtabEntrySize = 0;
438  Elf64_Dyn *entrySizeDyn = Dynamic;
439  while (entrySizeDyn->d_tag != DT_NULL)
440  {
441  if (entrySizeDyn->d_tag == DT_SYMENT)
442  {
443  symtabEntrySize = entrySizeDyn->d_un.d_val;
444  break;
445  }
446  entrySizeDyn++;
447  }
448 
449  if (symtabEntrySize == 0)
450  {
451  fixme("No information about symbol entry size");
452  break;
453  }
454 
455  size_t numSymbols = Dynamic->d_un.d_val / symtabEntrySize;
456 
457  for (size_t i = 0; i < numSymbols; i++)
458  {
459  Elf64_Sym *s = &_SymTab[i];
460  if (s->st_name == 0)
461  continue;
462 
463 #ifdef DEBUG
464  const char *SymbolName = (const char *)(DynStr + s->st_name);
465  debug("%d: Symbol %s at %#lx", i, SymbolName, s->st_value);
466 #endif
471  // if (strcmp(SymbolName, "DriverProbe") == 0)
472  // {
473  // Drivers[MajorIDCounter].Probe = (int (*)())(BaseAddress + s->st_value);
474  // debug("Found probe function at %#lx", Drivers[MajorIDCounter].Probe);
475  // }
476  }
477  break;
478  }
479  default:
480  {
481  fixme("Unhandled dynamic tag: %#lx",
482  Dynamic->d_tag);
483  break;
484  }
485  }
486  Dynamic++;
487  }
488  }
489  }
490 
491  EntryPoint = ELFHeader.e_entry;
492  EntryPoint += BaseAddress;
493 
494  debug("Driver %s has entry point %#lx and base %#lx",
495  rDrv->node->FullPath, EntryPoint, BaseAddress);
496
