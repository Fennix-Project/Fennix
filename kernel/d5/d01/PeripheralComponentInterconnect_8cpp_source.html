33  const char *u8ToHexString(uint8_t Value)
34  {
35  static char Buffer[3];
36  memset(Buffer, 0, 3);
37  for (size_t i = 0; i < 2; i++)
38  {
39  uint8_t Digit = (Value >> (4 - (i * 4))) & 0xF;
40  if (Digit < 10)
41  Buffer[i] = s_cst(char, '0' + Digit);
42  else
43  Buffer[i] = s_cst(char, 'A' + (Digit - 10));
44  }
45  return Buffer;
46  }
47 
48  const char *u32ToHexString(uint32_t Value)
49  {
50  static char Buffer[9];
51  memset(Buffer, 0, 9);
52  for (size_t i = 0; i < 8; i++)
53  {
54  uint8_t Digit = (Value >> (28 - (i * 4))) & 0xF;
55  if (Digit < 10)
56  Buffer[i] = s_cst(char, '0' + Digit);
57  else
58  Buffer[i] = s_cst(char, 'A' + (Digit - 10));
59  }
60  return Buffer;
61  }

791  void PCI::MapPCIAddresses(PCIDeviceHeader *PCIDevice, Memory::PageTable *Table)
792  {
793  debug("Header Type: %d", PCIDevice->HeaderType);
794  switch (PCIDevice->HeaderType)
795  {
796  case 0: /* PCI Header 0 */
797  {
798  uint32_t BAR[6] = {0};
799  size_t BARsSize[6] = {0};
800 
801  BAR[0] = ((PCIHeader0 *)PCIDevice)->BAR0;
802  BAR[1] = ((PCIHeader0 *)PCIDevice)->BAR1;
803  BAR[2] = ((PCIHeader0 *)PCIDevice)->BAR2;
804  BAR[3] = ((PCIHeader0 *)PCIDevice)->BAR3;
805  BAR[4] = ((PCIHeader0 *)PCIDevice)->BAR4;
806  BAR[5] = ((PCIHeader0 *)PCIDevice)->BAR5;
807 
808  debug("Type: %d; IOBase: %#lx; MemoryBase: %#lx",
809  BAR[0] & 1,
810  BAR[1] & (~3),
811  BAR[0] & (~15));
812 
813  /* BARs Size */
814  for (short i = 0; i < 6; i++)
815  {
816  if (BAR[i] == 0)
817  continue;
818 
819  if ((BAR[i] & 1) == 0) /* Memory Base */
820  {
821  ((PCIHeader0 *)PCIDevice)->BAR0 = 0xFFFFFFFF;
822  size_t size = ((PCIHeader0 *)PCIDevice)->BAR0;
823  ((PCIHeader0 *)PCIDevice)->BAR0 = BAR[i];
824  BARsSize[i] = size & (~15);
825  BARsSize[i] = ~BARsSize[i] + 1;
826  BARsSize[i] = BARsSize[i] & 0xFFFFFFFF;
827  debug("BAR%d %#lx size: %d", i, BAR[i], BARsSize[i]);
828  }
829  else if ((BAR[i] & 1) == 1) /* I/O Base */
830  {
831  ((PCIHeader0 *)PCIDevice)->BAR1 = 0xFFFFFFFF;
832  size_t size = ((PCIHeader0 *)PCIDevice)->BAR1;
833  ((PCIHeader0 *)PCIDevice)->BAR1 = BAR[i];
834  BARsSize[i] = size & (~3);
835  BARsSize[i] = ~BARsSize[i] + 1;
836  BARsSize[i] = BARsSize[i] & 0xFFFF;
837  debug("BAR%d %#lx size: %d", i, BAR[i], BARsSize[i]);
838  }
839  }
840 
841  /* Mapping the BARs */
842  for (short i = 0; i < 6; i++)
843  {
844  if (BAR[i] == 0)
845  continue;
846 
847  if ((BAR[i] & 1) == 0) /* Memory Base */
848  {
849  uintptr_t BARBase = BAR[i] & (~15);
850  size_t BARSize = BARsSize[i];
851 
852  debug("Mapping BAR%d %#lx-%#lx", i, BARBase, BARBase + BARSize);
853  Memory::Virtual(Table).Map((void *)BARBase, (void *)BARBase, BARSize,
854  Memory::PTFlag::RW | Memory::PTFlag::PWT);
855  }
856  else if ((BAR[i] & 1) == 1) /* I/O Base */
857  {
858  uintptr_t BARBase = BAR[i] & (~3);
859  size_t BARSize = BARsSize[i];
860 
861  debug("Mapping BAR%d %#x-%#x", i, BARBase, BARBase + BARSize);
862  Memory::Virtual(Table).Map((void *)BARBase, (void *)BARBase, BARSize,
863  Memory::PTFlag::RW | Memory::PTFlag::PWT);
864  }
865  }
866  break;
867  }
868  case 1: /* PCI Header 1 (PCI-to-PCI Bridge) */
869  {
870  fixme("PCI Header 1 (PCI-to-PCI Bridge) not implemented yet");
871  break;
872  }
873  case 2: /* PCI Header 2 (PCI-to-CardBus Bridge) */
874  {
875  fixme("PCI Header 2 (PCI-to-CardBus Bridge) not implemented yet");
876  break;
877  }
878  default:
879  {
880  error("Unknown header type %d", PCIDevice->HeaderType);
881  return;
882  }
883  }
