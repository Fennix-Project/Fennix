70  bool MemMgr::Add(void *Address, size_t Count)
71  {
72  SmartLock(MgrLock);
73  if (Address == nullptr)
74  {
75  error("Address is null!");
76  return false;
77  }
78 
79  if (Count == 0)
80  {
81  error("Count is 0!");
82  return false;
83  }
84 
85  for (size_t i = 0; i < AllocatedPagesList.size(); i++)
86  {
87  if (AllocatedPagesList[i].Address == Address)
88  {
89  error("Address already exists!");
90  return false;
91  }
92  else if ((uintptr_t)Address < (uintptr_t)AllocatedPagesList[i].Address)
93  {
94  if ((uintptr_t)Address + (Count * PAGE_SIZE) > (uintptr_t)AllocatedPagesList[i].Address)
95  {
96  error("Address intersects with an allocated page!");
97  return false;
98  }
99  }
100  else
101  {
102  if ((uintptr_t)AllocatedPagesList[i].Address + (AllocatedPagesList[i].PageCount * PAGE_SIZE) > (uintptr_t)Address)
103  {
104  error("Address intersects with an allocated page!");
105  return false;
106  }
107  }
108  }
109 
110  if (this->Directory)
111  {
112  char FileName[64];
113 #if defined(a64) || defined(aa64)
114  sprintf(FileName, "%lx-%ld", (uintptr_t)Address, Count);
115 #elif defined(a32)
116  sprintf(FileName, "%x-%ld", (uintptr_t)Address, Count);
117 #endif
118  VirtualFileSystem::Node *n = vfs->Create(FileName, VirtualFileSystem::NodeFlags::FILE, this->Directory);
119  if (n)
120  {
121  n->Address = (uintptr_t)Address;
122  n->Length = Count * PAGE_SIZE;
123  n->Operator = &mem_op;
124  }
125  }
126 
127  AllocatedPagesList.push_back({Address, Count});
128  return true;
129  }

174  void MemMgr::FreePages(void *Address, size_t Count)
175  {
176  SmartLock(MgrLock);
177  forItr(itr, AllocatedPagesList)
178  {
179  if (itr->Address == Address)
180  {
187  if (itr->PageCount != Count)
188  {
189  error("Page count mismatch! (Allocated: %lld, Requested: %lld)", itr->PageCount, Count);
190  return;
191  }
192 
193  KernelAllocator.FreePages(Address, Count);
194 
195  Memory::Virtual vmm = Memory::Virtual(this->Table);
196  for (size_t i = 0; i < Count; i++)
197  {
198  void *AddressToMap = (void *)((uintptr_t)Address + (i * PAGE_SIZE));
199  vmm.Remap(AddressToMap, AddressToMap, Memory::PTFlag::RW);
200  // vmm.Unmap((void *)((uintptr_t)Address + (i * PAGE_SIZE)));
201  }
202 
203  if (this->Directory)
204  {
205  char FileName[64];

207  sprintf(FileName, "%lx-%ld", (uintptr_t)Address, Count);
