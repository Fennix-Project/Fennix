44 __no_sanitize("alignment") size_t MaxNameLength(FileNode *nodes)
45 {
46  size_t maxLength = 0;
47 
48  kdirent *dirBuffer = new kdirent[16];
49  ssize_t read = 0;
50  off_t offset = 0;
51  while ((read = nodes->ReadDir(dirBuffer, sizeof(kdirent) * 16, offset, LONG_MAX)) > 0)
52  {
53  if (read / sizeof(kdirent) == 0)
54  break;
55 
56  off_t bufOffset = 0;
57  debug("There are %ld entries in this directory", read / sizeof(kdirent));
58  for (size_t i = 0; i < read / sizeof(kdirent); i++)
59  {
60  kdirent *dirent = (kdirent *)((uintptr_t)dirBuffer + bufOffset);
61  if (dirent->d_reclen == 0)
62  break;
63  bufOffset += dirent->d_reclen;
64  maxLength = std::max(maxLength, strlen(dirent->d_name));
65  debug("dirent->d_name: %s (max length: %ld)", dirent->d_name, maxLength);
66  }
67  offset += read / sizeof(kdirent);
68  }
69  delete[] dirBuffer;
70  return maxLength;
71 }
