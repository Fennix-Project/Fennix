358 EXTERNC __no_stack_protector NIF void Entry(BootInfo *Info)
359 {
360  trace("Hello, World!");
361 
362  if (strcmp(CPU::Hypervisor(), x86_CPUID_VENDOR_TCG) == 0)
363  {
364  debug("\n\n----------------------------------------\nDEBUGGER DETECTED\n----------------------------------------\n\n");
365  DebuggerIsAttached = true;
366  }
367 
368  memcpy(&bInfo, Info, sizeof(BootInfo));
369  debug("BootInfo structure is at %p", &bInfo);
370 
371  // https://wiki.osdev.org/Calling_Global_Constructors
372  trace("There are %d constructors to call",
373  __init_array_end - __init_array_start);
374  for (CallPtr *func = __init_array_start; func != __init_array_end; func++)
375  (*func)();
376 
377 #ifdef a86
378  if (!bInfo.SMBIOSPtr)
379  {
380  trace("SMBIOS was not provided by the bootloader. Trying to find it manually.");
381  for (uintptr_t i = 0xF0000; i < 0x100000; i += 16)
382  {
383  if (memcmp((void *)i, "_SM_", 4) == 0 ||
384  memcmp((void *)i, "_SM3_", 5) == 0)
385  {
386  bInfo.SMBIOSPtr = (void *)i;
387  trace("Found SMBIOS at %#lx", i);
388  }
389  }
390  }
391 
392  if (!bInfo.RSDP)
393  {
394  trace("RSDP was not provided by the bootloader. Trying to find it manually.");
395  /* FIXME: Not always shifting by 4 will work. */
396  uintptr_t EBDABase = (uintptr_t)mminw((void *)0x40E) << 4;
397 
398  for (uintptr_t ptr = EBDABase;
399  ptr < 0x100000; /* 1MB */
400  ptr += 16)
401  {
402  if (unlikely(ptr == EBDABase + 0x400))
403  {
404  trace("EBDA is full. Trying to find RSDP in the BIOS area.");
405  break;
406  }
407 
408  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
409  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
410  {
411  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
412  trace("Found RSDP at %#lx", rsdp);
413  }
414  }
415 
416  for (uintptr_t ptr = 0xE0000;
417  ptr < 0x100000; /* 1MB */
418  ptr += 16)
419  {
420  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
421  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
422  {
423  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
424  trace("Found RSDP at %#lx", rsdp);
425  }
426  }
427  }
428 #endif
429 
430  InitializeMemoryManagement();
431 
432  void *KernelStackAddress = KernelAllocator.RequestPages(TO_PAGES(STACK_SIZE));
433  uintptr_t KernelStack = (uintptr_t)KernelStackAddress + STACK_SIZE - 0x10;
434  debug("Kernel stack: %#lx-%#lx", KernelStackAddress, KernelStack);
435 #if defined(a64)
436  asmv("mov %0, %%rsp"
437  :
438  : "r"(KernelStack)
439  : "memory");
440  asmv("mov $0, %rbp");
441 #elif defined(a32)
442  asmv("mov %0, %%esp"
443  :
444  : "r"(KernelStack)
445  : "memory");
446  asmv("mov $0, %ebp");
447 #elif defined(aa64)
448 #warning "Kernel stack is not set!"
449 #endif
450

455  */
