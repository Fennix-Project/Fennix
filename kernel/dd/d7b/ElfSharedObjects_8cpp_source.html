75  bool AddLibrary(char *Identifier,
76  VirtualFileSystem::File &ExFile,
77  const Memory::Virtual &vmm)
78  {
79  SmartLock(ExecuteServiceLock);
80  SharedLibrary sl;
81 
82  foreach (auto lib in Libs)
83  {
84  if (strcmp(lib.Identifier, Identifier) == 0)
85  {
86  debug("Library %s already loaded", Identifier);
87  lib.RefCount++;
88  return true;
89  }
90  }
91 
92  PCB *Process = TaskManager->GetCurrentProcess();
93  ELFObject *obj = new ELFObject(vfs->GetPathFromNode(ExFile.GetNode()).get(), Process, true);
94  if (!obj->IsValid())
95  {
96  error("Failed to load dynamic ELF");
97  return false;
98  }
99  ELFBaseLoad bl = obj->GetBaseLoadInfo();
100 
101  strncpy(sl.Identifier, Identifier, sizeof(sl.Identifier) - 1);
102  char *AbsolutePath = vfs->GetPathFromNode(ExFile.GetNode()).get();
103  strncpy(sl.Path, AbsolutePath, sizeof(sl.Path) - 1);
104  sl.Timeout = TimeManager->CalculateTarget(10, Time::Units::Minutes);
105  sl.RefCount = 0;
106  sl.MemoryImage = (uintptr_t)bl.MemoryImage;
107  sl.Length = ExFile.GetLength();
108  Libs.push_back(sl);
109  debug("Library %s loaded at %#lx", Identifier, sl.MemoryImage);
110 
111  if (bl.InstructionPointer)
112  {
113  TCB *Thread = TaskManager->CreateThread(Process,
114  bl.InstructionPointer,
115  nullptr,
116  nullptr,
117  bl.auxv,
118  Process->Info.Architecture,
119  Process->Info.Compatibility,
120  true);
121  Thread->Rename(Identifier);
122  Thread->Status = TaskStatus::Ready;
123 
124  foreach (Memory::MemMgr::AllocatedPages p in bl.TmpMem->GetAllocatedPagesList())
125  {
126  Thread->Memory->Add(p.Address, p.PageCount);
127  bl.TmpMem->DetachAddress(p.Address);
128  }
129  }
130  else
131  {
132  foreach (Memory::MemMgr::AllocatedPages p in bl.TmpMem->GetAllocatedPagesList())
133  {
134  /* FIXME: MEMORY LEAK */

136  bl.TmpMem->DetachAddress(p.Address);
137  fixme("Potential memory leak. (%#lx - %ld)",
138  p.Address, p.PageCount);
139  }
140  }
141  return true;
142  }
