169 void BootLogoAnimationThread()
170 {
171  char BootAnimPath[16];
172  while (FrameCount < 27)
173  {
174  sprintf(BootAnimPath, "%d.tga", FrameCount);
175  File ba = bootanim_vfs->Open(BootAnimPath);
176  if (!ba.IsOK())
177  {
178  bootanim_vfs->Close(ba);
179  debug("Failed to load boot animation frame %s", BootAnimPath);
180  break;
181  }
182 
183  FrameSizes[FrameCount] = s_cst(uint32_t, ba.node->Length);
184  Frames[FrameCount] = new uint8_t[ba.node->Length];
185  memcpy((void *)Frames[FrameCount], (void *)ba.node->Address, ba.node->Length);
186  bootanim_vfs->Close(ba);
187  FrameCount++;
188  }
189 
190  uint32_t DispX = Display->GetBuffer(1)->Width;
191  uint32_t DispY = Display->GetBuffer(1)->Height;
192 
193  for (size_t i = 1; i < FrameCount; i++)
194  {
195  int x, y, channels;
196 
197  if (!stbi_info_from_memory((uint8_t *)Frames[i], FrameSizes[i], &x, &y, &channels))
198  continue;
199 
200  uint8_t *img = stbi_load_from_memory((uint8_t *)Frames[i], FrameSizes[i], &x, &y, &channels, STBI_rgb_alpha);
201 
202  if (img == NULL)
203  continue;
204 
205  int offsetX = DispX / 2 - x / 2;
206  int offsetY = DispY / 2 - y / 2;
207 
208  for (int i = 0; i < x * y; i++)
209  {
210  uint32_t pixel = ((uint32_t *)img)[i];
211  int r = (pixel >> 16) & 0xFF;
212  int g = (pixel >> 8) & 0xFF;
213  int b = (pixel >> 0) & 0xFF;
214  int a = (pixel >> 24) & 0xFF;
215 
216  if (a != 0xFF)
217  {
218  r = (r * a) / 0xFF;
219  g = (g * a) / 0xFF;
220  b = (b * a) / 0xFF;
221  }
222 
223  Display->SetPixel((i % x) + offsetX, (i / x) + offsetY, (r << 16) | (g << 8) | (b << 0), 1);
224  }
225 
226  free(img);
227  Display->SetBuffer(1);
228  TaskManager->Sleep(50);
229  }
230 
231  int brightness = 100;
232  while (brightness >= 0)
233  {
234  brightness -= 10;
235  Display->SetBrightness(brightness, 1);
236  Display->SetBuffer(1);
237  TaskManager->Sleep(5);
238  }
239 }
