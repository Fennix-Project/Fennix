18 PAGE_TABLE_SIZE = 0x4

20 .code32
21 .section .bootstrap.data, "a"
22 .align 0x1000
23 .global BootPageTable
24 BootPageTable:
25  .space 0x10000 /* 0x4000 bytes will be used in UpdatePageTable */
26 
27 .section .bootstrap.text, "a"
28 .global UpdatePageTable
29 UpdatePageTable:
30  mov $(BootPageTable + 0x0000), %edi /* First PML4E */
31  mov $(BootPageTable + 0x1000), %eax /* First PDPTE */
32  or $0x3, %eax /* Bitwise OR on eax (PDPTE) with 11b (Present, Write) */
33  mov %eax, (%edi) /* Write 11b to PML4E */
34 
35  mov $(BootPageTable + 0x1000), %edi /* First PDPTE */
36  mov $(BootPageTable + 0x2000), %eax /* First PDE */
37  or $0x3, %eax /* Bitwise OR on eax (PDE) with 11b (Present, Write) */
38 
39  mov $PAGE_TABLE_SIZE, %ecx /* For loop instruction */
40  mov $0x0, %ebx /* Value to store in the next 4 bytes */
41  .FillPageTableLevel3:
42  mov %eax, (%edi) /* Store modified PDE in PDPTE */
43  mov %ebx, 0x4(%edi) /* Store the ebx value in the next 4 bytes */
44  add $0x1000, %eax /* Increment (page size) */
45  adc $0x0, %ebx /* Add 0 to carry flag */
46  add $0x8, %edi /* Add 8 to edi (next PDE) */
47  loop .FillPageTableLevel3 /* Loop until ecx is 0 */
48 
49  mov $(BootPageTable + 0x2000), %edi /* First PDE */
50  mov $0x83, %eax /* Present, Write, Large Page */
51 
52  mov $(512 * PAGE_TABLE_SIZE), %ecx /* For loop instruction */
53  mov $0x0, %ebx /* Value to store in the next 4 bytes */
54  .FillPageTableLevel2:
55  mov %eax, (%edi) /* Store modified PDE in PDPTE */
56  mov %ebx, 0x4(%edi) /* Store the ebx value in the next 4 bytes */
57  add $0x200000, %eax /* Increment (page size) */
58  adc $0x0, %ebx /* Add 0 (carry flag) to ebx to increment if there was a carry */
59  add $0x8, %edi /* Add 8 to edi (next PDE) */
60  loop .FillPageTableLevel2 /* Loop until ecx is 0 */
61 
62  ret
