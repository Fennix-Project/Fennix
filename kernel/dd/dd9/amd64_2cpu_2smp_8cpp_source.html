51 nsa CPUData *GetCurrentCPU()
52 {
53  if (unlikely(!Interrupts::apic[0]))
54  return &CPUs[0]; /* No APIC means we are on the BSP. */
55 
56  APIC::APIC *apic = (APIC::APIC *)Interrupts::apic[0];
57  int CoreID = 0;
58  if (CPUEnabled.load(std::memory_order_acquire) == true)
59  {
60  Memory::SwapPT swap =
61  Memory::SwapPT(KernelPageTable, thisPageTable);
62 
63  if (apic->x2APIC)
64  CoreID = int(CPU::x64::rdmsr(CPU::x64::MSR_X2APIC_APICID));
65  else
66  CoreID = apic->Read(APIC::APIC_ID) >> 24;
67  }
68 
69  if (unlikely((&CPUs[CoreID])->IsActive != true))
70  {
71  error("CPU %d is not active!", CoreID);
72  assert((&CPUs[0])->IsActive == true); /* We can't continue without the BSP. */
73  return &CPUs[0];
74  }
75 
76  assert((&CPUs[CoreID])->Checksum == CPU_DATA_CHECKSUM); /* This should never happen. */
77  return &CPUs[CoreID];
78 }

101  void Initialize(void *_madt)
102  {
103  if (!_madt)
104  {
105  error("MADT is NULL");
106  return;
107  }
108 
109  ACPI::MADT *madt = (ACPI::MADT *)_madt;
110 
111  if (madt->lapic.size() < 1)
112  {
113  error("No CPUs found!");
114  return;
115  }
116 
117  int Cores = madt->CPUCores + 1;
118 
119  if (Config.Cores > madt->CPUCores + 1)
120  KPrint("More cores requested than available. Using %d cores",
121  madt->CPUCores + 1);
122  else if (Config.Cores != 0)
123  Cores = Config.Cores;
124 
125  CPUCores = Cores;
126 
127  uint64_t TrampolineLength = (uintptr_t)&_trampoline_end -
128  (uintptr_t)&_trampoline_start;
129  Memory::Virtual().Map(0x0, 0x0, Memory::PTFlag::RW);
130  /* We reserved the TRAMPOLINE_START address inside Physical class. */
131  Memory::Virtual().Map((void *)TRAMPOLINE_START,
132  (void *)TRAMPOLINE_START,
133  TrampolineLength, Memory::PTFlag::RW);
134  memcpy((void *)TRAMPOLINE_START, &_trampoline_start, TrampolineLength);
135  debug("Trampoline address: %#lx-%#lx",
136  TRAMPOLINE_START,
137  TRAMPOLINE_START + TrampolineLength);
138 
139  void *CPUTmpStack = KernelAllocator.RequestPages(TO_PAGES(STACK_SIZE + 1));
140  asmv("sgdt [0x580]");
141  asmv("sidt [0x590]");
142  VPOKE(uintptr_t, STACK) = (uintptr_t)CPUTmpStack + STACK_SIZE;
143  VPOKE(uintptr_t, PAGE_TABLE) = (uintptr_t)KernelPageTable;
144  VPOKE(uintptr_t, START_ADDR) = (uintptr_t)&StartCPU;
145 
146  for (int i = 0; i < Cores; i++)
147  {
148  ACPI::MADT::LocalAPIC *lapic = madt->lapic[i];
149  APIC::APIC *apic = (APIC::APIC *)Interrupts::apic[0];
150 
151  debug("Initializing CPU %d", lapic->APICId);
152  uint8_t APIC_ID = 0;
153  if (apic->x2APIC)
154  APIC_ID = uint8_t(CPU::x64::rdmsr(CPU::x64::MSR_X2APIC_APICID));
155  else
156  APIC_ID = uint8_t(apic->Read(APIC::APIC_ID) >> 24);
157 
158  if (APIC_ID != lapic->APICId)
159  {
160  VPOKE(int, CORE) = i;
161  if (!apic->x2APIC)
162  {
163  APIC::InterruptCommandRegister icr{};
164  icr.MT = APIC::INIT;
165  icr.DES = lapic->APICId;
166  apic->ICR(icr);
167  }
168 
169  apic->SendInitIPI(lapic->APICId);
170  TimeManager->Sleep(20, Time::Units::Milliseconds);
171  apic->SendStartupIPI(lapic->APICId, TRAMPOLINE_START);
172  debug("Waiting for CPU %d to load...", lapic->APICId);
173 
174  uint64_t Timeout = TimeManager->CalculateTarget(2, Time::Units::Seconds);
175  while (CPUEnabled.load(std::memory_order_acquire) == false)
176  {
177  if (TimeManager->GetCounter() > Timeout)
178  {
179  error("CPU %d failed to load!", lapic->APICId);
180  KPrint("\eFF8C19CPU \e8888FF%d \eFF8C19failed to load!",
181  lapic->APICId);
182  break;
183  }
184  CPU::Pause();
185  }
186  trace("CPU %d loaded.", lapic->APICId);
187  CPUEnabled.store(false, std::memory_order_release);
188  }
189  else
190  KPrint("\e058C19CPU \e8888FF%d \e058C19is the BSP", lapic->APICId);
191  }
192 
193  KernelAllocator.FreePages(CPUTmpStack, TO_PAGES(STACK_SIZE + 1));
194  /* We are going to unmap the page after we are done with it. */
195  Memory::Virtual().Unmap(0x0);
196  CPUEnabled.store(true, std::memory_order_release);
197  }
198 }
