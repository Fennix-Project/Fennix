73 uint64_t GetUsage(uint64_t OldSystemTime, Tasking::TaskInfo *Info)
74 {
75  /* https://github.com/reactos/reactos/blob/560671a784c1e0e0aa7590df5e0598c1e2f41f5a/base/applications/taskmgr/perfdata.c#L347 */
76  if (Info->OldKernelTime || Info->OldUserTime)
77  {
78  uint64_t SystemTime = TimeManager->GetCounter() - OldSystemTime;
79  uint64_t CurrentTime = Info->KernelTime + Info->UserTime;
80  uint64_t OldTime = Info->OldKernelTime + Info->OldUserTime;
81  uint64_t CpuUsage = (CurrentTime - OldTime) / SystemTime;
82  CpuUsage = CpuUsage * 100;
83 
84  // debug("CurrentTime: %ld OldTime: %ld Time Diff: %ld Usage: %ld%%",
85  // CurrentTime, OldTime, SystemTime, CpuUsage);
86 
87  Info->OldKernelTime = Info->KernelTime;
88  Info->OldUserTime = Info->UserTime;
89  return CpuUsage;
90  }
91  Info->OldKernelTime = Info->KernelTime;
92  Info->OldUserTime = Info->UserTime;
93  return 0;
94 }
