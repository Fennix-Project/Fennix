37  __no_sanitize("alignment") void Physical::ReserveEssentials()
38  {
39  debug("Reserving pages...");
40  /* The bootloader won't give us the entire mapping, so we
41  reserve everything and then unreserve the usable pages. */
42  this->ReservePages(0, TO_PAGES(bInfo.Memory.Size));
43  debug("Unreserving usable pages...");
44 
45  for (uint64_t i = 0; i < bInfo.Memory.Entries; i++)
46  {
47  if (bInfo.Memory.Entry[i].Type == Usable)
48  {
49  if (uintptr_t(bInfo.Memory.Entry[i].BaseAddress) <= 0xFFFFF)
50  continue;
51 
52  this->UnreservePages(bInfo.Memory.Entry[i].BaseAddress,
53  TO_PAGES(bInfo.Memory.Entry[i].Length));
54  }
55  }
56 
57  debug("Reserving 0x0-0xFFFFF range...");
58  // this->ReservePage((void *)0x0); /* Trampoline stack, gdt, idt, etc... */
59  // this->ReservePages((void *)0x2000, 4); /* TRAMPOLINE_START */
60 
61  /* Reserve the lower part of memory. (0x0-0xFFFFF)
62  This includes: BIOS, EBDA, VGA, SMP, etc...
63  https://wiki.osdev.org/Memory_Map_(x86)
64  */
65  this->ReservePages((void *)0x0, TO_PAGES(0xFFFFF));
66 
67  debug("Reserving bitmap region %#lx-%#lx...",
68  PageBitmap.Buffer,
69  (void *)((uintptr_t)PageBitmap.Buffer + PageBitmap.Size));
70 
71  this->ReservePages(PageBitmap.Buffer, TO_PAGES(PageBitmap.Size));
72

74  bInfo.Kernel.PhysicalBase,
75  (void *)((uintptr_t)bInfo.Kernel.PhysicalBase + bInfo.Kernel.Size));
76 
77  this->ReservePages(bInfo.Kernel.PhysicalBase, TO_PAGES(bInfo.Kernel.Size));
78 
79  debug("Reserving kernel file and symbols...");
80  if (bInfo.Kernel.FileBase)
81  this->ReservePages(bInfo.Kernel.FileBase, TO_PAGES(bInfo.Kernel.Size));
82 
83  if (bInfo.Kernel.Symbols.Num &&
84  bInfo.Kernel.Symbols.EntSize &&
85  bInfo.Kernel.Symbols.Shndx)
86  {
87  char *sections = reinterpret_cast<char *>(bInfo.Kernel.Symbols.Sections);

105

107  str->sh_type == SHT_STRTAB)
108  {
109  Symbols = (Elf_Sym *)sym->sh_addr;
110  StringAddress = (uint8_t *)str->sh_addr;
111  SymbolSize = (int)sym->sh_size;
112  StringSize = (int)str->sh_size;
113  debug("Symbol table found, %d entries",

115  break;
116  }
117  }
118

120  {
121  debug("Reserving symbol table region %#lx-%#lx...",
122  Symbols, (void *)((uintptr_t)Symbols + SymbolSize));
123  this->ReservePages(Symbols, TO_PAGES(SymbolSize));
124  }
125

128  debug("Reserving string table region %#lx-%#lx...",
129  StringAddress, (void *)((uintptr_t)StringAddress + StringSize));

143  (void *)((uintptr_t)bInfo.Modules[i].Address + bInfo.Modules[i].Size));

146  TO_PAGES(bInfo.Modules[i].Size));

148

174  }

191  // TODO: Should I be concerned about unaligned memory access?
192  ACPI::ACPI::ACPIHeader *SDTHdr = nullptr;
193  if (XSDT)
194  SDTHdr =
195  (ACPI::ACPI::ACPIHeader *)(*(uint64_t *)((uint64_t)ACPIPtr +
196  sizeof(ACPI::ACPI::ACPIHeader) +
197  (t * 8)));
198  else
199  SDTHdr =
200  (ACPI::ACPI::ACPIHeader *)(*(uint32_t *)((uint64_t)ACPIPtr +
201  sizeof(ACPI::ACPI::ACPIHeader) +
202  (t * 4)));
203 #pragma GCC diagnostic pop
204  this->ReservePages(SDTHdr, TO_PAGES(SDTHdr->Length));
205  }
206 
207 #elif defined(aa64)
208 #endif
209  }
