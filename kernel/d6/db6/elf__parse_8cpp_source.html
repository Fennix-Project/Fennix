61  Elf64_Sym *ELFLookupSymbol(Elf64_Ehdr *Header, std::string Name)
62  {
63  Elf64_Shdr *SymbolTable = nullptr;
64  Elf64_Shdr *StringTable = nullptr;
65 
66  for (Elf64_Half i = 0; i < Header->e_shnum; i++)
67  {
68  Elf64_Shdr *shdr = GetELFSection(Header, i);
69  switch (shdr->sh_type)
70  {
71  case SHT_SYMTAB:
72  SymbolTable = shdr;
73  StringTable = GetELFSection(Header, shdr->sh_link);
74  break;
75  default:
76  {
77  break;
78  }
79  }
80  }
81 
82  if (SymbolTable == nullptr || StringTable == nullptr)
83  return nullptr;
84 
85  for (size_t i = 0; i < (SymbolTable->sh_size / sizeof(Elf64_Sym)); i++)
86  {
87  Elf64_Sym *Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
88  char *String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
89  if (strcmp(String, Name.c_str()) == 0)
90  return Symbol;
91  }
92  return nullptr;
93  }

146  uintptr_t ELFGetSymbolValue(Elf64_Ehdr *Header, uint64_t Table, uint64_t Index)
147  {
148 #if defined(a64)
149  if (Table == SHN_UNDEF || Index == SHN_UNDEF)
150  return 0;
151  Elf64_Shdr *SymbolTable = GetELFSection(Header, Table);
152 
153  uint64_t STEntries = SymbolTable->sh_size / SymbolTable->sh_entsize;
154  if (Index >= STEntries)
155  {
156  error("Symbol index out of range %d-%u.", Table, Index);
157  return 0xdead;
158  }
159 
160  uint64_t SymbolAddress = (uint64_t)Header + SymbolTable->sh_offset;
161  Elf64_Sym *Symbol = &((Elf64_Sym *)SymbolAddress)[Index];
162 
163  if (Symbol->st_shndx == SHN_UNDEF)
164  {
165  Elf64_Shdr *StringTable = GetELFSection(Header, SymbolTable->sh_link);
166  const char *Name = (const char *)Header + StringTable->sh_offset + Symbol->st_name;
167 
168  void *Target = (void *)ELFLookupSymbol(Header, Name)->st_value;
169  if (Target == nullptr)
170  {
171  if (ELF64_ST_BIND(Symbol->st_info) & STB_WEAK)
172  return 0;
173  else
174  {

176  return 0xdead;
177  }
178  }
179  else

181  }
182  else if (Symbol->st_shndx == SHN_ABS)
183  return Symbol->st_value;
184  else

186  Elf64_Shdr *Target = GetELFSection(Header, Symbol->st_shndx);
187  return (uintptr_t)Header + Symbol->st_value + Target->sh_offset;
