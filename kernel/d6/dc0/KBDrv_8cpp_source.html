51 static inline int GetLetterFromScanCode(uint8_t ScanCode)
52 {
53  if (ScanCode & 0x80)
54  {
55  switch (ScanCode)
56  {
57  case KEY_U_LSHIFT:
58  LowerCase = true;
59  return KEY_INVALID;
60  case KEY_U_RSHIFT:
61  LowerCase = true;
62  return KEY_INVALID;
63  default:
64  return KEY_INVALID;
65  }
66  }
67  else
68  {
69  switch (ScanCode)
70  {
71  case KEY_D_RETURN:
72  return '\n';
73  case KEY_D_LSHIFT:
74  LowerCase = false;
75  return KEY_INVALID;
76  case KEY_D_RSHIFT:
77  LowerCase = false;
78  return KEY_INVALID;
79  case KEY_D_BACKSPACE:
80  return ScanCode;
81  default:
82  {
83  if (ScanCode > 0x39)
84  break;
85  if (LowerCase)
86  return sc_ascii_low[ScanCode];
87  else
88  return sc_ascii_high[ScanCode];
89  }
90  }
91  }
92  return KEY_INVALID;
93 }
