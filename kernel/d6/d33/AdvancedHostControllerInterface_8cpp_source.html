44  PortType CheckPortType(HBAPort *Port)
45  {
46  uint32_t SataStatus = Port->SataStatus;
47  uint8_t InterfacePowerManagement = (SataStatus >> 8) & 0b111;
48  uint8_t DeviceDetection = SataStatus & 0b111;
49 
50  if (DeviceDetection != HBA_PORT_DEV_PRESENT)
51  return PortType::None;
52  if (InterfacePowerManagement != HBA_PORT_IPM_ACTIVE)
53  return PortType::None;
54 
55  switch (Port->Signature)
56  {
57  case SATA_SIG_ATAPI:
58  return PortType::SATAPI;
59  case SATA_SIG_ATA:
60  return PortType::SATA;
61  case SATA_SIG_PM:
62  return PortType::PM;
63  case SATA_SIG_SEMB:
64  return PortType::SEMB;
65  default:
66  return PortType::None;
67  }
68  }

132  bool Port::ReadWrite(uint64_t Sector, uint32_t SectorCount, uint8_t *Buffer, bool Write)
133  {
134  if (this->PortNumber == PortType::SATAPI && Write)
135  {
136  error("SATAPI port does not support write.");
137  return false;
138  }
139 
140  uint32_t SectorL = (uint32_t)Sector;
141  uint32_t SectorH = (uint32_t)(Sector >> 32);
142 
143  HBAPortPtr->InterruptStatus = (uint32_t)-1; // Clear pending interrupt bits
144 
145  HBACommandHeader *CommandHeader = reinterpret_cast<HBACommandHeader *>(HBAPortPtr->CommandListBase);
146  CommandHeader->CommandFISLength = sizeof(FIS_REG_H2D) / sizeof(uint32_t);
147  if (Write)
148  CommandHeader->Write = 1;
149  else
150  CommandHeader->Write = 0;
151  CommandHeader->PRDTLength = 1;
152 
153  HBACommandTable *CommandTable = reinterpret_cast<HBACommandTable *>(CommandHeader->CommandTableBaseAddress);
154  memset(CommandTable, 0, sizeof(HBACommandTable) + (CommandHeader->PRDTLength - 1) * sizeof(HBAPRDTEntry));
155 
156  CommandTable->PRDTEntry[0].DataBaseAddress = (uint32_t)(uint64_t)Buffer;
157  CommandTable->PRDTEntry[0].DataBaseAddressUpper = (uint32_t)((uint64_t)Buffer >> 32);
158 
159 #pragma GCC diagnostic push
160 /* conversion from ‘uint32_t’ {aka ‘unsigned int’} to ‘unsigned int:22’ may change value */
161 #pragma GCC diagnostic ignored "-Wconversion"
162  CommandTable->PRDTEntry[0].ByteCount = (SectorCount << 9) - 1; /* 512 bytes per sector */
163 #pragma GCC diagnostic pop
164 
165  CommandTable->PRDTEntry[0].InterruptOnCompletion = 1;
166 
167  FIS_REG_H2D *CommandFIS = (FIS_REG_H2D *)(&CommandTable->CommandFIS);
168 
169  CommandFIS->FISType = FIS_TYPE_REG_H2D;
170  CommandFIS->CommandControl = 1;
171  if (Write)
172  CommandFIS->Command = ATA_CMD_WRITE_DMA_EX;
173  else
174  CommandFIS->Command = ATA_CMD_READ_DMA_EX;
175 
176  CommandFIS->LBA0 = (uint8_t)SectorL;
177  CommandFIS->LBA1 = (uint8_t)(SectorL >> 8);
178  CommandFIS->LBA2 = (uint8_t)(SectorL >> 16);
179  CommandFIS->LBA3 = (uint8_t)SectorH;
180  CommandFIS->LBA4 = (uint8_t)(SectorH >> 8);
181  CommandFIS->LBA5 = (uint8_t)(SectorH >> 16);
182 
183  CommandFIS->DeviceRegister = 1 << 6; // LBA mode
184  CommandFIS->CountLow = SectorCount & 0xFF;
185  CommandFIS->CountHigh = (SectorCount >> 8) & 0xFF;
186 
187  uint64_t Spin = 0;
188 
189  while ((HBAPortPtr->TaskFileData & (ATA_DEV_BUSY | ATA_DEV_DRQ)) && Spin < 1000000)
190  Spin++;
191  if (Spin == 1000000)
192  {
193  error("Port not responding.");
194  return false;
195  }
196 
197  HBAPortPtr->CommandIssue = 1;
198 
199  Spin = 0;
200  int TryCount = 0;
201 
202  while (true)
203  {
204  if (Spin > 100000000)
205  {
206  error("Port %d not responding. (%d)", this->PortNumber, TryCount);
207  Spin = 0;
208  TryCount++;
209  if (TryCount > 10)
210  return false;
211  }
212  if (HBAPortPtr->CommandIssue == 0)
213  break;
214  Spin++;
215  if (HBAPortPtr->InterruptStatus & HBA_PxIS_TFES)
216  {
217  error("Error reading/writing (%d).", Write);
218  return false;
219  }
220  }
221 
222  return true;
223  }

236  int CallbackHandler(KernelCallback *Data)
237  {
238  switch (Data->Reason)
239  {
240  case AcknowledgeReason:
241  {
242  debug("Kernel acknowledged the driver.");
243  break;
244  }
245  case ConfigurationReason:
246  {
247  debug("Driver received configuration data.");
248  PCIBaseAddress = reinterpret_cast<PCIDeviceHeader *>(Data->RawPtr);
249  ABAR = reinterpret_cast<HBAMemory *>(((PCIHeader0 *)PCIBaseAddress)->BAR5);
250  KAPI.Memory.Map((void *)ABAR, (void *)ABAR, (1 << 1));
251 
252  uint32_t PortsImplemented = ABAR->PortsImplemented;
253  for (int i = 0; i < 32; i++)
254  {
255  if (PortsImplemented & (1 << i))
256  {
257  PortType portType = CheckPortType(&ABAR->Ports[i]);
258  if (portType == PortType::SATA || portType == PortType::SATAPI)
259  {
260  trace("%s drive found at port %d", PortTypeName[portType], i);
261  Ports[PortCount] = new Port(portType, &ABAR->Ports[i], PortCount);
262  PortCount++;
263  }
264  else
265  {
266  if (portType != PortType::None)
267  warn("Unsupported drive type %s found at port %d", PortTypeName[portType], i);
268  }
269  }
270  }
271 
272  for (int i = 0; i < PortCount; i++)
273  Ports[i]->Configure();
274  break;
275  }
276  case QueryReason:
277  {
278  Data->DiskCallback.Fetch.Ports = PortCount;
279  Data->DiskCallback.Fetch.BytesPerSector = 512;
280  break;
281  }
282  case StopReason:
283  {
284  // TODO: Stop the driver.
285  debug("Driver stopped.");
286  break;
287  }
288  case SendReason:
289  case ReceiveReason:
290  {
291  Ports[Data->DiskCallback.RW.Port]->ReadWrite(Data->DiskCallback.RW.Sector,
292  (uint32_t)Data->DiskCallback.RW.SectorCount,
293  Data->DiskCallback.RW.Buffer,
294  Data->DiskCallback.RW.Write);
295  break;
296  }
297  default:
298  {
299  warn("Unknown reason.");
300  break;
301  }
302  }
303  return OK;
304  }
