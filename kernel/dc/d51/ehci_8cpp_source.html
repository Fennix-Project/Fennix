60  : Interrupts::Handler(dev)
61 {
62  Header = (PCIDeviceHeader *)dev.Header;
63 
64  switch (Header->HeaderType)

66  case 128:
67  {
68  ExPrint(ERROR_COLOR "Unknown header type %d! Guessing PCI Header 0\n" DEFAULT_COLOR,

74  PCI::PCIHeader0 *hdr = (PCI::PCIHeader0 *)Header;
