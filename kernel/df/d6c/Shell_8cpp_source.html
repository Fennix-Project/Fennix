49 char GetLetterFromScanCode(uint8_t ScanCode)
50 {
51  if (ScanCode & 0x80)
52  {
53  switch (ScanCode)
54  {
55  case KEY_U_LSHIFT:
56  LowerCase = true;
57  return 0;
58  case KEY_U_RSHIFT:
59  LowerCase = true;
60  return 0;
61  default:
62  return 0;
63  }
64  }
65  else
66  {
67  switch (ScanCode)
68  {
69  case KEY_D_RETURN:
70  return '\n';
71  case KEY_D_LSHIFT:
72  LowerCase = false;
73  return 0;
74  case KEY_D_RSHIFT:
75  LowerCase = false;
76  return 0;
77  case KEY_D_BACKSPACE:
78  return ScanCode;
79  default:
80  {
81  if (ScanCode > 0x39)
82  break;
83  if (LowerCase)
84  return sc_ascii_low[ScanCode];
85  else
86  return sc_ascii_high[ScanCode];
87  }
88  }
89  }
90  return 0;
91 }

172 
173  while (true)
174  {
175  size_t BackspaceCount = 0;
176  Buffer.clear();
177 
178  VirtualFileSystem::Node *cwd = thisProcess->CurrentWorkingDirectory;
179  if (!cwd)

195  switch (callback.InputCallback.Keyboard.Key)
196  {

200  continue;

206  }

208  {

215  Display->SetBuffer(0);
216  continue;
217  }
218  case KEY_D_UP:
219  {
220  if (History.size() == 0 ||
221  HistoryIndex == 0)
222  continue;
223

225

227  Display->Print('\b', 0);

231

233  Display->Print(Buffer[i], 0);

291

293  {

295  {
296  Buffer = commands[i].Name;

298  Display->Print(Buffer[i], 0);
299  BackspaceCount = Buffer.size();
300  Display->SetBuffer(0);
301  break;
302  }
303  }
304  continue;
305  }
306  default:
307  break;
308  }
309 
310  if (c == 0)
311  continue;
312 
313  if (CtrlDown)
314  {

336  continue;
337  }
338  }
339 
340  Display->Print(c, 0);

342  {
343  if (Buffer.length() > 0)
344  {
345  std::string *hBuff = new std::string(Buffer.c_str());
346  History.push_back(hBuff);
347  HistoryIndex = History.size();
348  }
349  break;
350  }
351 
352  Buffer += c;
353  BackspaceCount++;
354  Display->SetBuffer(0);
355  }
356  SecLoopEnd:
357 
358  if (Buffer.length() == 0)
359  continue;
360 
361  bool Found = false;
362  for (size_t i = 0; i < sizeof(commands) / sizeof(Command); i++)
363  {
364  std::string cmd_extracted;
365  for (size_t i = 0; i < Buffer.length(); i++)
366  {

368  break;

370  }
371 
372  debug("cmd: %s, array[%d]: %s", cmd_extracted.c_str(), i, commands[i].Name);
373  if (strncmp(commands[i].Name, cmd_extracted.c_str(), cmd_extracted.size()) == 0)

376  continue;
377

382  for (size_t i = strlen(cmd_name) + 1; i < Buffer.length(); i++)
383  arg_only += Buffer[i];
384

386  commands[i].Function(arg_only.c_str());
387  else
388  {
389  std::string cmd_only;

391  {

393  break;
