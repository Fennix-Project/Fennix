158  DriverCode Driver::LoadDriver(uintptr_t DriverAddress, uintptr_t Size)
159  {
160  Fex *DrvHdr = (Fex *)DriverAddress;
161  if (DrvHdr->Magic[0] != 'F' || DrvHdr->Magic[1] != 'E' || DrvHdr->Magic[2] != 'X' || DrvHdr->Magic[3] != '\0')
162  {
163  if (Size > 0x1000)
164  {
165  Fex *ElfDrvHdr = (Fex *)(DriverAddress + 0x1000);
166  if (ElfDrvHdr->Magic[0] != 'F' || ElfDrvHdr->Magic[1] != 'E' || ElfDrvHdr->Magic[2] != 'X' || ElfDrvHdr->Magic[3] != '\0')
167  return DriverCode::INVALID_FEX_HEADER;
168  else
169  {
170  debug("Fex Magic: \"%s\"; Type: %d; OS: %d; EntryPoint: %#lx", ElfDrvHdr->Magic, ElfDrvHdr->Type, ElfDrvHdr->OS, ElfDrvHdr->EntryPoint);
171 
172  if (ElfDrvHdr->Type == FexFormatType::FexFormatType_Driver)
173  {
174  FexExtended *ElfDrvExtHdr = (FexExtended *)((uintptr_t)ElfDrvHdr + EXTENDED_SECTION_ADDRESS);
175  debug("Name: \"%s\"; Type: %d; Callback: %#lx", ElfDrvExtHdr->Driver.Name, ElfDrvExtHdr->Driver.Type, ElfDrvExtHdr->Driver.Callback);
176 
177  if (ElfDrvExtHdr->Driver.Bind.Type == DriverBindType::BIND_PCI)
178  return this->DriverLoadBindPCI(ElfDrvExtHdr, DriverAddress, Size, true);
179  else if (ElfDrvExtHdr->Driver.Bind.Type == DriverBindType::BIND_INTERRUPT)
180  return this->DriverLoadBindInterrupt(ElfDrvExtHdr, DriverAddress, Size, true);
181  else if (ElfDrvExtHdr->Driver.Bind.Type == DriverBindType::BIND_PROCESS)
182  return this->DriverLoadBindProcess(ElfDrvExtHdr, DriverAddress, Size, true);
183  else if (ElfDrvExtHdr->Driver.Bind.Type == DriverBindType::BIND_INPUT)
184  return this->DriverLoadBindInput(ElfDrvExtHdr, DriverAddress, Size, true);
185  else
186  error("Unknown driver bind type: %d", ElfDrvExtHdr->Driver.Bind.Type);
187  }
188  else
189  return DriverCode::NOT_DRIVER;
190  }
191  }
192  else
193  return DriverCode::INVALID_FEX_HEADER;
194  }
195  debug("Fex Magic: \"%s\"; Type: %d; OS: %d; EntryPoint: %#lx", DrvHdr->Magic, DrvHdr->Type, DrvHdr->OS, DrvHdr->EntryPoint);

197  if (DrvHdr->Type == FexFormatType::FexFormatType_Driver)

201

218  Driver::Driver()

224  fixme("Loading driver config file: %s", DriverConfigFile.c_str());
225 
226  std::shared_ptr<VirtualFileSystem::File> DriverDirectory = vfs->Open(Config.DriverDirectory);
227  if (DriverDirectory->Status == VirtualFileSystem::FileStatus::OK)
228  {
229  foreach (auto driver in DriverDirectory->node->Children)
230  if (driver->Flags == VirtualFileSystem::NodeFlags::FILE)
231  if (cwk_path_has_extension(driver->Name))
232  {
233  const char *extension;
234  size_t extension_length;
235  cwk_path_get_extension(driver->Name, &extension, &extension_length);
236  debug("Driver: %s; Extension: %s", driver->Name, extension);
237  if (strcmp(extension, ".fex") == 0 || strcmp(extension, ".elf") == 0)
238  {
239  uintptr_t ret = this->LoadDriver(driver->Address, driver->Length);
240  char RetString[128];
241  if (ret == DriverCode::OK)
242  strncpy(RetString, "\e058C19OK", 10);

247  KPrint("%s %s", driver->Name, RetString);

249  }

253  KPrint("\eE85230Failed to open driver directory: %s! (Status: %#lx)", Config.DriverDirectory, DriverDirectory->Status);
