51  void DSDT::OnInterruptReceived(CPU::x64::TrapFrame *Frame)
52  {
53  debug("SCI Handle Triggered");
54  uint16_t Event = 0;
55  {
56  uint16_t a = 0, b = 0;
57  if (acpi->FADT->PM1aEventBlock)
58  {
59  a = inw(s_cst(uint16_t, acpi->FADT->PM1aEventBlock));
60  outw(s_cst(uint16_t, acpi->FADT->PM1aEventBlock), a);
61  }
62  if (acpi->FADT->PM1bEventBlock)
63  {
64  b = inw(s_cst(uint16_t, acpi->FADT->PM1bEventBlock));
65  outw(s_cst(uint16_t, acpi->FADT->PM1bEventBlock), b);
66  }
67  Event = a | b;
68  }
69 
70  debug("SCI Event: %#lx", Event);
71  if (Event & ACPI_BUSMASTER)
72  {
73  fixme("ACPI Busmaster");
74  }
75  else if (Event & ACPI_GLOBAL)
76  {
77  fixme("ACPI Global");
78  }
79  else if (Event & ACPI_POWER_BUTTON)
80  {
81  if (TaskManager)
82  {
83  TaskManager->CreateThread(TaskManager->CreateProcess(nullptr,
84  "Shutdown",
85  Tasking::TaskTrustLevel::Kernel),
86  (Tasking::IP)KST_Shutdown);
87  }
88  else
89  KernelShutdownThread(false);
90  }
91  else if (Event & ACPI_SLEEP_BUTTON)
92  {
93  fixme("ACPI Sleep Button");
94  }
95  else if (Event & ACPI_RTC_ALARM)
96  {
97  fixme("ACPI RTC Alarm");
98  }
99  else if (Event & ACPI_PCIE_WAKE)
100  {
101  fixme("ACPI PCIe Wake");
102  }
103  else if (Event & ACPI_WAKE)
104  {
105  fixme("ACPI Wake");
106  }
107  else if (Event & ACPI_TIMER)
108  {
109  fixme("ACPI Timer");
110  }
111  else
112  {
113  error("ACPI unknown event %#lx on CPU %d", Event, GetCurrentCPU()->ID);
114  CPU::Stop();
115  }
116  UNUSED(Frame);
117  }

119  void DSDT::Shutdown()
120  {
121  trace("Shutting down...");
122  if (SCI_EN == 1)
123  {
124  outw(s_cst(uint16_t, acpi->FADT->PM1aControlBlock),
125  s_cst(uint16_t,
126  (inw(s_cst(uint16_t,
127  acpi->FADT->PM1aControlBlock)) &
128  0xE3FF) |
129  ((SLP_TYPa << 10) | ACPI_SLEEP)));
130 
131  if (acpi->FADT->PM1bControlBlock)
132  outw(s_cst(uint16_t, acpi->FADT->PM1bControlBlock),
133  s_cst(uint16_t,
134  (inw(
135  s_cst(uint16_t, acpi->FADT->PM1bControlBlock)) &
136  0xE3FF) |
137  ((SLP_TYPb << 10) | ACPI_SLEEP)));
138 
139  outw(s_cst(uint16_t, PM1a_CNT), SLP_TYPa | SLP_EN);
140  if (PM1b_CNT)
141  outw(s_cst(uint16_t, PM1b_CNT), SLP_TYPb | SLP_EN);

145  void DSDT::Reboot()
146  {
147  trace("Rebooting...");
148  switch (acpi->FADT->ResetReg.AddressSpace)
149  {
150  case ACPI_GAS_MMIO:
151  {
152  *(uint8_t *)(acpi->FADT->ResetReg.Address) = acpi->FADT->ResetValue;
153  break;
154  }
155  case ACPI_GAS_IO:
156  {
157  outb(s_cst(uint16_t, acpi->FADT->ResetReg.Address), acpi->FADT->ResetValue);
158  break;
159  }
160  case ACPI_GAS_PCI:
161  {
162  fixme("ACPI_GAS_PCI not supported.");
163  /*
164  seg - 0
165  bus - 0
166  dev - (FADT->ResetReg.Address >> 32) & 0xFFFF
167  function - (FADT->ResetReg.Address >> 16) & 0xFFFF
168  offset - FADT->ResetReg.Address & 0xFFFF
169  value - FADT->ResetValue
170  */
171  break;
172  }
173  default:
174  {
175  error("Unknown reset register address space: %d", acpi->FADT->ResetReg.AddressSpace);
176  break;
177  }
178  }

181  DSDT::DSDT(ACPI *acpi) : Interrupts::Handler(acpi->FADT->SCI_Interrupt)
182  {
183  this->acpi = acpi;
184  uint64_t Address = ((IsCanonical(acpi->FADT->X_Dsdt) && acpi->XSDTSupported) ? acpi->FADT->X_Dsdt : acpi->FADT->Dsdt);
185  uint8_t *S5Address = (uint8_t *)(Address) + 36;
186  ACPI::ACPI::ACPIHeader *Header = (ACPI::ACPI::ACPIHeader *)Address;
187  size_t Length = Header->Length;
188  while (Length-- > 0)
189  {
190  if (!memcmp(S5Address, "_S5_", 4))
191  break;
192  S5Address++;
193  }
194  if (Length <= 0)
195  {
196  warn("_S5 not present in ACPI");
197  return;
198  }
199  if ((*(S5Address - 1) == 0x08 || (*(S5Address - 2) == 0x08 && *(S5Address - 1) == '\\')) && *(S5Address + 4) == 0x12)
200  {
201  S5Address += 5;
202  S5Address += ((*S5Address & 0xC0) >> 6) + 2;
203  if (*S5Address == 0x0A)
204  S5Address++;
205  SLP_TYPa = s_cst(uint16_t, *(S5Address) << 10);
206  S5Address++;
207  if (*S5Address == 0x0A)
208  S5Address++;
209  SLP_TYPb = s_cst(uint16_t, *(S5Address) << 10);
210  SMI_CMD = acpi->FADT->SMI_CommandPort;
211  ACPI_ENABLE = acpi->FADT->AcpiEnable;
212  ACPI_DISABLE = acpi->FADT->AcpiDisable;
213  PM1a_CNT = acpi->FADT->PM1aControlBlock;
214  PM1b_CNT = acpi->FADT->PM1bControlBlock;
215  PM1_CNT_LEN = acpi->FADT->PM1ControlLength;
216  SLP_EN = 1 << 13;
217  SCI_EN = 1;
218  trace("ACPI Shutdown is supported");
219  ACPIShutdownSupported = true;
220 
221  uint16_t value = ACPI_POWER_BUTTON | ACPI_SLEEP_BUTTON | ACPI_WAKE;
222  {
223  uint16_t a = s_cst(uint16_t, acpi->FADT->PM1aEventBlock + (acpi->FADT->PM1EventLength / 2));
224  uint16_t b = s_cst(uint16_t, acpi->FADT->PM1bEventBlock + (acpi->FADT->PM1EventLength / 2));
225  debug("SCI Event: %#llx [a:%#x b:%#x]", value, a, b);
226  if (acpi->FADT->PM1aEventBlock)
227  outw(a, value);
228  if (acpi->FADT->PM1bEventBlock)
229  outw(b, value);
230  }
231 
232  {
233  uint16_t a = 0, b = 0;
234  if (acpi->FADT->PM1aEventBlock)
235  {
236  a = inw(s_cst(uint16_t, acpi->FADT->PM1aEventBlock));
237  outw(s_cst(uint16_t, acpi->FADT->PM1aEventBlock), a);
238  }
239  if (acpi->FADT->PM1bEventBlock)
240  {
241  b = inw(s_cst(uint16_t, acpi->FADT->PM1bEventBlock));
242  outw(s_cst(uint16_t, acpi->FADT->PM1bEventBlock), b);
243  }
244  }
245  ((APIC::APIC *)Interrupts::apic[0])->RedirectIRQ(0, acpi->FADT->SCI_Interrupt, 1);
246  return;
247  }
248  warn("Failed to parse _S5 in ACPI");
249  SCI_EN = 0;
250  }
