68 nsa bool WriteDiagDataToNode(vfs::RefNode *refFile)
69 {
70  uintptr_t KStart = (uintptr_t)&_kernel_start;
71  uintptr_t kEnd = (uintptr_t)&_kernel_end;
72  size_t kSize = kEnd - KStart;
73 
74  size_t fileSize = sizeof(DiagnosticFile) + kSize;
75  uint8_t *buf = (uint8_t *)KernelAllocator.RequestPages(TO_PAGES(fileSize));
76  if (!buf)
77  {
78  ExPrint("\eFF0000Failed to allocate memory for diagnostic data\n");
79  Display->UpdateBuffer();
80  return false;
81  }
82 
83  DiagnosticFile *file = (DiagnosticFile *)buf;
84  file->Header = {};
85  file->Header.Length = uint32_t(fileSize);
86  file->Header.Is64Bit = sizeof(void *) == 8;
87 
88  file->Data.Frame = *ExFrame;
89  file->Data.KernelMemoryLength = uint32_t(kSize);
90  memcpy(file->Data.KernelMemory, (void *)KStart, kSize);
91 
92  ExPrint("\eFAFAFAWriting to %s\n", refFile->node->FullPath);
93  size_t w = refFile->write(buf, fileSize);
94  if (w != fileSize)
95  {
96  debug("%d out of %d bytes written", w, fileSize);
97  ExPrint("\eFF0000Failed to write diagnostic data to file: %s\n",
98  strerror((int)w));
99  Display->UpdateBuffer();
100  return false;
101  }
102 
103  return true;
104 }
