145  extern "C" SafeFunction void MainInterruptHandler(void *Data)
146  {
147 #if defined(a64)
148  CPU::x64::TrapFrame *Frame = (CPU::x64::TrapFrame *)Data;
149  // debug("IRQ%ld", Frame->InterruptNumber - 32);
150 
151  memmove(InterruptFrames + 1, InterruptFrames, sizeof(InterruptFrames) - sizeof(InterruptFrames[0]));
152  InterruptFrames[0] = (void *)Frame->rip;
153 
154  CPUData *CoreData = GetCurrentCPU();
155  int Core = 0;
156  if (likely(CoreData != nullptr))
157  Core = CoreData->ID;
158 
159  /* If this is false, we have a big problem. */
160  if (likely(Frame->InterruptNumber < CPU::x86::IRQ223 && Frame->InterruptNumber > CPU::x86::ISR0))
161  {
162  /* Halt core interrupt */
163  if (unlikely(Frame->InterruptNumber == CPU::x86::IRQ29))
164  CPU::Stop();
165 
166  bool InterruptHandled = false;
167  foreach (auto ev in RegisteredEvents)
168  {
169 #if defined(a86)
170  if ((ev.ID + CPU::x86::IRQ0) == static_cast<int>(Frame->InterruptNumber))
171 #elif defined(aa64)
172  if (ev.ID == static_cast<int>(Frame->InterruptNumber))
173 #endif
174  {
175  ((Handler *)ev.Data)->OnInterruptReceived(Frame);
176  InterruptHandled = true;
177  }
178  }
179 
180  if (!InterruptHandled)
181  {
182  error("IRQ%ld is unhandled on CPU %d.", Frame->InterruptNumber - 32, Core);
183  if (Frame->InterruptNumber == CPU::x86::IRQ1)
184  {
185  uint8_t scancode = inb(0x60);
186  warn("IRQ1 is the keyboard interrupt. Scancode: %#x", scancode);
187  }
188  }
189 
190  if (likely(apic[Core]))
191  {
192  ((APIC::APIC *)Interrupts::apic[Core])->EOI();
193  // TODO: Handle PIC too
194  return;
195  }
196  // TODO: PIC
197  }
198 #elif defined(a32)
199  void *Frame = Data;
200 #elif defined(aa64)
201  CPU::aarch64::TrapFrame *Frame = (CPU::aarch64::TrapFrame *)Data;
202 #endif
203  error("HALT HALT HALT HALT HALT HALT HALT HALT HALT");
204  CPU::Stop();
205  }

207  Handler::Handler(int InterruptNumber)
208  {
209  foreach (auto ev in RegisteredEvents)
210  {

212  {
213  warn("IRQ%d is already registered.", InterruptNumber);

216

218  this->InterruptNumber = InterruptNumber;
