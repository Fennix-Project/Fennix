67 {

72  uint32_t ax;

74  };

77  uint32_t bx;

87  uint32_t dx;
88  uint16_t port;
89  };

91  uint32_t di;
92  } cmd{};
93 
94 #define VMWARE_MAGIC 0x564D5868 /* hXMV */
95 #define VMWARE_PORT 0x5658
96 #define CMD_GETVERSION 0xA
97 
98  cmd.bx = ~VMWARE_MAGIC;
99  cmd.command = CMD_GETVERSION;
100  cmd.magic = VMWARE_MAGIC;
101  cmd.port = VMWARE_PORT;
102 
103  asmv("in %%dx, %0"
104  : "+a"(cmd.ax),
105  "+b"(cmd.bx),
106  "+c"(cmd.cx),
107  "+d"(cmd.dx),
108  "+S"(cmd.si),
109  "+D"(cmd.di));
110 
111  if (cmd.bx != VMWARE_MAGIC || cmd.ax == 0xFFFFFFFF)
112  return false;
113 
114  debug("VMware backdoor version: %d.%d.%d",
115  cmd.ax >> 16,
116  (cmd.ax >> 8) & 0xFF,
117  cmd.ax & 0xFF);
118 
119  return true;
120 }

122 bool DetectByCPUID()
123 {
124  bool IsVM = false;
125  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_INTEL) == 0)
126  {
127  CPU::x86::Intel::CPUID0x00000001 cpuid00000001;
128  cpuid00000001.Get();
129 
130  if (cpuid00000001.ECX.Hypervisor == 1)
131  {
132  debug("Intel: Hypervisor: %d",
133  cpuid00000001.ECX.Hypervisor);
134  IsVM = true;
135  }
136  }
137  else if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
138  {
139  CPU::x86::AMD::CPUID0x00000001 cpuid00000001;

142  if (cpuid00000001.ECX.Hypervisor == 1)
143  {
144  debug("AMD: Hypervisor: %d",
145  cpuid00000001.ECX.Hypervisor);

147  }

152 bool DetectByHPET()

157

161 
162  ACPI::ACPI::HPETHeader *HPET = (ACPI::ACPI::HPETHeader *)hpet;
163 
164  if (strstr((const char *)HPET->Header.OEMID, "BOCHS BXPC") != NULL)
165  goto Yes;
166 
167  if (strstr((const char *)HPET->Header.OEMID, "VBOX") != NULL)
168  goto Yes;
169 
170  if (strstr((const char *)HPET->Header.OEMID, "VMWAREVMW") != NULL)
171  goto Yes;
172 
173  return false;
174 
175 Yes:
176  debug("HPET: %s", HPET->Header.OEMID);
177  return true;
