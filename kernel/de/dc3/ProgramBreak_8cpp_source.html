27  void *ProgramBreak::brk(void *Address)
28  {
29  if (HeapStart == 0x0 || Break == 0x0)
30  {
31  error("HeapStart or Break is 0x0");
32  return (void *)-EAGAIN;
33  }
34 
35  /* Get the current program break. */
36  if (Address == nullptr)
37  return (void *)Break;
38 
39  /* Check if the address is valid. */
40  if (Address < (void *)HeapStart)
41  return (void *)-ENOMEM;
42 
43  Virtual vmm = Virtual(this->Table);
44 
45  if (Address > (void *)Break)
46  {
47  /* Allocate more memory. */
48  size_t Pages = TO_PAGES(uintptr_t(Address) - Break);
49  void *Allocated = mm->RequestPages(Pages);
50  if (Allocated == nullptr)
51  return (void *)-ENOMEM;
52 
53  /* Map the allocated pages. */
54  for (size_t i = 0; i < Pages; i++)
55  {
56  void *VirtAddr = (void *)(Break + (i * PAGE_SIZE));
57  void *PhysAddr = (void *)(uintptr_t(Allocated) + (i * PAGE_SIZE));
58  vmm.Map(VirtAddr, PhysAddr, RW | US);
59  }
60 
61  Break = (uint64_t)Address;
62  return (void *)Break;
63  }
64  else if (Address < (void *)Break)
65  {
66  /* Free memory. */
67  size_t Pages = TO_PAGES(uintptr_t(Address) - Break);
68  mm->FreePages((void *)Break, Pages);
69 
70  /* Unmap the freed pages. */
71  for (size_t i = 0; i < Pages; i++)
72  {
73  uint64_t Page = Break - (i * 0x1000);
74  vmm.Unmap((void *)Page);
75  }
76

78  return (void *)Break;
79  }
