55

61  Tasking::PCB *pcb = thisProcess;

63 
64  for (uintptr_t i = uintptr_t(addr);

66  i += PAGE_SIZE)
67  {
68  if (likely(!vmm.Check((void *)i, Memory::G)))
69  {
70  Memory::PageTableEntry *pte = vmm.GetPTE(addr);
71  if (!pte->Present ||
72  (!pte->UserSupervisor && p_Read) ||
73  (!pte->ReadWrite && p_Write))
74  {

79

85 #if defined(a64)
