111  void Display::SetBrightness(int Value, int Index)
112  {
113  if (unlikely(this->Buffers[Index].Checksum != 0xBBFFE515A117E))
114  {
115  debug("Invalid buffer %d", Index);
116  return;
117  }
118 
119  if (Value > 100)
120  Value = 100;
121  else if (Value < 0)
122  Value = 0;
123 
124  uint32_t *pixel = (uint32_t *)this->Buffers[Index].Buffer;
125 
126  for (uint32_t y = 0; y < this->Buffers[Index].Height; y++)
127  {
128  for (uint32_t x = 0; x < this->Buffers[Index].Width; x++)
129  {
130  uint32_t color = pixel[y * this->Buffers[Index].Width + x];
131 
132  uint8_t r = color & 0xff;
133  uint8_t g = (color >> 8) & 0xff;
134  uint8_t b = (color >> 16) & 0xff;
135 
136  r = s_cst(uint8_t, (r * Value) / 100);
137  g = s_cst(uint8_t, (g * Value) / 100);
138  b = s_cst(uint8_t, (b * Value) / 100);
139 
140  pixel[y * this->Buffers[Index].Width + x] = (b << 16) | (g << 8) | r;
141  }
142  }
143  this->Buffers[Index].Brightness = s_cst(char, Value);
144  }

235  char Display::Print(char Char, int Index, bool WriteToUART)
236  {
237  if (unlikely(this->Buffers[Index].Checksum != 0xBBFFE515A117E))
238  return 0;
239 
240  // SmartLock(PrintLock);
241 
242  if (this->ColorIteration)
243  {
244  // RRGGBB
245  if (Char >= '0' && Char <= '9')
246  this->Buffers[Index].Color = (this->Buffers[Index].Color << 4) | (Char - '0');
247  else if (Char >= 'a' && Char <= 'f')
248  this->Buffers[Index].Color = (this->Buffers[Index].Color << 4) | (Char - 'a' + 10);
249  else if (Char >= 'A' && Char <= 'F')
250  this->Buffers[Index].Color = (this->Buffers[Index].Color << 4) | (Char - 'A' + 10);
251  else
252  this->Buffers[Index].Color = 0xFFFFFF;
253  if (WriteToUART)
254  UniversalAsynchronousReceiverTransmitter::UART(UniversalAsynchronousReceiverTransmitter::COM1).Write(Char);
255  this->ColorPickerIteration++;
256  if (this->ColorPickerIteration == 6)
257  {
258  this->ColorPickerIteration = 0;
259  if (WriteToUART)
260  UniversalAsynchronousReceiverTransmitter::UART(UniversalAsynchronousReceiverTransmitter::COM1).Write(']');
261  this->ColorIteration = false;
262  }
263  return Char;
264  }
265 
266  if (WriteToUART)
267  UniversalAsynchronousReceiverTransmitter::UART(UniversalAsynchronousReceiverTransmitter::COM1).Write(Char);
268 
269  switch (Char)
270  {
271  case '\e':
272  {
273  if (WriteToUART)
274  UniversalAsynchronousReceiverTransmitter::UART(UniversalAsynchronousReceiverTransmitter::COM1).Write('[');
275  this->ColorIteration = true;
276  return Char;
277  }
278  case '\b':
279  {
280  switch (this->CurrentFont->GetInfo().Type)
281  {
282  case FontType::PCScreenFont1:
283  {
284  fixme("PCScreenFont1");
285  break;
286  }
287  case FontType::PCScreenFont2:
288  {
289  uint32_t fonthdrWidth = this->CurrentFont->GetInfo().PSF2Font->Header->width;
290  uint32_t fonthdrHeight = this->CurrentFont->GetInfo().PSF2Font->Header->height;
291 
292  for (unsigned long Y = this->Buffers[Index].CursorY; Y < this->Buffers[Index].CursorY + fonthdrHeight; Y++)
293  for (unsigned long X = this->Buffers[Index].CursorX - fonthdrWidth; X < this->Buffers[Index].CursorX; X++)
294  *(uint32_t *)((uintptr_t)this->Buffers[Index].Buffer +
295  (Y * this->Buffers[Index].Width + X) * (this->framebuffer.BitsPerPixel / 8)) = 0;
296  break;
297  }
298  default:
299  warn("Unsupported font type");
300  break;
301  }
302 
303  if (this->Buffers[Index].CursorX > 0)
304  this->Buffers[Index].CursorX -= this->GetCurrentFont()->GetInfo().Width;
305 
306  return Char;
307  }
308  case '\t':
309  {
310  this->Buffers[Index].CursorX = (this->Buffers[Index].CursorX + 8) & ~(8 - 1);
311  return Char;
312  }
313  case '\r':
314  {
315  this->Buffers[Index].CursorX = 0;
316  return Char;
317  }
318  case '\n':
319  {
320  this->Buffers[Index].CursorX = 0;
321  this->Buffers[Index].CursorY += this->GetCurrentFont()->GetInfo().Height;
322  return Char;
323  }
324  default:
325  break;
326  }
327 
328  uint32_t FontHeight = this->GetCurrentFont()->GetInfo().Height;
329 
330  if (this->Buffers[Index].CursorX + this->GetCurrentFont()->GetInfo().Width >= this->Buffers[Index].Width)
331  {
332  this->Buffers[Index].CursorX = 0;
333  this->Buffers[Index].CursorY += FontHeight;
334  }
335 
336  if (this->Buffers[Index].CursorY + FontHeight >= this->Buffers[Index].Height)
337  {
338  if (!this->Buffers[Index].DoNotScroll)
339  {
340  this->Buffers[Index].CursorY -= FontHeight;
341  this->Scroll(Index, 1);
342  }
343  }
344 
345  switch (this->CurrentFont->GetInfo().Type)
346  {
347  case FontType::PCScreenFont1:
348  {
349  uint32_t *PixelPtr = (uint32_t *)this->Buffers[Index].Buffer;
350  char *FontPtr = (char *)this->CurrentFont->GetInfo().PSF1Font->GlyphBuffer + (Char * this->CurrentFont->GetInfo().PSF1Font->Header->charsize);
351  for (uint64_t Y = this->Buffers[Index].CursorY; Y < this->Buffers[Index].CursorY + 16; Y++)
352  {
353  for (uint64_t X = this->Buffers[Index].CursorX; X < this->Buffers[Index].CursorX + 8; X++)
354  if ((*FontPtr & (0b10000000 >> (X - this->Buffers[Index].CursorX))) > 0)
355  *(unsigned int *)(PixelPtr + X + (Y * this->Buffers[Index].Width)) = this->Buffers[Index].Color;
356  FontPtr++;
357  }
358  this->Buffers[Index].CursorX += 8;
359 
360  break;
361  }
362  case FontType::PCScreenFont2:
363  {
364  // if (this->CurrentFont->PSF2Font->GlyphBuffer == (uint16_t *)0x01) // HAS UNICODE TABLE
365  // Char = this->CurrentFont->PSF2Font->GlyphBuffer[Char];
366  int BytesPerLine = (this->CurrentFont->GetInfo().PSF2Font->Header->width + 7) / 8;
367  char *FontAddress = (char *)this->CurrentFont->GetInfo().StartAddress;
368  uint32_t FontHeaderSize = this->CurrentFont->GetInfo().PSF2Font->Header->headersize;
369  uint32_t FontCharSize = this->CurrentFont->GetInfo().PSF2Font->Header->charsize;
370  uint32_t FontLength = this->CurrentFont->GetInfo().PSF2Font->Header->length;
371  char *FontPtr = FontAddress + FontHeaderSize + (Char > 0 && (uint32_t)Char < FontLength ? Char : 0) * FontCharSize;
372 
373  uint32_t FontHdrWidth = this->CurrentFont->GetInfo().PSF2Font->Header->width;
374  uint32_t FontHdrHeight = this->CurrentFont->GetInfo().PSF2Font->Header->height;
375 
376  for (size_t Y = this->Buffers[Index].CursorY; Y < this->Buffers[Index].CursorY + FontHdrHeight; Y++)
377  {
378  for (size_t X = this->Buffers[Index].CursorX; X < this->Buffers[Index].CursorX + FontHdrWidth; X++)
379  {
380  if ((*FontPtr & (0b10000000 >> (X - this->Buffers[Index].CursorX))) > 0)
381  {
382  void *FramebufferAddress = (void *)((uintptr_t)this->Buffers[Index].Buffer +
383  (Y * this->Buffers[Index].Width + X) *
384  (this->framebuffer.BitsPerPixel / 8));
385  *(uint32_t *)FramebufferAddress = this->Buffers[Index].Color;
386  }
387  }
388  FontPtr += BytesPerLine;
389  }
390  this->Buffers[Index].CursorX += FontHdrWidth;
391  break;
392  }
393  default:
394  warn("Unsupported font type");
395  break;
396  }
397  return Char;
