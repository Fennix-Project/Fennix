53  void DHCP::CreatePacket(DHCPHeader *Packet, uint8_t MessageType, uint32_t RequestIP)
54  {
55  Packet->Opcode = b8(DHCP_OP_BOOTREQUEST);
56  Packet->HardwareType = b8(1);
57  Packet->HardwareAddressLength = b8(6);
58  Packet->Hops = b8(0);
59  Packet->TransactionID = b32(DHCP_TRANSACTION_ID);
60  Packet->Flags = b16(0x40);
61  uint48_t InterfaceMAC = b48(Interface->MAC.ToHex());
62  memcpy(Packet->ClientHardwareAddress, &InterfaceMAC, sizeof(InterfaceMAC));
63 
64  uint8_t *Ptr = Packet->Options;
65  *((uint32_t *)(Ptr)) = b32(0x63825363); // Magic Cookie
66  Ptr += 4;
67 
68  *(Ptr++) = DHCP_OPTION_MESSAGE_TYPE;
69  *(Ptr++) = DHCP_MESSAGE_TYPE_DISCOVER;
70  *(Ptr++) = MessageType;
71 
72  *(Ptr++) = DHCP_OPTION_CLIENT_IDENTIFIER;
73  *(Ptr++) = 0x07;
74  *(Ptr++) = 0x01;
75  memcpy(Ptr, &InterfaceMAC, sizeof(InterfaceMAC));
76  Ptr += 6;
77 
78  *(Ptr++) = DHCP_OPTION_REQUESTED_IP;
79  *(Ptr++) = 0x04;
80  *((uint32_t *)(Ptr)) = b32(0x0a00020e);
81  memcpy((uint32_t *)(Ptr), &RequestIP, 4);
82  Ptr += 4;
83 
84  *(Ptr++) = DHCP_OPTION_HOST_NAME;
85  char *HostName = (char *)KERNEL_NAME;
86  *(Ptr++) = s_cst(uint8_t, 1 + strlen(HostName));
87  memcpy(Ptr, HostName, strlen(HostName));
88  Ptr += strlen(HostName);
89 
90  *(Ptr++) = DHCP_OPTION_PAD;
91 
92  *(Ptr++) = DHCP_OPTION_PARAMETER_REQUEST_LIST;
93  *(Ptr++) = DHCP_OPTION_COOKIE_SERVER;
94  *(Ptr++) = DHCP_OPTION_SUBNETMASK;
95  *(Ptr++) = DHCP_OPTION_ROUTER;
96  *(Ptr++) = DHCP_OPTION_DOMAIN_NAME_SERVER;
97  *(Ptr++) = DHCP_OPTION_DOMAIN_NAME;
98  *(Ptr++) = DHCP_OPTION_NETBIOS_NAME_SERVERS;
99  *(Ptr++) = DHCP_OPTION_NETBIOS_NODE_TYPE;
100  *(Ptr++) = DHCP_OPTION_NETBIOS_SCOPE;
101  *(Ptr++) = DHCP_OPTION_MAX_MESSAGE_SIZE;
102 
103  *(Ptr++) = DHCP_OPTION_END;

106  void DHCP::Request()
107  {
108  netdbg("Requesting IP address");
109  DHCPHeader packet;
110  memset(&packet, 0, sizeof(DHCPHeader));
111 
112  CreatePacket(&packet, DHCP_MESSAGE_TYPE_DISCOVER, 0x00000000);

140  void *DHCP::GetOption(DHCPHeader *Packet, uint8_t Type)
141  {
142  uint8_t *Option = Packet->Options + 4;
143  uint8_t Current = *Option;
144  while (Current != 0xFF)
145  {
146  uint8_t OptionLength = *(Option + 1);
147  if (Current == Type)
148  return Option + 2;
149  Option += (2 + OptionLength);
150  Current = *Option;
151  }
152  warn("Option %#x not found", Type);
153  return nullptr;
154  }

156  void DHCP::OnUDPPacketReceived(NetworkUDP::Socket *Socket, uint8_t *Data, uint64_t Length)
157  {
158  DHCPHeader *Packet = (DHCPHeader *)Data;
159  uint8_t *MessageType = (uint8_t *)GetOption(Packet, DHCP_OPTION_MESSAGE_TYPE);
160 
161  switch (*MessageType)
162  {
163  case DHCP_OPTION_TIME_OFFSET:
164  {
165  InternetProtocol ReqIP;
166  ReqIP.v4.FromHex(b32(Packet->YourIP));
167  netdbg("Received DHCP offer for IP %s", ReqIP.v4.ToStringLittleEndian());
168  this->Request(ReqIP);
169  break;
170  }
171  case DHCP_OPTION_NAME_SERVER:
172  this->IP.v4.FromHex(b32(Packet->YourIP));
173  this->Gateway.v4.FromHex(b32((uint32_t)(*(uintptr_t *)GetOption(Packet, DHCP_OPTION_ROUTER))));
174  this->DomainNameSystem.v4.FromHex(b32((uint32_t)(*(uintptr_t *)GetOption(Packet, DHCP_OPTION_DOMAIN_NAME_SERVER))));
175  this->SubNetworkMask.v4.FromHex(b32((uint32_t)(*(uintptr_t *)GetOption(Packet, DHCP_OPTION_SUBNETMASK))));
176  this->Received = true;
177  netdbg("Received DHCP ACK for IP %s", this->IP.v4.ToStringLittleEndian());
178  break;
179  default:
180  warn("Received unknown message type %#x", *MessageType);
181  break;
182  }
183  }
