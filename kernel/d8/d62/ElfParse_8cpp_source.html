62  Elf64_Sym *ELFLookupSymbol(Elf64_Ehdr *Header, const char *Name)
63  {
64  Elf64_Shdr *SymbolTable = nullptr;
65  Elf64_Shdr *StringTable = nullptr;
66 
67  for (Elf64_Half i = 0; i < Header->e_shnum; i++)
68  {
69  Elf64_Shdr *shdr = GetELFSection(Header, i);
70  switch (shdr->sh_type)
71  {
72  case SHT_SYMTAB:
73  SymbolTable = shdr;
74  StringTable = GetELFSection(Header, shdr->sh_link);
75  break;
76  default:
77  {
78  break;
79  }
80  }
81  }
82 
83  if (SymbolTable == nullptr || StringTable == nullptr)
84  return nullptr;
85 
86  for (size_t i = 0; i < (SymbolTable->sh_size / sizeof(Elf64_Sym)); i++)
87  {
88  Elf64_Sym *Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
89  char *String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
90  if (strcmp(String, Name) == 0)
91  return Symbol;
92  }
93  return nullptr;
94  }

96  Elf64_Sym ELFLookupSymbol(VirtualFileSystem::File &ElfFile, const char *Name)
97  {
98  off_t OldOffset = vfs->Seek(ElfFile, 0, SEEK_CUR);
99 
100  Elf64_Ehdr Header;
101  vfs->Seek(ElfFile, 0, SEEK_SET);
102  vfs->Read(ElfFile, (uint8_t *)&Header, sizeof(Elf64_Ehdr));
103 
104  Elf64_Shdr SymbolTable;
105  Elf64_Shdr StringTable;
106 
107  for (Elf64_Half i = 0; i < Header.e_shnum; i++)
108  {
109  Elf64_Shdr shdr;
110  vfs->Seek(ElfFile, Header.e_shoff + (i * sizeof(Elf64_Shdr)), SEEK_SET);
111  vfs->Read(ElfFile, (uint8_t *)&shdr, sizeof(Elf64_Shdr));
112 
113  switch (shdr.sh_type)
114  {
115  case SHT_SYMTAB:
116  SymbolTable = shdr;
117  vfs->Seek(ElfFile, Header.e_shoff + (shdr.sh_link * sizeof(Elf64_Shdr)), SEEK_SET);
118  vfs->Read(ElfFile, (uint8_t *)&StringTable, sizeof(Elf64_Shdr));
119  break;
120  default:
121  {
122  break;
123  }
124  }
125  }
126 
127  if (SymbolTable.sh_name == 0 ||
128  StringTable.sh_name == 0)
129  {
130  error("Symbol table not found.");
131  vfs->Seek(ElfFile, OldOffset, SEEK_SET);
132  return {};
133  }
134 
135  for (size_t i = 0; i < (SymbolTable.sh_size / sizeof(Elf64_Sym)); i++)
136  {
137  // Elf64_Sym *Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
138  Elf64_Sym Symbol;
139  vfs->Seek(ElfFile, SymbolTable.sh_offset + (i * sizeof(Elf64_Sym)), SEEK_SET);
140  vfs->Read(ElfFile, (uint8_t *)&Symbol, sizeof(Elf64_Sym));
141 
142  // char *String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
143  char String[256];
144  vfs->Seek(ElfFile, StringTable.sh_offset + Symbol.st_name, SEEK_SET);
145  vfs->Read(ElfFile, (uint8_t *)&String, 256);
146 
147  if (strcmp(String, Name) == 0)
148  {
149  vfs->Seek(ElfFile, OldOffset, SEEK_SET);
150  return Symbol;
151  }
152  }
153  error("Symbol not found.");
154  vfs->Seek(ElfFile, OldOffset, SEEK_SET);
155  return {};
156  }

158  uintptr_t ELFGetSymbolValue(Elf64_Ehdr *Header, uint64_t Table, uint64_t Index)
159  {
160 #if defined(a64)
161  if (Table == SHN_UNDEF || Index == SHN_UNDEF)
162  return 0;
163  Elf64_Shdr *SymbolTable = GetELFSection(Header, Table);
164 
165  uint64_t STEntries = SymbolTable->sh_size / SymbolTable->sh_entsize;
166  if (Index >= STEntries)
167  {
168  error("Symbol index out of range %d-%u.", Table, Index);
169  return 0xdead;
170  }
171 
172  uint64_t SymbolAddress = (uint64_t)Header + SymbolTable->sh_offset;
173  Elf64_Sym *Symbol = &((Elf64_Sym *)SymbolAddress)[Index];
174 
175  if (Symbol->st_shndx == SHN_UNDEF)
176  {
177  Elf64_Shdr *StringTable = GetELFSection(Header, SymbolTable->sh_link);
178  const char *Name = (const char *)Header + StringTable->sh_offset + Symbol->st_name;
179 
180  void *Target = (void *)ELFLookupSymbol(Header, Name)->st_value;
181  if (Target == nullptr)
182  {
183  if (ELF64_ST_BIND(Symbol->st_info) & STB_WEAK)
184  return 0;
185  else
186  {
187  error("Undefined external symbol \"%s\".", Name);
188  return 0xdead;
189  }
190  }
191  else
192  return (uintptr_t)Target;
193  }
194  else if (Symbol->st_shndx == SHN_ABS)
195  return Symbol->st_value;
196  else
197  {
198  Elf64_Shdr *Target = GetELFSection(Header, Symbol->st_shndx);

204  void CopyLOADSegments(VirtualFileSystem::File &ElfFile, uintptr_t HdrsBase, uintptr_t PhysicalBase)
205  {
206  off_t OldOffset = vfs->Seek(ElfFile, 0, SEEK_CUR);
207 
208  Elf64_Ehdr ELFHeader;
209  vfs->Seek(ElfFile, 0, SEEK_SET);
210  vfs->Read(ElfFile, (uint8_t *)&ELFHeader, sizeof(Elf64_Ehdr));
211 
212  Elf64_Phdr ProgramHeaders;
213  for (Elf64_Half i = 0; i < ELFHeader.e_phnum; i++)
214  {
215  vfs->Seek(ElfFile, ELFHeader.e_phoff + (i * sizeof(Elf64_Phdr)), SEEK_SET);
216  vfs->Read(ElfFile, (uint8_t *)&ProgramHeaders, sizeof(Elf64_Phdr));
217  if (ProgramHeaders.p_type != PT_LOAD)
218  continue;
219 
220  uintptr_t SegmentDestination = (ProgramHeaders.p_vaddr - HdrsBase) + PhysicalBase;
221 
222  debug("Copying segment to %#lx (%ld file bytes, %ld mem bytes)",
223  SegmentDestination,
224  ProgramHeaders.p_filesz, ProgramHeaders.p_memsz);
225 
226  if (ProgramHeaders.p_filesz > 0)
227  {
228  vfs->Seek(ElfFile, ProgramHeaders.p_offset, SEEK_SET);
229  vfs->Read(ElfFile, (uint8_t *)SegmentDestination, ProgramHeaders.p_filesz);
230  }
231 
232  if (ProgramHeaders.p_memsz - ProgramHeaders.p_filesz > 0)
233  memset((void *)(SegmentDestination + ProgramHeaders.p_filesz),
234  0,
235  ProgramHeaders.p_memsz - ProgramHeaders.p_filesz);
236  }
237 
238  vfs->Seek(ElfFile, OldOffset, SEEK_SET);
239  }

241  void GetBaseAndSize(VirtualFileSystem::File &ElfFile, uintptr_t &Base, size_t &Size)
242  {
243  Base = UINTPTR_MAX;
244  Size = 0;
245 
246  off_t OldOffset = vfs->Seek(ElfFile, 0, SEEK_CUR);
247 
248  Elf64_Ehdr ELFHeader;
249  vfs->Seek(ElfFile, 0, SEEK_SET);
250  vfs->Read(ElfFile, (uint8_t *)&ELFHeader, sizeof(Elf64_Ehdr));
251 
252  Elf64_Phdr ProgramHeaders;
253  vfs->Seek(ElfFile, ELFHeader.e_phoff, SEEK_SET);
254  vfs->Read(ElfFile, (uint8_t *)&ProgramHeaders, sizeof(Elf64_Phdr));
255 
256  for (Elf64_Half i = 0; i < ELFHeader.e_phnum; i++)
257  {
258  Base = MIN(Base, ProgramHeaders.p_vaddr);
259  uintptr_t SegmentEnd = ProgramHeaders.p_vaddr - Base + ProgramHeaders.p_memsz;
260  Size = MAX(Size, SegmentEnd);
261 
262  vfs->Seek(ElfFile, sizeof(Elf64_Phdr), SEEK_CUR);
263  vfs->Read(ElfFile, (uint8_t *)&ProgramHeaders, sizeof(Elf64_Phdr));
264  }
265  debug("Base: %#lx, Size: %#lx (%ld, %ld KB)", Base, Size, Size, TO_KB(Size));
266  vfs->Seek(ElfFile, OldOffset, SEEK_SET);
267  }
