224 std::atomic<bool> ExceptionLock = false;
225 nsa void HandleException(CPU::ExceptionFrame *Frame)
226 {
227  /* We don't need to restore the page table
228  because the ExceptionHandlerStub will
229  do it for us if we return. */
230  CPU::PageTable(KernelPageTable);
231  InitFont();
232 
233  /* First check the exception */
234  if (unlikely(Frame->InterruptNumber == CPU::x86::DoubleFault))
235  {
236  ExceptionLock.store(true, std::memory_order_release);
237  HandleUnrecoverableException(Frame);
238  }
239 
240  if (Frame->cs == GDT_USER_CODE && Frame->cs == GDT_USER_DATA)
241  {
242  if (UserModeExceptionHandler(Frame))
243  goto ExceptionExit;
244 
245  CPUData *core = GetCurrentCPU();
246  if (likely(core->CurrentThread->Security.IsCritical == false))
247  TaskManager->Yield();
248 
249  error("Critical thread \"%s\"(%d) died",
250  core->CurrentThread->Name,
251  core->CurrentThread->ID);
252  }
253 
254  debug("-----------------------------------------------------------------------------------");
255  error("Exception: %#x", Frame->InterruptNumber);
256  debug("%ld MiB / %ld MiB (%ld MiB Reserved)",
257  TO_MiB(KernelAllocator.GetUsedMemory()),
258  TO_MiB(KernelAllocator.GetTotalMemory()),
259  TO_MiB(KernelAllocator.GetReservedMemory()));
260 
261  if (ExceptionLock.load(std::memory_order_acquire))
262  HandleExceptionInsideException(Frame);
263  ExceptionLock.store(true, std::memory_order_release);
264 
265  {
266  const char msg[] = "Entering in panic mode...";
267  size_t msgLen = strlen(msg);
268  size_t msgPixels = msgLen * CrashFont->GetInfo().Width;
269  uint32_t x = uint32_t((Display->GetWidth - msgPixels) / 2);
270  uint32_t y = uint32_t((Display->GetHeight - CrashFont->GetInfo().Height) / 2);
271  Display->SetBufferCursor(x, y);
272  Display->PrintString("\eFF2525");
273  Display->PrintString(msg, CrashFont);
274  Display->UpdateBuffer();
275  }
276 
277  if (DriverManager)
278  DriverManager->Panic();
279  if (TaskManager)
280  TaskManager->Panic();
281  Interrupts::RemoveAll();
282  HaltAllCores();
283  ForceUnlock = true;
284 
285  DisplayCrashScreen(Frame);
286  CPU::Stop();
287 
288 ExceptionExit:
289  ExceptionLock.store(false, std::memory_order_release);
290 }
