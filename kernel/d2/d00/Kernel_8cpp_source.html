588 EXTERNC __no_stack_protector NIF void Entry(BootInfo *Info)
589 {
590  trace("Hello, World!");
591 
592  if (strcmp(CPU::Hypervisor(), x86_CPUID_VENDOR_TCG) == 0)
593  {
594  debug("\n\n----------------------------------------\nDEBUGGER DETECTED\n----------------------------------------\n\n");
595  DebuggerIsAttached = true;
596  }
597 
598  memcpy(&bInfo, Info, sizeof(BootInfo));
599  debug("BootInfo structure is at %p", &bInfo);
600 
601  // https://wiki.osdev.org/Calling_Global_Constructors
602  trace("There are %d constructors to call", __init_array_end - __init_array_start);
603  for (CallPtr *func = __init_array_start; func != __init_array_end; func++)
604  (*func)();
605 
606 #ifdef a86
607  if (!bInfo.SMBIOSPtr)
608  {
609  trace("SMBIOS was not provided by the bootloader. Trying to find it manually.");
610  for (uintptr_t i = 0xF0000; i < 0x100000; i += 16)
611  {
612  if (memcmp((void *)i, "_SM_", 4) == 0 || memcmp((void *)i, "_SM3_", 5) == 0)
613  {
614  bInfo.SMBIOSPtr = (void *)i;
615  trace("Found SMBIOS at %#lx", i);
616  }
617  }
618  }
619 
620  if (!bInfo.RSDP)
621  {
622  trace("RSDP was not provided by the bootloader. Trying to find it manually.");
623  /* FIXME: Not always shifting by 4 will work. */
624  uintptr_t EBDABase = (uintptr_t)mminw((void *)0x40E) << 4;
625 
626  for (uintptr_t ptr = EBDABase;
627  ptr < 0x100000; /* 1MB */
628  ptr += 16)
629  {
630  if (unlikely(ptr == EBDABase + 0x400))
631  {
632  trace("EBDA is full. Trying to find RSDP in the BIOS area.");
633  break;
634  }
635 
636  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
637  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
638  {
639  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
640  trace("Found RSDP at %#lx", rsdp);
641  }
642  }
643 
644  for (uintptr_t ptr = 0xE0000;
645  ptr < 0x100000; /* 1MB */
646  ptr += 16)
647  {
648  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
649  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
650  {
651  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
652  trace("Found RSDP at %#lx", rsdp);
653  }
654  }
655  }
656 #endif
657 
658  InitializeMemoryManagement();
659 
660  void *KernelStackAddress = KernelAllocator.RequestPages(TO_PAGES(STACK_SIZE));
661  uintptr_t KernelStack = (uintptr_t)KernelStackAddress + STACK_SIZE - 0x10;
662  debug("Kernel stack: %#lx-%#lx", KernelStackAddress, KernelStack);
663 #if defined(a64)
664  asmv("mov %0, %%rsp"
665  :
666  : "r"(KernelStack)
667  : "memory");

669 #elif defined(a32)
