318 {
319  char GetScanCode(uint8_t ScanCode, bool Upper)
320  {
321  ScanCode &= 0x7F; /* Remove KEY_PRESSED bit */
322  if (ScanCode >= sizeof(ScanCodeConversionTableLower))
323  {
324  warn("Unknown scancode %x", ScanCode);
325  return 0x00;
326  }
327 
328  // debug("Scancode %x (%s)", ScanCode, ScanCodeDebugNames[ScanCode]);
329  return Upper
330  ? ScanCodeConversionTableUpper[ScanCode]
331  : ScanCodeConversionTableLower[ScanCode];
332  }
333 
334  bool IsValidChar(uint8_t ScanCode)
335  {
336  ScanCode &= 0x7F; /* Remove KEY_PRESSED bit */
337  if (ScanCode >= sizeof(ScanCodeConversionTableLower))
338  return false;
339 
340  if (ScanCode > KEY_M)
341  {
342  if (ScanCode < KEYPAD_7)
343  return false; /* F1 - F12 */
344 
345  switch (ScanCode)
346  {
347  case KEY_MINUS:
348  case KEY_EQUAL:
349  case KEY_LEFT_BRACKET:
350  case KEY_RIGHT_BRACKET:
351  case KEY_RETURN:
352  case KEY_SEMICOLON:
353  case KEY_APOSTROPHE:
354  case KEY_BACK_TICK:
355  case KEY_BACKSLASH:
356  case KEY_COMMA:
357  case KEY_PERIOD:
358  case KEY_SLASH:
359  case KEY_SPACE:
360  return true;
361 
362  default:
363  return false;
364  }
365  }
366  return true;
367  }
368 }
