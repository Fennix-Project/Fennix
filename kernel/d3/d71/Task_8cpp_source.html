736 #endif
737  switch (Parent->Security.TrustLevel)
738  {
739  case TaskTrustLevel::System:
740  warn("Trust level not supported.");

742  case TaskTrustLevel::Idle:
743  case TaskTrustLevel::Kernel:

758 #elif defined(__aarch64__)

762  case TaskTrustLevel::User:
763  {
764  Thread->Stack = new Memory::StackGuard(true, Parent->PageTable);
765  Thread->Memory = new Memory::Tracker(Parent->PageTable);
766 #if defined(__amd64__)
767  SecurityManager.TrustToken(Thread->Security.UniqueToken, TokenTrustLevel::Untrusted);
768  Thread->GSBase = 0;
769  Thread->FSBase = 0;
770  Thread->Registers.cs = GDT_USER_CODE;
771  Thread->Registers.ss = GDT_USER_DATA;
772  Thread->Registers.rflags.AlwaysOne = 1;
773  // Thread->Registers.rflags.PF = 1;
774  // Thread->Registers.rflags.SF = 1;
775  // Thread->Registers.rflags.IOPL = 3;
776  Thread->Registers.rflags.IF = 1;
777  Thread->Registers.rflags.ID = 1;
778  Thread->Registers.rsp = ((uint64_t)Thread->Stack->GetStackTop());
779 
780  uint64_t ArgvSize = 0;
781  if (argv)
782  while (argv[ArgvSize] != nullptr)
783  ArgvSize++;
784 
785  uint64_t EnvpSize = 0;
786  if (envp)
787  while (envp[EnvpSize] != nullptr)
788  EnvpSize++;
789 
790  debug("ArgvSize: %d", ArgvSize);
791  debug("EnvpSize: %d", EnvpSize);
792 
793  /* https://articles.manugarg.com/aboutelfauxiliaryvectors.html */
794  /* https://refspecs.linuxbase.org/elf/x86_64-abi-0.99.pdf#figure.3.9 */
795  // rsp is the top of the stack
796  char *Stack = (char *)Thread->Stack->GetStackPhysicalTop();
797  // Temporary stack pointer for strings
798  char *StackStrings = (char *)Stack;
799  char *StackStringsVirtual = (char *)Thread->Stack->GetStackTop();
800 
801  // Store string pointers for later
802  uint64_t ArgvStrings[ArgvSize];
803  uint64_t EnvpStrings[EnvpSize];
804 
805  for (uint64_t i = 0; i < ArgvSize; i++)
806  {
807  // Subtract the length of the string and the null terminator
808  StackStrings -= strlen(argv[i]) + 1;
809  StackStringsVirtual -= strlen(argv[i]) + 1;
810  // Store the pointer to the string
811  ArgvStrings[i] = (uint64_t)StackStringsVirtual;
812  // Copy the string to the stack
813  strcpy(StackStrings, argv[i]);
814  }
815 
816  for (uint64_t i = 0; i < EnvpSize; i++)
817  {
818  // Subtract the length of the string and the null terminator
819  StackStrings -= strlen(envp[i]) + 1;
820  StackStringsVirtual -= strlen(envp[i]) + 1;
821  // Store the pointer to the string
822  EnvpStrings[i] = (uint64_t)StackStringsVirtual;
823  // Copy the string to the stack
824  strcpy(StackStrings, envp[i]);
825  }
826 
827  // Align the stack to 16 bytes
828  StackStrings -= (uint64_t)StackStrings & 0xF;
829  // Set "Stack" to the new stack pointer
830  Stack = (char *)StackStrings;
831  // If argv and envp sizes are odd then we need to align the stack
832  Stack -= (ArgvSize + EnvpSize) % 2;
833 
834  // We need 8 bit pointers for the stack from here
835  uint64_t *Stack64 = (uint64_t *)Stack;
836 
837  // Store the null terminator
838  Stack64--;
839  *Stack64 = AT_NULL;
840 
841  // Store auxillary vector
842  foreach (AuxiliaryVector var in auxv)
843  {
844  // Subtract the size of the auxillary vector
845  Stack64 -= sizeof(Elf64_auxv_t) / sizeof(uint64_t);
846  // Store the auxillary vector
847  POKE(Elf64_auxv_t, Stack64) = var.archaux;
848  // TODO: Store strings to the stack
849  }
850 
851  // Store the null terminator
852  Stack64--;
853  *Stack64 = AT_NULL;
854 
855  // Store EnvpStrings[] to the stack
856  Stack64 -= EnvpSize; // (1 Stack64 = 8 bits; Stack64 = 8 * EnvpSize)
857  for (uint64_t i = 0; i < EnvpSize; i++)
858  {
859  *(Stack64 + i) = (uint64_t)EnvpStrings[i];
860  debug("EnvpStrings[%d]: %#lx", i, EnvpStrings[i]);
861  }
862 
863  // Store the null terminator
864  Stack64--;
865  *Stack64 = AT_NULL;
866 
867  // Store ArgvStrings[] to the stack
868  Stack64 -= ArgvSize; // (1 Stack64 = 8 bits; Stack64 = 8 * ArgvSize)
869  for (uint64_t i = 0; i < ArgvSize; i++)
870  {
871  *(Stack64 + i) = (uint64_t)ArgvStrings[i];
872  debug("ArgvStrings[%d]: %#lx", i, ArgvStrings[i]);
873  }
874 
875  // Store the argc
876  Stack64--;
877  *Stack64 = ArgvSize;
878 
879  // Set "Stack" to the new stack pointer
880  Stack = (char *)Stack64;
881 
882  /* We need the virtual address but because we are in the kernel we can't use the process page table.
883  So we modify the physical address and store how much we need to subtract to get the virtual address for RSP. */
884  uint64_t SubtractStack = (uint64_t)Thread->Stack->GetStackPhysicalTop() - (uint64_t)Stack;
885  debug("SubtractStack: %#lx", SubtractStack);
886 
887  // Set the stack pointer to the new stack
888  Thread->Registers.rsp = ((uint64_t)Thread->Stack->GetStackTop() - SubtractStack);
889 
890 #ifdef DEBUG
891  DumpData("Stack Data", (void *)((uint64_t)Thread->Stack->GetStackPhysicalTop() - (uint64_t)SubtractStack), SubtractStack);
892 #endif
893 
894  Thread->Registers.rdi = (uint64_t)ArgvSize; // argc
895  Thread->Registers.rsi = (uint64_t)(Thread->Registers.rsp + 8); // argv
896  Thread->Registers.rcx = (uint64_t)EnvpSize; // envc
897  Thread->Registers.rdx = (uint64_t)(Thread->Registers.rsp + 8 + (8 * ArgvSize) + 8); // envp
898 
899  /* We need to leave the libc's crt to make a syscall when the Thread is exited or we are going to get GPF or PF exception. */
900 
901  Memory::Virtual uva = Memory::Virtual(Parent->PageTable);
902  if (!uva.Check((void *)Offset, Memory::PTFlag::US))
903  {
904  error("Offset is not user accessible");
905  uva.Map((void *)Offset, (void *)Offset, Memory::PTFlag::RW | Memory::PTFlag::US); // We try one more time.
906  }
907 #elif defined(__i386__)
908 #elif defined(__aarch64__)
909 #endif
910 #ifdef DEBUG_SCHEDULER
911  DumpData(Thread->Name, Thread->Stack, STACK_SIZE);
912 #endif
913  break;
914  }
915  default:
916  {
917  error("Unknown elevation.");
918  delete Thread->Stack;
919  this->NextTID--;
920  delete Thread;
921  return nullptr;
922  }
923  }
924 
925  Thread->Security.TrustLevel = Parent->Security.TrustLevel;
926  // Thread->Security.UniqueToken = SecurityManager.CreateToken();
927 
928  Thread->Info = {};
929  Thread->Info.SpawnTime = CPU::Counter();
930  Thread->Info.Year = 0;
931  Thread->Info.Month = 0;
932  Thread->Info.Day = 0;
933  Thread->Info.Hour = 0;
934  Thread->Info.Minute = 0;
935  Thread->Info.Second = 0;
936  for (int i = 0; i < MAX_CPU; i++)
937  {
938  Thread->Info.Usage[i] = 0;
939  Thread->Info.Affinity[i] = true;
940  }
941  Thread->Info.Priority = 10;
942  Thread->Info.Architecture = Architecture;
943  Thread->Info.Compatibility = Compatibility;
944 
945  debug("Thread offset is %#lx (EntryPoint: %#lx) => RIP: %#lx", Thread->Offset, Thread->EntryPoint, Thread->Registers.rip);
946  if (Parent->Security.TrustLevel == TaskTrustLevel::User)
947  debug("Thread stack region is %#lx-%#lx (U) and rsp is %#lx", Thread->Stack->GetStackBottom(), Thread->Stack->GetStackTop(), Thread->Registers.rsp);
948  else

950  debug("Created thread \"%s\"(%d) in process \"%s\"(%d)",
951  Thread->Name, Thread->ID,
952  Thread->Parent->Name, Thread->Parent->ID);
953 
954  Parent->Threads.push_back(Thread);
