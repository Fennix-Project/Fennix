25  bool Virtual::Check(void *VirtualAddress, PTFlag Flag, MapType Type)
26  {
27  // 0x1000 aligned
28  uintptr_t Address = (uintptr_t)VirtualAddress;
29  Address &= 0xFFFFF000;
30 
31  PageMapIndexer Index = PageMapIndexer(Address);
32  PageDirectoryEntry *PDE = &this->Table->Entries[Index.PDEIndex];
33  PageTableEntryPtr *PTE = nullptr;
34 
35  if ((PDE->raw & Flag) > 0)
36  {
37  if (Type == MapType::FourMiB && PDE->PageSize)
38  return true;
39 
40  PTE = (PageTableEntryPtr *)((uintptr_t)PDE->GetAddress() << 12);
41  if (PTE)
42  {
43  if ((PTE->Entries[Index.PTEIndex].Present))
44  return true;
45  }
46  }
47  return false;
48  }

130  void Virtual::Map(void *VirtualAddress, void *PhysicalAddress, uint64_t Flags, MapType Type)
131  {
132  SmartLock(this->MemoryLock);
133  if (unlikely(!this->Table))
134  {
135  error("No page table");
136  return;
137  }
138 
139  Flags |= PTFlag::P;
140 
141  PageMapIndexer Index = PageMapIndexer((uintptr_t)VirtualAddress);
142  // Clear any flags that are not 1 << 0 (Present) - 1 << 5 (Accessed) because rest are for page table entries only
143  uint64_t DirectoryFlags = Flags & 0x3F;
144 
145  PageDirectoryEntry *PDE = &this->Table->Entries[Index.PDEIndex];
146  if (Type == MapType::FourMiB)
147  {
148  PDE->raw |= (uintptr_t)Flags;
149  PDE->PageSize = true;
150  PDE->SetAddress((uintptr_t)PhysicalAddress >> 12);
151  debug("Mapped 4MB page at %p to %p", VirtualAddress, PhysicalAddress);
152  return;
153  }
154 
155  PageTableEntryPtr *PTEPtr = nullptr;
156  if (!PDE->Present)
157  {
158  PTEPtr = (PageTableEntryPtr *)KernelAllocator.RequestPages(TO_PAGES(sizeof(PageTableEntryPtr) + 1));
159  memset(PTEPtr, 0, sizeof(PageTableEntryPtr));
160  PDE->Present = true;
161  PDE->SetAddress((uintptr_t)PTEPtr >> 12);
162  }
163  else
164  PTEPtr = (PageTableEntryPtr *)(PDE->GetAddress() << 12);
165  PDE->raw |= (uintptr_t)DirectoryFlags;
166 
167  PageTableEntry *PTE = &PTEPtr->Entries[Index.PTEIndex];
168  PTE->Present = true;
169  PTE->raw |= (uintptr_t)Flags;
170  PTE->SetAddress((uintptr_t)PhysicalAddress >> 12);
171  CPU::x32::invlpg(VirtualAddress);
172 
173 #ifdef DEBUG
174 /* https://stackoverflow.com/a/3208376/9352057 */
175 #define BYTE_TO_BINARY_PATTERN "%c%c%c%c%c%c%c%c"
176 #define BYTE_TO_BINARY(byte) \
177  (byte & 0x80 ? '1' : '0'), \
178  (byte & 0x40 ? '1' : '0'), \
179  (byte & 0x20 ? '1' : '0'), \
180  (byte & 0x10 ? '1' : '0'), \
181  (byte & 0x08 ? '1' : '0'), \
182  (byte & 0x04 ? '1' : '0'), \
183  (byte & 0x02 ? '1' : '0'), \
184  (byte & 0x01 ? '1' : '0')
185 
186  if (!this->Check(VirtualAddress, (PTFlag)Flags, Type)) // quick workaround just to see where it fails
187  warn("Failed to map v:%#lx p:%#lx with flags: " BYTE_TO_BINARY_PATTERN, VirtualAddress, PhysicalAddress, BYTE_TO_BINARY(Flags));
188 #endif
189  }

191  void Virtual::Unmap(void *VirtualAddress, MapType Type)
192  {
193  SmartLock(this->MemoryLock);
194  if (!this->Table)
195  {
196  error("No page table");
197  return;
198  }
199 
200  PageMapIndexer Index = PageMapIndexer((uintptr_t)VirtualAddress);
201  PageDirectoryEntry *PDE = &this->Table->Entries[Index.PDEIndex];
202  if (!PDE->Present)
203  {
204  warn("Page %#lx not present", PDE->GetAddress());
205  return;
206  }
207 
208  if (Type == MapType::FourMiB && PDE->PageSize)
209  {
210  PDE->Present = false;
211  return;
212  }
213 
214  PageTableEntryPtr *PTEPtr = (PageTableEntryPtr *)((uintptr_t)PDE->Address << 12);
215  PageTableEntry PTE = PTEPtr->Entries[Index.PTEIndex];
216  if (!PTE.Present)
217  {
218  warn("Page %#lx not present", PTE.GetAddress());
219  return;
220  }
221 
222  PTE.Present = false;
223  PTEPtr->Entries[Index.PTEIndex] = PTE;
224  CPU::x32::invlpg(VirtualAddress);
225  }
