47  void SetEntry(uint8_t Index, void (*Base)(),
48  InterruptStackTableType InterruptStackTable,
49  GateType Gate, PrivilegeLevelType Ring,
50  bool Present, uint16_t SegmentSelector)
51  {
52  switch (Gate)
53  {
54  case CALL_GATE_64BIT:
55  {
56  CallGate gate{
57  .TargetOffsetLow = s_cst(uint16_t, ((uint64_t)Base & 0xFFFF)),
58  .TargetSelector = SegmentSelector,
59  .Reserved0 = 0,
60  .Type = Gate,
61  .Zero0 = 0,
62  .DescriptorPrivilegeLevel = Ring,
63  .Present = Present,
64  .TargetOffsetMiddle = s_cst(uint16_t, ((uint64_t)Base >> 16)),
65  .TargetOffsetHigh = s_cst(uint32_t, ((uint64_t)Base >> 32)),
66  .Reserved1 = 0,
67  .Zero1 = 0,
68  .Reserved2 = 0,
69  };
70  Entries[Index].Call = gate;
71  break;
72  }
73  case INTERRUPT_GATE_64BIT:
74  {
75  InterruptGate gate{
76  .TargetOffsetLow = s_cst(uint16_t, ((uint64_t)Base & 0xFFFF)),
77  .TargetSelector = SegmentSelector,
78  .InterruptStackTable = InterruptStackTable,
79  .Reserved0 = 0,
80  .Type = Gate,
81  .Zero = 0,
82  .DescriptorPrivilegeLevel = Ring,
83  .Present = Present,
84  .TargetOffsetMiddle = s_cst(uint16_t, ((uint64_t)Base >> 16)),
85  .TargetOffsetHigh = s_cst(uint32_t, ((uint64_t)Base >> 32)),
86  .Reserved1 = 0,
87  };
88  Entries[Index].Interrupt = gate;
89  break;
90  }
91  case TRAP_GATE_64BIT:
92  {
93  TrapGate gate{
94  .TargetOffsetLow = s_cst(uint16_t, ((uint64_t)Base & 0xFFFF)),
95  .TargetSelector = SegmentSelector,
96  .InterruptStackTable = InterruptStackTable,
97  .Reserved0 = 0,
98  .Type = Gate,
99  .Zero = 0,
100  .DescriptorPrivilegeLevel = Ring,
101  .Present = Present,
102  .TargetOffsetMiddle = s_cst(uint16_t, ((uint64_t)Base >> 16)),
103  .TargetOffsetHigh = s_cst(uint32_t, ((uint64_t)Base >> 32)),
104  .Reserved1 = 0,
105  };
106  Entries[Index].Trap = gate;
107  break;
108  }
109  case LDT_64BIT:
110  case AVAILABLE_64BIT_TSS:
111  case BUSY_64BIT_TSS:
112  default:
113  {
114  assert(!"Unsupported gate type");
115  break;
116  }
117  }
