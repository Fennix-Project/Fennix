130 uintptr_t HandleNativeSyscalls(SysFrm *Frame)
131 {
132 #if defined(a64)
133  if (unlikely(Frame->rax > sc_MaxSyscall))
134  {
135  fixme("Syscall %ld not implemented.", Frame->rax);
136  return -ENOSYS;
137  }
138 
139  SyscallData Syscall = NativeSyscallsTable[Frame->rax];
140 
141  uintptr_t (*call)(SysFrm *, uintptr_t, ...) =
142  r_cst(uintptr_t(*)(SysFrm *, uintptr_t, ...),
143  Syscall.Handler);
144 
145  if (unlikely(!call))
146  {
147  error("Syscall %s(%d) not implemented.",
148  Syscall.Name, Frame->rax);
149  return -ENOSYS;
150  }
151 
152  int euid = thisProcess->Security.Effective.UserID;
153  int egid = thisProcess->Security.Effective.GroupID;
154  int reqID = Syscall.RequiredID;
155  if (euid > reqID || egid > reqID)
156  {
157  warn("Process %s(%d) tried to access a system call \"%s\" with insufficient privileges.",
158  thisProcess->Name, thisProcess->ID, Syscall.Name);
159  debug("Required: %d; Effective u:%d, g:%d", reqID, euid, egid);
160  return -EPERM;
161  }
162 
163  debug("> [%d:\"%s\"]( %#lx %#lx %#lx %#lx %#lx %#lx )",
164  Frame->rax, Syscall.Name,
165  Frame->rdi, Frame->rsi, Frame->rdx,
166  Frame->r10, Frame->r8, Frame->r9);
167 
168  long sc_ret = call(Frame,
169  Frame->rdi, Frame->rsi, Frame->rdx,
170  Frame->r10, Frame->r8, Frame->r9);
171 
172  debug("< [%d:\"%s\"] = %d",
173  Frame->rax, Syscall.Name, sc_ret);
174  return sc_ret;
175 #elif defined(a32)
176  if (unlikely(Frame->eax > sc_MaxSyscall))
177  {
178  fixme("Syscall %ld not implemented.", Frame->eax);
179  return -ENOSYS;
180  }
181 
182  SyscallData Syscall = NativeSyscallsTable[Frame->eax];
183 
184  uintptr_t (*call)(SysFrm *, uintptr_t, ...) =
185  r_cst(uintptr_t(*)(SysFrm *, uintptr_t, ...),
186  Syscall.Handler);
187 
188  if (unlikely(!call))
189  {
190  error("Syscall %s(%d) not implemented.",
191  Syscall.Name, Frame->eax);
192  return -ENOSYS;
193  }
194 
195  int euid = thisProcess->Security.Effective.UserID;
196  int egid = thisProcess->Security.Effective.GroupID;
197  int reqID = Syscall.RequiredID;
198  if (euid > reqID || egid > reqID)
199  {
200  warn("Process %s(%d) tried to access a system call \"%s\" with insufficient privileges.",
201  thisProcess->Name, thisProcess->ID, Syscall.Name);
202  debug("Required: %d; Effective u:%d, g:%d", reqID, euid, egid);
203  return -EPERM;
204  }
205 
206  debug("> [%d:\"%s\"]( %#x %#x %#x %#x %#x %#x )",
207  Frame->eax, Syscall.Name,
208  Frame->ebx, Frame->ecx, Frame->edx,
209  Frame->esi, Frame->edi, Frame->ebp);
210 
211  int sc_ret = call(Frame,
212  Frame->ebx, Frame->ecx, Frame->edx,
213  Frame->esi, Frame->edi, Frame->ebp);
214 
215  debug("< [%d:\"%s\"] = %d",
216  Frame->eax, Syscall.Name, sc_ret);
217  return sc_ret;
218 #elif defined(aa64)
219  return -ENOSYS;
220 #endif
221 }
