34 void sys_fork_return()
35 {
36 #if defined(a64)
37  asmv("movq $0, %rax\n"); /* Return 0 */
38  asmv("movq %r8, %rsp\n"); /* Restore stack pointer */
39  asmv("movq %r8, %rbp\n"); /* Restore base pointer */
40  asmv("swapgs\n"); /* Swap GS back to the user GS */
41  asmv("sti\n"); /* Enable interrupts */
42  asmv("sysretq\n"); /* Return to rcx address in user mode */
43 #elif defined(a32)
44 #warning "sys_fork not implemented for i386"
45 #endif
46  __builtin_unreachable();
47 }
