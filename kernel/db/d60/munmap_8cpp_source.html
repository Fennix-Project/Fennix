32 /* https://pubs.opengroup.org/onlinepubs/009604499/functions/munmap.html */
33 int sys_munmap(SysFrm *,
34  void *addr, size_t len)
35 {
36  if (uintptr_t(addr) % PAGE_SIZE)
37  return -EINVAL;
38 
39  if (len == 0)
40  return -EINVAL;
41 
42  PCB *pcb = thisProcess;
43  VirtualMemoryArea *vma = pcb->vma;
44  Virtual vmm = Virtual(pcb->PageTable);
45 
46  for (uintptr_t i = uintptr_t(addr);
47  i < uintptr_t(addr) + len;
48  i += PAGE_SIZE)
49  {
50  if (likely(!vmm.Check((void *)i, G)))
51  vmm.Remap((void *)i, (void *)i, P | RW);
52  else
53  warn("%p is a global page", (void *)i);
54  }
55 
56  /* TODO: Check if the page is allocated
57  and not only mapped */
58  vma->FreePages((void *)addr, TO_PAGES(len) + 1);
59  return 0;
60 }
