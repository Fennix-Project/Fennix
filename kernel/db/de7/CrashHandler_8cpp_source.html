104  SafeFunction char *TrimWhiteSpace(char *str)
105  {
106  char *end;
107  while (*str == ' ')
108  str++;
109  if (*str == 0)
110  return str;
111  end = str + strlen(str) - 1;
112  while (end > str && *end == ' ')
113  end--;
114  *(end + 1) = 0;
115  return str;
