93  off_t ReferenceNode::Seek(off_t _Offset, int Whence)
94  {
95  if (this->SymlinkTo)
96  return this->SymlinkTo->Seek(_Offset, Whence);
97 
98  if (!this->node->Operator)
99  {
100  errno = EFAULT;
101  return -1;
102  }
103 
104  SmartLock(RefNodeLock);
105 
106  if (this->node->Operator->Seek)
107  {
108  off_t RefOffset = off_t(this->Offset.load());
109  debug("The node has a seek function");
110  return this->node->Operator->Seek(this->node, _Offset, Whence, RefOffset);
111  }
112 
113  debug("Current offset is %d", this->Offset.load());
114  switch (Whence)
115  {
116  case SEEK_SET:
117  {
118  if (_Offset > this->node->Length)
119  {
120  errno = EINVAL;
121  return -1;
122  }
123 
124  if (_Offset < 0)
125  {
126  fixme("Negative offset %d is not implemented", _Offset);
127  _Offset = 0;
128  }
129 
130  if (_Offset > this->node->Length)
131  {
132  fixme("Offset %d is bigger than file size %d",
133  _Offset, this->node->Length);
134  _Offset = this->node->Length;
135  }
136 
137  this->Offset.store(_Offset);
138  break;
139  }
140  case SEEK_CUR:
141  {
142  off_t NewOffset = off_t(this->Offset.load()) + _Offset;
143  if (NewOffset > this->node->Length ||
144  NewOffset < 0)
145  {
146  errno = EINVAL;
147  return -1;
148  }
149 
150  this->Offset.store(NewOffset);
151  break;
152  }
153  case SEEK_END:
154  {
155  off_t NewOffset = this->node->Length + _Offset;
156  if (NewOffset > this->node->Length ||
157  NewOffset < 0)
158  {
159  errno = EINVAL;
160  return -1;
161  }
162 
163  this->Offset.store(NewOffset);
164  break;
165  }
166  default:
167  {
168  error("Invalid whence!");
169  errno = EINVAL;
170  return -1;
171  }
172  }
173 
174  off_t RetOffset = off_t(this->Offset.load());
175  debug("( %d %ld %s[%d] ) -> %d",
176  _Offset, this->Offset.load(),
177  SeekStrings[Whence], Whence,
178  RetOffset);
179  return RetOffset;
180  }
