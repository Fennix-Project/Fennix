38  std::string Virtual::GetPathFromNode(Node *File)
39  {
40  vfsdbg("GetPathFromNode( Node: \"%s\" )", File->Name);
41  SmartLock(VirtualLock);
42 
43  Node *Parent = File;
44  char **Path = nullptr;
45  size_t PathSize = 0;
46  std::string FinalPath;
47 
48  while (Parent != FileSystemRoot && Parent != nullptr)
49  {
50  if (File == FileSystemRoot->Children[0])
51  {
52  FinalPath = "/";
53  break;
54  }
55 
56  bool Found = false;
57  foreach (const auto &Children in FileSystemRoot->Children)
58  if (Children == Parent)
59  {
60  Found = true;
61  break;
62  }
63 
64  if (Found)
65  break;
66 
67  if (strlen(Parent->Name) == 0)
68  break;
69 
70  char **new_path = new char *[PathSize + 1];
71  if (Path != nullptr)
72  {
73  memcpy(new_path, Path, sizeof(char *) * PathSize);
74  delete[] Path;
75  }
76 
77  Path = new_path;
78  Path[PathSize] = (char *)Parent->Name;
79  PathSize++;
80  new_path = new char *[PathSize + 1];
81  memcpy(new_path, Path, sizeof(char *) * PathSize);
82  delete[] Path;
83  Path = new_path;
84  Path[PathSize] = (char *)"/";
85  PathSize++;
86  Parent = Parent->Parent;
87  }
88 
89  for (size_t i = PathSize - 1; i < PathSize; i--)
90  {
91  if (Path[i] == nullptr)
92  continue;
93  FinalPath += Path[i];
94  }
95 
96  FinalPath += "\0";
97  delete[] Path, Path = nullptr;
98  vfsdbg("GetPathFromNode()->\"%s\"", FinalPath.c_str());
99  return FinalPath;
100  }

102  Node *Virtual::GetNodeFromPath_Unsafe(const char *Path, Node *Parent)
103  {
104  vfsdbg("GetNodeFromPath( Path: \"%s\" Parent: \"%s\" )",
105  Path, Parent ? Parent->Name : "(null)");
106 
107  if (strcmp(Path, "/") == 0)
108  return FileSystemRoot->Children[0]; // 0 - filesystem root
109 
110  if (strcmp(Path, ".") == 0)
111  return Parent;
112 
113  if (strcmp(Path, "..") == 0)
114  {
115  if (Parent)
116  {
117  if (Parent->Parent)
118  return Parent->Parent;
119  else
120  return Parent;
121  }
122  else
123  return nullptr;
124  }
125 
126  Node *ReturnNode = Parent;
127  bool IsAbsolutePath = cwk_path_is_absolute(Path);
128 
129  if (!ReturnNode)
130  ReturnNode = FileSystemRoot->Children[0]; // 0 - filesystem root
131 
132  if (IsAbsolutePath)
133  ReturnNode = FileSystemRoot->Children[0]; // 0 - filesystem root
134 
135  cwk_segment segment;
136  if (unlikely(!cwk_path_get_first_segment(Path, &segment)))
137  {
138  error("Path doesn't have any segments.");
139  errno = ENOENT;
140  return nullptr;
141  }
142 
143  do
144  {
145  char *SegmentName = new char[segment.end - segment.begin + 1];
146  memcpy(SegmentName, segment.begin, segment.end - segment.begin);
147  vfsdbg("GetNodeFromPath()->SegmentName: \"%s\"", SegmentName);
148  GetNodeFromPathNextParent:
149  foreach (auto Child in ReturnNode->Children)
150  {
151  vfsdbg("comparing \"%s\" with \"%s\"",
152  Child->Name, SegmentName);
153  if (strcmp(Child->Name, SegmentName) == 0)
154  {
155  ReturnNode = Child;
156  goto GetNodeFromPathNextParent;
157  }
158  }
159  delete[] SegmentName;
160  } while (cwk_path_get_next_segment(&segment));
161 
162  const char *basename;
163  cwk_path_get_basename(Path, &basename, nullptr);
164  vfsdbg("BaseName: \"%s\" NodeName: \"%s\"",
165  basename, ReturnNode->Name);
166 
167  if (strcmp(basename, ReturnNode->Name) == 0)
168  {
169  vfsdbg("GetNodeFromPath()->\"%s\"", ReturnNode->Name);
170  return ReturnNode;
171  }
172 
173  vfsdbg("GetNodeFromPath()->\"(null)\"");
174  errno = ENOENT;
175  return nullptr;
176  }

373  {

394  VirtualLock.Lock(__FUNCTION__);
395 
396  cwk_segment segment;
397  if (!cwk_path_get_first_segment(CleanPath.c_str(), &segment))
398  {
399  error("Path doesn't have any segments.");
400  goto CreatePathError;
401  }
402 
403  do
404  {
405  char *SegmentName = new char[segment.end - segment.begin + 1];
406  memcpy(SegmentName, segment.begin, segment.end - segment.begin);
407  vfsdbg("SegmentName: \"%s\"", SegmentName);
408

410  {
411  if (GetChild(SegmentName, RootNode) != nullptr)
412  {

418

446  int Virtual::Delete(const char *Path, bool Recursive, Node *Parent)
447  {
448  vfsdbg("Virtual::Delete( Path: \"%s\" Parent: \"%s\" )",
449  Path, Parent ? Parent->Name : "(null)");
450 
451  if (isempty((char *)Path))
452  {
453  errno = EINVAL;
454  return -1;
455  }
456 
457  if (Parent == nullptr)
458  Parent = FileSystemRoot;
459 
460  std::string CleanPath = this->NormalizePath(Path, Parent);
461  vfsdbg("CleanPath: \"%s\"", CleanPath.c_str());
462 
463  if (!PathExists(CleanPath.c_str(), Parent))
464  {
465  vfsdbg("Path %s doesn't exist.", CleanPath.c_str());
466  errno = ENOENT;
467  return -1;
468  }
469 
470  Node *NodeToDelete = GetNodeFromPath(CleanPath.c_str(), Parent);
471 
472  if (NodeToDelete->Flags == NodeFlags::DIRECTORY)
473  {
474  SmartLock(VirtualLock);
475  if (Recursive)
476  {
477  foreach (auto Child in NodeToDelete->Children)
478  {
479  VirtualLock.Unlock();
480  int Status = Delete(GetPathFromNode(Child).c_str(), true);
481  VirtualLock.Lock(__FUNCTION__);
482  if (Status != 0)
483  {
484  error("Failed to delete child %s with status %d. (%s)",
485  Child->Name, Status, Path);
486  errno = EIO;
487  return -1;
488  }
489  }
490  }
491  else if (NodeToDelete->Children.size() > 0)
492  {
493  error("Directory %s is not empty.", CleanPath.c_str());
494  errno = ENOTEMPTY;
495  return -1;
496  }
497  }
498 
499  SmartLock(VirtualLock);
500  Node *ParentNode = GetParent(CleanPath.c_str(), Parent);
501  if (RemoveChild(NodeToDelete->Name, ParentNode) != 0)
502  {
503  error("Failed to remove child %s from parent %s. (%s)", NodeToDelete->Name, ParentNode->Name, Path);
504  errno = EIO;
505  return -1;
506  }
507 
508  debug("Deleted %s", CleanPath.c_str());
509  vfsdbg("Virtual::Delete()->OK");
510  return 0;
511  }
