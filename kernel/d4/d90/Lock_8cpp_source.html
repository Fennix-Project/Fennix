110 void LockClass::DeadLock(SpinLockData &Lock)
111 {

113  {
114  warn("Unlocking lock '%s' which it was held by '%s'...",
115  Lock.AttemptingToGet, Lock.CurrentHolder);
116  this->DeadLocks = 0;

124  CCore = CoreData->ID;
125 
126  warn("Potential deadlock in lock '%s' held by '%s'! %ld %s in queue. Interrupts are %s. Core %ld held by %ld. (%ld times happened)",
127  Lock.AttemptingToGet, Lock.CurrentHolder,
128  Lock.Count, Lock.Count > 1 ? "locks" : "lock",
129  CPU::Interrupts(CPU::Check) ? "enabled" : "disabled",

150 int LockClass::Lock(const char *FunctionName)
151 {
152  LockData.AttemptingToGet = FunctionName;
153  LockData.StackPointerAttempt = (uintptr_t)__builtin_frame_address(0);
154 
155 Retry:
156  int i = 0;
157  while (IsLocked.exchange(true, std::memory_order_acquire) &&
158  ++i < (DebuggerIsAttached ? DEADLOCK_TIMEOUT_DEBUGGER : DEADLOCK_TIMEOUT))
159  {
160  this->Yield();
161  }
162 
163  if (i >= (DebuggerIsAttached ? DEADLOCK_TIMEOUT_DEBUGGER : DEADLOCK_TIMEOUT))
164  {
165  DeadLock(LockData);
166  goto Retry;
167  }
168 
169  LockData.Count++;
170  LockData.CurrentHolder = FunctionName;
171  LockData.StackPointerHolder = (uintptr_t)__builtin_frame_address(0);
172 
173  CPUData *CoreData = GetCurrentCPU();
174  if (CoreData != nullptr)
175  LockData.Core = CoreData->ID;
176 
177  LocksCount++;
178 
179  __sync;
180  return 0;
181 }
