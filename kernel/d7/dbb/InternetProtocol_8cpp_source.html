37  void IPv4::Send(uint8_t *Data, size_t Length, uint8_t Protocol, InternetProtocol DestinationIP)
38  {
39  netdbg("Sending %ld bytes to %s", Length, DestinationIP.v4.ToStringLittleEndian());
40  IPv4Packet *Packet = (IPv4Packet *)kmalloc(Length + sizeof(IPv4Header));
41 
42  /* This part is the most confusing one for me. */
43  Packet->Header.Version = b8(4);
44  Packet->Header.IHL = b8(sizeof(IPv4Header) / 4);
45  Packet->Header.TypeOfService = b8(0);
46  /* We don't byteswap. */
47  Packet->Header.TotalLength = s_cst(uint16_t, Length + sizeof(IPv4Header));
48  Packet->Header.TotalLength = s_cst(uint16_t, ((Packet->Header.TotalLength & 0xFF00) >> 8) | ((Packet->Header.TotalLength & 0x00FF) << 8));
49 
50  Packet->Header.Identification = b16(0x0000);
51  Packet->Header.Flags = b8(0x0);
52  Packet->Header.FragmentOffset = b8(0x0);
53  Packet->Header.TimeToLive = b8(64);
54  Packet->Header.Protocol = b8(Protocol);
55  Packet->Header.DestinationIP = b32(DestinationIP.v4.ToHex());
56  Packet->Header.SourceIP = b32(Ethernet->GetInterface()->IP.v4.ToHex());
57  Packet->Header.HeaderChecksum = b16(0x0);
58  Packet->Header.HeaderChecksum = CalculateChecksum((uint16_t *)Packet, sizeof(IPv4Header));
59 
60  memcpy(Packet->Data, Data, Length);
61  InternetProtocol DestinationRoute = DestinationIP;
62  if ((DestinationIP.v4.ToHex() & SubNetworkMaskIP.v4.ToHex()) != (b32(Packet->Header.SourceIP) & SubNetworkMaskIP.v4.ToHex()))
63  DestinationRoute = SubNetworkMaskIP;
64 
65  Ethernet->Send(MediaAccessControl().FromHex(ARP->Resolve(DestinationRoute)), this->GetFrameType(), (uint8_t *)Packet, Length + sizeof(IPv4Header));
66  kfree(Packet);
67  }

71  __no_sanitize("alignment") bool IPv4::OnEthernetPacketReceived(uint8_t *Data, size_t Length)
72  {
73  IPv4Packet *Packet = (IPv4Packet *)Data;
74  netdbg("Received %d bytes [Protocol %ld]", Length, Packet->Header.Protocol);
75  if (Length < sizeof(IPv4Header))
76  {
77  warn("Packet too short");
78  return false;
79  }
80 
81  bool Reply = false;
82 
83  if (b32(Packet->Header.DestinationIP) == Ethernet->GetInterface()->IP.v4.ToHex() || b32(Packet->Header.DestinationIP) == 0xFFFFFFFF || Ethernet->GetInterface()->IP.v4.ToHex() == 0)
84  {
85  size_t TotalLength = Packet->Header.TotalLength;
86  if (TotalLength > Length)
87  TotalLength = Length;
88 
89  foreach (auto Event in RegisteredEvents)
90  if (Packet->Header.Protocol == Event->GetProtocol())
91  {
92  InternetProtocol SourceIP;
93  InternetProtocol DestinationIP;
94 
95  SourceIP.v4 = SourceIP.v4.FromHex(b32(Packet->Header.SourceIP));
96  DestinationIP.v4 = DestinationIP.v4.FromHex(b32(Packet->Header.DestinationIP));
97 
98  if (Event->OnIPv4PacketReceived(SourceIP, DestinationIP, (uint8_t *)((uint64_t)Data + 4 * Packet->Header.IHL), TotalLength - 4 * Packet->Header.IHL))
99  Reply = true;
100  }
101  }
102  else
103  {
104  netdbg("Not for us. We are %s but this is for %s",
105  Ethernet->GetInterface()->IP.v4.ToHex(),
106  Packet->Header.DestinationIP);
107  }
108 
109  if (Reply)
110  {
111  uint32_t SwapIP = Packet->Header.DestinationIP;
112  Packet->Header.DestinationIP = Packet->Header.SourceIP;
113  Packet->Header.SourceIP = SwapIP;
114  Packet->Header.TimeToLive = 0x40;
115  Packet->Header.HeaderChecksum = 0x0;
116  Packet->Header.HeaderChecksum = CalculateChecksum((uint16_t *)Data, 4 * Packet->Header.TotalLength);
117  NIManager->Send(Ethernet->GetInterface(), (uint8_t *)Data, Length);
118  }
119  return Reply;
120  }
