120 extern "C" int __cxa_atexit(void (*f)(void *), void *objptr, void *dso)
121 {
122  fixme("__cxa_atexit( %p %p %p ) triggered.", f, objptr, dso);
123  if (__atexit_func_count >= ATEXIT_MAX_FUNCS)
124  return -1;
125  __atexit_funcs[__atexit_func_count].destructor_func = f;
126  __atexit_funcs[__atexit_func_count].obj_ptr = objptr;
127  __atexit_funcs[__atexit_func_count].dso_handle = dso;
128  __atexit_func_count++;
129  return 0;
130 }

132 extern "C" void __cxa_finalize(void *f)
133 {
134  fixme("__cxa_finalize( %p ) triggered.", f);
135  uarch_t i = __atexit_func_count;
136  if (!f)
137  {
138  while (i--)
139  if (__atexit_funcs[i].destructor_func)
140  (*__atexit_funcs[i].destructor_func)(__atexit_funcs[i].obj_ptr);
141 
142  return;
143  }
144 
145  while (i--)
146  if (__atexit_funcs[i].destructor_func == f)
147  {
148  (*__atexit_funcs[i].destructor_func)(__atexit_funcs[i].obj_ptr);
149  __atexit_funcs[i].destructor_func = 0;
150  }
151 }

166 extern "C" void *__cxa_allocate_exception(size_t thrown_size) throw()
167 {
168  fixme("__cxa_allocate_exception( %d ) triggered.", thrown_size);
169 
170  size_t real_size = align_exception_allocation_size(thrown_size + sizeof(__cxa_exception), alignof(__cxa_exception));
171 
172  __cxa_exception *header = (__cxa_exception *)kmalloc(real_size);
173  if (!header)
174  {
175  error("Failed to allocate exception.");
176  return nullptr;
177  }
178 
179  header->referenceCount = 1;
180  header->exceptionType = nullptr;
181  header->exceptionDestructor = nullptr;
182  header->unexpectedHandler = {.unexpected = unexpected_header_stub};
183  header->terminateHandler = {.handler = terminate_header_stub};
184  header->nextException = nullptr;
185  header->handlerCount = -1;
186  header->handlerSwitchValue = 0;
187  header->actionRecord = nullptr;
188  header->languageSpecificData = nullptr;
189  header->catchTemp = 0;
190  header->adjustedPtr = nullptr;
191 
192  return header + 1;
193 }
