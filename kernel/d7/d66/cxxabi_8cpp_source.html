44 extern "C" int __cxa_atexit(void (*f)(void *), void *objptr, void *dso)
45 {
46  if (KernelSymbolTable)
47  {
48  debug("Registering atexit function for \"%s\" with destructor \"%s\"",
49  KernelSymbolTable->GetSymbol((uintptr_t)objptr),
50  KernelSymbolTable->GetSymbol((uintptr_t)f));
51  }
52  else
53  {
54  debug("Registering atexit function for %p with destructor %p",
55  objptr, f);
56  }
57 
58  if (__atexit_func_count >= ATEXIT_MAX_FUNCS)
59  return -1;
60  __atexit_funcs[__atexit_func_count].destructor_func = f;
61  __atexit_funcs[__atexit_func_count].obj_ptr = objptr;
62  __atexit_funcs[__atexit_func_count].dso_handle = dso;
63  __atexit_func_count++;
64  return 0;
65 }

67 extern "C" void __cxa_finalize(void *f)
68 {
69  func("%p", f);
70  uarch_t i = __atexit_func_count;
71  if (f == nullptr)
72  {
73  while (i--)
74  {
75  if (__atexit_funcs[i].destructor_func)
76  {
77  if (KernelSymbolTable)
78  {
79  debug("Calling atexit function \"%s\"",
80  KernelSymbolTable->GetSymbol((uintptr_t)__atexit_funcs[i].destructor_func));
81  }
82  else
83  {
84  debug("Calling atexit function %p",
85  __atexit_funcs[i].destructor_func);
86  }
87  (*__atexit_funcs[i].destructor_func)(__atexit_funcs[i].obj_ptr);
88  }
89  }
90  return;
91  }
92 
93  while (i--)
94  {
95  if (__atexit_funcs[i].destructor_func == f)
96  {
97  (*__atexit_funcs[i].destructor_func)(__atexit_funcs[i].obj_ptr);
98  __atexit_funcs[i].destructor_func = 0;
99  }
100  }
101 }

169 {
