130  uint32_t BlendColors(uint32_t c1, uint32_t c2, float t)
131  {
132  uint8_t r1 = (c1 >> 16) & 0xFF;
133  uint8_t g1 = (c1 >> 8) & 0xFF;
134  uint8_t b1 = c1 & 0xFF;
135  uint8_t r2 = (c2 >> 16) & 0xFF;
136  uint8_t g2 = (c2 >> 8) & 0xFF;
137  uint8_t b2 = c2 & 0xFF;
138 
139  uint8_t r = (uint8_t)(r1 + t * (r2 - r1));
140  uint8_t g = (uint8_t)(g1 + t * (g2 - g1));
141  uint8_t b = (uint8_t)(b1 + t * (b2 - b1));
142 
143  return (r << 16) | (g << 8) | b;
144  }
