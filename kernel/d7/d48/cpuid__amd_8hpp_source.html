32 #if defined(a86)
33 #define __amd_cpuid_init(leaf) \
34  CPUID##leaf() \
35  { \
36  asmv("cpuid" \
37  : "=a"(EAX.raw), "=b"(EBX.raw), \
38  "=c"(ECX.raw), "=d"(EDX.raw) \
39  : "a"(leaf)); \
40  if (!EAX.raw && !EBX.raw && !ECX.raw && !EDX.raw) \
41  warn("cpuid not supported"); \
42  }
43 
44 #define __amd_cpuid_init2(leaf, leaf2, suffix) \
45  CPUID##leaf##suffix() \
46  { \
47  asmv("cpuid" \
48  : "=a"(EAX.raw), "=b"(EBX.raw), \
49  "=c"(ECX.raw), "=d"(EDX.raw) \
50  : "a"(leaf), "c"(leaf2)); \
51  if (!EAX.raw && !EBX.raw && !ECX.raw && !EDX.raw) \
52  warn("cpuid not supported"); \
53  }

58  assert(!"cpuid not implemented for this architecture"); \
59  }
