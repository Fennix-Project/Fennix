149  bool Interrupts(InterruptsType Type)
150  {
151  switch (Type)
152  {
153  case Check:
154  {
155  uintptr_t Flags;
156 #if defined(a64)
157  asmv("pushfq");
158  asmv("popq %0"
159  : "=r"(Flags));
160  return Flags & (1 << 9);
161 #elif defined(a32)
162  asmv("pushfl");
163  asmv("popl %0"
164  : "=r"(Flags));
165  return Flags & (1 << 9);
166 #elif defined(aa64)
167  asmv("mrs %0, cpsr"
168  : "=r"(Flags));
169  return Flags & (1 << 7);
170 #endif
171  }
172  case Enable:
173  {
174 #if defined(a86)
175  asmv("sti");
176 #elif defined(aa64)
177  asmv("cpsie i");
178 #endif
179  return true;
180  }
181  case Disable:
182  {
183 #if defined(a86)
184  asmv("cli");
185 #elif defined(aa64)
186  asmv("cpsid i");
187 #endif
188  return true;
189  }
190  default:
191  assert(!"Unknown InterruptsType");
192  break;
193  }
194  return false;
195  }

276  void InitializeFeatures(int Core)
277  {
278  static int BSP = 0;
279  SupportedFeat feat = GetCPUFeat();
280 
281  CR0 cr0 = readcr0();
282  CR4 cr4 = readcr4();
283 
284  if (Config.SIMD == false)
285  {
286  debug("Disabling SSE support...");
287  feat.SSE = false;
288  }
289 
290  if (feat.PGE)
291  {
292  debug("Enabling global pages support...");
293  if (!BSP)
294  KPrint("Global Pages is supported.");
295  cr4.PGE = true;
296  }
297 
298  bool SSEEnableAfter = false;
299 
300  /* Not sure if my code is not working properly or something else is the issue. */
301  if ((strcmp(Hypervisor(), x86_CPUID_VENDOR_VIRTUALBOX) != 0) &&
302  feat.SSE)
303  {
304  debug("Enabling SSE support...");
305  if (!BSP)
306  KPrint("SSE is supported.");
307  cr0.EM = false;
308  cr0.MP = true;
309  cr4.OSFXSR = true;
310  cr4.OSXMMEXCPT = true;
311 
312  CPUData *CoreData = GetCPU(Core);
313  CoreData->Data.FPU.mxcsr = 0b0001111110000000;
314  CoreData->Data.FPU.mxcsrmask = 0b1111111110111111;
315  CoreData->Data.FPU.fcw = 0b0000001100111111;
316  fxrstor(&CoreData->Data.FPU);
317 
318  SSEEnableAfter = true;
319  }
320 
321  /* More info in AMD64 Architecture Programmer's Manual
322  Volume 2: 3.1.1 CR0 Register */
323 
324  /* Not Write-Through
325  This is ignored on recent processors.
326  */
327  cr0.NW = false;
328 
329  /* Cache Disable
330  Wether the CPU should cache memory or not.
331  If it's enabled, PWT and PCD are ignored.
332  */
333  cr0.CD = false;
334 
335  /* Write Protect
336  When set, the supervisor can't write to read-only pages.
337  */
338  cr0.WP = true;
339 
340  /* Alignment check
341  The CPU checks the alignment of memory operands
342  and generates #AC if the alignment is incorrect.
343 
344  The condition for an alignment check is:
345  - The AM flag in CR0 is set.
346  - The AC flag in the RFLAGS register is set.
347  - CPL is 3.
348  */
349  cr0.AM = true;
350 
351  debug("Updating CR0...");
352  writecr0(cr0);
353  debug("Updated CR0.");
354 
355  debug("CPU Prevention Features:%s%s%s",
356  feat.SMEP ? " SMEP" : "",
357  feat.SMAP ? " SMAP" : "",
358  feat.UMIP ? " UMIP" : "");
359  /* User-Mode Instruction Prevention
360  This prevents user-mode code from executing these instructions:
361  SGDT, SIDT, SLDT, SMSW, STR
362  If any of these instructions are executed with CPL > 0, a #GP is generated.
363  */
364  // cr4.UMIP = feat.UMIP;
365 
366  /* Supervisor Mode Execution Prevention
367  This prevents user-mode code from executing code in the supervisor mode.
368  */
369  cr4.SMEP = feat.SMEP;
370 
371  /* Supervisor Mode Access Prevention
372  This prevents supervisor-mode code from accessing user-mode pages.
373  */
374  cr4.SMAP = feat.SMAP;
375 
376  debug("Updating CR4...");
377  writecr4(cr4);
378  debug("Updated CR4.");
379 
380  debug("Enabling PAT support...");
381  wrmsr(MSR_CR_PAT, 0x6 | (0x0 << 8) | (0x1 << 16));
382  if (!BSP++)
383  trace("Features for BSP initialized.");
384  if (SSEEnableAfter)
385  {
386  SSEEnabled = true;
387  debug("SSE support enabled.");
388  }
389  }
