73  bool AddLibrary(char *Identifier, void *ElfImage, size_t Length, const Memory::Virtual &pV)
74  {
75  SmartLock(ExecuteServiceLock);
76  SharedLibraries sl;
77 
78  foreach (auto lib in Libs)
79  {
80  if (strcmp(lib.Identifier, Identifier) == 0)

97  Memory::Virtual ncpV = pV;

101  {
102  uintptr_t BaseAddress = UINTPTR_MAX;
103  Elf64_Phdr ItrProgramHeader;
104 
105  for (Elf64_Half i = 0; i < ((Elf64_Ehdr *)LibFile)->e_phnum; i++)
106  {
107  memcpy(&ItrProgramHeader, (uint8_t *)LibFile + ((Elf64_Ehdr *)LibFile)->e_phoff + ((Elf64_Ehdr *)LibFile)->e_phentsize * i, sizeof(Elf64_Phdr));

109  }
110 
111  for (Elf64_Half i = 0; i < ((Elf64_Ehdr *)LibFile)->e_phnum; i++)
112  {
