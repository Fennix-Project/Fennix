25 void SearchForInitrd()
26 {
27  for (size_t i = 0; i < MAX_MODULES; i++)
28  {
29  uintptr_t initrdAddress = (uintptr_t)bInfo.Modules[i].Address;
30 
31  if (!initrdAddress)
32  continue;
33 
34  if (strcmp(bInfo.Modules[i].CommandLine, "initrd") != 0)
35  continue;
36 
37  KPrint("initrd found at %#lx", initrdAddress);
38 
39  Memory::Virtual vmm;
40  if (!vmm.Check((void *)initrdAddress))
41  {
42  warn("Initrd is not mapped!");
43  vmm.Map((void *)initrdAddress, (void *)initrdAddress,
44  bInfo.Modules[i].Size, Memory::RW);
45  }
46 
47  if (TestAndInitializeUSTAR(initrdAddress, bInfo.Modules[i].Size))
48  continue; /* Maybe add another root? */
49  }
50 }
