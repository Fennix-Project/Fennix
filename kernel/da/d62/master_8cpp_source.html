134  int MasterDeviceFile::ReportKeyEvent(maj_t ID, min_t MinorID, uint8_t ScanCode)
135  {
136  debug("New key event: %02x", ScanCode);
137  if (this->SlavesMap.find(ID) == this->SlavesMap.end())
138  return -EINVAL;
139 
140  std::unordered_map<min_t, SlaveDeviceFile *> *slave = this->SlavesMap[ID];
141  if ((*slave).find(MinorID) == (*slave).end())
142  return -EINVAL;
143 
144  /* We are master, keep a copy of the scancode and
145  converted key */
146 
147  if (RawKeyQueue.size() > 16)
148  RawKeyQueue.pop_front();
149  RawKeyQueue.push_back(ScanCode);
150 
151  if (KeyQueue.size() > 16)
152  KeyQueue.pop_front();
153 
154  switch (ScanCode & ~KEY_PRESSED)
155  {
156  case KEY_LEFT_SHIFT:
157  case KEY_RIGHT_SHIFT:
158  {
159  if (ScanCode & KEY_PRESSED)
160  UpperCase = true;
161  else
162  UpperCase = false;
163  break;
164  }
165  case KEY_CAPS_LOCK:
166  {
167  if (ScanCode & KEY_PRESSED)
168  CapsLock = !CapsLock;
169  break;
170  }
171  default:
172  break;
173  }
174  KeyQueue.push_back(GetScanCode(ScanCode, UpperCase || CapsLock));
175 
176  SlaveDeviceFile *sdf = (*slave)[MinorID];
177  return sdf->ReportKeyEvent(ScanCode);
178  }

239  dev_t MasterDeviceFile::Register(maj_t ID)
240  {
241  debug("Registering slave device %d", ID);
242  Slaves slave;
243  if (this->SlavesMap.find(ID) != this->SlavesMap.end())
244  slave = this->SlavesMap[ID];
245  else
246  slave = new std::unordered_map<min_t, SlaveDeviceFile *>();
247

251  this->SlaveParent,

253  this->Type);
254 
255  sdf->DeviceMajor = ID;
256  sdf->DeviceMinor = this->SlaveIDCounter;
257 
258  (*slave)[this->SlaveIDCounter] = sdf;
259  this->SlavesMap[ID] = slave;

263  int MasterDeviceFile::Unregister(maj_t ID, min_t MinorID)
264  {
265  debug("Unregistering slave device %d:%d", ID, MinorID);
266  if (this->SlavesMap.find(ID) == this->SlavesMap.end())
267  return -EINVAL;
268 
269  std::unordered_map<min_t, SlaveDeviceFile *> *slave = this->SlavesMap[ID];
270  if ((*slave).find(MinorID) == (*slave).end())
271  return -EINVAL;
272 
273  SlaveDeviceFile *sdf = (*slave)[MinorID];
274  delete sdf;
275  slave->erase(MinorID);
276  if (slave->empty())
277  {
278  delete slave;
279  this->SlavesMap.erase(ID);
280  }
281  return 0;
282  }
