387 void __ubsan_handle_type_mismatch_v1(struct type_mismatch_v1_data *type_mismatch, uintptr_t pointer)
388 {
389  struct source_location *location = &type_mismatch->location;
390  if (pointer == 0)
391  {
392  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
393  ubsan("Null pointer access.");
394  }
395  else if (type_mismatch->alignment != 0 && is_aligned(pointer, type_mismatch->alignment))
396  {
397  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
398  ubsan("Unaligned memory access %#lx.", pointer);
399  }
400  else
401  {
402  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
403  ubsan("%s address %#lx with insufficient space for object of type %s",
404  Type_Check_Kinds[type_mismatch->type_check_kind],
405  (void *)pointer, type_mismatch->type->name);
406  }
407 }
