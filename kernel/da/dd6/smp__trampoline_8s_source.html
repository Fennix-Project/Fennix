21 TRAMPOLINE_STACK = 0x570

29 /* ========== 16-bit ========== */
30 
31 .code16
32 .global _trampoline_start
33 _trampoline_start:
34  cli
35  cld
36  call Trampoline16
37 
38 Trampoline16:
39  mov $0x0, %ax
40  mov %ax, %ds
41  mov %ax, %es
42  mov %ax, %fs
43  mov %ax, %gs
44  mov %ax, %ss
45 
46  /* Load Protected Mode GDT */
47  lgdt [ProtectedMode_gdtr - _trampoline_start + TRAMPOLINE_START]
48 
49  /* Enable Protected Mode */
50  mov %cr0, %eax
51  or $0x1, %al
52  mov %eax, %cr0
53 
54  /* Jump to Protected Mode */
55  ljmp $0x8, $(Trampoline32 - _trampoline_start + TRAMPOLINE_START)
56 
57 /* ========== 32-bit ========== */
58 
59 .code32
60 Trampoline32:
61  mov $0x10, %bx
62  mov %bx, %ds
63  mov %bx, %es
64  mov %bx, %ss
65 
66  /* Set a page table */
67  mov [TRAMPOLINE_PAGE_TABLE], %eax
68  mov %eax, %cr3
69 
70  /* Enable PAE and PSE */
71  mov %cr4, %eax
72  or $0x20, %eax /* PAE */
73  or $0x80, %eax /* PSE */
74  mov %eax, %cr4
75 
76  /* Enable Long Mode */
77  mov $0xC0000080, %ecx
78  rdmsr
79  or $0x100, %eax /* LME */
80  wrmsr
81 
82  /* Enable paging */
83  mov %cr0, %eax
84  or $0x80000000, %eax /* PG */
85  mov %eax, %cr0
86 
87  /* Load Long Mode GDT */
88  lgdt [LongMode_gdtr - _trampoline_start + TRAMPOLINE_START]
89 
90  /* Jump to Long Mode */
91  ljmp $0x8, $(Trampoline64 - _trampoline_start + TRAMPOLINE_START)
92 
93 /* ========== 64-bit ========== */
94 
95 .code64
96 Trampoline64:
97  mov $0x10, %ax
98  mov %ax, %ds
99  mov %ax, %es
100  mov %ax, %ss
101 
102  mov $0x0, %ax
103  mov %ax, %fs
104  mov %ax, %gs
105 
106  /* Set custom GDT & IDT */
107  lgdt [TRAMPOLINE_GDT]
108  lidt [TRAMPOLINE_IDT]
109 
110  /* Set up stack */
111  mov [TRAMPOLINE_STACK], %rsp
112  mov $0x0, %rbp
113 
114  /* Reset RFLAGS */
115  push $0x0
116  popf
117 
118  /* Jump to TrampolinePrepareExit */
119  call TrampolineExit
120

122 TrampolineExit:
123  mov $StartCPU, %rax
