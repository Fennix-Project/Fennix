37  void *VirtualMemoryArea::RequestPages(size_t Count, bool User, bool Protect)
38  {
39  func("%lld, %s, %s", Count,
40  User ? "true" : "false",
41  Protect ? "true" : "false");
42 
43  void *Address = KernelAllocator.RequestPages(Count);
44  memset(Address, 0, Count * PAGE_SIZE);
45 
46  int Flags = PTFlag::RW;
47  if (User)
48  Flags |= PTFlag::US;
49  if (Protect)
50  Flags |= PTFlag::KRsv;
51 
52  Virtual vmm(this->Table);
53 
54  SmartLock(MgrLock);
55 
56  vmm.Map(Address, Address, FROM_PAGES(Count), Flags);
57  AllocatedPagesList.push_back({Address, Count, Protect});
58  debug("%#lx +{%#lx, %lld}", this, Address, Count);
59  return Address;
60  }

62  void VirtualMemoryArea::FreePages(void *Address, size_t Count)
63  {
64  func("%#lx, %lld", Address, Count);
65 
66  SmartLock(MgrLock);
67  forItr(itr, AllocatedPagesList)
68  {
69  if (itr->Address != Address)
70  continue;
71 
72  if (itr->Protected)
73  {
74  error("Address %#lx is protected", Address);
75  return;
76  }
77 
84  if (itr->PageCount != Count)
85  {
86  error("Page count mismatch! (Allocated: %lld, Requested: %lld)",
87  itr->PageCount, Count);
88  return;
89  }
90 
91  Virtual vmm(this->Table);
92  for (size_t i = 0; i < Count; i++)
93  {
94  void *AddressToMap = (void *)((uintptr_t)Address + (i * PAGE_SIZE));
95  vmm.Remap(AddressToMap, AddressToMap, PTFlag::RW);
96  }
97 
98  KernelAllocator.FreePages(Address, Count);
99  AllocatedPagesList.erase(itr);
100  debug("%#lx -{%#lx, %lld}", this, Address, Count);
101  return;
102  }
103  }

194  bool VirtualMemoryArea::HandleCoW(uintptr_t PFA)
195  {
196  func("%#lx", PFA);
197  Virtual vmm(this->Table);
198  PageTableEntry *pte = vmm.GetPTE((void *)PFA);
199 
200  debug("ctx: %#lx", this);
201 
202  if (!pte)
203  {
204  /* Unmapped page */
205  debug("PTE is null!");
206  return false;
207  }
208 
209  if (!pte->CopyOnWrite)
210  {
211  debug("PFA %#lx is not CoW (pt %#lx) (flags %#lx)",
212  PFA, this->Table, pte->raw);
213  return false;
214  }
215 
216  foreach (auto sr in SharedRegions)
217  {
218  uintptr_t Start = (uintptr_t)sr.Address;
219  uintptr_t End = (uintptr_t)sr.Address + sr.Length;
220  debug("Start: %#lx, End: %#lx (PFA: %#lx)",
221  Start, End, PFA);
222 
223  if (PFA >= Start && PFA < End)
224  {
225  if (sr.Shared)
226  {
227  fixme("Shared CoW");
228  return false;
229  }
230 
231  void *pAddr = this->RequestPages(1);
232  if (pAddr == nullptr)
233  return false;
234  memset(pAddr, 0, PAGE_SIZE);
235 
236  assert(pte->Present == true);
237  pte->ReadWrite = sr.Write;
238  pte->UserSupervisor = sr.Read;
239  pte->ExecuteDisable = sr.Exec;
240 
241  pte->CopyOnWrite = false;
242  debug("PFA %#lx is CoW (pt %#lx, flags %#lx)",
243  PFA, this->Table, pte->raw);
244 #if defined(a64)
245  CPU::x64::invlpg((void *)PFA);
246 #elif defined(a32)
247  CPU::x32::invlpg((void *)PFA);
248 #endif
249  return true;
250  }
251  }
252 
253  debug("%#lx not found in CoW regions", PFA);
254  return false;
255  }

272  void VirtualMemoryArea::Fork(VirtualMemoryArea *Parent)
273  {
274  func("%#lx", Parent);
275  assert(Parent);
276 
277  debug("parent apl:%d sr:%d [P:%#lx C:%#lx]",
278  Parent->AllocatedPagesList.size(), Parent->SharedRegions.size(),
279  Parent->Table, this->Table);
280  debug("ctx: this: %#lx parent: %#lx", this, Parent);
281 
282  Virtual vmm(this->Table);
283  SmartLock(MgrLock);
284  foreach (auto &ap in Parent->AllocatedPagesList)
285  {
286  if (ap.Protected)
287  {
288  debug("Protected %#lx-%#lx", ap.Address,
289  (uintptr_t)ap.Address + (ap.PageCount * PAGE_SIZE));
290  continue; /* We don't want to modify these pages. */
291  }
292 
293  MgrLock.Unlock();
294  void *Address = this->RequestPages(ap.PageCount);
295  MgrLock.Lock(__FUNCTION__);
296  if (Address == nullptr)
297  return;
298 
299  memcpy(Address, ap.Address, FROM_PAGES(ap.PageCount));
300 
301  for (size_t i = 0; i < ap.PageCount; i++)
302  {
303  void *AddressToMap = (void *)((uintptr_t)ap.Address + (i * PAGE_SIZE));
304  void *RealAddress = (void *)((uintptr_t)Address + (i * PAGE_SIZE));
305 
306 #if defined(a86)
307  PageTableEntry *pte = vmm.GetPTE(AddressToMap);
308  uintptr_t Flags = 0;
309  Flags |= pte->Present ? (uintptr_t)PTFlag::P : 0;
310  Flags |= pte->ReadWrite ? (uintptr_t)PTFlag::RW : 0;
311  Flags |= pte->UserSupervisor ? (uintptr_t)PTFlag::US : 0;
312  Flags |= pte->CopyOnWrite ? (uintptr_t)PTFlag::CoW : 0;
313  Flags |= pte->KernelReserve ? (uintptr_t)PTFlag::KRsv : 0;
314 
315  debug("Mapping %#lx to %#lx (flags %s/%s/%s/%s/%s)",
316  RealAddress, AddressToMap,
317  Flags & PTFlag::P ? "P" : "-",
318  Flags & PTFlag::RW ? "RW" : "-",
319  Flags & PTFlag::US ? "US" : "-",
320  Flags & PTFlag::CoW ? "CoW" : "-",
321  Flags & PTFlag::KRsv ? "KRsv" : "-");
322  MgrLock.Unlock();
323  this->Map(AddressToMap, RealAddress, PAGE_SIZE, Flags);
324  MgrLock.Lock(__FUNCTION__);
325 #else
326 #error "Not implemented"
327 #endif
328  }
329 
330  debug("Forked %#lx-%#lx", ap.Address,
331  (uintptr_t)ap.Address + (ap.PageCount * PAGE_SIZE));
332  }
333 
334  foreach (auto &sr in Parent->SharedRegions)
335  {
336  MgrLock.Unlock();
337  void *Address = this->CreateCoWRegion(sr.Address, sr.Length,
338  sr.Read, sr.Write, sr.Exec,
339  sr.Fixed, sr.Shared);
340  MgrLock.Lock(__FUNCTION__);
341  if (Address == nullptr)
342  return;
343  memcpy(Address, sr.Address, sr.Length);
344  debug("Forked CoW region %#lx-%#lx", sr.Address,
345  (uintptr_t)sr.Address + sr.Length);
346  }
347  }
