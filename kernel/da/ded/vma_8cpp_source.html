74  bool VirtualMemoryArea::Add(void *Address, size_t Count)
75  {
76  SmartLock(MgrLock);
77  function("%#lx, %lld", Address, Count);
78 
79  if (Address == nullptr)
80  {
81  error("Address is null!");
82  return false;
83  }
84 
85  if (Count == 0)
86  {
87  error("Count is 0!");
88  return false;
89  }
90 
91  for (size_t i = 0; i < AllocatedPagesList.size(); i++)
92  {
93  if (AllocatedPagesList[i].Address == Address)
94  {
95  error("Address already exists!");
96  return false;
97  }
98  else if ((uintptr_t)Address < (uintptr_t)AllocatedPagesList[i].Address)
99  {
100  if ((uintptr_t)Address + (Count * PAGE_SIZE) > (uintptr_t)AllocatedPagesList[i].Address)
101  {
102  error("Address intersects with an allocated page!");
103  return false;
104  }
105  }
106  else
107  {
108  if ((uintptr_t)AllocatedPagesList[i].Address + (AllocatedPagesList[i].PageCount * PAGE_SIZE) > (uintptr_t)Address)
109  {
110  error("Address intersects with an allocated page!");
111  return false;
112  }
113  }
114  }
115 
116  AllocatedPagesList.push_back({Address, Count});
117  return true;
118  }

120  void *VirtualMemoryArea::RequestPages(size_t Count, bool User)
121  {
122  SmartLock(MgrLock);
123  function("%lld, %s", Count, User ? "true" : "false");
124 
125  void *Address = KernelAllocator.RequestPages(Count);
126  memset(Address, 0, Count * PAGE_SIZE);
127  for (size_t i = 0; i < Count; i++)
128  {
129  int Flags = Memory::PTFlag::RW;
130  if (User)
131  Flags |= Memory::PTFlag::US;
132 
133  void *AddressToMap = (void *)((uintptr_t)Address + (i * PAGE_SIZE));
134 
135  Memory::Virtual vmm(this->Table);
136  vmm.Map(AddressToMap, AddressToMap, Flags);
137  }
138 
139  AllocatedPagesList.push_back({Address, Count});
140  return Address;
141  }

254  bool VirtualMemoryArea::HandleCoW(uintptr_t PFA)
255  {
256  function("%#lx", PFA);
257  Memory::Virtual vmm(this->Table);
258  Memory::PageTableEntry *pte = vmm.GetPTE((void *)PFA);
259 
260  if (!pte)
261  {
262  /* Unmapped page */
263  debug("PTE is null!");
264  return false;
265  }
266 
267  if (pte->CopyOnWrite == true)
268  {
269  foreach (auto sr in SharedRegions)
270  {
271  uintptr_t Start = (uintptr_t)sr.Address;
272  uintptr_t End = (uintptr_t)sr.Address + sr.Length;
273 
274  if (PFA >= Start && PFA < End)
275  {
276  debug("Start: %#lx, End: %#lx (PFA: %#lx)",
277  Start, End, PFA);
278 
279  if (sr.Shared)
280  {
281  fixme("Shared CoW");
282  return false;
283  }
284  else
285  {
286  void *pAddr = this->RequestPages(1);
287  if (pAddr == nullptr)
288  return false;
289  memset(pAddr, 0, PAGE_SIZE);
290 
291  assert(pte->Present == true);
292  pte->ReadWrite = sr.Write;
293  pte->UserSupervisor = sr.Read;
294  pte->ExecuteDisable = sr.Exec;
295 
296  pte->CopyOnWrite = false;
297  debug("PFA %#lx is CoW (pt %#lx, flags %#lx)",
298  PFA, this->Table, pte->raw);
299 #if defined(a64)
300  CPU::x64::invlpg((void *)PFA);
301 #elif defined(a32)
302  CPU::x32::invlpg((void *)PFA);
303 #endif
304  return true;
305  }
306  }
307  }
308  }
309 
310  debug("PFA %#lx is not CoW (pt %#lx)",
311  PFA, this->Table);
312  return false;
313  }

330  void VirtualMemoryArea::Fork(VirtualMemoryArea *Parent)
331  {
332  function("%#lx", Parent);
333 
334  if (Parent == nullptr)
335  {
336  error("Parent is null!");
337  return;
338  }
339 
340  if (Parent->Table == nullptr)
341  {
342  error("Parent's table is null!");
343  return;
344  }
345 
346  Memory::Virtual vmm(this->Table);
347  SmartLock(MgrLock);
348  foreach (auto &ap in Parent->GetAllocatedPagesList())
349  {
350  MgrLock.Unlock();
351  void *Address = this->RequestPages(ap.PageCount);
352  MgrLock.Lock(__FUNCTION__);
353  if (Address == nullptr)
354  return;
355 
356  memcpy(Address, ap.Address, ap.PageCount * PAGE_SIZE);
357 
358  // map these new allocated pages to be the same as the parent
359  for (size_t i = 0; i < ap.PageCount; i++)
360  {
361  void *AddressToMap = (void *)((uintptr_t)ap.Address + (i * PAGE_SIZE));
362  void *RealAddress = (void *)((uintptr_t)Address + (i * PAGE_SIZE));
363 
364 #if defined(a86)
365  Memory::PageTableEntry *pte = vmm.GetPTE(AddressToMap);
366  uintptr_t Flags = 0;
367  Flags |= pte->Present ? 1UL : 0;
368  Flags |= pte->ReadWrite ? 2UL : 0;
369  Flags |= pte->UserSupervisor ? 4UL : 0;
370  Flags |= pte->CopyOnWrite ? 512UL : 0;
371 
372  debug("Mapping %#lx to %#lx (flags %s/%s/%s/%s)",
373  RealAddress, AddressToMap,
374  Flags & PTFlag::P ? "P" : "-",
375  Flags & PTFlag::RW ? "RW" : "-",
376  Flags & PTFlag::US ? "US" : "-",
377  Flags & PTFlag::CoW ? "CoW" : "-");
378  vmm.Map(AddressToMap, RealAddress, Flags);
379 #else
380 #warning "Not implemented"
381 #endif
382  }
383  }
384 
385  foreach (auto &sr in Parent->GetSharedRegions())
386  {
387  MgrLock.Unlock();
388  void *Address = this->CreateCoWRegion(sr.Address, sr.Length,
389  sr.Read, sr.Write, sr.Exec,
390  sr.Fixed, sr.Shared);
391  MgrLock.Lock(__FUNCTION__);
392  if (Address == nullptr)
393  return;
394  memcpy(Address, sr.Address, sr.Length);
395  }
396  }
