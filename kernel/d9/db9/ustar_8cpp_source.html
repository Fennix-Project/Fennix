33  int USTAR::Lookup(struct Inode *_Parent, const char *Name, struct Inode **Result)
34  {
35  auto Parent = (USTARInode *)_Parent;
36 
37  const char *basename;
38  size_t length;
39  cwk_path_get_basename(Name, &basename, &length);
40  if (basename == NULL)
41  {
42  if (strcmp(Name, "/") == 0)
43  {
44  auto &it = Files.at(0);
45  *Result = &it->Node;
46  return 0;
47  }
48 
49  error("Invalid name %s", Name);
50  return -EINVAL;
51  }
52 
53  if (_Parent)
54  {
55  for (const auto &child : Parent->Children)
56  {
57  if (child->Deleted || strcmp(child->Name.c_str(), basename) != 0)
58  continue;
59 
60  *Result = &child->Node;
61  return 0;
62  }
63 
64  return -ENOENT;
65  }
66 
67  auto fileItr = Files.begin();
68  while (fileItr != Files.end())
69  {
70  USTARInode *node = fileItr->second;
71 
72  if (node->Deleted || strcmp(node->Name.c_str(), basename) != 0)
73  {
74  fileItr++;
75  continue;
76  }
77 
78  *Result = &fileItr->second->Node;
79  return 0;
80  }
81  return -ENOENT;
82  }

84  int USTAR::Create(struct Inode *_Parent, const char *Name, mode_t Mode, struct Inode **Result)
85  {
86  USTARInode *Parent = (USTARInode *)_Parent;
87 
88  Inode inode{};
89  inode.Mode = Mode;
90  inode.Device = this->DeviceID;
91  inode.RawDevice = 0;
92  inode.Index = NextInode;
93  inode.Offset = 0;
94  inode.PrivateData = this;
95  inode.Flags = I_FLAG_CACHE_KEEP;
96 
97  const char *basename;
98  size_t length;
99  cwk_path_get_basename(Name, &basename, &length);
100 
101  auto SetMode = [&](mode_t &Mode, FileHeader *header)
102  {
103  if (Mode & S_IFREG)
104  header->typeflag[0] = REGTYPE;
105  else if (Mode & S_IFLNK)
106  header->typeflag[0] = SYMTYPE;
107  else if (Mode & S_IFCHR)
108  header->typeflag[0] = CHRTYPE;
109  else if (Mode & S_IFBLK)
110  header->typeflag[0] = BLKTYPE;
111  else if (Mode & S_IFDIR)
112  header->typeflag[0] = DIRTYPE;
113  else if (Mode & S_IFIFO)
114  header->typeflag[0] = FIFOTYPE;
115 
116  mode_t final = 0;
117  if (Mode & S_ISUID)
118  final |= TSUID;
119  else if (Mode & S_ISGID)
120  final |= TSGID;
121  else if (Mode & S_ISVTX)
122  final |= TSVTX;
123  else if (Mode & S_IRUSR)
124  final |= TUREAD;
125  else if (Mode & S_IWUSR)
126  final |= TUWRITE;
127  else if (Mode & S_IXUSR)
128  final |= TUEXEC;
129  else if (Mode & S_IRGRP)
130  final |= TGREAD;
131  else if (Mode & S_IWGRP)
132  final |= TGWRITE;
133  else if (Mode & S_IXGRP)
134  final |= TGEXEC;
135  else if (Mode & S_IROTH)
136  final |= TOREAD;
137  else if (Mode & S_IWOTH)
138  final |= TOWRITE;
139  else if (Mode & S_IXOTH)
140  final |= TOEXEC;
141 
142  snprintf(header->mode, sizeof(header->mode), "%07o", final);
143  };
144 
145  FileHeader *hdr = new FileHeader{};
146  SetMode(inode.Mode, hdr);
147  strncpy(hdr->name, basename, sizeof(hdr->name));
148  strncpy(hdr->signature, TMAGIC, TMAGLEN);
149  strncpy(hdr->version, TVERSION, TVERSLEN);
150 
151  USTARInode *node = new USTARInode{.Node = inode,
152  .Header = hdr,
153  .Parent = Parent,
154  .Name{},
155  .Path{},
156  .Children{},
157  .Deleted = false,
158  .Checksum = INODE_CHECKSUM};
159 
160  node->Name.assign(basename, length);
161  node->Path.assign(Name, strlen(Name));
162 
163  Files.insert(std::make_pair(NextInode, node));
164  *Result = &Files.at(NextInode)->Node;
165  if (Parent)
166  Parent->Children.push_back(Files.at(NextInode));
167  NextInode++;
168  return 0;
169  }

171  ssize_t USTAR::Read(struct Inode *Node, void *Buffer, size_t Size, off_t Offset)
172  {
173  auto fileItr = Files.find(Node->Index);
174  assert(fileItr != Files.end());
175 
176  if (fileItr->second->Deleted)
177  return -ENOENT;
178 
179  USTARInode *node = fileItr->second;
180  size_t fileSize = GetSize(node->Header->size);
181 
182  if (Size <= 0)
183  {
184  debug("Size is less than or equal to 0");
185  Size = fileSize;
186  }
187 
188  if ((size_t)Offset > fileSize)
189  {
190  debug("Offset %d is greater than file size %d", Offset, fileSize);
191  return 0;
192  }
193 
194  if ((fileSize - Offset) == 0)
195  {
196  debug("Offset %d is equal to file size %d", Offset, fileSize);
197  return 0; /* EOF */
198  }
199 
200  if ((size_t)Offset + Size > fileSize)
201  {
202  debug("Offset %d + Size %d is greater than file size %d",
203  Offset, Size, fileSize);
204  Size = fileSize;
205  }
206 
207  memcpy(Buffer, (uint8_t *)((uintptr_t)node->Header + sizeof(FileHeader) + Offset), Size);
208  // debug("Read %d bytes from %d[%d]", Size, Node->Index, Offset);
209  return Size;
210  }

617  node->Name.assign((const char *)header->name, strlen(header->name));

622

624  Address += ((size / 512) + 1) * 512;
625  if (size % 512)
626  Address += 512;
627 
628  header = (FileHeader *)Address;
629  NextInode++;
630  }
631 
632  /* TODO: This code can be significantly optimized but good luck understanding it */
633  USTARInode *parent = nullptr;
634  std::vector<USTARInode *> parentStack;
635  std::vector<std::string *> pathStack;
636  for (auto &file : tmpNodes)
637  {
638  if (file->Path == "/") /* This is root / */
639  {
640  parentStack.push_back(file);
641  pathStack.push_back(&file->Path);
642  // debug("root / generated");
643  continue;
644  }
645 
646  /* pathStack is never empty */
647 
648  if (file->Path.back() == '/') /* This is a directory */
649  {
650  const char *path = file->Path.c_str();
651  size_t length;
652  /* This converts /one/two/path.txt to /one/two/ */
653  cwk_path_get_dirname(path, &length);
654  std::string dirName(path, length);
655 
656  /* Check if the directory is at the same level as the current directory */
657  if (dirName == *pathStack.back())
658  {
659  parent = parentStack.back();
660  parentStack.push_back(file);
661  pathStack.push_back(&file->Path);
662  parent->Children.push_back(file);
663  file->Parent = parent;
664  // debug("adding \"%s\" to \"%s\"", file->Path.c_str(), parent->Path.c_str());
665  continue;
666  }
667  else
668  {
669  /* Check if the directory is at a higher level */
670  for (size_t i = 0; i < parentStack.size(); i++)
671  {
672  if (dirName != *pathStack[i])
673  continue;
674 
675  /* Adjust vectors */
676  while (!parentStack.empty())
677  {
678  if (dirName == *pathStack.back())
679  break;
680 
681  // debug("popping \"%s\"", pathStack.back()->c_str());
682  parentStack.pop_back();
683  pathStack.pop_back();
684  }
685 
686  parent = parentStack.back();
687  parentStack.push_back(file);
688  pathStack.push_back(&file->Path);
689  parent->Children.push_back(file);
690  file->Parent = parent;
691  // debug("adding \"%s\" to \"%s\"", file->Path.c_str(), parent->Path.c_str());
692  goto foundEnd;
693  }
694 
695  // This is a new directory level
696  parentStack.pop_back();
697  pathStack.pop_back();
698  parent = parentStack.back();
699  parentStack.push_back(file);
700  pathStack.push_back(&file->Path);
701  parent->Children.push_back(file);
702  file->Parent = parent;
703  // debug("adding \"%s\" to \"%s\"", file->Path.c_str(), parent->Path.c_str());

706  }
707  }
