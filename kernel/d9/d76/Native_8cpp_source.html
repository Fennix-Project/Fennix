271 static int sys_fork(SysFrm *Frame)
272 {
273 #ifdef a32
274  return -ENOSYS;
275 #endif
276  PCB *Parent = thisThread->Parent;
277  TCB *Thread = thisThread;
278 
279  void *ProcSymTable = nullptr;
280  if (Parent->ELFSymbolTable)
281  ProcSymTable = Parent->ELFSymbolTable->GetImage();
282 
283  PCB *NewProcess =
284  TaskManager->CreateProcess(Parent,
285  Parent->Name,
286  Parent->Security.ExecutionMode,
287  ProcSymTable);
288 
289  if (!NewProcess)
290  {
291  error("Failed to create process for fork");
292  return -EAGAIN;
293  }
294 
295  NewProcess->IPC->Fork(Parent->IPC);
296 
297  TCB *NewThread =
298  TaskManager->CreateThread(NewProcess,
299  0,
300  nullptr,
301  nullptr,
302  std::vector<AuxiliaryVector>(),
303  Thread->Info.Architecture,
304  Thread->Info.Compatibility,
305  true);
306 
307  NewThread->Rename(Thread->Name);
308 
309  if (!NewThread)
310  {
311  error("Failed to create thread for fork");
312  return -EAGAIN;
313  }
314 
315  static int RetChild = 0;
316  static uint64_t ReturnAddress = 0;
317  static uint64_t ChildStackPointer = 0;
318 
319  TaskManager->UpdateFrame();
320 
321  if (RetChild--)
322  {
323  /* We can't just return 0; because the
324  CPUData->SystemCallStack is no
325  longer valid */
326 #if defined(a64) || defined(aa64)
327  asmv("movq %0, %%rcx\n"
328  :
329  : "r"(ReturnAddress));
330  asmv("mov %0, %%rsp\n"
331  :
332  : "r"(ChildStackPointer));
333  asmv("mov %0, %%rbp\n"
334  :
335  : "r"(ChildStackPointer));
336  asmv("movq $0, %rax\n"); /* Return 0 to the child */
337  asmv("swapgs\n"); /* Swap GS back to the user GS */
338  asmv("sti\n"); /* Enable interrupts */
339  asmv("sysretq\n"); /* Return to rcx address in user mode */
340 #elif defined(a32)
341  UNUSED(ReturnAddress);
342  UNUSED(ChildStackPointer);
343 #endif
344  }
345  RetChild = 1;
346  ReturnAddress = Frame->ReturnAddress;
347  ChildStackPointer = Frame->StackPointer;
348 
349  memcpy(NewThread->FPU, Thread->FPU, sizeof(CPU::x64::FXState));
350  NewThread->Stack->Fork(Thread->Stack);
351  NewThread->Info = Thread->Info;
352  NewThread->Registers = Thread->Registers;
353 
354  if (Thread->Security.IsCritical)
355  NewThread->SetCritical(true);
356 
357 #ifdef a86
358  NewThread->ShadowGSBase = Thread->ShadowGSBase;
359  NewThread->GSBase = Thread->GSBase;
360  NewThread->FSBase = Thread->FSBase;
361 #endif
362 
363  debug("Forked thread \"%s\"(%d) to \"%s\"(%d)",
364  Thread->Name, Thread->ID,
365  NewThread->Name, NewThread->ID);
366  NewThread->Status = Ready;
367  return (int)NewThread->ID;
368 }

659 uintptr_t HandleNativeSyscalls(SysFrm *Frame)
660 {
661 #if defined(a64)
662  if (unlikely(Frame->rax > sys_MaxSyscall))
663  {
664  fixme("Syscall %ld not implemented.", Frame->rax);
665  return -ENOSYS;
666  }
667 
668  SyscallData Syscall = NativeSyscallsTable[Frame->rax];
669 
670  uintptr_t (*call)(SysFrm *, uintptr_t, ...) =
671  r_cst(uintptr_t(*)(SysFrm *, uintptr_t, ...),
672  Syscall.Handler);
673 
674  if (unlikely(!call))
675  {
676  error("Syscall %s(%d) not implemented.",
677  Syscall.Name, Frame->rax);
678  return -ENOSYS;
679  }
680 
681  int euid = thisProcess->Security.Effective.UserID;
682  int egid = thisProcess->Security.Effective.GroupID;
683  int reqID = Syscall.RequiredID;
684  if (euid > reqID || egid > reqID)
685  {
686  warn("Process %s(%d) tried to access a system call \"%s\" with insufficient privileges.",
687  thisProcess->Name, thisProcess->ID, Syscall.Name);
688  debug("Required: %d; Effective u:%d, g:%d", reqID, euid, egid);
689  return -EPERM;
690  }
691 
692  debug("[%d:\"%s\"]->( %#lx %#lx %#lx %#lx %#lx %#lx )",
693  Frame->rax, Syscall.Name,
694  Frame->rdi, Frame->rsi, Frame->rdx,
695  Frame->r10, Frame->r8, Frame->r9);
696 
697  return call(Frame,
698  Frame->rdi, Frame->rsi, Frame->rdx,
699  Frame->r10, Frame->r8, Frame->r9);
700 #elif defined(a32)
701  return -ENOSYS;
702 #elif defined(aa64)
703  return -ENOSYS;
704 #endif
705 }
