64  nsa bool Task::InvalidPCB(PCB *pcb)
65  {
66  if (!pcb)
67  return true;
68 
69  /* Uninitialized pointers may have uintptr_t max value instead of nullptr. */
70  if (pcb >= (PCB *)(UINTPTR_MAX - 0x1FFE))
71  return true;
72 
73  /* In this section of the memory is reserved by the kernel. */
74  if (pcb < (PCB *)(0xFFFFF))
75  return true;
76 
77  /* Check if it's mapped. */
78  if (!Memory::Virtual().Check((void *)pcb))
79  return true;
80 
81  return false;
82  }

324  void Task::StartScheduler()
325  {
326 #if defined(a86)
327  if (Interrupts::apicTimer[0])
328  {
329  ((APIC::Timer *)Interrupts::apicTimer[0])->OneShot(CPU::x86::IRQ16, 100);
330 
331  /* FIXME: The kernel is not ready for multi-core tasking. */
332  return;
333 
334  APIC::InterruptCommandRegister icr{};
335  bool x2APIC = ((APIC::APIC *)Interrupts::apic[0])->x2APIC;
336 
337  if (likely(x2APIC))
338  {
339  icr.x2.VEC = s_cst(uint8_t, CPU::x86::IRQ16);
340  icr.x2.MT = APIC::Fixed;
341  icr.x2.L = APIC::Assert;
342  icr.x2.DES = 0xFFFFFFFF; /* Broadcast IPI to all local APICs. */
343  ((APIC::APIC *)Interrupts::apic[0])->ICR(icr);
344  }
345  else
346  {
347  icr.VEC = s_cst(uint8_t, CPU::x86::IRQ16);
348  icr.MT = APIC::Fixed;
349  icr.L = APIC::Assert;
350 
351  for (int i = 0; i < SMP::CPUCores; i++)
352  {
353  icr.DES = uint8_t(i);
354  ((APIC::APIC *)Interrupts::apic[i])->ICR(icr);
355  }
356  }
357  }
358 #elif defined(aa64)
359 #endif
360  debug("Tasking Started");
361  }

363  Task::Task(const IP EntryPoint) : Interrupts::Handler(16) /* IRQ16 */
364  {
365 #if defined(a64)
366  // Map the IRQ16 to the first CPU.
367  ((APIC::APIC *)Interrupts::apic[0])->RedirectIRQ(0, CPU::x86::IRQ16 - CPU::x86::IRQ0, 1);
368 #elif defined(a32)
369 #elif defined(aa64)
370 #endif
371  KPrint("Starting tasking instance %#lx with ip: %p (\e666666%s\eCCCCCC)",
372  this, EntryPoint, KernelSymbolTable->GetSymbol(EntryPoint));
373 
374 #if defined(a64)
375  TaskArchitecture Arch = TaskArchitecture::x64;
376 #elif defined(a32)
377  TaskArchitecture Arch = TaskArchitecture::x32;
378 #elif defined(aa64)
379  TaskArchitecture Arch = TaskArchitecture::ARM64;
380 #endif
381 
382  KernelProcess = CreateProcess(nullptr, "Kernel",
383  TaskExecutionMode::Kernel, true);
384  KernelProcess->PageTable = KernelPageTable;
385  TCB *kthrd = CreateThread(KernelProcess, EntryPoint,
386  nullptr, nullptr,
387  std::vector<AuxiliaryVector>(), Arch);
388  kthrd->Rename("Main Thread");
389  debug("Created Kernel Process: %s and Thread: %s",
390  KernelProcess->Name, kthrd->Name);
391 
392  bool MONITORSupported = false;
393  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
394  {
395  CPU::x86::AMD::CPUID0x00000001 cpuid;
396  MONITORSupported = cpuid.ECX.MONITOR;
397  }
398  else if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_INTEL) == 0)
399  {
400  CPU::x86::Intel::CPUID0x00000001 cpuid;
401  MONITORSupported = cpuid.ECX.MONITOR;
402  }
403 
404  if (MONITORSupported)
405  {
406  trace("CPU has MONITOR/MWAIT support.");
407  }
408 
409  if (!CPU::Interrupts(CPU::Check))
410  {
411  error("Interrupts are not enabled.");
412  CPU::Interrupts(CPU::Enable);
413  }
414 
415  IdleProcess = CreateProcess(nullptr, (char *)"Idle",
416  TaskExecutionMode::Kernel, true);
417  for (int i = 0; i < SMP::CPUCores; i++)
418  {
419  TCB *thd = CreateThread(IdleProcess, IP(IdleProcessLoop));
420  char IdleName[16];
421  sprintf(IdleName, "Idle Thread %d", i);
422  thd->Rename(IdleName);
423  thd->SetPriority(Idle);
424  for (int j = 0; j < MAX_CPU; j++)
425  thd->Info.Affinity[j] = false;
426  thd->Info.Affinity[i] = true;
427 
428  if (unlikely(i == 0))
429  IdleThread = thd;
430  }
431  debug("Tasking is ready");
432  }

434  Task::~Task()
435  {
436  debug("Destructor called");
437  {
438  SmartLock(TaskingLock);
439  foreach (PCB *Process in ProcessList)
440  {
441  foreach (TCB *Thread in Process->Threads)
442  {
443  if (Thread == GetCurrentCPU()->CurrentThread.load())
444  continue;
445  this->KillThread(Thread, KILL_SCHEDULER_DESTRUCTION);
446  }
447 
448  if (Process == GetCurrentCPU()->CurrentProcess.load())
449  continue;
450  this->KillProcess(Process, KILL_SCHEDULER_DESTRUCTION);
451  }
452  }
453 
454  debug("Waiting for processes to terminate");
455  uint64_t timeout = TimeManager->CalculateTarget(20, Time::Units::Seconds);
456  while (ProcessList.size() > 0)
457  {
458  trace("Waiting for %d processes to terminate", ProcessList.size());
459  int NotTerminated = 0;
460  foreach (PCB *Process in ProcessList)
461  {
462  trace("Process %s(%d) is still running (or waiting to be removed state %#lx)",
463  Process->Name, Process->ID, Process->State);
464 
465  if (Process->State == TaskState::Terminated)
466  {
467  debug("Process %s(%d) terminated", Process->Name, Process->ID);
468  continue;
469  }
470 
471  NotTerminated++;
472  }
473  if (NotTerminated == 1)
474  break;
475 
476  this->Sleep(1000);
477  debug("Current working process is %s(%d)",
478  GetCurrentProcess()->Name, GetCurrentProcess()->ID);
479 
480  if (TimeManager->GetCounter() > timeout)
481  {
482  error("Timeout waiting for processes to terminate");
483  break;
484  }
485 
486  TaskingScheduler_OneShot(100);
487  }
488 
489  debug("Tasking stopped");
490  }
