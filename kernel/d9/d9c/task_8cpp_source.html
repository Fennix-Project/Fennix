106  void Task::WaitForProcess(PCB *pcb)
107  {
108  if (pcb == nullptr)
109  return;
110 
111  if (pcb->State == TaskState::UnknownStatus)
112  return;
113 
114  debug("Waiting for process \"%s\"(%d)",
115  pcb->Name, pcb->ID);
116 
117  while (pcb->State != TaskState::Terminated &&
118  pcb->State != TaskState::Zombie &&
119  pcb->State != TaskState::CoreDump)
120  this->Yield();
121  }

171  {

174

198  void Task::SignalShutdown()
199  {
200  debug("Current process is %s(%d) and thread is %s(%d)",
201  GetCurrentProcess()->Name, GetCurrentProcess()->ID,
202  GetCurrentThread()->Name, GetCurrentThread()->ID);
203 
204  foreach (auto pcb in((Scheduler::Base *)Scheduler)->GetProcessList())
205  {
206  if (pcb->State == TaskState::Terminated ||
207  pcb->State == TaskState::Zombie)
208  continue;
209 
210  if (pcb == GetCurrentProcess())
211  continue;
212 
213  debug("Sending SIGTERM to process \"%s\"(%d)",
214  pcb->Name, pcb->ID);
215  pcb->SendSignal(SIGTERM);
216  }

249  Task::Task(const IP EntryPoint)
250  {
251  /* I don't know if this is the best way to do this. */
252  Scheduler::Custom *custom_sched = new Scheduler::Custom(this);
253  Scheduler::Base *sched = r_cst(Scheduler::Base *, custom_sched);
254  __sched_ctx = custom_sched;
255  Scheduler = sched;
256 
257  KernelProcess = CreateProcess(nullptr, "Kernel",
258  TaskExecutionMode::Kernel, true);
259  KernelProcess->PageTable = KernelPageTable;
260  TCB *kthrd = CreateThread(KernelProcess, EntryPoint,
261  nullptr, nullptr,
262  std::vector<AuxiliaryVector>(), GetKArch());
263  kthrd->Rename("Main Thread");
264  debug("Created Kernel Process: %s and Thread: %s",
265  KernelProcess->Name, kthrd->Name);
266 
267  if (!CPU::Interrupts(CPU::Check))
268  {
269  error("Interrupts are not enabled.");
270  CPU::Interrupts(CPU::Enable);
271  }
272 
273  ((Scheduler::Base *)Scheduler)->StartIdleProcess();
274  debug("Tasking is ready");
275  }
