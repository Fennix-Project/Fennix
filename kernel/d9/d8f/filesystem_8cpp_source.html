113  FileNode *Virtual::GetByPath(const char *Path, FileNode *Parent)
114  {
115  debug("GetByPath: %s", Path);
116  if (Parent == nullptr || this->PathIsAbsolute(Path))
117  Parent = thisProcess ? thisProcess->Info.RootNode : this->GetRoot(0);
118 
119  if (strcmp(Path, ".") == 0)
120  return Parent;
121 
122  if (strcmp(Path, "..") == 0)
123  return Parent->Parent ? Parent->Parent : Parent;
124 
125  FileNode *fn = this->CacheRecursiveSearch(Parent, Path, this->PathIsRelative(Path));
126  if (fn)
127  return fn;
128 
129  if (strncmp(Path, "\x06root-", 6) == 0) /* FIXME: deduce the index */
130  {
131  Path += 7;
132  while (*Path != '\0' && *Path != '\x06')
133  Path++;
134  if (*Path == '\x06')
135  Path++;
136  }
137 
138  FileNode *__Parent = CacheSearchReturnLast(Parent, &Path);
139 
140  struct cwk_segment segment;
141  if (!cwk_path_get_first_segment(Path, &segment))
142  {
143  auto it = DeviceMap.find(Parent->Node->Device);
144  if (unlikely(it == DeviceMap.end()))
145  ReturnLogError(nullptr, "Device %d not found", Parent->Node->Device);
146 
147  if (it->second.fsi->Ops.Lookup == NULL)
148  ReturnLogError(nullptr, "Lookup not supported for %d", it->first);
149 
150  Inode *Node = NULL;
151  int ret = it->second.fsi->Ops.Lookup(Parent->Node, Path, &Node);
152  if (ret < 0)
153  ReturnLogError(nullptr, "Lookup for \"%s\"(%d) failed with %d", Path, it->first, ret);
154 
155  if (Parent->Node == Node) /* root / */
156  {
157  debug("Returning root (%#lx)", Node);
158  return Parent;
159  }
160  ReturnLogError(nullptr, "Path has no segments");
161  }
162 
163  Inode *Node = NULL;
164  do
165  {
166  auto it = DeviceMap.find(__Parent->Node->Device);
167  if (unlikely(it == DeviceMap.end()))
168  ReturnLogError(nullptr, "Device %d not found", __Parent->Node->Device);
169 
170  if (it->second.fsi->Ops.Lookup == NULL)
171  ReturnLogError(nullptr, "Lookup not supported for %d", it->first);
172 
173  std::string segmentName(segment.begin, segment.size);
174  int ret = it->second.fsi->Ops.Lookup(__Parent->Node, segmentName.c_str(), &Node);
175  if (ret < 0)
176  ReturnLogError(nullptr, "Lookup for \"%s\"(%d) failed with %d", segmentName.c_str(), it->first, ret);
177  __Parent = this->CreateCacheNode(__Parent, Node, segmentName.c_str(), 0);
178  } while (cwk_path_get_next_segment(&segment));
179 
180  FileNode *ret = __Parent;
181  if (!ret->IsDirectory())
182  return ret;
183 
184  auto it = DeviceMap.find(__Parent->Node->Device);
185  if (unlikely(it == DeviceMap.end()))
186  ReturnLogError(nullptr, "Device %d not found", __Parent->Node->Device);
187 
188  size_t dirAllocLen = sizeof(struct kdirent) + strlen(Path);
189  struct kdirent *dirent = (struct kdirent *)malloc(dirAllocLen);
190  size_t offset = 2; /* Skip . and .. */
191  while (it->second.fsi->Ops.ReadDir(Node, dirent, dirAllocLen, offset++, 1) > 0)
192  {
193  Inode *ChildNode = NULL;
194  int luRet = it->second.fsi->Ops.Lookup(Node, dirent->d_name, &ChildNode);
195  if (luRet < 0)
196  {
197  debug("Lookup for %d failed with %d", it->first, luRet);
198  break;
199  }
200 
201  this->CreateCacheNode(ret, ChildNode, dirent->d_name, 0);
202  }
203  free(dirent);
204  return ret;
205  }
