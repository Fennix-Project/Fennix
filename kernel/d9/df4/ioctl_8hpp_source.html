25 #define _IOC_NRBITS 8
26 #define _IOC_TYPEBITS 8
27 #define _IOC_SIZEBITS 14
28 #define _IOC_DIRBITS 2
29 
30 #define _IOC_NRMASK \
31  ((1 << _IOC_NRBITS) - 1)
32 #define _IOC_TYPEMASK \
33  ((1 << _IOC_TYPEBITS) - 1)
34 #define _IOC_SIZEMASK \
35  ((1 << _IOC_SIZEBITS) - 1)
36 #define _IOC_DIRMASK \
37  ((1 << _IOC_DIRBITS) - 1)
38 
39 #define _IOC_NRSHIFT 0
40 #define _IOC_TYPESHIFT \
41  (_IOC_NRSHIFT + _IOC_NRBITS)
42 #define _IOC_SIZESHIFT \
43  (_IOC_TYPESHIFT + _IOC_TYPEBITS)
44 #define _IOC_DIRSHIFT \
45  (_IOC_SIZESHIFT + _IOC_SIZEBITS)
46 
47 #define _IOC(dir, type, nr, size) \
48  (((dir) << _IOC_DIRSHIFT) | \
49  ((type) << _IOC_TYPESHIFT) | \
50  ((nr) << _IOC_NRSHIFT) | \
51  ((size) << _IOC_SIZESHIFT))
52 
53 #define _IOC_NONE 0U
54 #define _IOC_WRITE 1U
55 #define _IOC_READ 2U
56 
57 #define _IOC_TYPECHECK(t) (sizeof(t))
58 
59 #define _IO(type, nr) \
60  _IOC(_IOC_NONE, (type), (nr), 0)
61 #define _IOR(type, nr, size) \
62  _IOC(_IOC_READ, (type), (nr), (_IOC_TYPECHECK(size)))
63 #define _IOW(type, nr, size) \
64  _IOC(_IOC_WRITE, (type), (nr), (_IOC_TYPECHECK(size)))
65 #define _IOWR(type, nr, size) \
66  _IOC(_IOC_READ | _IOC_WRITE, (type), (nr), (_IOC_TYPECHECK(size)))
67 #define _IOR_BAD(type, nr, size) \
68  _IOC(_IOC_READ, (type), (nr), sizeof(size))
69 #define _IOW_BAD(type, nr, size) \
70  _IOC(_IOC_WRITE, (type), (nr), sizeof(size))
71 #define _IOWR_BAD(type, nr, size) \
72  _IOC(_IOC_READ | _IOC_WRITE, (type), (nr), sizeof(size))
