83  void *Physical::RequestPage()
84  {
85  SmartLock(this->MemoryLock);
86 
87  for (; PageBitmapIndex < PageBitmap.Size * 8; PageBitmapIndex++)
88  {
89  if (PageBitmap[PageBitmapIndex] == true)
90  continue;
91 
92  this->LockPage((void *)(PageBitmapIndex * PAGE_SIZE));
93  return (void *)(PageBitmapIndex * PAGE_SIZE);
94  }
95 
96  if (this->SwapPage((void *)(PageBitmapIndex * PAGE_SIZE)))
97  {
98  this->LockPage((void *)(PageBitmapIndex * PAGE_SIZE));
99  return (void *)(PageBitmapIndex * PAGE_SIZE);
100  }
101 
102  if (TaskManager && !TaskManager->IsPanic())
103  {
104  error("Out of memory! Killing current process...");
105  TaskManager->KillProcess(thisProcess, Tasking::KILL_OOM);
106  TaskManager->Yield();
107  }
108 
109  error("Out of memory! (Free: %ld MiB; Used: %ld MiB; Reserved: %ld MiB)",
110  TO_MiB(FreeMemory.load()), TO_MiB(UsedMemory.load()), TO_MiB(ReservedMemory.load()));
111  KPrint("Out of memory! (Free: %ld MiB; Used: %ld MiB; Reserved: %ld MiB)",
112  TO_MiB(FreeMemory.load()), TO_MiB(UsedMemory.load()), TO_MiB(ReservedMemory.load()));
113  debug("Raw values: free %#lx used %#lx reserved %#lx",
114  FreeMemory.load(), UsedMemory.load(), ReservedMemory.load());
115  CPU::Stop();
116  __builtin_unreachable();
117  }

119  void *Physical::RequestPages(size_t Count)
120  {
121  SmartLock(this->MemoryLock);
122 
123  for (; PageBitmapIndex < PageBitmap.Size * 8; PageBitmapIndex++)
124  {
125  if (PageBitmap[PageBitmapIndex] == true)
126  continue;
127 
128  for (uint64_t Index = PageBitmapIndex; Index < PageBitmap.Size * 8; Index++)
129  {
130  if (PageBitmap[Index] == true)
131  continue;
132 
133  for (size_t i = 0; i < Count; i++)
134  {
135  if (PageBitmap[Index + i] == true)
136  goto NextPage;
137  }
138 
139  this->LockPages((void *)(Index * PAGE_SIZE), Count);
140  return (void *)(Index * PAGE_SIZE);
141 
142  NextPage:
143  Index += Count;
144  continue;
145  }
146  }
147 
148  if (this->SwapPages((void *)(PageBitmapIndex * PAGE_SIZE), Count))
149  {
150  this->LockPages((void *)(PageBitmapIndex * PAGE_SIZE), Count);
151  return (void *)(PageBitmapIndex * PAGE_SIZE);
152  }
153 
154  if (TaskManager && !TaskManager->IsPanic())
155  {
156  error("Out of memory! Killing current process...");
157  TaskManager->KillProcess(thisProcess, Tasking::KILL_OOM);
158  TaskManager->Yield();
159  }
160 
161  error("Out of memory! (Free: %ld MiB; Used: %ld MiB; Reserved: %ld MiB)",
162  TO_MiB(FreeMemory.load()), TO_MiB(UsedMemory.load()), TO_MiB(ReservedMemory.load()));
163  KPrint("Out of memory! (Free: %ld MiB; Used: %ld MiB; Reserved: %ld MiB)",
164  TO_MiB(FreeMemory.load()), TO_MiB(UsedMemory.load()), TO_MiB(ReservedMemory.load()));
165  debug("Raw values: free %#lx used %#lx reserved %#lx",
166  FreeMemory.load(), UsedMemory.load(), ReservedMemory.load());

171  void Physical::FreePage(void *Address)
172  {
173  SmartLock(this->MemoryLock);
174 
175  if (unlikely(Address == nullptr))
176  {
177  warn("Null pointer passed to FreePage.");
178  return;
179  }
180 
181  size_t Index = (size_t)Address / PAGE_SIZE;
182 
183  if (unlikely(PageBitmap[Index] == false))
184  {
185  warn("Tried to free an already free page. (%p)",
186  Address);
187  return;
188  }
189 
190  if (PageBitmap.Set(Index, false))
191  {
192  FreeMemory.fetch_add(PAGE_SIZE);
193  UsedMemory.fetch_sub(PAGE_SIZE);
194  if (PageBitmapIndex > Index)
195  PageBitmapIndex = Index;
196  }
197  }

320  void Physical::Init()
321  {
322  SmartLock(this->MemoryLock);
323 
324  uint64_t MemorySize = bInfo.Memory.Size;
325  debug("Memory size: %lld bytes (%ld pages)",
326  MemorySize, TO_PAGES(MemorySize));
327  TotalMemory.store(MemorySize);
328  FreeMemory.store(MemorySize);
329 
330  size_t BitmapSize = (size_t)(MemorySize / PAGE_SIZE) / 8 + 1;
331  uintptr_t BitmapAddress = 0x0;
332  size_t BitmapAddressSize = 0;
333 
334  FindBitmapRegion(BitmapAddress, BitmapAddressSize);
335  if (BitmapAddress == 0x0)
336  {
337  error("No free memory found!");
338  CPU::Stop();
339  }
340 
341  debug("Initializing Bitmap at %p-%p (%d Bytes)",
342  BitmapAddress,
343  (void *)(BitmapAddress + BitmapSize),
344  BitmapSize);
345 
346  PageBitmap.Size = BitmapSize;
347  PageBitmap.Buffer = (uint8_t *)BitmapAddress;
348  for (size_t i = 0; i < BitmapSize; i++)
349  *(uint8_t *)(PageBitmap.Buffer + i) = 0;
350 
351  ReserveEssentials();
352  }
