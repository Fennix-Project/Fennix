123  nsa void Init(int Core)
124  {
125  GDTEntries[Core] = GDTEntriesTemplate;
126  gdt[Core] =
127  {
128  .Limit = sizeof(GlobalDescriptorTableEntries) - 1,
129  .BaseAddress = &GDTEntries[Core],
130  };
131 
132  debug("GDT: %#lx", &gdt[Core]);
133  debug("GDT KERNEL CODE %#lx", GDT_KERNEL_CODE);
134  debug("GDT KERNEL DATA %#lx", GDT_KERNEL_DATA);
135  debug("GDT USER CODE %#lx", GDT_USER_CODE);
136  debug("GDT USER DATA %#lx", GDT_USER_DATA);
137  debug("GDT TSS %#lx", GDT_TSS);
138 
139  CPU::x64::lgdt(&gdt[Core]);
140 
141  asmv("movq %%rsp, %%rax\n"
142  "pushq $16\n"
143  "pushq %%rax\n"
144  "pushfq\n"
145  "pushq $8\n"
146  "pushq $1f\n"
147  "iretq\n"
148  "1:\n"
149  "movw $16, %%ax\n"
150  "movw %%ax, %%ds\n"
151  "movw %%ax, %%es\n" ::
152  : "memory", "rax");
153 
154  CPUStackPointer[Core] = StackManager.Allocate(STACK_SIZE);
155  memset(CPUStackPointer[Core], 0, STACK_SIZE);
156  debug("CPU %d Stack Pointer: %#lx-%#lx (%d pages)", Core,
157  CPUStackPointer[Core], (uintptr_t)CPUStackPointer[Core] + STACK_SIZE,
158  TO_PAGES(STACK_SIZE + 1));
159 
160  uintptr_t Base = (uintptr_t)&tss[Core];
161  size_t Limit = Base + sizeof(TaskStateSegment);
162  SystemSegmentDescriptor *tssDesc = &gdt[Core].BaseAddress->TaskStateSegment;
163  tssDesc->SegmentLimitLow = Limit & 0xFFFF;
164  tssDesc->BaseAddressLow = Base & 0xFFFF;
165  tssDesc->BaseAddressMiddle = (Base >> 16) & 0xFF;
166  tssDesc->Type = AVAILABLE_64BIT_TSS;
167  tssDesc->Zero0 = 0;
168  tssDesc->DescriptorPrivilegeLevel = 0;
169  tssDesc->Present = 1;
170  tssDesc->Available = 0;
171  tssDesc->Reserved0 = 0;
172  tssDesc->Granularity = 0;
173  tssDesc->BaseAddressHigh = (Base >> 24) & 0xFF;
174  tssDesc->BaseAddressHigher = s_cst(uint32_t, (Base >> 32) & 0xFFFFFFFF);
175  tssDesc->Reserved1 = 0;
176  tssDesc->Zero1 = 0;
177  tssDesc->Reserved2 = 0;
178 
179  tss[Core].IOMapBaseAddressOffset = sizeof(TaskStateSegment);
180  tss[Core].StackPointer[0] = (uint64_t)CPUStackPointer[Core] + STACK_SIZE;
181  tss[Core].StackPointer[1] = 0x0;
182  tss[Core].StackPointer[2] = 0x0;
183 
184  for (size_t i = 0; i < sizeof(tss[Core].InterruptStackTable) / sizeof(tss[Core].InterruptStackTable[7]); i++)
185  {
186  void *NewStack = StackManager.Allocate(STACK_SIZE);
187  tss[Core].InterruptStackTable[i] = (uint64_t)NewStack + STACK_SIZE;
188  memset((void *)(tss[Core].InterruptStackTable[i] - STACK_SIZE), 0, STACK_SIZE);
189  debug("IST-%d: %#lx-%#lx", i, NewStack, (uintptr_t)NewStack + STACK_SIZE);
190  }
191 
192  CPU::x64::ltr(GDT_TSS);
193  debug("Global Descriptor Table initialized");
194  }

196  nsa void SetKernelStack(void *Stack)
197  {
198  long CPUID = GetCurrentCPU()->ID;
199  if (Stack != nullptr)
200  tss[CPUID].StackPointer[0] = (uint64_t)Stack;
201  else
202  tss[CPUID].StackPointer[0] = (uint64_t)CPUStackPointer[CPUID] + STACK_SIZE;
203 
204  /*
205  FIXME: There's a bug in kernel which if
206  we won't update "tss[CPUID].StackPointer[0]"
207  with the current stack pointer, the kernel
208  will crash.
209  */
210  asmv("mov %%rsp, %0"
211  : "=r"(tss[CPUID].StackPointer[0]));
212  }
