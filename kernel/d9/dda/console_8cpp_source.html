51  __no_sanitize("undefined") char FontRenderer::Paint(long CellX, long CellY, char Char, uint32_t Foreground, uint32_t Background)
52  {
53  uint64_t x = CellX * CurrentFont->GetInfo().Width;
54  uint64_t y = CellY * CurrentFont->GetInfo().Height;
55 
56  switch (CurrentFont->GetInfo().Type)
57  {
58  case Video::FontType::PCScreenFont1:
59  {
60  uint32_t *PixelPtr = (uint32_t *)Display->GetBuffer;
61  char *FontPtr = (char *)CurrentFont->GetInfo().PSF1Font->GlyphBuffer + (Char * CurrentFont->GetInfo().PSF1Font->Header->charsize);
62  for (uint64_t Y = 0; Y < 16; Y++)
63  {
64  for (uint64_t X = 0; X < 8; X++)
65  {
66  if ((*FontPtr & (0b10000000 >> X)) > 0)
67  *(unsigned int *)(PixelPtr + (x + X) + ((y + Y) * Display->GetWidth)) = Foreground;
68  else
69  *(unsigned int *)(PixelPtr + (x + X) + ((y + Y) * Display->GetWidth)) = Background;
70  }
71  FontPtr++;
72  }
73  break;
74  }
75  case Video::FontType::PCScreenFont2:
76  {
77  Video::FontInfo fInfo = CurrentFont->GetInfo();
78 
79  int BytesPerLine = (fInfo.PSF2Font->Header->width + 7) / 8;
80  char *FontAddress = (char *)fInfo.StartAddress;
81  uint32_t FontHeaderSize = fInfo.PSF2Font->Header->headersize;
82  uint32_t FontCharSize = fInfo.PSF2Font->Header->charsize;
83  uint32_t FontLength = fInfo.PSF2Font->Header->length;
84  char *FontPtr = FontAddress + FontHeaderSize + (Char > 0 && (uint32_t)Char < FontLength ? Char : 0) * FontCharSize;
85 
86  uint32_t FontHdrWidth = fInfo.PSF2Font->Header->width;
87  uint32_t FontHdrHeight = fInfo.PSF2Font->Header->height;
88 
89  for (uint32_t Y = 0; Y < FontHdrHeight; Y++)
90  {
91  for (uint32_t X = 0; X < FontHdrWidth; X++)
92  {
93  void *FramebufferAddress = (void *)((uintptr_t)Display->GetBuffer +
94  ((y + Y) * Display->GetWidth + (x + X)) *
95  (Display->GetFramebufferStruct().BitsPerPixel / 8));
96 
97  if ((*FontPtr & (0b10000000 >> (X % 8))) > 0)
98  *(uint32_t *)FramebufferAddress = Foreground;
99  else
100  *(uint32_t *)FramebufferAddress = Background;
101  }
102  FontPtr += BytesPerLine;
103  }
104  break;
105  }
106  default:
107  warn("Unsupported font type");
108  break;
109  }
110  return Char;
111  }

167  auto parseColor = [getColorComponent](const char *colorStr) -> unsigned int
168  {
169  int index = 0;
170  int r = getColorComponent(colorStr, index);
171  if (colorStr[index] == ',')
172  ++index;
173  int g = getColorComponent(colorStr, index);
174  if (colorStr[index] == ',')
175  ++index;
176  int b = getColorComponent(colorStr, index);
177  return (r << 16) | (g << 8) | b;
178  };
