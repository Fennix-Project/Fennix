69 EXTERNC void *__rpmalloc_mmap(void *addr, size_t length, int, int, int fd, off_t offset)
70 {
71  assert(addr == 0 && fd == -1 && offset == 0);
72 
73  void *ptr = KernelAllocator.RequestPages(TO_PAGES(length));
74  debug("Requested %d pages, got %p", TO_PAGES(length), ptr);
75  if (ptr == nullptr)
76  return MAP_FAILED;
77  return ptr;
78 }
