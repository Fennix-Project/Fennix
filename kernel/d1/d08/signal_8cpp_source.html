302  bool Signal::HandleSignal(CPU::TrapFrame *tf)
303  {
304  SmartLock(SignalLock);
305  if (SignalQueue.empty())
306  return false;
307 
308  /* We don't want to do this in kernel mode */
309  if (unlikely(tf->cs != GDT_USER_CODE &&
310  tf->cs != GDT_USER_DATA))
311  {
312  // debug("Not user-mode");
313  return false;
314  }
315 
316  debug("We have %d signals to handle", SignalQueue.size());
317 
318  SignalInfo sigI = SignalQueue.front();
319  SignalQueue.erase(SignalQueue.begin());
320 
321  uintptr_t _p_rsp = ((PCB *)ctx)->PageTable->Get(tf->rsp);
322  uint64_t paRsp = _p_rsp;
323  paRsp &= ~0xF; /* Align */
324  paRsp -= 128; /* Red zone */
325 
326  /* Calculate the virtual rsp */
327  uintptr_t _v_rsp = tf->rsp;
328  _v_rsp &= ~0xF; /* Align */
329  _v_rsp -= 128; /* Red zone */
330  uint64_t *vRsp = (uint64_t *)(_v_rsp - sizeof(StackInfo));
331  vRsp--; /* Alignment */
332  vRsp--; /* Handler Address */
333  assert(!((uintptr_t)vRsp & 0xF));
334 
335  /* Add the stack info */
336  StackInfo si{};
337  CPU::x64::fxsave(&si.fx);
338  si.tf = *tf;
339  si.GSBase = CPU::x64::rdmsr(CPU::x64::MSR_GS_BASE);
340  si.FSBase = CPU::x64::rdmsr(CPU::x64::MSR_FS_BASE);
341  si.ShadowGSBase = CPU::x64::rdmsr(CPU::x64::MSR_SHADOW_GS_BASE);
342  debug("gs: %#lx fs: %#lx shadow: %#lx",
343  si.GSBase, si.FSBase, si.ShadowGSBase);
344  si.SignalMask = SignalMask.load();
345  si.Compatibility = ((PCB *)ctx)->Info.Compatibility;
346 
347  /* Copy the stack info */
348  uint64_t *pRsp = (uint64_t *)(paRsp - sizeof(StackInfo));
349  memcpy(pRsp, &si, sizeof(StackInfo));
350 
351  /* Set the handler address */
352  pRsp--; /* Alignment */
353  pRsp--;
354  *pRsp = uint64_t(SignalAction[sigI.sig].__sa_handler.sa_handler);
355 
356  assert(!((uintptr_t)pRsp & 0xF));
357 
358 #ifdef DEBUG
359  DumpData("Stack Data", (void *)pRsp,
360  paRsp - uint64_t(pRsp));
361  debug("initial stack tf->rsp: %#lx after: %#lx",
362  tf->rsp, uint64_t(vRsp));
363  debug("sig: %d -> %d", sigI.sig, CTLif(sigI.sig));
364 #endif
365 
366  tf->rsp = uint64_t(vRsp);
367  tf->rip = uint64_t(TrampAddr);
368  tf->rdi = CTLif(sigI.sig);
369  tf->rsi = uint64_t(sigI.val.sival_ptr);
370  return true;
371  }

373  void Signal::RestoreHandleSignal(SyscallsFrame *sf)
374  {
375  SmartLock(SignalLock);
376  debug("Restoring signal handler");
377  gsTCB *gs = (gsTCB *)CPU::x64::rdmsr(CPU::x64::MSR_GS_BASE);
378  uint64_t *sp = (uint64_t *)((PCB *)ctx)->PageTable->Get(gs->TempStack);
379  sp++; /* Alignment */
380  sp++; /* Handler Address */
381 
382  assert(!((uintptr_t)sp & 0xF));
383 
384  StackInfo *si = (StackInfo *)sp;
385  assert(si != nullptr);
386 
387  sf->r15 = si->tf.r15;
388  sf->r14 = si->tf.r14;
389  sf->r13 = si->tf.r13;
390  sf->r12 = si->tf.r12;
391  sf->r11 = si->tf.r11;
392  sf->r10 = si->tf.r10;
393  sf->r9 = si->tf.r9;
394  sf->r8 = si->tf.r8;
395  sf->rbp = si->tf.rbp;
396  sf->rdi = si->tf.rdi;
397  sf->rsi = si->tf.rsi;
398  sf->rdx = si->tf.rdx;
399  sf->rcx = si->tf.rcx;
400  sf->rbx = si->tf.rbx;
401  sf->rax = si->tf.rax;
402  sf->Flags = si->tf.rflags.raw;
403  sf->ReturnAddress = si->tf.rip;
404  gs->TempStack = si->tf.rsp;
405 
406  SignalMask.store(si->SignalMask);
407 
408  CPU::x64::fxrstor(&si->fx);
409  CPU::x64::wrmsr(CPU::x64::MSR_GS_BASE, si->ShadowGSBase);
410  CPU::x64::wrmsr(CPU::x64::MSR_FS_BASE, si->FSBase);
411  CPU::x64::wrmsr(CPU::x64::MSR_SHADOW_GS_BASE, si->GSBase);
412  debug("gs: %#lx fs: %#lx shadow: %#lx",
413  si->GSBase, si->FSBase, si->ShadowGSBase);
414 
415  // ((PCB *)ctx)->GetContext()->Yield();
416  // __builtin_unreachable();
417  /* Return because we will restore at sysretq */
418  }

486  int Signal::SendSignal(int sig, union sigval val)
487  {
488  PCB *pcb = (PCB *)ctx;
489  sig = CTNif(sig);
490  LastSignal = (Signals)sig;
491 
492  debug("Sending signal %s to %s(%d)",
493  sigStr[sig], pcb->Name, pcb->ID);
494 
495  if (SignalAction[sig].__sa_handler.sa_handler)
496  {
497  if (pcb->Security.ExecutionMode == Kernel)
498  {
499  info("Kernel processes cannot have signal handlers");
500  return -EINVAL;
501  }
502 
503  debug("sa_handler: %#lx",
504  SignalAction[sig].__sa_handler.sa_handler);
505  debug("Adding signal %s to queue", sigStr[sig]);
506  goto CompleteSignal;
507  }
508 
509  debug("Signal disposition: %s", dispStr[sigDisp[sig]]);
510  switch (sigDisp[sig])
511  {
512  case SIG_TERM:
513  {
514  if (unlikely(pcb->Security.IsCritical))
515  {
516  debug("Critical process %s received signal %s(%d): Terminated",
517  pcb->Name, sigStr[sig], sig);
518  // int3;
519  }
520 
521  pcb->SetExitCode(MakeExitCode(sig));
522  debug("We have %d watchers", this->Watchers.size());
523  if (this->Watchers.size() > 0)
524  pcb->SetState(Zombie);
525  else
526  pcb->SetState(Terminated);
527  break;
528  }
529  case SIG_IGN:
530  {
531  debug("Ignoring signal %d", sig);
532  return 0;
533  }
534  case SIG_CORE:
535  {
536  fixme("core dump");
537 
538  if (unlikely(pcb->Security.IsCritical))
539  {
540  debug("Critical process %s received signal %s(%d): Core dumped",
541  pcb->Name, sigStr[sig], sig);
542  // int3;
543  }
544 
545  pcb->SetExitCode(MakeExitCode(sig));
546  debug("We have %d watchers", this->Watchers.size());
547  if (this->Watchers.size() > 0)
548  pcb->SetState(CoreDump);
549  else
550  pcb->SetState(Terminated);
551  break;
552  }
553  case SIG_STOP:
554  {
555  pcb->SetState(Stopped);
556  break;
557  }
558  case SIG_CONT:
559  {
560  pcb->SetState(Ready);
561  break;
562  }
563  default:
564  assert(!"Invalid signal disposition");
565  return -EINVAL;
566  };
567 
568  CompleteSignal:
569  if (pcb->Security.ExecutionMode == Kernel)
570  {
571  debug("Kernel process %s received signal %s(%d)! Ignoring... (with exceptions)",
572  pcb->Name, sigStr[sig], sig);
573  return 0;
574  }
575 
576  if (unlikely(TrampAddr == nullptr))
577  {
578  debug("Trampoline not set up yet");
579  switch (thisThread->Info.Compatibility)
580  {
581  case Native:
582  {
583  debug("%#lx - %#lx",
584  &_sig_native_trampoline_end,
585  &_sig_native_trampoline_start);
586 
587  TrampSz = (size_t)&_sig_native_trampoline_end -
588  (size_t)&_sig_native_trampoline_start;
589  TrampAddr = pcb->vma->RequestPages(TO_PAGES(TrampSz), true);
590  memcpy((void *)TrampAddr,
591  (void *)&_sig_native_trampoline_start,
592  TrampSz);
593  debug("Trampoline at %#lx with size %lld",
594  TrampAddr, TrampSz);
595  break;
596  }
597  case Linux:
598  {
599  debug("%#lx - %#lx",
600  &_sig_linux_trampoline_end,
601  &_sig_linux_trampoline_start);
602 
603  TrampSz = (size_t)&_sig_linux_trampoline_end -
604  (size_t)&_sig_linux_trampoline_start;
605  TrampAddr = pcb->vma->RequestPages(TO_PAGES(TrampSz), true);
606  memcpy((void *)TrampAddr,
607  (void *)&_sig_linux_trampoline_start,
608  TrampSz);
609  debug("Trampoline at %#lx with size %lld",
610  TrampAddr, TrampSz);
611  break;
612  }
613  default:
614  {
615  assert(!"Not implemented");
616  break;
617  }
618  }
619  }
620 
621  debug("Signal %s(%d) completed", sigStr[sig], sig);
622  if (sigDisp[sig] != SIG_IGN)
623  {
624  foreach (auto info in Watchers)
625  {
626  Signal *who = (Signal *)info.val.sival_ptr;
627  assert(who != nullptr);
628  debug("Sending SIGCHLD to %s(%d)",
629  ((PCB *)who->GetContext())->Name,
630  ((PCB *)who->GetContext())->ID);
631  who->SendSignal(SIGCHLD, val);
632  }
633  }
634 
635  debug("Adding signal to queue");
636  SignalQueue.push_back({.sig = sig, .val = val});
637  return 0;
638  }

640  int Signal::WaitAnySignal()
641  {
642  /* Sleep until a signal that terminated or invokes
643  the signal catch function */
644 
645  debug("Waiting for any signal");
646 
647  size_t oldSize = SignalQueue.size();
648  Reset:
649  while (SignalQueue.size() == oldSize)
650  TaskManager->Yield();
651 
652  if (SignalQueue.size() > oldSize)
653  {
654  debug("Added signal to queue %d > %d",
655  SignalQueue.size(), oldSize);
656  oldSize = SignalQueue.size();
657  goto Reset;
658  }
659 
660  debug("Signal received");
661  return -EINTR;
662  }
