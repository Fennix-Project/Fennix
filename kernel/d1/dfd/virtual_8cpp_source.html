35  int __vfs_Lookup(struct Inode *_Parent, const char *Name, struct Inode **Result)
36  {
37  vfsInode *Parent = (vfsInode *)_Parent;
38 
39  if (!S_ISDIR(Parent->Node.Mode))
40  return -ENOTDIR;
41 
42  assert(Parent->Node.Flags & I_FLAG_MOUNTPOINT);
43 
44  if (Parent->Children.empty())
45  return -ENOENT;
46 
47  off_t offset = 0;
48  for (const auto &Root : Parent->Children)
49  {
50  char rootName[128]{};
51  snprintf(rootName, sizeof(rootName), "\x06root-%ld\x06", offset);
52 
53  if (strcmp(rootName, Name) == 0)
54  {
55  *Result = Root;
56  return 0;
57  }
58  offset++;
59  }
60 
61  return -ENOENT;
62  }

72  ssize_t __vfs_Readdir(struct Inode *_Node, struct kdirent *Buffer, size_t Size, off_t Offset, off_t Entries)
73  {
74  if (_Node->GetMinor() != 0)
75  {
76  debug("_Node->GetMinor() != 0");
77  return -ENOENT;
78  }
79 
80  assert(_Node->Flags & I_FLAG_MOUNTPOINT);
81 
82  fixme("maybe wrong implementation of readdir");
83 
84  size_t totalSize = 0;
85  off_t entriesSkipped = 0;
86  struct kdirent *ent = nullptr;
87  vfsInode *Node = (vfsInode *)_Node;
88  off_t entries = 0;
89  foreach (const auto &Root in Node->Children)
90  {
91  if (entries >= Entries)
92  break;
93 
94  uint16_t reclen = (uint16_t)(offsetof(struct kdirent, d_name) + strlen("root") + 1);
95 
96  if (Offset > entriesSkipped)
97  {
98  entriesSkipped++;
99  continue;
100  }
101 
102  if (totalSize + reclen >= Size)
103  break;
104 
105  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
106  ent->d_ino = Root->Index;
107  ent->d_off = Root->Offset;
108  ent->d_reclen = reclen;
109  ent->d_type = IFTODT(Root->Mode);
110  strncpy(ent->d_name, "root", strlen("root"));
111 
112  totalSize += reclen;
113  entries++;
114  }
115 
116  if (ent)
117  ent->d_off = INT32_MAX;
118 
119  if (totalSize + sizeof(struct kdirent) >= Size)
120  return totalSize;
121 
122  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
123  ent->d_ino = 0;
124  ent->d_off = 0;
125  ent->d_reclen = 0;
126  ent->d_type = DT_UNKNOWN;
127  ent->d_name[0] = '\0';
128  return totalSize;
129  }

209  int Virtual::UnregisterFileSystem(dev_t Device)
210  {
211  auto it = DeviceMap.find(Device);
212  if (it == DeviceMap.end())
213  ReturnLogError(-ENOENT, "Device %d not found", Device);
214 
215  if (it->second.fsi->SuperOps.Synchronize)
216  it->second.fsi->SuperOps.Synchronize(it->second.fsi, NULL);
217  if (it->second.fsi->SuperOps.Destroy)
218  it->second.fsi->SuperOps.Destroy(it->second.fsi);
219  DeviceMap.erase(it);
220  return 0;
