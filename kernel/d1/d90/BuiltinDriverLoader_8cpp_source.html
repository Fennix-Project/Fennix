163 std::vector<PCI::PCIDeviceHeader *> FindPCI(uint16_t VendorID[16], uint16_t DeviceID[16])
164 {
165  for (uint16_t Vidx = 0; Vidx < 16; Vidx++)
166  {
167  for (uint16_t Didx = 0; Didx < 16; Didx++)
168  {
169  if (VendorID[Vidx] == 0 || DeviceID[Didx] == 0)
170  break;
171 
172  std::vector<PCI::PCIDeviceHeader *> devices = PCIManager->FindPCIDevice(VendorID[Vidx], DeviceID[Didx]);
173  if (devices.size() == 0)
174  continue;
175 
176  return devices;
177  }
178  }
179 
180  warn("No PCI device found");
181  return std::vector<PCI::PCIDeviceHeader *>();
182 }
