114 static inline int ConvertFileFlags(const char *Mode)
115 {
116  int Flags = 0;
117 
118  if (strchr(Mode, '+'))
119  Flags |= O_RDWR;
120  else if (*Mode == 'r')
121  Flags |= O_RDONLY;
122  else
123  Flags |= O_WRONLY;
124 
125  if (strchr(Mode, 'x'))
126  Flags |= O_EXCL;
127 
128  if (strchr(Mode, 'e'))
129  Flags |= O_CLOEXEC;
130 
131  if (*Mode != 'r')
132  Flags |= O_CREAT;
133 
134  if (*Mode == 'w')
135  Flags |= O_TRUNC;
136 
137  if (*Mode == 'a')
138  Flags |= O_APPEND;
139 
140  return Flags;
141 }
