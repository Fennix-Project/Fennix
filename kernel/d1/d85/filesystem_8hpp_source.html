124 static inline int ConvertFileFlags(const char *Mode)
125 {
126  int Flags = 0;
127 
128  if (strchr(Mode, '+'))
129  Flags |= O_RDWR;
130  else if (*Mode == 'r')
131  Flags |= O_RDONLY;
132  else
133  Flags |= O_WRONLY;
134 
135  if (strchr(Mode, 'x'))
136  Flags |= O_EXCL;
137 
138  if (strchr(Mode, 'e'))
139  Flags |= O_CLOEXEC;
140 
141  if (*Mode != 'r')
142  Flags |= O_CREAT;
143 
144  if (*Mode == 'w')
145  Flags |= O_TRUNC;
146 
147  if (*Mode == 'a')
148  Flags |= O_APPEND;
149 
150  return Flags;
151 }
