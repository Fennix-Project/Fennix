82 static int
83 getstat (fname, flags, sp)
84  const char *fname;
85  int flags;
86  struct stat *sp;
87 {
88  intmax_t lfd;
89  int fd, r;
90 
91  if (strncmp (fname, "/dev/fd/", 8) == 0)
92  {
93  if ((legal_number(fname + 8, &lfd) == 0) || (int)lfd != lfd)
94  {
95  errno = EINVAL;
96  return -1;
97  }
98  fd = lfd;
99  r = fstat(fd, sp);
100  }
101 #ifdef HAVE_LSTAT
102  else if (flags & 1)
103  r = lstat(fname, sp);
104 #endif
105  else
106  r = stat(fname, sp);
107 
108  return r;
109 }

285 statval (which, fname, flags, fmt, sp)
286  int which;
287  char *fname;
288  int flags;
289  char *fmt;
290  struct stat *sp;
291 {
292  int temp;
293 
294  switch (which)
295  {
296  case ST_NAME:
297  return savestring (fname);
298  case ST_DEV:
299  return itos (sp->st_dev);
300  case ST_INO:
301  return itos (sp->st_ino);
302  case ST_MODE:
303  return (statmode (sp->st_mode));
304  case ST_NLINK:
305  return itos (sp->st_nlink);
306  case ST_UID:
307  return itos (sp->st_uid);
308  case ST_GID:
309  return itos (sp->st_gid);
310  case ST_RDEV:
311  return itos (sp->st_rdev);
312  case ST_SIZE:
313  return itos (sp->st_size);
314  case ST_ATIME:
315  return ((flags & 2) ? stattime (sp->st_atime, fmt) : itos (sp->st_atime));
316  case ST_MTIME:
317  return ((flags & 2) ? stattime (sp->st_mtime, fmt) : itos (sp->st_mtime));
318  case ST_CTIME:
319  return ((flags & 2) ? stattime (sp->st_ctime, fmt) : itos (sp->st_ctime));
320  case ST_BLKSIZE:
321  return itos (sp->st_blksize);
322  case ST_BLOCKS:
323  return itos (sp->st_blocks);
324  case ST_CHASELINK:
325  return (statlink (fname, sp));
326  case ST_PERMS:
327  temp = sp->st_mode & (S_IRWXU|S_IRWXG|S_IRWXO|S_ISUID|S_ISGID);
328  return (flags & 2) ? statperms (temp) : octalperms (temp);
329  default:
330  return savestring ("42");
331  }

396  return (EXECUTION_FAILURE);
397  }
398 
399  list = loptend;

401  {
402  builtin_usage ();
403  return (EX_USAGE);
404  }
405 
406 
407 #if 0
408  unbind_variable (aname);
409 #endif
410  fname = list->word->word;
411 
412  if (getstat (fname, flags, &st) < 0)
413  {
414  builtin_error ("%s: cannot stat: %s", fname, strerror (errno));
415  return (EXECUTION_FAILURE);

417

419  if (v == 0)

423  }

425  {
426  builtin_error ("%s: cannot assign file status information", aname);
427  unbind_variable (aname);

429  }
