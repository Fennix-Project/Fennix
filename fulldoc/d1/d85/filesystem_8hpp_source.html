165 static inline int ConvertFileFlags(const char *Mode)
166 {
167  int Flags = 0;
168 
169  if (strchr(Mode, '+'))
170  Flags |= O_RDWR;
171  else if (*Mode == 'r')
172  Flags |= O_RDONLY;
173  else
174  Flags |= O_WRONLY;
175 
176  if (strchr(Mode, 'x'))
177  Flags |= O_EXCL;
178 
179  if (strchr(Mode, 'e'))
180  Flags |= O_CLOEXEC;
181 
182  if (*Mode != 'r')
183  Flags |= O_CREAT;
184 
185  if (*Mode == 'w')
186  Flags |= O_TRUNC;
187 
188  if (*Mode == 'a')
189  Flags |= O_APPEND;
190 
191  return Flags;
192 }
