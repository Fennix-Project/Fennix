140 EXTERNC int cxx_Probe()
141 {
142  uint16_t VendorIDs[] = {0x8086, /* Intel */
143  0x15AD, /* VMware */
144  PCI_END};
145  uint16_t DeviceIDs[] = {0x9D71 /* Sunrise Point-LP HD Audio */,
146  0x2668 /* ICH6 */,
147  0x293E /* ICH9 */,
148  PCI_END};
149  Devices = GetPCIDevices(VendorIDs, DeviceIDs);
150  if (Devices == nullptr)
151  {
152  KernelLog("No HDA device found.");
153  return -ENODEV;
154  }
155 
156  PCIArray *ctx = Devices;
157  bool Found = false;
158  size_t Count = 0;
159  while (ctx != nullptr)
160  {
161  if (Count++ > sizeof(Drivers) / sizeof(HDADevice *))
162  break;
163 
164  PCIHeader0 *PCIBaseAddress = (PCIHeader0 *)ctx->Device->Header;
165  uint32_t PCIBAR0 = PCIBaseAddress->BAR0;
166  uint8_t Type = PCIBAR0 & 1;
167  if (Type == 1)
168  {
169  KernelLog("Device %x:%x.%d BAR0 is I/O.",
170  PCIBaseAddress->Header.VendorID,
171  PCIBaseAddress->Header.DeviceID,
172  PCIBaseAddress->Header.ProgIF);
173  continue;
174  }
175 
176  Found = true;
177  ctx = (PCIArray *)ctx->Next;
178  }
179 
180  if (!Found)
181  {
182  KernelLog("No valid HDA device found.");
183  return -EINVAL;
184  }
185  return 0;
