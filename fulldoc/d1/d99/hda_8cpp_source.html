119 EXTERNC int cxx_Probe()
120 {
121  uint16_t VendorIDs[] = {0x8086, /* Intel */
122  0x15AD, /* VMware */
123  PCI_END};
124  uint16_t DeviceIDs[] = {0x9D71 /* Sunrise Point-LP HD Audio */,
125  0x2668 /* ICH6 */,
126  0x293E /* ICH9 */,
127  PCI_END};
128  Devices = FindPCIDevices(VendorIDs, DeviceIDs);
129  if (Devices == nullptr)
130  {
131  Log("No HDA device found.");
132  return -ENODEV;
133  }
134 
135  PCIArray *ctx = Devices;
136  bool Found = false;
137  size_t Count = 0;
138  while (ctx != nullptr)
139  {
140  if (Count++ > sizeof(Drivers) / sizeof(HDADevice *))
141  break;
142 
143  PCIHeader0 *PCIBaseAddress = (PCIHeader0 *)ctx->Device->Header;
144  uint32_t PCIBAR0 = PCIBaseAddress->BAR0;
145  uint8_t Type = PCIBAR0 & 1;
146  if (Type == 1)
147  {
148  Log("Device %x:%x.%d BAR0 is I/O.",
149  PCIBaseAddress->Header.VendorID,
150  PCIBaseAddress->Header.DeviceID,
151  PCIBaseAddress->Header.ProgIF);
152  continue;
153  }
154 
155  Found = true;
156  ctx = (PCIArray *)ctx->Next;
157  }
158 
159  if (!Found)
160  {
161  Log("No valid HDA device found.");
162  return -EINVAL;
163  }
164  return 0;
165 }
