214  void Signal::InitTrampoline()
215  {
216  if (unlikely(TrampAddr))
217  return;
218 
219  PCB *pcb = (PCB *)ctx;
220 
221  debug("Trampoline not set up yet");
222  switch (pcb->Info.Compatibility)
223  {
224  case Native:
225  {
226  debug("%#lx - %#lx",
227  &_sig_native_trampoline_end,
228  &_sig_native_trampoline_start);
229 
230  TrampSz = (size_t)&_sig_native_trampoline_end -
231  (size_t)&_sig_native_trampoline_start;
232  TrampAddr = pcb->vma->RequestPages(TO_PAGES(TrampSz), true);
233  memcpy((void *)TrampAddr,
234  (void *)&_sig_native_trampoline_start,
235  TrampSz);
236  debug("Trampoline at %#lx with size %lld",
237  TrampAddr, TrampSz);
238  break;
239  }
240  case Linux:
241  {
242  debug("%#lx - %#lx",
243  &_sig_linux_trampoline_end,
244  &_sig_linux_trampoline_start);
245 
246  TrampSz = (size_t)&_sig_linux_trampoline_end -
247  (size_t)&_sig_linux_trampoline_start;
248  TrampAddr = pcb->vma->RequestPages(TO_PAGES(TrampSz), true);
249  memcpy((void *)TrampAddr,
250  (void *)&_sig_linux_trampoline_start,
251  TrampSz);
252  debug("Trampoline at %#lx with size %lld",
253  TrampAddr, TrampSz);
254  break;
255  }
256  case Windows:
257  {
258  assert(!"Windows compatibility not implemented");
259  break;
260  }
261  default:
262  /* Process not fully initalized. */
263  return;
264  }
265  }

347  bool Signal::HandleSignal(CPU::SchedulerFrame *tf, void *thread)
348  {
349  /* We don't want to do this in kernel mode */
350  if (tf->cs != GDT_USER_CODE)
351  return false;
352 
353  if (Queue.empty())
354  return false;
355 
356  debug("We have %d signals to handle", Queue.size());
357 
358  SmartLock(SignalLock);
359  SignalInfo sigI = GetAvailableSignal(thread);
360  if (sigI.sig == SIG_NULL)
361  return false;
362 
363  uintptr_t _p_rsp = ((PCB *)ctx)->PageTable->Get(tf->rsp);
364  uint64_t paRsp = _p_rsp;
365  paRsp &= ~0xF; /* Align */
366  paRsp -= 128; /* Red zone */
367 
368  /* Calculate the virtual rsp */
369  uintptr_t _v_rsp = tf->rsp;
370  _v_rsp &= ~0xF; /* Align */
371  _v_rsp -= 128; /* Red zone */
372  uint64_t *vRsp = (uint64_t *)(_v_rsp - sizeof(StackInfo));
373  vRsp--; /* Alignment */
374  vRsp--; /* Handler Address */
375  assert(!((uintptr_t)vRsp & 0xF));
376 
377  /* Add the stack info */
378  StackInfo si{};
379  CPU::x64::fxsave(&si.fx);
380  si.tf = *tf;
381  si.GSBase = CPU::x64::rdmsr(CPU::x64::MSR_GS_BASE);
382  si.FSBase = CPU::x64::rdmsr(CPU::x64::MSR_FS_BASE);
383  si.ShadowGSBase = CPU::x64::rdmsr(CPU::x64::MSR_SHADOW_GS_BASE);
384  si.SignalMask = ((TCB *)thread)->Signals.Mask.to_ulong();
385  si.Compatibility = ((PCB *)ctx)->Info.Compatibility;
386 
387  debug("gs: %#lx fs: %#lx shadow: %#lx",
388  si.GSBase, si.FSBase, si.ShadowGSBase);
389 
390  /* Copy the stack info */
391  uint64_t *pRsp = (uint64_t *)(paRsp - sizeof(StackInfo));
392  memcpy(pRsp, &si, sizeof(StackInfo));
393 
394  /* Set the handler address */
395  pRsp--; /* Alignment */
396  pRsp--;
397  *pRsp = uint64_t(sa[sigI.sig].sa_handler.Handler);
398 
399  assert(!((uintptr_t)pRsp & 0xF));
400 
401  int cSig = LinuxSig() ? ConvertSignalToLinux((Signals)sigI.sig) : sigI.sig;
402 
403 #ifdef DEBUG
404  DumpData("Stack Data", (void *)pRsp,
405  paRsp - uint64_t(pRsp));
406  debug("initial stack tf->rsp: %#lx after: %#lx",
407  tf->rsp, uint64_t(vRsp));
408  debug("sig: %d -> %d", sigI.sig, cSig);
409 #endif
410 
411  tf->rsp = uint64_t(vRsp);
412  tf->rip = uint64_t(TrampAddr);
413 
414  /* void func(int signo); */
415  /* void func(int signo, siginfo_t *info, void *context); */
416  tf->rdi = cSig;
417  if (sa[sigI.sig].Flags & SA_SIGINFO)
418  {
419  fixme("SA_SIGINFO not implemented");
420  siginfo_t *info = 0;
421  void *context = 0;
422  tf->rsi = uint64_t(info);
423  tf->rdx = uint64_t(context);
424  tf->rcx = 0;
425  tf->r8 = 0;
426  tf->r9 = 0;
427  }
428  else
429  {
430  tf->rsi = 0;
431  tf->rdx = 0;
432  tf->rcx = 0;
433  tf->r8 = 0;
434  tf->r9 = 0;
435  }
436 
437  ((TCB *)thread)->Signals.Mask = sa[sigI.sig].Mask;
438  assert(TrampAddr != nullptr);
439  return true;
440  }

490  int Signal::SetAction(Signals sig, const SignalAction *act)
491  {
492  SmartLock(SignalLock);
493  if ((size_t)sig > sizeof(sa) / sizeof(sa[0]))
494  {
495  debug("Invalid signal: %d", sig);
496  return -EINVAL;
497  }
498 
499  if ((long)act->sa_handler.Disposition == SAD_IGN)
500  {
501  Disposition[sig] = SIG_IGN;
502  debug("Set disposition for %s to SIG_IGN", sigStr[sig]);
503  debug("Discarding pending signals %s", sigStr[sig]);
504 
505  bool found = false;
506  forItr(itr, Queue)
507  {
508  if (itr->sig == sig)
509  {
510  Queue.erase(itr);
511  found = true;
512  break;
513  }
514  }
515 
516  if (!found)
517  {
518  debug("No pending signal %s", sigStr[sig]);
519  }
520  }
521 
522  if ((long)act->sa_handler.Disposition == SAD_DFL)
523  {
524  Disposition[sig] = GetDefaultSignalDisposition(sig);
525  debug("Set disposition for %s to %s (default)", sigStr[sig],
526  dispStr[Disposition[sig]]);
527  }
528 
529  sa[sig].sa_handler.Handler = act->sa_handler.Handler;
530  sa[sig].Mask = act->Mask;
531  sa[sig].Flags = act->Flags;
532  sa[sig].Restorer = act->Restorer;
533 
534  debug("Set action for %s with handler %#lx, mask %#lx and flags %#lx",
535  sigStr[sig], sa[sig].sa_handler.Handler, sa[sig].Mask, sa[sig].Flags);
536  return 0;
537  }

557  int Signal::SendSignal(Signals sig, sigval val, pid_t tid)
558  {
559  SmartLock(SignalLock);
560  PCB *pcb = (PCB *)ctx;
561  LastSignal = sig;
562 
563  debug("Sending signal %s to %s(%d)",
564  sigStr[sig], pcb->Name, pcb->ID);
565 
566  if (sa[sig].sa_handler.Handler)
567  {
568  if (pcb->Security.ExecutionMode == Kernel)
569  {
570  info("Kernel processes cannot have signal handlers");
571  return -EINVAL;
572  }
573 
574  if (sa[sig].sa_handler.Disposition == SAD_IGN)
575  {
576  debug("Ignoring signal %s", sigStr[sig]);
577  return 0;
578  }
579 
580  debug("sa_handler: %#lx", sa[sig].sa_handler.Handler);
581  debug("Adding signal %s to queue", sigStr[sig]);
582  goto CompleteSignal;
583  }
584  else
585  {
586  debug("No handler for signal %s", sigStr[sig]);
587  }
588 
589  if (thisThread->Signals.Mask.test(sig - 1))
590  {
591  debug("Signal %s is blocked by thread mask",
592  sigStr[sig]);
593  return 0;
594  }
595 
596  debug("Signal disposition: %s", dispStr[Disposition[sig]]);
597  switch (Disposition[sig])
598  {
599  case SIG_TERM:
600  {
601  if (unlikely(pcb->Security.IsCritical))
602  {
603  debug("Critical process %s received signal %s(%d): Terminated",
604  pcb->Name, sigStr[sig], sig);
605  // int3;
606  }
607 
608  pcb->SetExitCode(MakeExitCode(sig));
609  debug("We have %d watchers", this->Watchers.size());
610  if (this->Watchers.size() > 0)
611  pcb->SetState(Zombie);
612  else
613  pcb->SetState(Terminated);
614  return val.sival_int == Tasking::KILL_CRASH ? -EFAULT : 0;
615  }
616  case SIG_IGN:
617  {
618  debug("Ignoring signal %d", sig);
619  return 0;
620  }
621  case SIG_CORE:
622  {
623  fixme("core dump");
624 
625  if (unlikely(pcb->Security.IsCritical))
626  {
627  debug("Critical process %s received signal %s(%d): Core dumped",
628  pcb->Name, sigStr[sig], sig);
629  // int3;
630  }
631 
632  pcb->SetExitCode(MakeExitCode(sig));
633  debug("We have %d watchers", this->Watchers.size());
634  if (this->Watchers.size() > 0)
635  pcb->SetState(CoreDump);
636  else
637  pcb->SetState(Terminated);
638  return val.sival_int == Tasking::KILL_CRASH ? -EFAULT : 0;
639  }
640  case SIG_STOP:
641  {
642  debug("Stopping process %s(%d) with signal %s(%d)",
643  pcb->Name, pcb->ID, sigStr[sig], sig);
644  pcb->SetState(Stopped);
645  return 0;
646  }
647  case SIG_CONT:
648  {
649  debug("Continuing process %s(%d) with signal %s(%d)",
650  pcb->Name, pcb->ID, sigStr[sig], sig);
651  pcb->SetState(Ready);
652  return 0;
653  }
654  default:
655  assert(!"Invalid signal disposition");
656  return -EINVAL;
657  };
658 
659  CompleteSignal:
660  if (pcb->Security.ExecutionMode == Kernel)
661  {
662  debug("Kernel process %s received signal %s(%d)! Ignoring... (with exceptions)",
663  pcb->Name, sigStr[sig], sig);
664  return 0;
665  }
666 
667  this->InitTrampoline();
668 
669  debug("Signal %s(%d) completed", sigStr[sig], sig);
670  if (Disposition[sig] != SIG_IGN)
671  {
672  foreach (auto info in Watchers)
673  {
674  Signal *who = (Signal *)info.val.sival_ptr;
675  assert(who != nullptr);
676  debug("Sending SIGCHLD to %s(%d)",
677  ((PCB *)who->GetContext())->Name,
678  ((PCB *)who->GetContext())->ID);
679  who->SendSignal(SIGCHLD, val);
680  }
681  }
682 
683  debug("Adding signal to queue");
684  Queue.push_back({.sig = sig, .val = val, .tid = tid});
685  return 0;
686  }

688  int Signal::WaitAnySignal()
689  {
690  debug("Waiting for any signal");
691 
692  size_t oldSize = Queue.size();
693  Reset:
694  while (Queue.size() == oldSize)
695  TaskManager->Yield();
696 
697  if (Queue.size() > oldSize)
698  {
