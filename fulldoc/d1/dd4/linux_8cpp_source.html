277 /* https://man7.org/linux/man-pages/man2/mmap.2.html */
278 static void *linux_mmap(void *addr, size_t length, int prot,
279  int flags, int fildes, off_t offset)
280 {
281  static_assert(PROT_NONE == sc_PROT_NONE);
282  static_assert(PROT_READ == sc_PROT_READ);
283  static_assert(PROT_WRITE == sc_PROT_WRITE);
284  static_assert(PROT_EXEC == sc_PROT_EXEC);
285 
286  int new_flags = 0;
287  if (flags & MAP_SHARED)
288  {
289  new_flags |= sc_MAP_SHARED;
290  flags &= ~MAP_SHARED;
291  }
292  if (flags & MAP_PRIVATE)
293  {
294  new_flags |= sc_MAP_PRIVATE;
295  flags &= ~MAP_PRIVATE;
296  }
297  if (flags & MAP_FIXED)
298  {
299  new_flags |= sc_MAP_FIXED;
300  flags &= ~MAP_FIXED;
301  }
302  if (flags & MAP_ANONYMOUS)
303  {
304  new_flags |= sc_MAP_ANONYMOUS;
305  flags &= ~MAP_ANONYMOUS;
306  }
307  if (flags)
308  fixme("unhandled flags: %#x", flags);
309 
310  return sys_mmap(thisFrame, addr, length, prot,
311  new_flags, fildes, offset);
312 }

353 /* https://man7.org/linux/man-pages/man2/readv.2.html */
354 static ssize_t linux_readv(int fildes, const struct iovec *iov, int iovcnt)
355 {
356  size_t iovec_size = sizeof(struct iovec) * iovcnt;
357  Tasking::PCB *pcb = thisProcess;
358  Memory::SmartHeap sh(iovec_size, pcb->vma);
359  {
360  Memory::SwapPT swap(pcb->PageTable);
361  memcpy(sh, iov, iovec_size);
362  }
363  iov = (struct iovec *)sh.Get();
364 
365  ssize_t Total = 0;
366  for (int i = 0; i < iovcnt; i++)
367  {
368  debug("%d: iov[%d]: %p %d", fildes, i, iov[i].iov_base, iov[i].iov_len);
369  ssize_t n = linux_read(fildes, iov[i].iov_base, iov[i].iov_len);
370  if (n < 0)
371  return n;
372  debug("n: %d", n);
373 
374  Total += n;
375  if (n < (ssize_t)iov[i].iov_len)
376  {
377  debug("break");
378  break;
379  }
380  }
381  debug("readv: %d", Total);
382  return Total;
383 }

452 /* https://man7.org/linux/man-pages/man2/wait4.2.html */
453 static pid_t linux_wait4(pid_t pid, int *wstatus, int options, struct rusage *rusage)
454 {
455  static_assert(sizeof(struct rusage) < PAGE_SIZE);
456 
457  Tasking::PCB *pcb = thisProcess;
458  Memory::Virtual vmm = Memory::Virtual(pcb->PageTable);
459 
460  if (!vmm.Check(rusage, Memory::US) && rusage != nullptr)
461  {
462  warn("Invalid address %#lx", rusage);
463  return -EFAULT;
464  }
465 
466  if (pid == -1)
467  pid = pcb->ID + 1; /* TODO: Wait for any child process */
468 
469  Tasking::PCB *tPcb = pcb->GetContext()->GetProcessByID(pid);
470 
471  if (!tPcb)
472  {
473  warn("Invalid PID %d", pid);
474  return -ECHILD;
475  }
476 
477  tPcb->KeepInMemory = true;
478 
479  Tasking::TaskState state = tPcb->State;
480  debug("Waiting for %d(%#lx) state %d", pid, tPcb, state);
481  while (tPcb->State == state)
482  pcb->GetContext()->Yield();
483  debug("Waited for %d(%#lx) state %d", pid, tPcb, state);
484 
485  if (wstatus)
486  {
487  int status = tPcb->ExitCode.load();
488 
489  Memory::SwapPT swap(pcb->PageTable);
490  *wstatus = status;
491  }
492 
493  if (rusage != nullptr)
494  {
495  size_t kTime = tPcb->Info.KernelTime;
496  size_t uTime = tPcb->Info.UserTime;
497  size_t _maxrss = tPcb->GetSize();
498 
499  {
500  Memory::SwapPT swap(pcb->PageTable);
501 
502  rusage->ru_utime.tv_sec = uTime / 1000000000000000; /* Seconds */
503  rusage->ru_utime.tv_usec = uTime / 1000000000; /* Microseconds */
504 
505  rusage->ru_stime.tv_sec = kTime / 1000000000000000; /* Seconds */
506  rusage->ru_stime.tv_usec = kTime / 1000000000; /* Microseconds */
507 
508  rusage->ru_maxrss = _maxrss;
509  /* TODO: The rest of the fields */
510  }
511  }
512 
513  tPcb->KeepInMemory = false;
514  debug("%d", pid);
515  return pid;
516 }

527 static int linux_arch_prctl(int code, unsigned long addr)
528 {

531 
532  if (!vmm.Check((void *)addr))
533  {
534  warn("Invalid address %#lx", addr);
535  return -EFAULT;
536  }
537 
538  if (!vmm.Check((void *)addr, Memory::US))
539  {
540  warn("Address %#lx is not user accessible", addr);
541  return -EPERM;
542  }
543 
544  switch (code)
545  {
546  case ARCH_SET_GS:
547  {
548 #if defined(a64)
549  CPU::x64::wrmsr(CPU::x64::MSRID::MSR_GS_BASE, addr);
550 #elif defined(a32)

552 #endif

554  }
555  case ARCH_SET_FS:

559 #elif defined(a32)
560  CPU::x32::wrmsr(CPU::x32::MSRID::MSR_FS_BASE, addr);
561 #endif
562  return 0;
563  }
564  case ARCH_GET_FS:
565  {
566 #if defined(a64)
567  *r_cst(uint64_t *, addr) =
568  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_FS_BASE);
569 #elif defined(a32)
570  *r_cst(uint64_t *, addr) =
571  CPU::x32::rdmsr(CPU::x32::MSRID::MSR_FS_BASE);
572 #endif
573  return 0;
574  }
575  case ARCH_GET_GS:
576  {
577 #if defined(a64)
578  *r_cst(uint64_t *, addr) =
579  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_GS_BASE);
580 #elif defined(a32)
581  *r_cst(uint64_t *, addr) =
582  CPU::x32::rdmsr(CPU::x32::MSRID::MSR_GS_BASE);
583 #endif
584  return 0;
585  }
586  case ARCH_GET_CPUID:
587  case ARCH_SET_CPUID:
588  case ARCH_GET_XCOMP_SUPP:
589  case ARCH_GET_XCOMP_PERM:
590  case ARCH_REQ_XCOMP_PERM:
591  case ARCH_GET_XCOMP_GUEST_PERM:
592  case ARCH_REQ_XCOMP_GUEST_PERM:
593  case ARCH_XCOMP_TILECFG:
594  case ARCH_XCOMP_TILEDATA:
595  case ARCH_MAP_VDSO_X32:
596  case ARCH_MAP_VDSO_32:
597  case ARCH_MAP_VDSO_64:
598  case ARCH_GET_UNTAG_MASK:
599  case ARCH_ENABLE_TAGGED_ADDR:
600  case ARCH_GET_MAX_TAG_BITS:
601  case ARCH_FORCE_TAGGED_SVA:
602  {
603  fixme("Code %#lx not implemented", code);
604  return -ENOSYS;
605  }
606  default:
607  {
608  warn("Invalid code %#lx", code);
609  return -EINVAL;
610  }
611  }
612 }
