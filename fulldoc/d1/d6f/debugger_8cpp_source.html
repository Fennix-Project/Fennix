28 static inline NIF void uart_wrapper(char c, void *unused)
29 {
30  static int once = 0;
31  if (unlikely(!once++))
32  {
33  uint8_t com = inb(0x3F8);
34  if (com != 0xFF)
35  {
36  outb(s_cst(uint16_t, 0x3F8 + 1), 0x00); // Disable all interrupts
37  outb(s_cst(uint16_t, 0x3F8 + 3), 0x80); // Enable DLAB (set baud rate divisor)
38  outb(s_cst(uint16_t, 0x3F8 + 0), 0x1); // Set divisor to 1 (lo byte) 115200 baud
39  outb(s_cst(uint16_t, 0x3F8 + 1), 0x0); // (hi byte)
40  outb(s_cst(uint16_t, 0x3F8 + 3), 0x03); // 8 bits, no parity, one stop bit
41  outb(s_cst(uint16_t, 0x3F8 + 2), 0xC7); // Enable FIFO, clear them, with 14-byte threshold
42  outb(s_cst(uint16_t, 0x3F8 + 4), 0x0B); // IRQs enabled, RTS/DSR set
43 
44  /* FIXME https://wiki.osdev.org/Serial_Ports */
45  // outb(s_cst(uint16_t, 0x3F8 + 0), 0x1E);
46  // outb(s_cst(uint16_t, 0x3F8 + 0), 0xAE);
47  // Check if the serial port is faulty.
48  // if (inb(s_cst(uint16_t, 0x3F8 + 0)) != 0xAE)
49  // {
50  // static int once = 0;
51  // if (!once++)
52  // warn("Serial port %#llx is faulty.", 0x3F8);
53  // // serialports[0x3F8] = false; // ignore for now
54  // // return;
55  // }
56 
57  // Set to normal operation mode.
58  outb(s_cst(uint16_t, 0x3F8 + 4), 0x0F);
59  serialports[0] = true;
60  }
61  }
62 
63  if (likely(serialports[0]))
64  {
65  while ((inb(s_cst(uint16_t, 0x3F8 + 5)) & 0x20) == 0)
66  ;
67  outb(0x3F8, c);
68  }
69  UNUSED(unused);
70 }

72 static inline NIF bool WritePrefix(DebugLevel Level, const char *File, int Line, const char *Function, const char *Format, va_list args)
73 {
74  const char *DbgLvlString;
75  switch (Level)
76  {
77  case DebugLevelError:
78  DbgLvlString = "ERROR";
79  break;
80  case DebugLevelWarning:
81  DbgLvlString = "WARN ";
82  break;
83  case DebugLevelInfo:
84  DbgLvlString = "INFO ";
85  break;
86  case DebugLevelDebug:
87  DbgLvlString = "DEBUG";
88  break;
89  case DebugLevelTrace:
90  DbgLvlString = "TRACE";
91  break;
92  case DebugLevelFixme:
93  DbgLvlString = "FIXME";
94  break;
95  case DebugLevelStub:
96  fctprintf(uart_wrapper, nullptr, "STUB | %s>%s() is stub\n", File, Function);
97  return false;
98  case DebugLevelFunction:
99  fctprintf(uart_wrapper, nullptr, "FUNC | %s>%s( ", File, Function);
100  vfctprintf(uart_wrapper, nullptr, Format, args);
101  fctprintf(uart_wrapper, nullptr, " )\n");
102  return false;
103  case DebugLevelUbsan:
104  {
105  DbgLvlString = "UBSAN";
106  fctprintf(uart_wrapper, nullptr, "%s| ", DbgLvlString);
107  return true;
108  }
109  default:
110  DbgLvlString = "UNKNW";
111  break;
112  }
113  fctprintf(uart_wrapper, nullptr, "%s| %s>%s:%d: ", DbgLvlString, File, Function, Line);
114  return true;
115 }
