78  {
79  switch (Data->Reason)
80  {
81  case AcknowledgeReason:

83  debug("Kernel acknowledged the driver.");
84  break;
85  }
86  case ConfigurationReason:
87  {
88  debug("Module received configuration data.");
89  PCIBaseAddress = reinterpret_cast<PCIDeviceHeader *>(Data->RawPtr);
90  if (PCIBaseAddress->VendorID == 0x1022 && PCIBaseAddress->DeviceID == 0x2000)
91  {
92  trace("Found AMD PCNET.");
93  uint32_t PCIBAR = ((PCIHeader0 *)PCIBaseAddress)->BAR0;
94  BAR.Type = PCIBAR & 1;
95  BAR.IOBase = (uint16_t)(PCIBAR & (~3));
96  BAR.MemoryBase = PCIBAR & (~15);
97  }
98  else
99  return DEVICE_NOT_SUPPORTED;

101  }
102  case QueryReason:
103  {
104  memcpy(Data->NetworkCallback.Fetch.Name, (void *)"AMD PCNET", 10);
105  Data->NetworkCallback.Fetch.MAC = MAC.ToHex();

108  case SendReason:
109  {
110  break;
111  }
112  case StopReason:
113  {
114  // TODO: Stop the driver.

116  break;
117  }
118  default:
119  {
