433 bool InitializePort(uint16_t Port)
434 {
435  ECS;
436  LCR lcr = {0};
437  IER ier = {0};
438  FCR fcr = {0};
439  MCR mcr = {0};
440 
441  outb(Port + 3, lcr.raw);
442  outb(Port + 1, ier.raw);
443 
444  lcr.DLAB = 1;
445  outb(Port + 3, lcr.raw);
446 
447  outb(Port + 0, RATE_115200_LO);
448  outb(Port + 1, RATE_115200_HI);
449 
450  lcr.DLAB = 0;
451  lcr.WordLength = 0b11;
452  outb(Port + 3, lcr.raw);
453 
454  fcr.FIFO = 1;
455  fcr.ClearRX = 1;
456  fcr.ClearTX = 1;
457  fcr.TriggerLevel = 0b11;
458  outb(Port + 2, fcr.raw);
459 
460  mcr.DataTerminalReady = 1;
461  mcr.RequestToSend = 1;
462  mcr.Out2 = 1;
463  mcr.Loopback = 1;
464  outb(Port + 4, mcr.raw);
465 
466  /* Test the serial port */
467  outb(Port + 0, 0x48);
468  uint8_t result = inb(Port + 0);
469  if (result != 0x48)
470  {
471  /* FIXME: DETECT BAUD RATE
472  Do multiple test to check if the output is garbage.
473  If so, reduce the baud rate until it works. */
474 
475  LCS;
476  KernelPrint("Port %#X test failed!", Port);
477  return false;
478  }
479 
480  /* Set normal operation mode */
481  mcr.DataTerminalReady = 1;
482  mcr.RequestToSend = 1;
483  mcr.Out1 = 1;
484  mcr.Out2 = 1;
485  mcr.Loopback = 0;
486  outb(Port + 4, mcr.raw);
487 
488  /* Enable interrupts on receive */
489  ier.InterruptOnReceive = 1;
490  outb(Port + 1, ier.raw);
491  RegisterInterruptHandler(3, UartCOM24);
492  RegisterInterruptHandler(4, UartCOM13);
493 
494  LCS;
495  KernelPrint("Port %#X initialized", Port);
496  return true;
497 }
