32 void FAST_FUNC setup_environment(const char *shell, int flags, const struct passwd *pw)
33 {
34  if (!shell || !shell[0])
35  shell = DEFAULT_SHELL;
36 
37  /* Change the current working directory to be the home directory
38  * of the user */
39  if (flags & SETUP_ENV_CHDIR) {
40  if (chdir_or_warn(pw->pw_dir) != 0) {
41  xchdir((flags & SETUP_ENV_TO_TMP) ? "/tmp" : "/");
42  }
43  }
44 
45  if (flags & SETUP_ENV_CLEARENV) {
46  const char *term;
47 
48  /* Leave TERM unchanged. Set HOME, SHELL, USER, LOGNAME, PATH.
49  * Unset all other environment variables. */
50  term = getenv("TERM");
51  clearenv();
52  if (term)
53  xsetenv("TERM", term);
54  xsetenv("PATH", (pw->pw_uid ? bb_default_path : bb_default_root_path));
55  goto shortcut;
56  // No, gcc (4.2.1) is not clever enough to do it itself.
57  //xsetenv("USER", pw->pw_name);
58  //xsetenv("LOGNAME", pw->pw_name);
59  //xsetenv("HOME", pw->pw_dir);
60  //xsetenv("SHELL", shell);
61  } else
62  if (flags & (SETUP_ENV_CHANGEENV|SETUP_ENV_CHANGEENV_LOGNAME)) {
63  /* Set HOME, SHELL, and if not becoming a super-user
64  * or if SETUP_ENV_CHANGEENV_LOGNAME, USER and LOGNAME. */
65  if ((flags & SETUP_ENV_CHANGEENV_LOGNAME) || pw->pw_uid != 0) {
66  shortcut:
67  xsetenv("USER", pw->pw_name);
68  xsetenv("LOGNAME", pw->pw_name);
69  }
70  xsetenv("HOME", pw->pw_dir);
71  xsetenv("SHELL", shell);
72  }
73 }
