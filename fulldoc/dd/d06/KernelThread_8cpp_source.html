281 void BootLogoAnimationThread()
282 {
283  char BootAnimPath[16];
284  while (FrameCount < 27)
285  {
286  sprintf(BootAnimPath, "/etc/boot/%ld.tga", FrameCount);
287  File ba = vfs->Open(BootAnimPath);
288  if (!ba.IsOK())
289  {
290  vfs->Close(ba);
291  debug("Failed to load boot animation frame %s", BootAnimPath);
292  break;
293  }
294 
295  FrameSizes[FrameCount] = s_cst(uint32_t, ba.GetLength());
296  Frames[FrameCount] = new uint8_t[ba.GetLength()];
297  vfs->Read(ba, Frames[FrameCount], ba.GetLength());
298  vfs->Close(ba);
299  FrameCount++;
300  }
301 
302  uint32_t DispX = Display->GetBuffer(1)->Width;
303  uint32_t DispY = Display->GetBuffer(1)->Height;
304 
305  for (size_t i = 1; i < FrameCount; i++)
306  {
307  int x, y, channels;
308 
309  if (!stbi_info_from_memory((uint8_t *)Frames[i], FrameSizes[i], &x, &y, &channels))
310  continue;
311 
312  uint8_t *img = stbi_load_from_memory((uint8_t *)Frames[i], FrameSizes[i], &x, &y, &channels, STBI_rgb_alpha);
313 
314  if (img == NULL)
315  continue;
316 
317  int offsetX = DispX / 2 - x / 2;
318  int offsetY = DispY / 2 - y / 2;
319 
320  for (int i = 0; i < x * y; i++)
321  {
322  uint32_t pixel = ((uint32_t *)img)[i];
323  int r = (pixel >> 16) & 0xFF;
324  int g = (pixel >> 8) & 0xFF;
325  int b = (pixel >> 0) & 0xFF;
326  int a = (pixel >> 24) & 0xFF;
327 
328  if (a != 0xFF)
329  {
330  r = (r * a) / 0xFF;
331  g = (g * a) / 0xFF;
332  b = (b * a) / 0xFF;
333  }
334 
335  Display->SetPixel((i % x) + offsetX, (i / x) + offsetY, (r << 16) | (g << 8) | (b << 0), 1);
336  }
337 
338  free(img);
339  Display->SetBuffer(1);
340  TaskManager->Sleep(50);
341  }
342 
343  int brightness = 100;
344  while (brightness >= 0)
345  {
346  brightness -= 10;
347  Display->SetBrightness(brightness, 1);
348  Display->SetBuffer(1);
349  TaskManager->Sleep(5);
350  }
351 }
