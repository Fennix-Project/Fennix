318 void BootLogoAnimationThread()
319 {
320  char BootAnimPath[16];
321  while (FrameCount < 27)
322  {
323  sprintf(BootAnimPath, "/etc/boot/%ld.tga", FrameCount);
324  RefNode *frame = vfs->Open(BootAnimPath);
325  if (!frame)
326  {
327  debug("Failed to load boot animation frame %s", BootAnimPath);
328  break;
329  }
330 
331  FrameSizes[FrameCount] = s_cst(uint32_t, frame->Length);
332  Frames[FrameCount] = new uint8_t[frame->Length];
333  frame->Read(Frames[FrameCount], frame->Length);
334  delete frame;
335  FrameCount++;
336  }
337 
338  uint32_t DispX = Display->GetBuffer(1)->Width;
339  uint32_t DispY = Display->GetBuffer(1)->Height;
340 
341  for (size_t i = 1; i < FrameCount; i++)
342  {
343  int x, y, channels;
344 
345  if (!stbi_info_from_memory((uint8_t *)Frames[i], FrameSizes[i],
346  &x, &y, &channels))
347  continue;
348 
349  uint8_t *img = stbi_load_from_memory((uint8_t *)Frames[i],
350  FrameSizes[i], &x, &y,
351  &channels, STBI_rgb_alpha);
352 
353  if (img == NULL)
354  continue;
355 
356  int offsetX = DispX / 2 - x / 2;
357  int offsetY = DispY / 2 - y / 2;
358 
359  for (int i = 0; i < x * y; i++)
360  {
361  uint32_t pixel = ((uint32_t *)img)[i];
362  int r = (pixel >> 16) & 0xFF;
363  int g = (pixel >> 8) & 0xFF;
364  int b = (pixel >> 0) & 0xFF;
365  int a = (pixel >> 24) & 0xFF;
366 
367  if (a != 0xFF)
368  {
369  r = (r * a) / 0xFF;
370  g = (g * a) / 0xFF;
371  b = (b * a) / 0xFF;
372  }
373 
374  Display->SetPixel((i % x) + offsetX, (i / x) + offsetY,
375  (r << 16) | (g << 8) | (b << 0), 1);
376  }
377 
378  free(img);
379  Display->SetBuffer(1);
380  TaskManager->Sleep(50);
381  }
382 
383  int brightness = 100;
384  while (brightness >= 0)
385  {
386  brightness -= 10;
387  Display->SetBrightness(brightness, 1);
388  Display->SetBuffer(1);
389  TaskManager->Sleep(5);
390  }
391 }
