36  __no_sanitize("alignment") void Physical::ReserveEssentials()
37  {
38  debug("Reserving pages...");
39  /* The bootloader won't give us the entire mapping, so we
40  reserve everything and then unreserve the usable pages. */
41  this->ReservePages(0, TO_PAGES(bInfo.Memory.Size));
42  debug("Unreserving usable pages...");
43 
44  for (uint64_t i = 0; i < bInfo.Memory.Entries; i++)
45  {
46  if (bInfo.Memory.Entry[i].Type == Usable)
47  this->UnreservePages(bInfo.Memory.Entry[i].BaseAddress, TO_PAGES(bInfo.Memory.Entry[i].Length));
48  }
49 
50  debug("Reserving 0x0-0xFFFFF range...");
51  // this->ReservePage((void *)0x0); /* Trampoline stack, gdt, idt, etc... */
52  // this->ReservePages((void *)0x2000, 4); /* TRAMPOLINE_START */
53 
54  /* Reserve the lower part of memory. (0x0-0xFFFFF)
55  This includes: BIOS, EBDA, VGA, SMP, etc...
56  https://wiki.osdev.org/Memory_Map_(x86)
57  */
58  this->ReservePages((void *)0x0, TO_PAGES(0xFFFFF));
59 
60  debug("Reserving bitmap region %#lx-%#lx...", PageBitmap.Buffer, (void *)((uintptr_t)PageBitmap.Buffer + PageBitmap.Size));

62

64  this->ReservePages(bInfo.Kernel.PhysicalBase, TO_PAGES(bInfo.Kernel.Size));
65
