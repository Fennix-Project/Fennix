76 EXTERNC void _KPrint(const char *Format, va_list Args)
77 {
78  SmartLock(KernelLock);
79 
80  if (TimeManager)
81  {
82  uint64_t Nanoseconds = TimeManager->GetNanosecondsSinceClassCreation();
83  if (Nanoseconds != 0)
84  {
85 #if defined(a64)
86  printf("\eCCCCCC[\e00AEFF%lu.%07lu\eCCCCCC] ",
87  Nanoseconds / 10000000, Nanoseconds % 10000000);
88 #elif defined(a32)
89  printf("\eCCCCCC[\e00AEFF%llu.%07llu\eCCCCCC] ",
90  Nanoseconds / 10000000, Nanoseconds % 10000000);
91 #elif defined(aa64)
92  printf("\eCCCCCC[\e00AEFF%lu.%07lu\eCCCCCC] ",
93  Nanoseconds / 10000000, Nanoseconds % 10000000);
94 #endif
95  }
96  }
97 
98  vprintf(Format, Args);
99  printf("\eCCCCCC\n");
100  if (!Config.Quiet && Display)
101  Display->SetBuffer(0);
102 }

262 EXTERNC __no_stack_protector NIF void Entry(BootInfo *Info)
263 {
264  trace("Hello, World!");
265 
266  if (strcmp(CPU::Hypervisor(), x86_CPUID_VENDOR_TCG) == 0)
267  {
268  info("\n\n----------------------------------------\nDEBUGGER DETECTED\n----------------------------------------\n\n");
269  DebuggerIsAttached = true;
270  }
271 
272  memcpy(&bInfo, Info, sizeof(BootInfo));
273  debug("BootInfo structure is at %p", &bInfo);
274 
275  // https://wiki.osdev.org/Calling_Global_Constructors
276  trace("There are %d constructors to call",
277  __init_array_end - __init_array_start);
278  for (CallPtr *func = __init_array_start; func != __init_array_end; func++)
279  (*func)();
280 
281 #ifdef a86
282  if (!bInfo.SMBIOSPtr)
283  {
284  trace("SMBIOS was not provided by the bootloader. Trying to find it manually.");
285  for (uintptr_t i = 0xF0000; i < 0x100000; i += 16)
286  {
287  if (memcmp((void *)i, "_SM_", 4) == 0 ||
288  memcmp((void *)i, "_SM3_", 5) == 0)
289  {
290  bInfo.SMBIOSPtr = (void *)i;
291  trace("Found SMBIOS at %#lx", i);
292  }
293  }
294  }
295 
296  if (!bInfo.RSDP)
297  {
298  trace("RSDP was not provided by the bootloader. Trying to find it manually.");
299  /* FIXME: Not always shifting by 4 will work. */
300  uintptr_t EBDABase = (uintptr_t)mminw((void *)0x40E) << 4;
301 
302  for (uintptr_t ptr = EBDABase;
303  ptr < 0x100000; /* 1MB */
304  ptr += 16)
305  {
306  if (unlikely(ptr == EBDABase + 0x400))
307  {
308  trace("EBDA is full. Trying to find RSDP in the BIOS area.");
309  break;
310  }
311 
312  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
313  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
314  {
315  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
316  trace("Found RSDP at %#lx", rsdp);
317  }
318  }
319 
320  for (uintptr_t ptr = 0xE0000;
321  ptr < 0x100000; /* 1MB */
322  ptr += 16)
323  {
324  BootInfo::RSDPInfo *rsdp = (BootInfo::RSDPInfo *)ptr;
325  if (memcmp(rsdp->Signature, "RSD PTR ", 8) == 0)
326  {
327  bInfo.RSDP = (BootInfo::RSDPInfo *)rsdp;
328  trace("Found RSDP at %#lx", rsdp);
329  }
330  }
331  }
332 #endif
333 
334  InitializeMemoryManagement();
335 
336  void *KernelStackAddress = KernelAllocator.RequestPages(TO_PAGES(STACK_SIZE));
337  uintptr_t KernelStack = (uintptr_t)KernelStackAddress + STACK_SIZE - 0x10;
338  debug("Kernel stack: %#lx-%#lx", KernelStackAddress, KernelStack);
339 #if defined(a64)
340  asmv("mov %0, %%rsp"
341  :
342  : "r"(KernelStack)
343  : "memory");
344  asmv("mov $0, %rbp");
345 #elif defined(a32)
346  asmv("mov %0, %%esp"
347  :
348  : "r"(KernelStack)
349  : "memory");
350  asmv("mov $0, %ebp");
351 #elif defined(aa64)
352 #warning "Kernel stack is not set!"
353 #endif
354 
355 #ifdef DEBUG
356  /* I had to do this because KernelAllocator
357  * is a global constructor but we need
358  * memory management to be initialized first.
359  */
360  TestMemoryAllocation();
361  TestString();
362  Test_std();
363 #endif
364  EnableProfiler = true;
365  Main();
366 }
