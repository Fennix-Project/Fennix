78 //
79 // A fairly efficient way to draw a number
80 // based on differences from the old number.
81 // Note: worth the trouble?
82 //
83 void
84 STlib_drawNum
85 ( st_number_t* n,
86  boolean refresh )
87 {
88 
89  int numdigits = n->width;
90  int num = *n->num;
91 
92  int w = SHORT(n->p[0]->width);
93  int h = SHORT(n->p[0]->height);
94  int x = n->x;
95 
96  int neg;
97 
98  n->oldnum = *n->num;
99 
100  neg = num < 0;
101 
102  if (neg)
103  {
104  if (numdigits == 2 && num < -9)
105  num = -9;
106  else if (numdigits == 3 && num < -99)
107  num = -99;
108 
109  num = -num;
110  }
111 
112  // clear the area
113  x = n->x - numdigits*w;
114 
115  if (n->y - ST_Y < 0)
116  I_Error("drawNum: n->y - ST_Y < 0");
117 
118  V_CopyRect(x, n->y - ST_Y, st_backing_screen, w*numdigits, h, x, n->y);
119 
120  // if non-number, do not draw it
121  if (num == 1994)
122  return;
123 
124  x = n->x;
125 
126  // in the special case of 0, you draw 0
127  if (!num)
128  V_DrawPatch(x - w, n->y, n->p[ 0 ]);
129 
130  // draw the new number
131  while (num && numdigits--)
132  {
133  x -= w;
134  V_DrawPatch(x, n->y, n->p[ num % 10 ]);
135  num /= 10;
136  }
137 
138  // draw a minus sign if necessary
139  if (neg)
140  V_DrawPatch(x - 8, n->y, sttminus);
141 }

208 {
209  int w;
210  int h;
211  int x;
212  int y;
213 
214  if (*mi->on && (mi->oldinum != *mi->inum || refresh) && (*mi->inum != -1))

218  x = mi->x - SHORT(mi->p[mi->oldinum]->leftoffset);
219  y = mi->y - SHORT(mi->p[mi->oldinum]->topoffset);
220  w = SHORT(mi->p[mi->oldinum]->width);

222

224  I_Error("updateMultIcon: y - ST_Y < 0");
225 
226  V_CopyRect(x, y-ST_Y, st_backing_screen, w, h, x, y);
227  }
228  V_DrawPatch(mi->x, mi->y, mi->p[*mi->inum]);
