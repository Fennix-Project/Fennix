81 static byte saveg_read8(void)
82 {
83  byte result;
84 
85  if (fread(&result, 1, 1, save_stream) < 1)

96  return result;
97 }

150 static void saveg_read_pad(void)
151 {
152  unsigned long pos;
153  int padding;
154  int i;
155 
156  pos = ftell(save_stream);
157 
158  padding = (4 - (pos & 3)) & 3;
159 
160  for (i=0; i<padding; ++i)
161  {
162  saveg_read8();
163  }
164 }

173 
174  padding = (4 - (pos & 3)) & 3;
175 
176  for (i=0; i<padding; ++i)
177  {
178  saveg_write8(0);
179  }
180 }
