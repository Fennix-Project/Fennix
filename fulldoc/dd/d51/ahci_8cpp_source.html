588  void StartCMD()
589  {
590  while (HBAPortPtr->CommandStatus & HBA_PxCMD_CR)
591  Yield();
592  HBAPortPtr->CommandStatus |= HBA_PxCMD_FRE;
593  HBAPortPtr->CommandStatus |= HBA_PxCMD_ST;
594  }
595 
596  void StopCMD()
597  {
598  HBAPortPtr->CommandStatus &= ~HBA_PxCMD_ST;
599  HBAPortPtr->CommandStatus &= ~HBA_PxCMD_FRE;
600  while (true)
601  {
602  if (HBAPortPtr->CommandStatus & HBA_PxCMD_FR)
603  continue;
604  if (HBAPortPtr->CommandStatus & HBA_PxCMD_CR)
605  continue;
606  break;
607  }
608  }

610  void Configure()
611  {
612  this->StopCMD();
613  void *CmdBase = AllocateMemory(1);
614  HBAPortPtr->CommandListBase = (uint32_t)(uint64_t)CmdBase;
615  HBAPortPtr->CommandListBaseUpper = (uint32_t)((uint64_t)CmdBase >> 32);
616  MemorySet(reinterpret_cast<void *>(HBAPortPtr->CommandListBase), 0, 1024);
617 
618  void *FISBase = AllocateMemory(1);
619  HBAPortPtr->FISBaseAddress = (uint32_t)(uint64_t)FISBase;
620  HBAPortPtr->FISBaseAddressUpper = (uint32_t)((uint64_t)FISBase >> 32);
621  MemorySet(FISBase, 0, 256);
622 
623  HBACommandHeader *CommandHeader = (HBACommandHeader *)((uint64_t)HBAPortPtr->CommandListBase + ((uint64_t)HBAPortPtr->CommandListBaseUpper << 32));
624  for (int i = 0; i < 32; i++)

642  bool ReadWrite(uint64_t Sector, uint32_t SectorCount, uint8_t *Buffer, bool Write)
643  {

645  Write == true)

648  return false;
649  }

652  uint32_t SectorH = (uint32_t)(Sector >> 32);

655

668  CommandTable->PRDTEntry[0].DataBaseAddressUpper = (uint32_t)((uint64_t)Buffer >> 32);

671 /* conversion from 'uint32_t' {aka 'unsigned int'} to 'unsigned int:22' may change value */
672 #pragma GCC diagnostic ignored "-Wconversion"
673  CommandTable->PRDTEntry[0].ByteCount = (SectorCount << 9) - 1; /* 512 bytes per sector */
674 #pragma GCC diagnostic pop

679

681  CommandFIS->CommandControl = 1;
682  if (Write)
683  CommandFIS->Command = ATA_CMD_WRITE_DMA_EX;
684  else
685  CommandFIS->Command = ATA_CMD_READ_DMA_EX;
686 
687  CommandFIS->LBA0 = (uint8_t)SectorL;
688  CommandFIS->LBA1 = (uint8_t)(SectorL >> 8);
689  CommandFIS->LBA2 = (uint8_t)(SectorL >> 16);
690  CommandFIS->LBA3 = (uint8_t)SectorH;
691  CommandFIS->LBA4 = (uint8_t)(SectorH >> 8);
692  CommandFIS->LBA5 = (uint8_t)(SectorH >> 16);
693 
694  CommandFIS->DeviceRegister = 1 << 6; // LBA mode
695  CommandFIS->CountLow = SectorCount & 0xFF;
696  CommandFIS->CountHigh = (SectorCount >> 8) & 0xFF;
697 
698  uint64_t Spin = 0;
699 
700  while ((HBAPortPtr->TaskFileData & (ATA_DEV_BUSY | ATA_DEV_DRQ)) && Spin < 1000000)
701  Spin++;
702 
703  if (Spin == 1000000)
704  {
705  KernelLog("Port not responding.");
706  return false;
707  }
708 
709  HBAPortPtr->CommandIssue = 1;
710 
711  Spin = 0;
712  int TryCount = 0;
713 
714  while (true)
715  {
716  if (Spin > 100000000)
717  {
718  KernelLog("Port %d not responding. (%d)",
719  this->PortNumber, TryCount);
720 
721  Spin = 0;
722  TryCount++;
723  if (TryCount > 10)
724  return false;
725  }
726 
727  if (HBAPortPtr->CommandIssue == 0)
728  break;
729  Spin++;
730 
731  if (HBAPortPtr->InterruptStatus & HBA_PxIS_TFES)
732  {
733  KernelLog("Error reading/writing (%d).", Write);
734  return false;
735  }
736  }
737

739  }
