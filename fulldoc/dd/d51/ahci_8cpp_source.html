185  void StartCMD()
186  {
187  while (HBAPortPtr->CommandStatus & HBA_PxCMD_CR)
188  Yield();
189  HBAPortPtr->CommandStatus |= HBA_PxCMD_FRE;
190  HBAPortPtr->CommandStatus |= HBA_PxCMD_ST;
191  }

193  void StopCMD()
194  {
195  HBAPortPtr->CommandStatus &= ~HBA_PxCMD_ST;
196  HBAPortPtr->CommandStatus &= ~HBA_PxCMD_FRE;
197  while (true)
198  {
199  if (HBAPortPtr->CommandStatus & HBA_PxCMD_FR)
200  continue;
201  if (HBAPortPtr->CommandStatus & HBA_PxCMD_CR)
202  continue;
203  break;
204  }
205  }

207  void Configure()
208  {
209  this->StopCMD();
210  void *CmdBase = AllocateMemory(1);
211  HBAPortPtr->CommandListBase = (uint32_t)(uint64_t)CmdBase;
212  HBAPortPtr->CommandListBaseUpper = (uint32_t)((uint64_t)CmdBase >> 32);
213  MemorySet(reinterpret_cast<void *>(HBAPortPtr->CommandListBase), 0, 1024);
214 
215  void *FISBase = AllocateMemory(1);

217  HBAPortPtr->FISBaseAddressUpper = (uint32_t)((uint64_t)FISBase >> 32);
218  MemorySet(FISBase, 0, 256);
219

237  bool ReadWrite(uint64_t Sector, uint32_t SectorCount, uint8_t *Buffer, bool Write)
238  {
239  if (this->AHCIPortType == PortType::SATAPI &&
240  Write == true)
241  {
242  Log("SATAPI port does not support write.");
243  return false;
244  }
245 
246  uint32_t SectorL = (uint32_t)Sector;
247  uint32_t SectorH = (uint32_t)(Sector >> 32);
248 
249  HBAPortPtr->InterruptStatus = 0xFFFFFFFF; /* Clear pending interrupt bits */
250 
251  HBACommandHeader *CommandHeader = reinterpret_cast<HBACommandHeader *>(HBAPortPtr->CommandListBase);
252  CommandHeader->CommandFISLength = sizeof(FIS_REG_H2D) / sizeof(uint32_t);
253  if (Write)
254  CommandHeader->Write = 1;
255  else
256  CommandHeader->Write = 0;
257  CommandHeader->PRDTLength = 1;
258 
259  HBACommandTable *CommandTable = reinterpret_cast<HBACommandTable *>(CommandHeader->CommandTableBaseAddress);
260  MemorySet(CommandTable, 0, sizeof(HBACommandTable) + (CommandHeader->PRDTLength - 1) * sizeof(HBAPRDTEntry));
261 
262  CommandTable->PRDTEntry[0].DataBaseAddress = (uint32_t)(uint64_t)Buffer;
263  CommandTable->PRDTEntry[0].DataBaseAddressUpper = (uint32_t)((uint64_t)Buffer >> 32);
264 
265 #pragma GCC diagnostic push
266 /* conversion from 'uint32_t' {aka 'unsigned int'} to 'unsigned int:22' may change value */
267 #pragma GCC diagnostic ignored "-Wconversion"
268  CommandTable->PRDTEntry[0].ByteCount = (SectorCount << 9) - 1; /* 512 bytes per sector */
269 #pragma GCC diagnostic pop
270 
271  CommandTable->PRDTEntry[0].InterruptOnCompletion = 1;
272 
273  FIS_REG_H2D *CommandFIS = (FIS_REG_H2D *)(&CommandTable->CommandFIS);
274 
275  CommandFIS->FISType = FIS_TYPE_REG_H2D;
276  CommandFIS->CommandControl = 1;
277  if (Write)
278  CommandFIS->Command = ATA_CMD_WRITE_DMA_EX;
279  else
280  CommandFIS->Command = ATA_CMD_READ_DMA_EX;
281 
282  CommandFIS->LBA0 = (uint8_t)SectorL;
283  CommandFIS->LBA1 = (uint8_t)(SectorL >> 8);
284  CommandFIS->LBA2 = (uint8_t)(SectorL >> 16);
285  CommandFIS->LBA3 = (uint8_t)SectorH;
286  CommandFIS->LBA4 = (uint8_t)(SectorH >> 8);
287  CommandFIS->LBA5 = (uint8_t)(SectorH >> 16);
288 
289  CommandFIS->DeviceRegister = 1 << 6; // LBA mode
290  CommandFIS->CountLow = SectorCount & 0xFF;
291  CommandFIS->CountHigh = (SectorCount >> 8) & 0xFF;
292 
293  uint64_t Spin = 0;
294 
295  while ((HBAPortPtr->TaskFileData & (ATA_DEV_BUSY | ATA_DEV_DRQ)) && Spin < 1000000)
296  Spin++;
297 
298  if (Spin == 1000000)
299  {
300  Log("Port not responding.");
301  return false;
302  }
303 
304  HBAPortPtr->CommandIssue = 1;
305 
306  Spin = 0;
307  int TryCount = 0;
308 
309  while (true)
310  {
311  if (Spin > 100000000)
312  {
313  Log("Port %d not responding. (%d)",
314  this->PortNumber, TryCount);
315 
316  Spin = 0;
317  TryCount++;
318  if (TryCount > 10)
319  return false;
320  }
321 
322  if (HBAPortPtr->CommandIssue == 0)
323  break;
324  Spin++;
325 
326  if (HBAPortPtr->InterruptStatus & HBA_PxIS_TFES)
327  {
328  Log("Error reading/writing (%d).", Write);
329  return false;
330  }
331  }
332 
333  return true;
334  }

346 PortType CheckPortType(HBAPort *Port)
347 {
348  uint32_t SataStatus = Port->SataStatus;
349  uint8_t InterfacePowerManagement = (SataStatus >> 8) & 0b111;
350  uint8_t DeviceDetection = SataStatus & 0b111;
351 
352  if (DeviceDetection != HBA_PORT_DEV_PRESENT)
353  return PortType::None;
354  if (InterfacePowerManagement != HBA_PORT_IPM_ACTIVE)
355  return PortType::None;
356 
357  switch (Port->Signature)
358  {
359  case SATA_SIG_ATAPI:
360  return PortType::SATAPI;
361  case SATA_SIG_ATA:
362  return PortType::SATA;
363  case SATA_SIG_PM:
364  return PortType::PM;
365  case SATA_SIG_SEMB:
366  return PortType::SEMB;
367  default:
368  return PortType::None;
369  }
370 }

422 EXTERNC int cxx_Initialize()
423 {
424  PCIArray *ctx = Devices;
425 
429  while (ctx != nullptr)
430  {
431  /* We don't use the interrupt handler now... maybe we will in the future */
432  // RegisterInterruptHandler(iLine(ctx->Device), (void *)OnInterruptReceived);
433 
434  InitializePCI(ctx->Device);
435  HBAMemory *HBA = (HBAMemory *)(uintptr_t)GetBAR(5, ctx->Device);
436 
437  uint32_t PortsImplemented = HBA->PortsImplemented;
438  Log("AHCI ports implemented: %x", PortsImplemented);
439  for (int i = 0; i < 32; i++)
440  {
441  if (PortCount > 64)
442  {
443  Log("There are more than 64 AHCI ports implemented");
444  break;
445  }
446 
447  if (PortsImplemented & (1 << i))
448  {
449  Log("Port %d implemented", i);
450  PortType portType = CheckPortType(&HBA->Ports[i]);
451  if (portType == PortType::SATA || portType == PortType::SATAPI)
452  {
453  KPrint("%s drive found at port %d", PortTypeName[portType], i);
454  Ports[PortCount] = new Port(portType, &HBA->Ports[i], PortCount);
455  dev_t ret = RegisterBlockDevice(ddt_SATA,
456  nullptr, nullptr,
457  drvRead, drvWrite,
458  nullptr);
459  if (ret != (dev_t)PortCount)
460  {
461  KPrint("Failed to register block device %d", ret);
462  return -EBADSLT;
463  }
464 
465  PortCount++;
466  Ports[PortCount] = nullptr;
467  }
468  else
469  {
470  if (portType != PortType::None)
471  {
472  KPrint("Unsupported drive type %s found at port %d",
473  PortTypeName[portType], i);
474  }
475  }
476  }
477  }
478 
479  ctx = (PCIArray *)ctx->Next;
480  }
481 
482  Log("Initializing AHCI ports");
483  for (int i = 0; i < PortCount; i++)
484  Ports[i]->Configure();
485 
486  return PortCount > 0 ? 0 : -ENODEV;
487 }
