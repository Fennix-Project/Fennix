192 static char *GetRegistryString(registry_value_t *reg_val)
193 {
194  HKEY key;
195  DWORD len;
196  DWORD valtype;
197  char *result;
198 
199  // Open the key (directory where the value is stored)
200 
201  if (RegOpenKeyEx(reg_val->root, reg_val->path,
202  0, KEY_READ, &key) != ERROR_SUCCESS)
203  {
204  return NULL;
205  }
206 
207  result = NULL;
208 
209  // Find the type and length of the string, and only accept strings.
210 
211  if (RegQueryValueEx(key, reg_val->value,
212  NULL, &valtype, NULL, &len) == ERROR_SUCCESS
213  && valtype == REG_SZ)
214  {
215  // Allocate a buffer for the value and read the value
216 
217  result = malloc(len);
218 
219  if (RegQueryValueEx(key, reg_val->value, NULL, &valtype,
220  (unsigned char *) result, &len) != ERROR_SUCCESS)
221  {
222  free(result);
223  result = NULL;
224  }
225  }
226 
227  // Close the key
228 
229  RegCloseKey(key);
230 
231  return result;
232 }

236 static void CheckUninstallStrings(void)
237 {
238  unsigned int i;
239 
240  for (i=0; i<arrlen(uninstall_values); ++i)
241  {
242  char *val;
243  char *path;
244  char *unstr;
245 
246  val = GetRegistryString(&uninstall_values[i]);
247 
248  if (val == NULL)
249  {
250  continue;
251  }
252 
253  unstr = strstr(val, UNINSTALLER_STRING);
254 
255  if (unstr == NULL)

257  free(val);
258  }
259  else
260  {
261  path = unstr + strlen(UNINSTALLER_STRING);
262 
263  AddIWADDir(path);
264  }
265  }

388 // Returns true if the specified path is a path to a file

391 static boolean DirIsFile(char *path, char *filename)
392 {
393  size_t path_len;
394  size_t filename_len;
395 
396  path_len = strlen(path);
397  filename_len = strlen(filename);
398 
399  return path_len >= filename_len + 1
400  && path[path_len - filename_len - 1] == DIR_SEPARATOR
401  && !strcasecmp(&path[path_len - filename_len], filename);
402 }

404 // Check if the specified directory contains the specified IWAD
405 // file, returning the full path to the IWAD if found, or NULL
406 // if not found.

408 static char *CheckDirectoryHasIWAD(char *dir, char *iwadname)
409 {
410  char *filename;
411 
412  // As a special case, the "directory" may refer directly to an
413  // IWAD file if the path comes from DOOMWADDIR or DOOMWADPATH.
414 
415  if (DirIsFile(dir, iwadname) && M_FileExists(dir))
416  {
417  return strdup(dir);
418  }
419 
420  // Construct the full path to the IWAD if it is located in
421  // this directory, and check if it exists.
422 
423  if (!strcmp(dir, "."))
424  {
425  filename = strdup(iwadname);
426  }
427  else
428  {
429  filename = M_StringJoin(dir, DIR_SEPARATOR_S, iwadname, NULL);
430  }
431 
432  printf("Trying IWAD file:%s\n", filename);
433 
434  if (M_FileExists(filename))
435  {
436  return filename;
437  }
438 
439  free(filename);
440 
441  return NULL;
442 }

444 // Search a directory to try to find an IWAD
445 // Returns the location of the IWAD if found, otherwise NULL.
446 
447 static char *SearchDirectoryForIWAD(char *dir, int mask, GameMission_t *mission)
448 {
449  char *filename;
450  size_t i;
451 
452  for (i=0; i<arrlen(iwads); ++i)
453  {
454  if (((1 << iwads[i].mission) & mask) == 0)
455  {
456  continue;
457  }
458 
459  filename = CheckDirectoryHasIWAD(dir, DEH_String(iwads[i].name));
460 
461  if (filename != NULL)
462  {
463  *mission = iwads[i].mission;
464 
465  return filename;
466  }
467  }
468 
469  return NULL;
470 }

472 // When given an IWAD with the '-iwad' parameter,
473 // attempt to identify it by its name.
474 
475 static GameMission_t IdentifyIWADByName(char *name, int mask)
476 {
477  size_t i;
478  GameMission_t mission;
479  char *p;
480 
481  p = strrchr(name, DIR_SEPARATOR);
482 
483  if (p != NULL)
484  {
485  name = p + 1;
486  }
487 
488  mission = none;
489 
490  for (i=0; i<arrlen(iwads); ++i)
491  {
492  // Check if the filename is this IWAD name.
493 
494  // Only use supported missions:
495 
496  if (((1 << iwads[i].mission) & mask) == 0)
497  continue;
498 
499  // Check if it ends in this IWAD name.
500 
501  if (!strcasecmp(name, iwads[i].name))
502  {
503  mission = iwads[i].mission;
504  break;
505  }
506  }
507 
508  return mission;
509 }

624 //
625 // Searches WAD search paths for an WAD with a specific filename.
626 //
627 
628 char *D_FindWADByName(char *name)
629 {
630  char *path;
631  int i;
632 
633  // Absolute path?
634 
635  if (M_FileExists(name))
636  {
637  return name;
638  }
639 
640  BuildIWADDirList();
641 
642  // Search through all IWAD paths for a file with the given name.
643 
644  for (i=0; i<num_iwad_dirs; ++i)
645  {
646  // As a special case, if this is in DOOMWADDIR or DOOMWADPATH,
647  // the "directory" may actually refer directly to an IWAD
648  // file.
649 
650  if (DirIsFile(iwad_dirs[i], name) && M_FileExists(iwad_dirs[i]))
651  {
652  return strdup(iwad_dirs[i]);
653  }
654 
655  // Construct a string for the full path
656 
657  path = M_StringJoin(iwad_dirs[i], DIR_SEPARATOR_S, name, NULL);
658 
659  if (M_FileExists(path))
660  {
661  return path;
662  }
663 
664  free(path);
665  }
666 
667  // File not found
668 
669  return NULL;
670 }

703 {

708

710

715  //
716

728  {
729  I_Error("IWAD file '%s' not found!", iwadfile);
730  }
731 
732  *mission = IdentifyIWADByName(result, mask);
733  }
734  else
735  {
736  // Search through the list and look for an IWAD
737 
738  printf("-iwad not specified, trying a few iwad names\n");
739 
740  result = NULL;
741 
742  BuildIWADDirList();
743 
744  for (i=0; result == NULL && i<num_iwad_dirs; ++i)
745  {
746  result = SearchDirectoryForIWAD(iwad_dirs[i], mask, mission);
747  }
748  }
749 
750  return result;
751 }
