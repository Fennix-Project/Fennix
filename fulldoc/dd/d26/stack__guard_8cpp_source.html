24  bool StackGuard::Expand(uintptr_t FaultAddress)
25  {
26  if (this->UserMode)
27  {
28  if (FaultAddress < (uintptr_t)this->StackBottom - 0x100 ||
29  FaultAddress > (uintptr_t)this->StackTop)
30  {
31  info("Fault address %#lx is not in range of stack %#lx - %#lx", FaultAddress,
32  (uintptr_t)this->StackBottom - 0x100, (uintptr_t)this->StackTop);
33  return false; /* It's not about the stack. */
34  }
35  else
36  {
37  void *AllocatedStack = this->vma->RequestPages(TO_PAGES(USER_STACK_SIZE) + 1);
38  debug("AllocatedStack: %#lx", AllocatedStack);
39  memset(AllocatedStack, 0, USER_STACK_SIZE);
40 
41  Virtual vmm = Virtual(this->vma->GetTable());
42  for (size_t i = 0; i < TO_PAGES(USER_STACK_SIZE); i++)
43  {
44  void *VirtualPage = (void *)((uintptr_t)this->StackBottom - (i * PAGE_SIZE));
45  void *PhysicalPage = (void *)((uintptr_t)AllocatedStack + (i * PAGE_SIZE));
46 
47  vmm.Map(VirtualPage, PhysicalPage, PTFlag::RW | PTFlag::US);
48  AllocatedPages ap = {
49  .PhysicalAddress = PhysicalPage,
50  .VirtualAddress = VirtualPage,
51  };
52  AllocatedPagesList.push_back(ap);
53  debug("Mapped %#lx to %#lx", PhysicalPage, VirtualPage);
54  }
55 
56  this->StackBottom = (void *)((uintptr_t)this->StackBottom - USER_STACK_SIZE);
57  this->Size += USER_STACK_SIZE;
58  debug("Stack expanded to %#lx", this->StackBottom);
59  this->Expanded = true;
60  return true;
61  }
62  }
63  else
64  {
65  fixme("Not implemented and probably not needed");
66  return false;
67  }
68  }

70  void StackGuard::Fork(StackGuard *Parent)
71  {
72  this->UserMode = Parent->GetUserMode();
73  this->StackBottom = Parent->GetStackBottom();
74  this->StackTop = Parent->GetStackTop();
75  this->StackPhysicalBottom = Parent->GetStackPhysicalBottom();
76  this->StackPhysicalTop = Parent->GetStackPhysicalTop();
77  this->Size = Parent->GetSize();
78  this->Expanded = Parent->IsExpanded();
79 
80  if (this->UserMode)
81  {
82  std::vector<AllocatedPages> ParentAllocatedPages = Parent->GetAllocatedPages();
83  Virtual vma = Virtual(this->vma->GetTable());
84  foreach (auto Page in ParentAllocatedPages)
85  {
86  void *NewPhysical = this->vma->RequestPages(1);
87  debug("Forking address %#lx to %#lx", Page.PhysicalAddress, NewPhysical);
88  memcpy(NewPhysical, Page.PhysicalAddress, PAGE_SIZE);
89  vma.Map(Page.VirtualAddress, NewPhysical, PTFlag::RW | PTFlag::US);
90 
91  AllocatedPages ap = {

94  };
95  AllocatedPagesList.push_back(ap);
96  debug("Mapped %#lx to %#lx", NewPhysical, Page.VirtualAddress);
97  }
98  }
99  else
100  {
101  fixme("Kernel mode stack fork not implemented");
102  }
