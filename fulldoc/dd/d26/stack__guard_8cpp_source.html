24  bool StackGuard::Expand(uintptr_t FaultAddress)
25  {
26  if (!this->UserMode)
27  assert(!"Kernel mode stack expansion not implemented");
28 
29  if (FaultAddress < USER_STACK_END ||
30  FaultAddress > USER_STACK_BASE)
31  {
32  info("Fault address %#lx is not in range of stack %#lx - %#lx",
33  FaultAddress, USER_STACK_END, USER_STACK_BASE);
34  return false; /* It's not about the stack. */
35  }
36 
37  uintptr_t roundFA = ROUND_DOWN(FaultAddress, PAGE_SIZE);
38  uintptr_t diff = (uintptr_t)this->StackBottom - roundFA;
39  size_t stackPages = TO_PAGES(diff);
40  stackPages = stackPages < 1 ? 1 : stackPages;
41 
42  debug("roundFA: %#lx, sb: %#lx, diff: %#lx, stackPages: %d",
43  roundFA, this->StackBottom, diff, stackPages);
44 
45  void *AllocatedStack = vma->RequestPages(stackPages);
46  debug("AllocatedStack: %#lx", AllocatedStack);
47 
48  for (size_t i = 0; i < stackPages; i++)
49  {
50  void *vAddress = (void *)((uintptr_t)this->StackBottom - (i * PAGE_SIZE));
51  void *pAddress = (void *)((uintptr_t)AllocatedStack + (i * PAGE_SIZE));
52 
53  vma->Map(vAddress, pAddress, PAGE_SIZE, PTFlag::RW | PTFlag::US);
54  AllocatedPages ap = {
55  .PhysicalAddress = pAddress,
56  .VirtualAddress = vAddress,
57  };
58  AllocatedPagesList.push_back(ap);
59  debug("Mapped %#lx to %#lx", pAddress, vAddress);
60  }
61 
62  this->StackBottom = (void *)((uintptr_t)this->StackBottom - (stackPages * PAGE_SIZE));
63  this->Size += stackPages * PAGE_SIZE;
64  debug("Stack expanded to %#lx", this->StackBottom);
65  this->Expanded = true;
66  return true;
67  }
