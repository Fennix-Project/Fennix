27  __no_sanitize("alignment") void *ACPI::FindTable(ACPI::ACPIHeader *ACPIHeader, char *Signature)
28  {
29  for (uint64_t t = 0; t < ((ACPIHeader->Length - sizeof(ACPI::ACPIHeader)) / (XSDTSupported ? 8 : 4)); t++)
30  {
31 #pragma GCC diagnostic push
32 #pragma GCC diagnostic ignored "-Wint-to-pointer-cast"
33 
34  // TODO: Should I be concerned about unaligned memory access?
35  ACPI::ACPIHeader *SDTHdr = nullptr;
36  if (XSDTSupported)
37  SDTHdr = (ACPI::ACPIHeader *)(*(uint64_t *)((uint64_t)ACPIHeader + sizeof(ACPI::ACPIHeader) + (t * 8)));
38  else
39  SDTHdr = (ACPI::ACPIHeader *)(*(uint32_t *)((uint64_t)ACPIHeader + sizeof(ACPI::ACPIHeader) + (t * 4)));
40 
41 #pragma GCC diagnostic pop
42 
43  size_t signLength = strlen(Signature);
44  for (size_t i = 0; i < signLength; i++)
45  {
46  if (SDTHdr->Signature[i] != Signature[i])
47  break;
48  if (i == 3)
49  {
50 #ifdef DEBUG
51  KPrint("ACPI: %s [%s:%s] found at address %#lx",
52  Signature,
53  SDTHdr->OEMID,
54  SDTHdr->OEMTableID,
55  (uintptr_t)SDTHdr);
56 #endif
57  trace("%s found at address %#lx", Signature, (uintptr_t)SDTHdr);
58 
59  Tables[Signature] = SDTHdr;
60  return SDTHdr;
61  }
62  }
63  }
64  // warn("%s not found!", Signature);
65  return nullptr;
66  }
