502 ini_t* ini_load( char const* data, void* memctx )
503  {
504  ini_t* ini;
505  char const* ptr;
506  int s;
507  char const* start;
508  char const* start2;
509  int l;
510 
511  ini = ini_create( memctx );
512 
513  ptr = data;
514  if( ptr )
515  {
516  s = 0;
517  while( *ptr )
518  {
519  /* trim leading whitespace */
520  while( *ptr && *ptr <=' ' )
521  ++ptr;
522 
523  /* done? */
524  if( !*ptr ) break;
525 
526  /* comment */
527  else if( *ptr == ';' )
528  {
529  while( *ptr && *ptr !='\n' )
530  ++ptr;
531  }
532  /* section */
533  else if( *ptr == '[' )
534  {
535  ++ptr;
536  start = ptr;
537  while( *ptr && *ptr !=']' && *ptr != '\n' )
538  ++ptr;
539 
540  if( *ptr == ']' )
541  {
542  s = ini_section_add( ini, start, (int)( ptr - start) );
543  ++ptr;
544  }
545  }
546  /* property */
547  else
548  {
549  start = ptr;
550  while( *ptr && *ptr !='=' && *ptr != '\n' )
551  ++ptr;
552 
553  if( *ptr == '=' )
554  {
555  l = (int)( ptr - start);
556  ++ptr;
557  while( *ptr && *ptr <= ' ' && *ptr != '\n' )
558  ptr++;
559  start2 = ptr;
560  while( *ptr && *ptr != '\n' )
561  ++ptr;
562  while( ptr >= start2 && *(--ptr) <= ' ' )
563  (void)ptr;
564  ptr++;
565  if( ptr == start2 )
566  ini_property_add( ini, s, start, l, "", 1 );
567  else
568  ini_property_add( ini, s, start, l, start2, (int)( ptr - start2 ) );
569  }
570  }
571  }
572  }
573 
574  return ini;
575  }

578 int ini_save( ini_t const* ini, char* data, int size )
579  {
580  int s;
581  int p;
582  int i;
583  int l;
584  char* n;
585  int pos;
586 
587  if( ini )
588  {
589  pos = 0;
590  for( s = 0; s < ini->section_count; ++s )
591  {
592  n = ini->sections[ s ].name_large ? ini->sections[ s ].name_large : ini->sections[ s ].name;
593  l = (int) INI_STRLEN( n );
594  if( l > 0 )
595  {
596  if( data && pos < size ) data[ pos ] = '[';
597  ++pos;
598  for( i = 0; i < l; ++i )
599  {
600  if( data && pos < size ) data[ pos ] = n[ i ];
601  ++pos;
602  }
603  if( data && pos < size ) data[ pos ] = ']';
604  ++pos;
605  if( data && pos < size ) data[ pos ] = '\n';
606  ++pos;
607  }
608 
609  for( p = 0; p < ini->property_count; ++p )
610  {
611  if( ini->properties[ p ].section == s )
612  {
613  n = ini->properties[ p ].name_large ? ini->properties[ p ].name_large : ini->properties[ p ].name;
614  l = (int) INI_STRLEN( n );
615  for( i = 0; i < l; ++i )
616  {
617  if( data && pos < size ) data[ pos ] = n[ i ];
618  ++pos;
619  }
620  if( data && pos < size ) data[ pos ] = '=';
621  ++pos;
622  n = ini->properties[ p ].value_large ? ini->properties[ p ].value_large : ini->properties[ p ].value;
623  l = (int) INI_STRLEN( n );
624  for( i = 0; i < l; ++i )
625  {
626  if( data && pos < size ) data[ pos ] = n[ i ];
627  ++pos;
628  }
629  if( data && pos < size ) data[ pos ] = '\n';
630  ++pos;
631  }
632  }
633 
634  if( pos > 0 )
635  {
636  if( data && pos < size ) data[ pos ] = '\n';
637  ++pos;
638  }
639  }
640 
641  if( data && pos < size ) data[ pos ] = '\0';
642  ++pos;
643 
644  return pos;
645  }

647  return 0;

782 int ini_section_add( ini_t* ini, char const* name, int length )
783  {
784  struct ini_internal_section_t* new_sections;
785 
786  if( ini && name )
787  {
788  if( length <= 0 ) length = (int) INI_STRLEN( name );
789  if( ini->section_count >= ini->section_capacity )
790  {
791  ini->section_capacity *= 2;
792  new_sections = (struct ini_internal_section_t*) INI_MALLOC( ini->memctx,
793  ini->section_capacity * sizeof( ini->sections[ 0 ] ) );
794  INI_MEMCPY( new_sections, ini->sections, ini->section_count * sizeof( ini->sections[ 0 ] ) );
795  INI_FREE( ini->memctx, ini->sections );
796  ini->sections = new_sections;
797  }
798 
799  ini->sections[ ini->section_count ].name_large = 0;
800  if( (size_t) length + 1 >= sizeof( ini->sections[ 0 ].name ) )
801  {
802  ini->sections[ ini->section_count ].name_large = (char*) INI_MALLOC( ini->memctx, (size_t) length + 1 );
803  INI_MEMCPY( ini->sections[ ini->section_count ].name_large, name, (size_t) length );
804  ini->sections[ ini->section_count ].name_large[ length ] = '\0';
805  }
806  else
807  {
808  INI_MEMCPY( ini->sections[ ini->section_count ].name, name, (size_t) length );
809  ini->sections[ ini->section_count ].name[ length ] = '\0';
810  }
811 
812  return ini->section_count++;
813  }
814  return INI_NOT_FOUND;

876  {
877  if( ini->sections[ section ].name_large ) INI_FREE( ini->memctx, ini->sections[ section ].name_large );
878  for( p = ini->property_count - 1; p >= 0; --p )
879  {
880  if( ini->properties[ p ].section == section )
881  {
882  if( ini->properties[ p ].value_large ) INI_FREE( ini->memctx, ini->properties[ p ].value_large );
883  if( ini->properties[ p ].name_large ) INI_FREE( ini->memctx, ini->properties[ p ].name_large );
884  ini->properties[ p ] = ini->properties[ --ini->property_count ];
885  }
886  }
887 
888  ini->sections[ section ] = ini->sections[ --ini->section_count ];
889 
890  for( p = 0; p < ini->property_count; ++p )
891  {
892  if( ini->properties[ p ].section == ini->section_count )
893  ini->properties[ p ].section = section;
