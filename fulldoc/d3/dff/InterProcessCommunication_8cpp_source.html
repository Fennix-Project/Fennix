213  int IPC::HandleSyscall(long Command, long Type, int ID, int Flags, void *Buffer, size_t Size)
214  {
215  switch (Command)
216  {
217  case IPC_CREATE:
218  {
219  char UniqueToken[16];
220  if (Buffer != nullptr)
221  strcpy(UniqueToken, (char *)Buffer);
222  else
223  snprintf(UniqueToken, 16, "IPC_%d", ID);
224  IPCHandle *Hnd = this->Create((IPCType)Type, UniqueToken);
225  this->Allocate(Hnd->ID, Size ? Size : PAGE_SIZE);
226  return Hnd->ID;
227  }
228  case IPC_READ:
229  return this->Read(ID, Buffer, Size);
230  case IPC_WRITE:
231  return TaskManager->GetProcessByID(Flags)->IPC->Write(ID, Buffer, Size);
232  case IPC_DELETE:
233  {
234  this->Deallocate(ID);
235  return this->Destroy(ID);
236  }
237  case IPC_WAIT:
238  return this->Wait(ID);
239  case IPC_LISTEN:
240  return this->Listen(ID, Flags);
241  default:
242  return IPCInvalidCommand;
243  }
244  return IPCError;
245  }
