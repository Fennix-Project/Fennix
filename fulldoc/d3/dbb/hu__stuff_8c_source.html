402 void HU_Ticker(void)
403 {
404 
405  int i, rc;
406  char c;
407 
408  // tick down message counter if message is up
409  if (message_counter && !--message_counter)
410  {
411  message_on = false;
412  message_nottobefuckedwith = false;
413  }
414 
415  if (showMessages || message_dontfuckwithme)
416  {
417 
418  // display message if necessary
419  if ((plr->message && !message_nottobefuckedwith)
420  || (plr->message && message_dontfuckwithme))
421  {
422  HUlib_addMessageToSText(&w_message, 0, plr->message);
423  plr->message = 0;
424  message_on = true;
425  message_counter = HU_MSGTIMEOUT;
426  message_nottobefuckedwith = message_dontfuckwithme;
427  message_dontfuckwithme = 0;
428  }
429 
430  } // else message_on = false;
431 
432  // check for incoming chat characters
433  if (netgame)
434  {
435  for (i=0 ; i<MAXPLAYERS; i++)
436  {
437  if (!playeringame[i])
438  continue;
439  if (i != consoleplayer
440  && (c = players[i].cmd.chatchar))
441  {
442  if (c <= HU_BROADCAST)
443  chat_dest[i] = c;
444  else
445  {
446  rc = HUlib_keyInIText(&w_inputbuffer[i], c);
447  if (rc && c == KEY_ENTER)
448  {
449  if (w_inputbuffer[i].l.len
450  && (chat_dest[i] == consoleplayer+1
451  || chat_dest[i] == HU_BROADCAST))
452  {
453  HUlib_addMessageToSText(&w_message,
454  DEH_String(player_names[i]),
455  w_inputbuffer[i].l.l);
456 
457  message_nottobefuckedwith = true;
458  message_on = true;
459  message_counter = HU_MSGTIMEOUT;
460  if ( gamemode == commercial )
461  S_StartSound(0, sfx_radio);
462  else
463  S_StartSound(0, sfx_tink);
464  }
465  HUlib_resetIText(&w_inputbuffer[i]);
466  }
467  }
468  players[i].cmd.chatchar = 0;
469  }
470  }
471  }

482 void HU_queueChatChar(char c)
483 {
484  if (((head + 1) & (QUEUESIZE-1)) == tail)
485  {
486  plr->message = DEH_String(HUSTR_MSGU);
487  }
488  else
489  {
490  chatchars[head] = c;
491  head = (head + 1) & (QUEUESIZE-1);
492  }
493 }
494 
495 char HU_dequeueChatChar(void)
496 {
497  char c;
498 
499  if (head != tail)
500  {
501  c = chatchars[tail];
502  tail = (tail + 1) & (QUEUESIZE-1);
503  }
504  else
505  {
506  c = 0;
507  }
508 
509  return c;
510 }

512 boolean HU_Responder(event_t *ev)
513 {
514 
515  static char lastmessage[HU_MAXLINELENGTH+1];
516  char* macromessage;
517  boolean eatkey = false;
518  static boolean altdown = false;
519  unsigned char c;
520  int i;
521  int numplayers;
522 
523  static int num_nobrainers = 0;
524 
525  numplayers = 0;
526  for (i=0 ; i<MAXPLAYERS ; i++)
527  numplayers += playeringame[i];
528 
529  if (ev->data1 == KEY_RSHIFT)
530  {
531  return false;
532  }
533  else if (ev->data1 == KEY_RALT || ev->data1 == KEY_LALT)
534  {
535  altdown = ev->type == ev_keydown;
536  return false;
537  }
538 
539  if (ev->type != ev_keydown)
540  return false;
541 
542  if (!chat_on)
543  {
544  if (ev->data1 == key_message_refresh)
545  {
546  message_on = true;
547  message_counter = HU_MSGTIMEOUT;
548  eatkey = true;
549  }
550  else if (netgame && ev->data2 == key_multi_msg)
551  {
552  eatkey = chat_on = true;
553  HUlib_resetIText(&w_chat);
554  HU_queueChatChar(HU_BROADCAST);
555  }
556  else if (netgame && numplayers > 2)
557  {
558  for (i=0; i<MAXPLAYERS ; i++)
559  {
560  if (ev->data2 == key_multi_msgplayer[i])
561  {
562  if (playeringame[i] && i!=consoleplayer)
563  {
564  eatkey = chat_on = true;
565  HUlib_resetIText(&w_chat);
566  HU_queueChatChar(i+1);
567  break;
568  }
569  else if (i == consoleplayer)
570  {
571  num_nobrainers++;
572  if (num_nobrainers < 3)
573  plr->message = DEH_String(HUSTR_TALKTOSELF1);
574  else if (num_nobrainers < 6)
575  plr->message = DEH_String(HUSTR_TALKTOSELF2);
576  else if (num_nobrainers < 9)
577  plr->message = DEH_String(HUSTR_TALKTOSELF3);
578  else if (num_nobrainers < 32)
579  plr->message = DEH_String(HUSTR_TALKTOSELF4);
580  else
581  plr->message = DEH_String(HUSTR_TALKTOSELF5);
582  }
583  }
584  }
585  }
586  }
587  else
588  {
589  // send a macro
590  if (altdown)
591  {
592  c = ev->data1 - '0';
593  if (c > 9)
594  return false;
595  // fprintf(stderr, "got here\n");
596  macromessage = chat_macros[c];
597 
598  // kill last message with a '\n'
599  HU_queueChatChar(KEY_ENTER); // DEBUG!!!
600 
601  // send the macro message
602  while (*macromessage)
603  HU_queueChatChar(*macromessage++);
604  HU_queueChatChar(KEY_ENTER);
605 
606  // leave chat mode and notify that it was sent
607  chat_on = false;
608  M_StringCopy(lastmessage, chat_macros[c], sizeof(lastmessage));
609  plr->message = lastmessage;
610  eatkey = true;
611  }
612  else
613  {
614  c = ev->data2;
615 
616  eatkey = HUlib_keyInIText(&w_chat, c);
617  if (eatkey)
618  {
619  // static unsigned char buf[20]; // DEBUG
620  HU_queueChatChar(c);
621 
622  // M_snprintf(buf, sizeof(buf), "KEY: %d => %d", ev->data1, c);
623  // plr->message = buf;
624  }
625  if (c == KEY_ENTER)
626  {
627  chat_on = false;
628  if (w_chat.l.len)
629  {
630  M_StringCopy(lastmessage, w_chat.l.l, sizeof(lastmessage));
631  plr->message = lastmessage;
632  }
633  }
634  else if (c == KEY_ESCAPE)
635  chat_on = false;
636  }
637  }
638 
639  return eatkey;
