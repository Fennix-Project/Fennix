183  __no_sanitize("undefined") void TCB::SetupUserStack_x86_64(const char **argv,
184  const char **envp,
185  const std::vector<AuxiliaryVector> &auxv)
186  {
187  size_t ArgvSize = 0;
188  if (argv)
189  while (argv[ArgvSize] != nullptr)
190  ArgvSize++;
191 
192  size_t EnvpSize = 0;
193  if (envp)
194  while (envp[EnvpSize] != nullptr)
195  EnvpSize++;
196 
197  debug("ArgvSize: %d", ArgvSize);
198  debug("EnvpSize: %d", EnvpSize);
199 
200  /* https://articles.manugarg.com/aboutelfauxiliaryvectors.html */
201  /* https://refspecs.linuxbase.org/elf/x86_64-abi-0.99.pdf#figure.3.9 */
202  /* rsp is the top of the stack */
203  char *Stack = (char *)this->Stack->GetStackPhysicalTop();
204  /* Temporary stack pointer for strings */
205  char *StackStrings = (char *)Stack;
206  char *StackStringsVirtual = (char *)this->Stack->GetStackTop();
207 
208  /* Store string pointers for later */
209  uintptr_t *ArgvStrings = nullptr;
210  uintptr_t *EnvpStrings = nullptr;
211  if (ArgvSize > 0)
212  ArgvStrings = new uintptr_t[ArgvSize];
213  if (EnvpSize > 0)
214  EnvpStrings = new uintptr_t[EnvpSize];
215 
216  for (size_t i = 0; i < ArgvSize; i++)
217  {
218  /* Subtract the length of the string and the null terminator */
219  StackStrings -= strlen(argv[i]) + 1;
220  StackStringsVirtual -= strlen(argv[i]) + 1;
221  /* Store the pointer to the string */
222  ArgvStrings[i] = (uintptr_t)StackStringsVirtual;
223  /* Copy the string to the stack */
224  strcpy(StackStrings, argv[i]);
225  debug("argv[%d]: %s", i, argv[i]);
226  }
227 
228  for (size_t i = 0; i < EnvpSize; i++)
229  {
230  /* Subtract the length of the string and the null terminator */
231  StackStrings -= strlen(envp[i]) + 1;
232  StackStringsVirtual -= strlen(envp[i]) + 1;
233  /* Store the pointer to the string */
234  EnvpStrings[i] = (uintptr_t)StackStringsVirtual;
235  /* Copy the string to the stack */
236  strcpy(StackStrings, envp[i]);
237  debug("envp[%d]: %s", i, envp[i]);
238  }
239 
240  /* Align the stack to 16 bytes */
241  StackStrings -= (uintptr_t)StackStrings & 0xF;
242  /* Set "Stack" to the new stack pointer */
243  Stack = (char *)StackStrings;
244  /* If argv and envp sizes are odd then we need to align the stack */
245  Stack -= (ArgvSize + EnvpSize) % 2;
246  debug("odd align: %#lx | %#lx -> %#lx",
247  (ArgvSize + EnvpSize) % 2,
248  StackStrings, Stack);
249 
250  /* We need 8 bit pointers for the stack from here */
251  uintptr_t *Stack64 = (uintptr_t *)Stack;
252  assert(Stack64 != nullptr);
253 
254  /* Store the null terminator */
255  StackPush(Stack64, uintptr_t, AT_NULL);
256 
257  /* auxv_array is initialized with auxv elements.
258  If the array is empty then we add a null terminator */
259  std::vector<AuxiliaryVector> auxv_array = auxv;
260  if (auxv_array.size() == 0)
261  auxv_array.push_back({.archaux = {.a_type = AT_NULL, .a_un = {.a_val = 0}}});
262 
263  /* Store auxillary vector */
264  foreach (AuxiliaryVector av in auxv_array)
265  {
266  /* Subtract the size of the auxillary vector */
267  Stack64 -= sizeof(Elf_auxv_t) / sizeof(uintptr_t);
268  /* Store the auxillary vector */
269  POKE(Elf_auxv_t, Stack64) = av.archaux;
270  /* TODO: Store strings to the stack */
271  }
272 
273  /* Store the null terminator */
274  StackPush(Stack64, uintptr_t, AT_NULL);
275 
276  /* Store EnvpStrings[] to the stack */
277  Stack64 -= EnvpSize; /* (1 Stack64 = 8 bits; Stack64 = 8 * EnvpSize) */
278  for (size_t i = 0; i < EnvpSize; i++)
279  {
280  *(Stack64 + i) = (uintptr_t)EnvpStrings[i];
281  debug("EnvpStrings[%d]: %#lx",
282  i, EnvpStrings[i]);
283  }
284 
285  /* Store the null terminator */
286  StackPush(Stack64, uintptr_t, AT_NULL);
287 
288  /* Store ArgvStrings[] to the stack */
289  Stack64 -= ArgvSize; /* (1 Stack64 = 8 bits; Stack64 = 8 * ArgvSize) */
290  for (size_t i = 0; i < ArgvSize; i++)
291  {
292  *(Stack64 + i) = (uintptr_t)ArgvStrings[i];
293  debug("ArgvStrings[%d]: %#lx",
294  i, ArgvStrings[i]);
295  }
296 
297  /* Store the argc */
298  StackPush(Stack64, uintptr_t, ArgvSize);
299 
300  /* Set "Stack" to the new stack pointer */
301  Stack = (char *)Stack64;
302 
303  /* We need the virtual address but because we are in the kernel we can't use the process page table.
304  So we modify the physical address and store how much we need to subtract to get the virtual address for RSP. */
305  uintptr_t SubtractStack = (uintptr_t)this->Stack->GetStackPhysicalTop() - (uintptr_t)Stack;
306  debug("SubtractStack: %#lx", SubtractStack);
307 
308  /* Set the stack pointer to the new stack */
309  uintptr_t StackPointerReg = ((uintptr_t)this->Stack->GetStackTop() - SubtractStack);
310  // assert(!(StackPointerReg & 0xF));
311 
312 #if defined(a64)
313  this->Registers.rsp = StackPointerReg;
314 #elif defined(a32)
315  this->Registers.esp = StackPointerReg;
316 #elif defined(aa64)
317  this->Registers.sp = StackPointerReg;
318 #endif
319 
320  if (ArgvSize > 0)
321  delete[] ArgvStrings;
322  if (EnvpSize > 0)
323  delete[] EnvpStrings;
324 
325 #ifdef DEBUG
326  DumpData("Stack Data",
327  (void *)((uintptr_t)this->Stack->GetStackPhysicalTop() - (uintptr_t)SubtractStack),
328  SubtractStack);
329 #endif
330 
331 #if defined(a64)
332  this->Registers.rdi = (uintptr_t)ArgvSize; // argc
333  this->Registers.rsi = (uintptr_t)(this->Registers.rsp + 8); // argv
334  this->Registers.rcx = (uintptr_t)EnvpSize; // envc
335  this->Registers.rdx = (uintptr_t)(this->Registers.rsp + 8 + (8 * ArgvSize) + 8); // envp
336 #elif defined(a32)
337  this->Registers.eax = (uintptr_t)ArgvSize; // argc
338  this->Registers.ebx = (uintptr_t)(this->Registers.esp + 4); // argv
339  this->Registers.ecx = (uintptr_t)EnvpSize; // envc
340  this->Registers.edx = (uintptr_t)(this->Registers.esp + 4 + (4 * ArgvSize) + 4); // envp
341 #elif defined(aa64)
342  this->Registers.x0 = (uintptr_t)ArgvSize; // argc
343  this->Registers.x1 = (uintptr_t)(this->Registers.sp + 8); // argv
344  this->Registers.x2 = (uintptr_t)EnvpSize; // envc
345  this->Registers.x3 = (uintptr_t)(this->Registers.sp + 8 + (8 * ArgvSize) + 8); // envp
346 #endif
347  }

363  void TCB::SetupThreadLocalStorage()
364  {
365  if (Parent->TLS.pBase == 0x0)
366  return;
367  this->TLS = Parent->TLS;
368 
369  debug("msz: %ld fsz: %ld",
370  this->TLS.Size, this->TLS.fSize);
371 
372  // size_t pOffset = this->TLS.vBase - std::tolower(this->TLS.vBase);
373  size_t tlsFullSize = sizeof(uintptr_t) + this->TLS.Size;
374  /* 2 guard pages */
375  size_t tlsPages = 1 + TO_PAGES(tlsFullSize) + 1;
376 
377  void *opTLS = this->vma->RequestPages(tlsPages);
378  void *pTLS = (void *)(PAGE_SIZE + uintptr_t(opTLS));
379  this->TLS.pBase = uintptr_t(pTLS);
380 
381  memcpy(pTLS, (void *)this->TLS.pBase,
382  this->TLS.Size);
383 
384  size_t restBytes = this->TLS.Size - this->TLS.fSize;
385  if (restBytes)
386  {
387  memset((void *)(uintptr_t(pTLS) + this->TLS.Size),
388  0, restBytes);
389  }
390 
391  Memory::Virtual vmm(this->Parent->PageTable);
392  /* Map guard pages */
393  vmm.Remap((void *)(uintptr_t(pTLS) - PAGE_SIZE), opTLS, Memory::P);
394  vmm.Remap((void *)(uintptr_t(pTLS) + this->TLS.Size), opTLS, Memory::P);
395  /* Map TLS */
396  vmm.Map(pTLS, pTLS, this->TLS.Size, Memory::RW | Memory::US);
397 
398  uintptr_t *pTLSPointer = (uintptr_t *)this->TLS.pBase + this->TLS.Size;
399  *pTLSPointer = this->TLS.pBase + this->TLS.Size;
400 
401  this->GSBase = r_cst(uintptr_t, pTLSPointer);
402  this->FSBase = r_cst(uintptr_t, pTLSPointer);
403  }

405  TCB::TCB(Task *ctx, PCB *Parent, IP EntryPoint,
406  const char **argv, const char **envp,
407  const std::vector<AuxiliaryVector> &auxv,
408  TaskArchitecture Architecture,
409  TaskCompatibility Compatibility,
410  bool ThreadNotReady)
411  {
412  debug("+ %#lx", this);
413 
414  assert(ctx != nullptr);
415  assert(Architecture >= _ArchitectureMin);
416  assert(Architecture <= _ArchitectureMax);
417  assert(Compatibility >= _CompatibilityMin);
418  assert(Compatibility <= _CompatibilityMax);
419 
420  if (Parent == nullptr)
421  {
422  this->Parent = ctx->GetCurrentProcess();
423  assert(this->Parent != nullptr);
424  }
425  else
426  this->Parent = Parent;
427 
428  this->ctx = ctx;
429  this->ID = ctx->NextTID++;
430 
431  if (this->Name)
432  delete[] this->Name;
433 
434  this->Name = new char[strlen(this->Parent->Name) + 1];
435  strcpy((char *)this->Name, this->Parent->Name);
436 
437  this->EntryPoint = EntryPoint;
438  this->ExitCode = KILL_CRASH;
439 
440  if (ThreadNotReady)
441  this->SetState(Waiting);
442  else
443  this->SetState(Ready);
444 
445  this->vma = new Memory::VirtualMemoryArea(this->Parent->PageTable);
446 
447 #if defined(a64)
448  this->Registers.rip = EntryPoint;
449 #elif defined(a32)
450  this->Registers.eip = EntryPoint;
451 #elif defined(aa64)
452  this->Registers.pc = EntryPoint;
453 #endif
454 
455  switch (this->Parent->Security.ExecutionMode)
456  {
457  case TaskExecutionMode::System:
458  fixme("System mode not supported.");
459  [[fallthrough]];
460  case TaskExecutionMode::Kernel:
461  {
462  this->Security.IsCritical = true;
463  this->Stack = new Memory::StackGuard(false, this->vma);
464 
465 #if defined(a64)
466  this->ShadowGSBase =
467  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_SHADOW_GS_BASE);
468  this->GSBase = CPU::x64::rdmsr(CPU::x64::MSRID::MSR_GS_BASE);
469  this->FSBase = CPU::x64::rdmsr(CPU::x64::MSRID::MSR_FS_BASE);
470  this->Registers.cs = GDT_KERNEL_CODE;
471  this->Registers.ss = GDT_KERNEL_DATA;
472  this->Registers.rflags.AlwaysOne = 1;
473  this->Registers.rflags.IF = 1;
474  this->Registers.rflags.ID = 1;
475  this->Registers.rsp = ((uintptr_t)this->Stack->GetStackTop());
476  POKE(uintptr_t, this->Registers.rsp) = (uintptr_t)ThreadDoExit;
477 #elif defined(a32)
478  this->Registers.cs = GDT_KERNEL_CODE;
479  this->Registers.r3_ss = GDT_KERNEL_DATA;
480  this->Registers.eflags.AlwaysOne = 1;
481  this->Registers.eflags.IF = 1;
482  this->Registers.eflags.ID = 1;
483  this->Registers.esp = ((uintptr_t)this->Stack->GetStackTop());
484  POKE(uintptr_t, this->Registers.esp) = (uintptr_t)ThreadDoExit;
485 #elif defined(aa64)
486  this->Registers.pc = EntryPoint;
487  this->Registers.sp = ((uintptr_t)this->Stack->GetStackTop());
488  POKE(uintptr_t, this->Registers.sp) = (uintptr_t)ThreadDoExit;
489 #endif
490  break;
491  }
492  case TaskExecutionMode::User:
493  {
494  this->Stack = new Memory::StackGuard(true, this->vma);
495 
496  gsTCB *gsT = (gsTCB *)this->vma->RequestPages(TO_PAGES(sizeof(gsTCB)));
497 
498  gsT->ScPages = TO_PAGES(STACK_SIZE);
499  gsT->SyscallStackBase = this->vma->RequestPages(gsT->ScPages);
500  gsT->SyscallStack = (uintptr_t)gsT->SyscallStackBase + STACK_SIZE - 0x10;
501  debug("New syscall stack created: %#lx (base: %#lx) with gs base at %#lx",
502  gsT->SyscallStack, gsT->SyscallStackBase, gsT);
503 
504  gsT->TempStack = 0x0;
505  gsT->t = this;
506 #if defined(a64)
507  this->ShadowGSBase = (uintptr_t)gsT;
508  this->GSBase = 0;
509  this->FSBase = 0;
510  this->Registers.cs = GDT_USER_CODE;
511  this->Registers.ss = GDT_USER_DATA;
512  this->Registers.rflags.AlwaysOne = 1;
513  this->Registers.rflags.IF = 1;
514  this->Registers.rflags.ID = 1;
515  /* We need to leave the libc's crt
516  to make a syscall when the Thread
517  is exited or we are going to get
518  an exception. */
519 
520  this->SetupUserStack_x86_64(argv, envp, auxv);
521 #elif defined(a32)
522  this->Registers.cs = GDT_USER_CODE;
523  this->Registers.r3_ss = GDT_USER_DATA;
524  this->Registers.eflags.AlwaysOne = 1;
525  this->Registers.eflags.IF = 1;
526  this->Registers.eflags.ID = 1;
527  /* We need to leave the libc's crt
528  to make a syscall when the Thread
529  is exited or we are going to get
530  an exception. */
531 
532  this->SetupUserStack_x86_32(argv, envp, auxv);
533 #elif defined(aa64)
534  this->SetupUserStack_aarch64(argv, envp, auxv);
535 #endif
536 #ifdef DEBUG_TASKING
537  DumpData(this->Name, this->Stack, STACK_SIZE);
538 #endif
539  break;
540  }
541  default:
542  assert(false);
543  }
544 
545  SetupThreadLocalStorage();
546 
547  this->Info.Architecture = Architecture;
548  this->Info.Compatibility = Compatibility;
549  this->Security.ExecutionMode =
550  this->Parent->Security.ExecutionMode;
551 
552  if (this->Parent->Threads.size() == 0)
553  {
554  this->Parent->Info.Architecture = this->Info.Architecture;
555  this->Parent->Info.Compatibility = this->Info.Compatibility;
556  }
557 
558  // TODO: Is really a good idea to use the FPU in kernel mode?
559  this->FPU.mxcsr = 0b0001111110000000;
560  this->FPU.mxcsrmask = 0b1111111110111111;
561  this->FPU.fcw = 0b0000001100111111;
562 
563 #ifdef DEBUG
564 #ifdef a64
565  debug("Thread EntryPoint: %#lx => RIP: %#lx",
566  this->EntryPoint, this->Registers.rip);
567  if (this->Parent->Security.ExecutionMode == TaskExecutionMode::User)
568  debug("Thread stack region is %#lx-%#lx (U) and rsp is %#lx",
569  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
570  this->Registers.rsp);
571  else
572  debug("Thread stack region is %#lx-%#lx (K) and rsp is %#lx",
573  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
574  this->Registers.rsp);
575 #elif defined(a32)
576  debug("Thread EntryPoint: %#lx => RIP: %#lx",
577  this->EntryPoint, this->Registers.eip);
578  if (Parent->Security.ExecutionMode == TaskExecutionMode::User)
579  debug("Thread stack region is %#lx-%#lx (U) and rsp is %#lx",
580  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
581  this->Registers.esp);
582  else
583  debug("Thread stack region is %#lx-%#lx (K) and rsp is %#lx",
584  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
585  this->Registers.esp);
586 #elif defined(aa64)
587 #endif
588  debug("Created %s thread \"%s\"(%d) in process \"%s\"(%d)",
589  this->Security.ExecutionMode == TaskExecutionMode::User ? "user" : "kernel",
590  this->Name, this->ID, this->Parent->Name,
591  this->Parent->ID);
592 #endif
593 
594  this->AllocatedMemory += sizeof(TCB);
595  this->AllocatedMemory += strlen(this->Parent->Name) + 1;
596  this->AllocatedMemory += sizeof(Memory::StackGuard);
597 
598  this->Info.SpawnTime = TimeManager->GetCounter();
599  this->Parent->Threads.push_back(this);
600 
601  if (this->Parent->Threads.size() == 1 &&
602  this->Parent->State == Waiting &&
603  ThreadNotReady == false)
604  {
605  this->Parent->SetState(Ready);
606  debug("Setting process \"%s\"(%d) to ready",
607  this->Parent->Name, this->Parent->ID);
608  }
609  }
610 
611  TCB::~TCB()
612  {
613  debug("- %#lx", this);
614 
615  /* Remove us from the process list so we
616  don't get scheduled anymore */
617  std::list<Tasking::TCB *> &Threads = this->Parent->Threads;
618  Threads.erase(std::find(Threads.begin(),
619  Threads.end(),
620  this));
621 
622  /* Free CPU Stack */
623  delete this->Stack;
624 
625  /* Free all allocated memory */
626  delete this->vma;
627 
628  /* Free Name */
629  delete[] this->Name;
630 
631  debug("Thread \"%s\"(%d) destroyed",
632  this->Name, this->ID);
633  }
