260 /*
261  * Shift the array A N elements to the left. Delete the first N elements
262  * and subtract N from the indices of the remaining elements. If FLAGS
263  * does not include AS_DISPOSE, this returns a null-terminated array of
264  * elements so the caller can dispose of the chain. If FLAGS includes
265  * AS_DISPOSE, this function disposes of the shifted-out elements and
266  * returns NULL.
267  */
268 ARRAY_ELEMENT **
269 array_shift(a, n, flags)
270 ARRAY *a;
271 int n, flags;
272 {
273  ARRAY_ELEMENT **r, *ae;
274  register arrayind_t ni, ri;
275  int i, j;
276 
277  if (a == 0 || array_empty(a) || n <= 0)
278  return ((ARRAY_ELEMENT **)NULL);
279 
280  r = (ARRAY_ELEMENT **)xmalloc ((n + 1) * sizeof (ARRAY_ELEMENT *));
281 
282  /* Easy case; shifting out all of the elements */
283  if (n >= a->num_elements) {
284  if (flags & AS_DISPOSE) {
285  array_flush (a);
286  return ((ARRAY_ELEMENT **)NULL);
287  }
288  for (ri = 0, i = a->first_index; i <= a->max_index; i++)
289  if (a->elements[i]) {
290  r[ri++] = a->elements[i];
291  a->elements[i] = 0;
292  }
293 
294  a->first_index = a->max_index = -1;
295  a->num_elements = 0;
296  r[ri] = (ARRAY_ELEMENT *)NULL;
297  return r;
298  }
299 
300  /* Shift out the first N elements, return them in R. Handle sparse
301  arrays by skipping over NULL array elements. */
302  for (i = a->first_index, ri = 0, j = 0; j < n; i++) {
303  if ((ae = a->elements[i]) == 0)
304  continue;
305  if (i > a->max_index)
306  break;
307  ni = i + n;
308  j++;
309  if (ae)
310  r[ri++] = a->elements[i];
311  a->elements[i] = a->elements[ni];
312  if (a->elements[i])
313  element_index(a->elements[i]) = i;
314  }
315  r[ri]= (ARRAY_ELEMENT *)NULL;
316 
317 #ifdef DEBUG
318 if (j < n)
319  itrace("array_shift: short count: j = %d n = %d", j, n);
320 #endif
321 
322  /* Now shift everything else, modifying the index in each element */
323  for (; i <= a->max_index; i++) {
324  ni = i + n;
325  a->elements[i] = (ni <= a->max_index) ? a->elements[ni] : (ARRAY_ELEMENT *)NULL;
326  if (a->elements[i])
327  element_index(a->elements[i]) = i;
328  }
329 
330  a->num_elements -= n; /* modify bookkeeping information */
331  if (a->num_elements == 0)
332  a->first_index = a->max_index == -1;
333  else {
334  a->max_index -= n;
335  for (i = 0; i <= a->max_index; i++)
336  if (a->elements[i])
337  break;
338  a->first_index = i;
339  }
340 
341  if (flags & AS_DISPOSE) {
342  for (i = 0; i < ri; i++)
343  array_dispose_element(r[i]);
344  free (r);
345  return ((ARRAY_ELEMENT **)NULL);
346  }
347 
348  return r;
349 }

352  * Shift array A right N indices. If S is non-null, it becomes the value of
353  * the new element 0. Returns the number of elements in the array after the
354  * shift.
355  */
356 int
357 array_rshift (a, n, s)
358 ARRAY *a;
359 int n;
360 char *s;
361 {
362  register ARRAY_ELEMENT *ae, *new;
363  arrayind_t ni, nsize;
364 
365  if (a == 0 || (array_empty(a) && s == 0))
366  return 0;
367  else if (n <= 0)
368  return (a->num_elements);
369 
370  if (n >= a->alloc_size)
371  array_expand(a, n);
372 
373  /* Shift right, adjusting the element indexes as we go */
374  for (ni = a->max_index; ni >= 0; ni--) {
375  a->elements[ni+n] = a->elements[ni];
376  if (a->elements[ni+n])
377  element_index(a->elements[ni+n]) = ni + n;
378  a->elements[ni] = (ARRAY_ELEMENT *)NULL;
379  }
380  a->max_index += n;
381 
382 #if 0
383  /* Null out all the old indexes we just copied from */
384  for (ni = a->first_index; ni >= 0 && ni < n; ni++)
385  a->elements[ni] = (ARRAY_ELEMENT *)NULL;
386 #endif
387  a->first_index += n;
388 
389  if (s) {
390  new = array_create_element(0, s);
391  a->elements[0] = new;
392  a->num_elements++;
393  a->first_index = 0;
394  if (array_num_elements(a) == 1) /* array was empty */
395  a->max_index = 0;
396  }
397

520  * Return a string whose elements are the members of array A beginning at
521  * index START and spanning NELEM members. Null elements are counted.
522  * Since arrays are sparse, unset array elements are not counted.
523  */
524 char *
525 array_subrange (a, start, nelem, starsub, quoted, pflags)
526 ARRAY *a;
527 arrayind_t start, nelem;
528 int starsub, quoted, pflags;
529 {
530  ARRAY *a2;
531  arrayind_t s, e;
532  int i;
533  char *t;
534  WORD_LIST *wl;
535 
536  if (array_empty (a) || start > array_max_index(a))
537  return ((char *)NULL);
538 
539  /*
540  * Find element with index START. If START corresponds to an unset
541  * element (arrays can be sparse), use the first element whose index
542  * is >= START. If START is < 0, we count START indices back from
543  * the end of A (not elements, even with sparse arrays -- START is an
544  * index).
545  */
546  for (s = start; a->elements[s] == 0 && s <= a->max_index; s++)
547  ;
548 
549  if (s > a->max_index)
550  return ((char *)NULL);
551 
552  /* Starting at S, take NELEM elements, inclusive. */
553  for (i = 0, e = s; e <= a->max_index && i < nelem; e++) {
554  if (a->elements[e]) /* arrays are sparse */
555  i++;
556  }
557 
558  a2 = array_slice(a, s, e);
559 
560  wl = array_to_word_list(a2);
561  array_dispose(a2);
562  if (wl == 0)
563  return (char *)NULL;
564  t = string_list_pos_params(starsub ? '*' : '@', wl, quoted, pflags); /* XXX */
565  dispose_words(wl);
566 
567  return t;
