90 //
91 // R_ClipSolidWallSegment
92 // Does handle solid walls,
93 // e.g. single sided LineDefs (middle texture)
94 // that entirely block the view.
95 //
96 void
97 R_ClipSolidWallSegment
98 ( int first,
99  int last )
100 {
101  cliprange_t* next;
102  cliprange_t* start;
103 
104  // Find the first range that touches the range
105  // (adjacent pixels are touching).
106  start = solidsegs;
107  while (start->last < first-1)
108  start++;
109 
110  if (first < start->first)
111  {
112  if (last < start->first-1)
113  {
114  // Post is entirely visible (above start),
115  // so insert a new clippost.
116  R_StoreWallRange (first, last);
117  next = newend;
118  newend++;
119 
120  while (next != start)
121  {
122  *next = *(next-1);
123  next--;
124  }
125  next->first = first;
126  next->last = last;
127  return;
128  }
129 
130  // There is a fragment above *start.
131  R_StoreWallRange (first, start->first - 1);
132  // Now adjust the clip size.
133  start->first = first;
134  }
135 
136  // Bottom contained in start?
137  if (last <= start->last)
138  return;
139 
140  next = start;
141  while (last >= (next+1)->first-1)
142  {
143  // There is a fragment between two posts.
144  R_StoreWallRange (next->last + 1, (next+1)->first - 1);
145  next++;
146 
147  if (last <= next->last)
148  {
149  // Bottom is contained in next.
150  // Adjust the clip size.
151  start->last = next->last;
152  goto crunch;
153  }
154  }
155 
156  // There is a fragment after *next.
157  R_StoreWallRange (next->last + 1, last);
158  // Adjust the clip size.
159  start->last = last;
160 
161  // Remove start+1 to next from the clip list,
162  // because start now covers their area.
163  crunch:
164  if (next == start)
165  {
166  // Post just extended past the bottom of one post.
167  return;
168  }
169 
170 
171  while (next++ != newend)
172  {
173  // Remove a post.
174  *++start = *next;
175  }
176 
177  newend = start+1;
178 }

183 // R_ClipPassWallSegment
184 // Clips the given range of columns,
185 // but does not includes it in the clip list.
186 // Does handle windows,
187 // e.g. LineDefs with upper and lower texture.
188 //
189 void
190 R_ClipPassWallSegment
191 ( int first,
192  int last )
193 {
194  cliprange_t* start;

196  // Find the first range that touches the range
197  // (adjacent pixels are touching).
198  start = solidsegs;
199  while (start->last < first-1)
200  start++;
201 
202  if (first < start->first)
203  {
204  if (last < start->first-1)
205  {
206  // Post is entirely visible (above start).
207  R_StoreWallRange (first, last);
208  return;
209  }
210 
211  // There is a fragment above *start.
212  R_StoreWallRange (first, start->first - 1);
213  }
214 
215  // Bottom contained in start?
216  if (last <= start->last)
217  return;
218 
219  while (last >= (start+1)->first-1)
220  {
221  // There is a fragment between two posts.
222  R_StoreWallRange (start->last + 1, (start+1)->first - 1);
223  start++;
224 
225  if (last <= start->last)
226  return;
227  }
228 
229  // There is a fragment after *next.
230  R_StoreWallRange (start->last + 1, last);

247 //
248 // R_AddLine
249 // Clips the given segment
250 // and adds any visible pieces to the line list.
251 //
252 void R_AddLine (seg_t* line)
253 {
254  int x1;
255  int x2;
256  angle_t angle1;
257  angle_t angle2;
258  angle_t span;
259  angle_t tspan;
260 
261  curline = line;
262 
263  // OPTIMIZE: quickly reject orthogonal back sides.
264  angle1 = R_PointToAngle (line->v1->x, line->v1->y);
265  angle2 = R_PointToAngle (line->v2->x, line->v2->y);
266 
267  // Clip to view edges.
268  // OPTIMIZE: make constant out of 2*clipangle (FIELDOFVIEW).
269  span = angle1 - angle2;
270 
271  // Back side? I.e. backface culling?
272  if (span >= ANG180)
273  return;
274 
275  // Global angle needed by segcalc.
276  rw_angle1 = angle1;
277  angle1 -= viewangle;
278  angle2 -= viewangle;
279 
280  tspan = angle1 + clipangle;
281  if (tspan > 2*clipangle)
282  {
283  tspan -= 2*clipangle;
284 
285  // Totally off the left edge?
286  if (tspan >= span)
287  return;
288 
289  angle1 = clipangle;
290  }
291  tspan = clipangle - angle2;
292  if (tspan > 2*clipangle)
293  {
294  tspan -= 2*clipangle;
295 
296  // Totally off the left edge?
297  if (tspan >= span)
298  return;
299  angle2 = -clipangle;
300  }
301 
302  // The seg is in the view range,
303  // but not necessarily visible.
304  angle1 = (angle1+ANG90)>>ANGLETOFINESHIFT;
305  angle2 = (angle2+ANG90)>>ANGLETOFINESHIFT;
306  x1 = viewangletox[angle1];
307  x2 = viewangletox[angle2];
308 
309  // Does not cross a pixel?
310  if (x1 == x2)
311  return;
312 
313  backsector = line->backsector;
314 
315  // Single sided line?
316  if (!backsector)
317  goto clipsolid;
318 
319  // Closed door.
320  if (backsector->ceilingheight <= frontsector->floorheight
321  || backsector->floorheight >= frontsector->ceilingheight)
322  goto clipsolid;
323 
324  // Window.
325  if (backsector->ceilingheight != frontsector->ceilingheight
326  || backsector->floorheight != frontsector->floorheight)
327  goto clippass;
328 
329  // Reject empty lines used for triggers
330  // and special events.
331  // Identical floor and ceiling on both sides,
332  // identical light levels on both sides,
333  // and no middle texture.
334  if (backsector->ceilingpic == frontsector->ceilingpic
335  && backsector->floorpic == frontsector->floorpic
336  && backsector->lightlevel == frontsector->lightlevel
337  && curline->sidedef->midtexture == 0)
338  {
339  return;
340  }
341 
342 
343  clippass:
344  R_ClipPassWallSegment (x1, x2-1);
345  return;
346 
347  clipsolid:
348  R_ClipSolidWallSegment (x1, x2-1);
349 }

374 boolean R_CheckBBox (fixed_t* bspcoord)
375 {
376  int boxx;
377  int boxy;
378  int boxpos;
379 
380  fixed_t x1;
381  fixed_t y1;
382  fixed_t x2;
383  fixed_t y2;
384 
385  angle_t angle1;
386  angle_t angle2;
387  angle_t span;
388  angle_t tspan;
389 
390  cliprange_t* start;
391 
392  int sx1;
393  int sx2;
394 
395  // Find the corners of the box
396  // that define the edges from current viewpoint.
397  if (viewx <= bspcoord[BOXLEFT])
398  boxx = 0;
399  else if (viewx < bspcoord[BOXRIGHT])
400  boxx = 1;
401  else
402  boxx = 2;
403 
404  if (viewy >= bspcoord[BOXTOP])
405  boxy = 0;
406  else if (viewy > bspcoord[BOXBOTTOM])
407  boxy = 1;
408  else
409  boxy = 2;
410 
411  boxpos = (boxy<<2)+boxx;
412  if (boxpos == 5)
413  return true;
414 
415  x1 = bspcoord[checkcoord[boxpos][0]];
416  y1 = bspcoord[checkcoord[boxpos][1]];
417  x2 = bspcoord[checkcoord[boxpos][2]];
418  y2 = bspcoord[checkcoord[boxpos][3]];
419 
420  // check clip list for an open space
421  angle1 = R_PointToAngle (x1, y1) - viewangle;
422  angle2 = R_PointToAngle (x2, y2) - viewangle;
423 
424  span = angle1 - angle2;
425 
426  // Sitting on a line?
427  if (span >= ANG180)
428  return true;
429 
430  tspan = angle1 + clipangle;
431 
432  if (tspan > 2*clipangle)
433  {
434  tspan -= 2*clipangle;
435 
436  // Totally off the left edge?
437  if (tspan >= span)
438  return false;
439 
440  angle1 = clipangle;
441  }
442  tspan = clipangle - angle2;
443  if (tspan > 2*clipangle)
444  {
445  tspan -= 2*clipangle;
446 
447  // Totally off the left edge?
448  if (tspan >= span)
449  return false;
450 
451  angle2 = -clipangle;
452  }
453 
454 
455  // Find the first clippost
456  // that touches the source post
457  // (adjacent pixels are touching).
458  angle1 = (angle1+ANG90)>>ANGLETOFINESHIFT;
459  angle2 = (angle2+ANG90)>>ANGLETOFINESHIFT;
460  sx1 = viewangletox[angle1];
461  sx2 = viewangletox[angle2];
462 
463  // Does not cross a pixel.
464  if (sx1 == sx2)
465  return false;
466  sx2--;
467 
468  start = solidsegs;
469  while (start->last < sx2)
470  start++;
471 
472  if (sx1 >= start->first
473  && sx2 <= start->last)
474  {
475  // The clippost contains the new span.
476  return false;
477  }
478 
479  return true;
480 }

541 // RenderBSPNode
542 // Renders all subsectors below a given node,
543 // traversing subtree recursively.
544 // Just call with BSP root.
545 void R_RenderBSPNode (int bspnum)
546 {
547  node_t* bsp;
548  int side;
549 
550  // Found a subsector?
551  if (bspnum & NF_SUBSECTOR)
552  {

555  else

557  return;
558  }

560  bsp = &nodes[bspnum];
561 
562  // Decide which side the view point is on.
563  side = R_PointOnSide (viewx, viewy, bsp);
564 
565  // Recursively divide front space.
566  R_RenderBSPNode (bsp->children[side]);
567 
568  // Possibly divide back space.
569  if (R_CheckBBox (bsp->bbox[side^1]))
570  R_RenderBSPNode (bsp->children[side^1]);
571 }
