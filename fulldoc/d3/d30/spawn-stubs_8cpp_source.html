54 static int child(void *args_vp) {
55  int i, ret;
56  struct sigaction sa = {};
57  struct args *args = (struct args *)args_vp;
58  int p = args->p[1];
59  const posix_spawn_file_actions_t *fa = args->fa;
60  const posix_spawnattr_t *__restrict attr = args->attr;
61  sigset_t hset;
62  bool use_execvpe = false;
63 
64  if(attr->__fn)
65  use_execvpe = true;
66 
67  close(args->p[0]);
68 
69  /* All signal dispositions must be either SIG_DFL or SIG_IGN
70  * before signals are unblocked. Otherwise a signal handler
71  * from the parent might get run in the child while sharing
72  * memory, with unpredictable and dangerous results. To
73  * reduce overhead, sigaction has tracked for us which signals
74  * potentially have a signal handler. */
75  __get_handler_set(&hset);
76  for(i = 1; i < NSIG; i++) {
77  if((attr->__flags & POSIX_SPAWN_SETSIGDEF) && sigismember(&attr->__def, i)) {
78  sa.sa_handler = SIG_DFL;
79  } else if(sigismember(&hset, i)) {
80  if (i - 32 < 3) {
81  sa.sa_handler = SIG_IGN;
82  } else {;
83  sigaction(i, 0, &sa);
84  if(sa.sa_handler == SIG_IGN)
85  continue;
86  sa.sa_handler = SIG_DFL;
87  }
88  } else {
89  continue;
90  }
91  sigaction(i, &sa, 0);
92  }
93 
94  if(attr->__flags & POSIX_SPAWN_SETSID) {
95  if((ret = setsid()) < 0)
96  goto fail;
97  }
98 
99  if(attr->__flags & POSIX_SPAWN_SETPGROUP) {
100  mlibc::infoLogger() << "mlibc: posix_spawn: ignoring SETPGROUP" << frg::endlog;
101  //if((ret = setpgid(0, attr->__pgrp)))
102  // goto fail;
103  }
104 
105  if(attr->__flags & POSIX_SPAWN_RESETIDS) {
106  if((ret = setgid(getgid())) || (ret = setuid(getuid())) )
107  goto fail;
108  }
109 
110  if(fa && fa->__actions) {
111  struct fdop *op;
112  int fd;
113  for(op = (struct fdop *)fa->__actions; op->next; op = op->next);
114  for(; op; op = op->prev) {
115  /* It's possible that a file operation would clobber
116  * the pipe fd used for synchronizing with the
117  * parent. To avoid that, we dup the pipe onto
118  * an unoccupied fd. */
119  if(op->fd == p) {
120  ret = dup(p);
121  if(ret < 0)
122  goto fail;
123  close(p);
124  p = ret;
125  }
126  switch(op->cmd) {
127  case FDOP_CLOSE:
128  close(op->fd);
129  break;
130  case FDOP_DUP2:
131  fd = op->srcfd;
132  if(fd == p) {
133  ret = -EBADF;
134  goto fail;
135  }
136  if(fd != op->fd) {
137  if((ret = dup2(fd, op->fd)) < 0)
138  goto fail;
139  } else {
140  ret = fcntl(fd, F_GETFD);
141  ret = fcntl(fd, F_SETFD, ret & ~FD_CLOEXEC);
142  if(ret < 0)
143  goto fail;
144  }
145  break;
146  case FDOP_OPEN:
147  fd = open(op->path, op->oflag, op->mode);
148  if((ret = fd) < 0)
149  goto fail;
150  if(fd != op->fd) {
151  if((ret = dup2(fd, op->fd)) < 0)
152  goto fail;
153  close(fd);
154  }
155  break;
156  case FDOP_CHDIR:
157  ret = chdir(op->path);
158  if(ret < 0)
159  goto fail;
160  break;
161  case FDOP_FCHDIR:
162  ret = fchdir(op->fd);
163  if(ret < 0)
164  goto fail;
165  break;
166  }
167  }
168  }
169 
170  /* Close-on-exec flag may have been lost if we moved the pipe
171  * to a different fd. */
172  fcntl(p, F_SETFD, FD_CLOEXEC);
173 
174  pthread_sigmask(SIG_SETMASK, (attr->__flags & POSIX_SPAWN_SETSIGMASK)
175  ? &attr->__mask : &args->oldmask, 0);
176 
177  if(use_execvpe)
178  execvpe(args->path, args->argv, args->envp);
179  else
180  execve(args->path, args->argv, args->envp);
181  ret = -errno;
182 
183 fail:
184  /* Since sizeof errno < PIPE_BUF, the write is atomic. */
185  ret = -ret;
186  if(ret)
187  while(write(p, &ret, sizeof ret) < 0);
188  _exit(127);
189 }
