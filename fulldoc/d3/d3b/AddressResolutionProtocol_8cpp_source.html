29  DiscoveredAddress *ARP::ManageDiscoveredAddresses(DAType Type, InternetProtocol IP, MediaAccessControl MAC)
30  {
31  switch (Type)
32  {
33  case DA_ADD:
34  {
35  DiscoveredAddress *tmp = new DiscoveredAddress;
36  tmp->IP = IP;
37  tmp->MAC = MAC;
38  DiscoveredAddresses.push_back(tmp);
39  netdbg("Added %s to discovered addresses", IP.v4.ToStringLittleEndian());
40  return tmp;
41  }
42  case DA_DEL:
43  {
44  for (size_t i = 0; i < DiscoveredAddresses.size(); i++)
45  {
46  if (DiscoveredAddresses[i]->IP.v4 == IP.v4)
47  {
48  DiscoveredAddress *tmp = DiscoveredAddresses[i];
49  netdbg("Removed %s from discovered addresses", IP.v4.ToStringLittleEndian());
50  delete tmp, tmp = nullptr;
51  DiscoveredAddresses.remove(i);
52  break;
53  }
54  }
55  return nullptr;
56  }
57  case DA_SEARCH:
58  {
59  for (size_t i = 0; i < DiscoveredAddresses.size(); i++)
60  {
61  if (DiscoveredAddresses[i]->IP.v4 == IP.v4)
62  {
63  netdbg("Found %s in discovered addresses", IP.v4.ToStringLittleEndian());
64  return DiscoveredAddresses[i];
65  }
66  }
67  return nullptr;
68  }
69  case DA_UPDATE:
70  {
71  for (size_t i = 0; i < DiscoveredAddresses.size(); i++)
72  {
73  if (DiscoveredAddresses[i]->IP.v4 == IP.v4)
74  {
75  DiscoveredAddresses[i]->MAC = MAC;
76  netdbg("Updated %s in discovered addresses", IP.v4.ToStringLittleEndian());
77  return DiscoveredAddresses[i];
78  }
79  }
80  return nullptr;
81  }
82  default:
83  {
84  return nullptr;
85  }
86  }
87  return nullptr;
88  }

123  uint48_t ARP::Resolve(InternetProtocol IP)
124  {
125  netdbg("Resolving %s", IP.v4.ToStringLittleEndian());
126  if (IP.v4 == 0xFFFFFFFF)
127  return 0xFFFFFFFFFFFF;
128 
129  /* If we can't find the MAC, return the default value which is 0xFFFFFFFFFFFF. */
130  uint48_t ret = this->Search(IP)->MAC.ToHex();
131  netdbg("Resolved %s to %lx", IP.v4.ToStringLittleEndian(), ret);
132 
133  if (ret == 0xFFFFFFFFFFFF)
134  {
135  netdbg("Sending request");
136  /* If we can't find the MAC, send a request.
137  Because we are going to send this over the network, we need to byteswap first.
138  This is actually very confusing for me. */
139  ARPHeader *Header = new ARPHeader;
140  Header->HardwareType = b16(ARPHardwareType::HTYPE_ETHERNET);
141  Header->ProtocolType = b16(NetworkEthernet::FrameType::TYPE_IPV4);
142  Header->HardwareSize = b8(6);
143  Header->ProtocolSize = b8(4);
144  Header->Operation = b16(ARPOperation::REQUEST);
145  Header->SenderMAC = b48(Ethernet->GetInterface()->MAC.ToHex());
146  Header->SenderIP = b32(Ethernet->GetInterface()->IP.v4.ToHex());
147  Header->TargetMAC = b48(0xFFFFFFFFFFFF);
148  Header->TargetIP = b32(IP.v4.ToHex());
149  netdbg("SIP: %s; TIP: %s", InternetProtocol().v4.FromHex(Header->SenderIP).ToStringLittleEndian(),
150  InternetProtocol().v4.FromHex(Header->TargetIP).ToStringLittleEndian());
151  /* Send the request to the broadcast MAC address. */
152  Ethernet->Send({.Address = {0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF}}, NetworkEthernet::FrameType::TYPE_ARP, (uint8_t *)Header, sizeof(ARPHeader));
153  delete Header, Header = nullptr;
154  }
155 
156  int RequestTimeout = 20;
157  debug("Waiting for response");
158  while (ret == 0xFFFFFFFFFFFF)
159  {
160  ret = this->Search(IP)->MAC.ToHex();
161  if (--RequestTimeout == 0)
162  {
163  warn("Request timeout.");
164  return 0;
165  }
166  netdbg("Still waiting...");
167  TaskManager->Sleep(1000);
168  }
169 
170  return ret;
171  }

179  Header->ProtocolType = b16(NetworkEthernet::FrameType::TYPE_IPV4);

181  Header->ProtocolSize = b8(0x4);

183  Header->SenderMAC = b48(Ethernet->GetInterface()->MAC.ToHex());
184  Header->SenderIP = b32(Ethernet->GetInterface()->IP.v4.ToHex());

191  bool ARP::OnEthernetPacketReceived(uint8_t *Data, size_t Length)
192  {
193  UNUSED(Length);
194  netdbg("Received packet");
195  ARPHeader *Header = (ARPHeader *)Data;
196 
197  InternetProtocol SenderIPv4;
198  SenderIPv4.v4 = SenderIPv4.v4.FromHex(b32(Header->SenderIP));
199 
200  /* We only support Ethernet and IPv4. */
201  if (b16(Header->HardwareType) != ARPHardwareType::HTYPE_ETHERNET || b16(Header->ProtocolType) != NetworkEthernet::FrameType::TYPE_IPV4)
202  {
203  warn("Invalid hardware/protocol type (%d/%d)", b16(Header->HardwareType), b16(Header->ProtocolType));
204  return false;
205  }
206 
207  InternetProtocol TmpIPStruct;
208  InternetProtocol().v4.FromHex(b32(Header->SenderIP));
209 
210  if (ManageDiscoveredAddresses(DA_SEARCH, TmpIPStruct, MediaAccessControl().FromHex(b48(Header->SenderMAC))) == nullptr)
211  {
212  netdbg("Discovered new address %s", SenderIPv4.v4.ToStringLittleEndian());
213  TmpIPStruct.v4.FromHex(b32(Header->SenderIP));
214  ManageDiscoveredAddresses(DA_ADD, TmpIPStruct, MediaAccessControl().FromHex(b48(Header->SenderMAC)));
215  }
216  else
217  {
218  netdbg("Updated address %s", SenderIPv4.v4.ToStringLittleEndian());
219  TmpIPStruct.v4.FromHex(b32(Header->SenderIP));
220  ManageDiscoveredAddresses(DA_UPDATE, TmpIPStruct, MediaAccessControl().FromHex(b48(Header->SenderMAC)));
221  }
222 
223  switch (b16(Header->Operation))
224  {
225  case ARPOperation::REQUEST:
226  netdbg("Received request from %s", SenderIPv4.v4.ToStringLittleEndian());
227  /* We need to byteswap before we send it back. */
228  Header->TargetMAC = b48(Header->SenderMAC);
229  Header->TargetIP = b32(Header->SenderIP);
230  Header->SenderMAC = b48(Ethernet->GetInterface()->MAC.ToHex());
231  Header->SenderIP = b32(Ethernet->GetInterface()->IP.v4.ToHex());
232  Header->Operation = b16(ARPOperation::REPLY);
233  Ethernet->Send(MediaAccessControl().FromHex(Header->TargetMAC), NetworkEthernet::FrameType::TYPE_ARP, (uint8_t *)Header, sizeof(ARPHeader));
234  netdbg("Sent request for %s", SenderIPv4.v4.ToStringLittleEndian());
235  break;
236  case ARPOperation::REPLY:
237  fixme("Received reply from %s", SenderIPv4.v4.ToStringLittleEndian());
238  break;
239  default:
240  warn("Invalid operation (%d)", b16(Header->Operation));
241  break;
242  }
243  return false;
244  }
