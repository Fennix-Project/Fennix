54  Elf64_Sym *ELFLookupSymbol(Elf64_Ehdr *Header, const char *Name)
55  {
56  Elf64_Shdr *SymbolTable = nullptr;
57  Elf64_Shdr *StringTable = nullptr;
58  Elf64_Sym *Symbol = nullptr;
59  char *String = nullptr;
60 
61  for (Elf64_Half i = 0; i < Header->e_shnum; i++)
62  {
63  Elf64_Shdr *shdr = GetELFSection(Header, i);
64  switch (shdr->sh_type)
65  {
66  case SHT_SYMTAB:
67  SymbolTable = shdr;
68  StringTable = GetELFSection(Header, shdr->sh_link);
69  break;
70  default:
71  {
72  break;
73  }
74  }
75  }
76 
77  if (SymbolTable == nullptr || StringTable == nullptr)
78  return nullptr;
79 
80  for (size_t i = 0; i < (SymbolTable->sh_size / sizeof(Elf64_Sym)); i++)
81  {
82  Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
83  String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
84  if (strcmp(String, Name) == 0)
85  return Symbol;
86  }
87  return nullptr;
88  }

90  uintptr_t ELFGetSymbolValue(Elf64_Ehdr *Header, uint64_t Table, uint64_t Index)
91  {
92  if (Table == SHN_UNDEF || Index == SHN_UNDEF)
93  return 0;
94  Elf64_Shdr *SymbolTable = GetELFSection(Header, Table);
95 
96  uint64_t STEntries = SymbolTable->sh_size / SymbolTable->sh_entsize;
97  if (Index >= STEntries)
98  {
99  error("Symbol index out of range %d-%u.", Table, Index);
100  return 0xdead;
101  }
102 
103  uint64_t SymbolAddress = (uint64_t)Header + SymbolTable->sh_offset;
104  Elf64_Sym *Symbol = &((Elf64_Sym *)SymbolAddress)[Index];
105 
106  if (Symbol->st_shndx == SHN_UNDEF)
107  {
108  Elf64_Shdr *StringTable = GetELFSection(Header, SymbolTable->sh_link);
109  const char *Name = (const char *)Header + StringTable->sh_offset + Symbol->st_name;
110 
111  void *Target = (void *)ELFLookupSymbol(Header, Name)->st_value;
112  if (Target == nullptr)
113  {
114  if (ELF64_ST_BIND(Symbol->st_info) & STB_WEAK)
115  return 0;
116  else
117  {
118  error("Undefined external symbol \"%s\".", Name);
119  return 0xdead;
120  }
121  }
122  else
123  return (uintptr_t)Target;
124  }
125  else if (Symbol->st_shndx == SHN_ABS)
126  return Symbol->st_value;
127  else
128  {
129  Elf64_Shdr *Target = GetELFSection(Header, Symbol->st_shndx);
130  return (uintptr_t)Header + Symbol->st_value + Target->sh_offset;
131  }
132  }

134  Elf64_Dyn *ELFGetDynamicTag(void *ElfFile, enum DynamicArrayTags Tag)
135  {
136  Elf64_Ehdr *ELFHeader = (Elf64_Ehdr *)ElfFile;
137 
138  Elf64_Phdr ItrPhdr;
139  for (Elf64_Half i = 0; i < ELFHeader->e_phnum; i++)
140  {
141  memcpy(&ItrPhdr, (uint8_t *)ElfFile + ELFHeader->e_phoff + ELFHeader->e_phentsize * i, sizeof(Elf64_Phdr));
142  if (ItrPhdr.p_type == PT_DYNAMIC)
143  {
144  Elf64_Dyn *Dynamic = (Elf64_Dyn *)((uint8_t *)ElfFile + ItrPhdr.p_offset);
145  for (size_t i = 0; i < ItrPhdr.p_filesz / sizeof(Elf64_Dyn); i++)
146  {
147  if (Dynamic[i].d_tag == Tag)
148  {
149  debug("Found dynamic tag %d at %#lx [d_val: %#lx].", Tag, &Dynamic[i], Dynamic[i].d_un.d_val);
150  return &Dynamic[i];
151  }
152  if (Dynamic[i].d_tag == DT_NULL)
153  {
154  debug("Reached end of dynamic tag list for tag %d.", Tag);
155  return nullptr;
156  }
157  }
158  }
159  }
160  debug("Dynamic tag %d not found.", Tag);
161  return nullptr;
162  }

227  uintptr_t LoadELFInterpreter(Memory::MemMgr *mem, Memory::Virtual &pV, const char *Interpreter)
228  {
229  if (GetBinaryType((char *)Interpreter) != BinaryType::BinTypeELF)
230  {
231  error("Interpreter \"%s\" is not an ELF file.", Interpreter);
232  return 0;
233  }
234 
235  /* No need to check if it's valid, the GetBinaryType() call above does that. */
236  VirtualFileSystem::File File = vfs->Open(Interpreter);
237 
238  Elf64_Ehdr *ELFHeader = (Elf64_Ehdr *)File.node->Address;
239 
240 #ifdef DEBUG
241  const char *InterpreterType[6] = {
242  "ET_NONE",
243  "ET_REL",
244  "ET_EXEC",
245  "ET_DYN",
246  "ET_CORE",
247  "ET_LOPROC - ET_HIPROC"};
248  Elf64_Half IntType = ELFHeader->e_type;
249  if (IntType > 5)
250  IntType = 5;
251  debug("Interpreter type: %s - %#x", InterpreterType[IntType], ELFHeader->e_type);
252 #endif
253 
254  uintptr_t BaseAddress = UINTPTR_MAX;
255  uint64_t ElfAppSize = 0;
256 
257  Elf64_Phdr ItrPhdr;
258 
259  /* Get base address */
260  for (Elf64_Half i = 0; i < ELFHeader->e_phnum; i++)
261  {
262  memcpy(&ItrPhdr,
263  (uint8_t *)File.node->Address + ELFHeader->e_phoff + ELFHeader->e_phentsize * i,
264  sizeof(Elf64_Phdr));
265 
266  BaseAddress = MIN(BaseAddress, ItrPhdr.p_vaddr);
267  }
268 
269  /* Get size */
270  for (Elf64_Half i = 0; i < ELFHeader->e_phnum; i++)
271  {
272  memcpy(&ItrPhdr,
273  (uint8_t *)File.node->Address + ELFHeader->e_phoff + ELFHeader->e_phentsize * i,
274  sizeof(Elf64_Phdr));
275 
276  uintptr_t SegmentEnd;
277  SegmentEnd = ItrPhdr.p_vaddr - BaseAddress + ItrPhdr.p_memsz;
278  ElfAppSize = MAX(ElfAppSize, SegmentEnd);
279  }
280 
281  MmImage MemoryImage = ELFCreateMemoryImage(mem, pV, (void *)File.node->Address, ElfAppSize);
282 
283  for (Elf64_Half i = 0; i < ELFHeader->e_phnum; i++)
284  {
285  memcpy(&ItrPhdr,
286  (uint8_t *)File.node->Address + ELFHeader->e_phoff + ELFHeader->e_phentsize * i,
287  sizeof(Elf64_Phdr));
288 
289  if (ItrPhdr.p_type == PT_LOAD)
290  {
291  debug("PT_LOAD - Offset: %#lx, VirtAddr: %#lx, FileSiz: %ld, MemSiz: %ld, Align: %#lx",
292  ItrPhdr.p_offset, ItrPhdr.p_vaddr,
293  ItrPhdr.p_filesz, ItrPhdr.p_memsz, ItrPhdr.p_align);
294  uintptr_t MAddr = (ItrPhdr.p_vaddr - BaseAddress) + (uintptr_t)MemoryImage.Physical;
295  fixme("Address: %#lx %s%s%s", MAddr,
296  (ItrPhdr.p_flags & PF_R) ? "R" : "",
297  (ItrPhdr.p_flags & PF_W) ? "W" : "",
298  (ItrPhdr.p_flags & PF_X) ? "X" : "");
299 
300  memcpy((void *)MAddr, (uint8_t *)File.node->Address + ItrPhdr.p_offset, ItrPhdr.p_filesz);
301  debug("memcpy: %#lx => %#lx (%ld bytes)", (uint8_t *)File.node->Address + ItrPhdr.p_offset, MAddr, ItrPhdr.p_filesz);
302  }
303  }
304 
305  vfs->Close(File);
306  debug("Interpreter entry point: %#lx (%#lx + %#lx)", (uintptr_t)MemoryImage.Physical + ELFHeader->e_entry,
307  (uintptr_t)MemoryImage.Physical, ELFHeader->e_entry);
308  return (uintptr_t)MemoryImage.Physical + ELFHeader->e_entry;
309  }
