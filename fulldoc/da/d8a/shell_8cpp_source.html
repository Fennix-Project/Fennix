82 void StartKernelShell()
83 {
84  if (ShellLock.Locked())
85  return;
86  SmartLock(ShellLock);
87 
88  debug("Starting kernel shell...");
89  KPrint("Starting kernel shell...");
90  thisThread->SetPriority(Tasking::TaskPriority::High);

92  std::string Buffer;
93  std::vector<std::string *> History;
94  size_t HistoryIndex = 0;
95  bool CtrlDown = false;
96  bool UpperCase = false;
97  bool TabDoublePress = false;
98 
99  int kfd = fopen("/dev/key", "r");
100  if (kfd < 0)
101  {
102  KPrint("Failed to open keyboard device! %s",
103  strerror(kfd));
104  return;
105  }
106 
107  printf("Using \eCA21F6/dev/key\eCCCCCC for keyboard input.\n");
108  while (true)
109  {
110  size_t BackspaceCount = 0;
111  Buffer.clear();
112 
113  vfs::Node *cwd = thisProcess->CurrentWorkingDirectory;
114  if (!cwd)
115  cwd = fs->GetNodeFromPath("/");
116 
117  printf("\e34C6EB%s@%s:%s$ \eCCCCCC",
118  "kernel",
119  "fennix",
120  cwd->FullPath);
121  Display->SetBuffer(0);
122 
123  uint8_t scBuf[2];
124  scBuf[1] = 0x00; /* Request scan code */
125  ssize_t nBytes;
126  while (true)
127  {
128  nBytes = fread(kfd, scBuf, 2);
129  if (nBytes == 0)
130  continue;
131  if (nBytes < 0)
132  {
133  KPrint("Failed to read from keyboard device: %s",
134  strerror((int)nBytes));
135  return;
136  }
137 
138  if (scBuf[0] == 0x00)
139  continue;
140 
141  uint8_t sc = scBuf[0];
142  switch (sc & ~KEY_PRESSED)
143  {
144  case KEY_LEFT_CTRL:
145  case KEY_RIGHT_CTRL:
146  {
147  if (sc & KEY_PRESSED)
148  CtrlDown = true;
149  else
150  CtrlDown = false;
151  continue;
152  }
153  case KEY_LEFT_SHIFT:
154  case KEY_RIGHT_SHIFT:
155  {
156  if (sc & KEY_PRESSED)
157  UpperCase = true;
158  else
159  UpperCase = false;
160  continue;
161  }
162  case KEY_BACKSPACE:
163  {
164  if (!(sc & KEY_PRESSED))
165  continue;
166 
167  if (BackspaceCount == 0)
168  continue;
169 
170  Display->Print('\b', 0);
171  Buffer.pop_back();
172  BackspaceCount--;
173  Display->SetBuffer(0);
174  continue;
175  }
176  case KEY_UP_ARROW:
177  {
178  if (!(sc & KEY_PRESSED))
179  continue;
180 
181  if (History.size() == 0 ||
182  HistoryIndex == 0)
183  continue;
184 
185  HistoryIndex--;
186 
187  for (size_t i = 0; i < Buffer.size(); i++)
188  Display->Print('\b', 0);
189  Display->SetBuffer(0);
190 
191  Buffer = History[HistoryIndex]->c_str();
192 
193  for (size_t i = 0; i < strlen(Buffer.c_str()); i++)
194  Display->Print(Buffer[i], 0);
195  BackspaceCount = Buffer.size();
196  Display->SetBuffer(0);
197  continue;
198  }
199  case KEY_DOWN_ARROW:
200  {
201  if (!(sc & KEY_PRESSED))
202  continue;
203 
204  if (History.size() == 0 ||
205  HistoryIndex == History.size())
206  continue;
207 
208  if (HistoryIndex == History.size() - 1)
209  {
210  HistoryIndex++;
211  for (size_t i = 0; i < Buffer.size(); i++)
212  Display->Print('\b', 0);
213  BackspaceCount = Buffer.size();
214  Display->SetBuffer(0);
215  continue;
216  }
217 
218  for (size_t i = 0; i < Buffer.size(); i++)
219  Display->Print('\b', 0);
220  Display->SetBuffer(0);
221 
222  HistoryIndex++;
223  Buffer = History[HistoryIndex]->c_str();
224 
225  for (size_t i = 0; i < strlen(Buffer.c_str()); i++)
226  Display->Print(Buffer[i], 0);
227 
228  BackspaceCount = Buffer.size();
229  Display->SetBuffer(0);
230  continue;
231  }
232  case KEY_TAB:
233  {
234  if (!(sc & KEY_PRESSED))
235  continue;
236 
237  if (!TabDoublePress)
238  {
239  TabDoublePress = true;
240  continue;
241  }
242  TabDoublePress = false;
243  if (Buffer.size() == 0)
244  {
245  for (size_t i = 0; i < sizeof(commands) / sizeof(commands[0]); i++)
246  printf("%s ", commands[i].Name);
247 
248  Display->Print('\n', 0);
249  Display->SetBuffer(0);
250  goto SecLoopEnd;
251  }
252 
253  for (size_t i = 0; i < Buffer.size(); i++)
254  Display->Print('\b', 0);
255  Display->SetBuffer(0);
256 
257  for (size_t i = 0; i < sizeof(commands) / sizeof(commands[0]); i++)
258  {
259  if (strncmp(Buffer.c_str(), commands[i].Name, Buffer.size()) == 0)
260  {
261  Buffer = commands[i].Name;
262  for (size_t i = 0; i < strlen(Buffer.c_str()); i++)
263  Display->Print(Buffer[i], 0);
264  BackspaceCount = Buffer.size();
265  Display->SetBuffer(0);
266  break;
267  }
268  }
269  continue;
270  }
271  default:
272  break;
273  }
274 
275  if (!(sc & KEY_PRESSED))
276  continue;
277 
278  if (!Driver::IsValidChar(sc))
279  continue;
280 
281  char c = Driver::GetScanCode(sc, UpperCase);
282 
283  if (CtrlDown)
284  {
285  switch (std::toupper((char)c))
286  {
287  case 'C':
288  {
289  Display->Print('^', 0);
290  Display->Print('C', 0);
291  Display->Print('\n', 0);
292  fixme("No SIGINT handler yet.");
293  Display->SetBuffer(0);
294  goto SecLoopEnd;
295  }
296  case 'D':
297  {
298  Display->Print('^', 0);
299  Display->Print('D', 0);
300  Display->Print('\n', 0);
301  fixme("No SIGKILL handler yet.");
302  Display->SetBuffer(0);
303  goto SecLoopEnd;
304  }
305  default:
306  continue;
307  }
308  }
309 
310  Display->Print(c, 0);
311  if (c == '\n')
312  {
313  if (Buffer.length() > 0)
314  {
315  std::string *hBuff = new std::string(Buffer.c_str());
316  History.push_back(hBuff);
317  HistoryIndex = History.size();
318  }
319  break;
320  }
321 
322  Buffer += c;
323  BackspaceCount++;
324  Display->SetBuffer(0);
325  }
326  SecLoopEnd:
327 
328  if (Buffer.length() == 0)
329  continue;
330 
331  bool Found = false;
332  for (size_t i = 0; i < sizeof(commands) / sizeof(Command); i++)
333  {
334  std::string cmd_extracted;
335  for (size_t i = 0; i < Buffer.length(); i++)
336  {
337  if (Buffer[i] == ' ')
338  break;
339  cmd_extracted += Buffer[i];
340  }
341 
342  // debug("cmd: %s, array[%d]: %s", cmd_extracted.c_str(), i, commands[i].Name);
343  if (strncmp(commands[i].Name, cmd_extracted.c_str(), cmd_extracted.size()) == 0)
344  {
345  if (strlen(commands[i].Name) != cmd_extracted.size())
346  continue;
347 
348  Found = true;
349 
350  std::string arg_only;
351  const char *cmd_name = commands[i].Name;
352  for (size_t i = strlen(cmd_name) + 1; i < Buffer.length(); i++)
353  arg_only += Buffer[i];
354 
355  if (commands[i].Function)
356  commands[i].Function(arg_only.c_str());
357  else
358  {
359  std::string cmd_only;
360  for (size_t i = 0; i < Buffer.length(); i++)
361  {
362  if (Buffer[i] == ' ')
363  break;
364  cmd_only += Buffer[i];
365  }
366  printf("%s: command not implemented\n",
367  cmd_only.c_str());
368  }
369  break;
370  }
371  }
372 
373  if (!Found)
374  {
375  std::string cmd_only;
376  for (size_t i = 0; i < Buffer.length(); i++)
377  {
378  if (Buffer[i] == ' ')
379  break;
380  cmd_only += Buffer[i];
381  }
382  printf("%s: command not found\n",
383  cmd_only.c_str());
384  }
385  }
