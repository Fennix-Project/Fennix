385 void __ubsan_handle_type_mismatch_v1(struct type_mismatch_v1_data *type_mismatch, uintptr_t pointer)
386 {
387  struct source_location *location = &type_mismatch->location;
388  if (pointer == 0)
389  {
390  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
391  ubsan("Null pointer access.");
392  }
393  else if (type_mismatch->alignment != 0 && is_aligned(pointer, type_mismatch->alignment))
394  {
395  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
396  ubsan("Unaligned memory access %#llx.", pointer);
397  }
398  else
399  {
400  ubsan("\t\tIn File: %s:%i:%i", location->file, location->line, location->column);
401  ubsan("%s address %#llx with insufficient space for object of type %s",
402  Type_Check_Kinds[type_mismatch->type_check_kind], (void *)pointer, type_mismatch->type->name);
403  }
404 }
