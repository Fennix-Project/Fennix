142  __no_sanitize("undefined") void TCB::SetupUserStack_x86_64(const char **argv,
143  const char **envp,
144  const std::vector<AuxiliaryVector> &auxv)
145  {
146  size_t ArgvSize = 0;
147  if (argv)
148  while (argv[ArgvSize] != nullptr)
149  ArgvSize++;
150 
151  size_t EnvpSize = 0;
152  if (envp)
153  while (envp[EnvpSize] != nullptr)
154  EnvpSize++;
155 
156  debug("ArgvSize: %d", ArgvSize);
157  debug("EnvpSize: %d", EnvpSize);
158 
159  /* https://articles.manugarg.com/aboutelfauxiliaryvectors.html */
160  /* https://refspecs.linuxbase.org/elf/x86_64-abi-0.99.pdf#figure.3.9 */
161  // rsp is the top of the stack
162  char *Stack = (char *)this->Stack->GetStackPhysicalTop();
163  // Temporary stack pointer for strings
164  char *StackStrings = (char *)Stack;
165  char *StackStringsVirtual = (char *)this->Stack->GetStackTop();
166 
167  // Store string pointers for later
168  uintptr_t *ArgvStrings = nullptr;
169  uintptr_t *EnvpStrings = nullptr;
170  if (ArgvSize > 0)
171  ArgvStrings = new uintptr_t[ArgvSize];
172  if (EnvpSize > 0)
173  EnvpStrings = new uintptr_t[EnvpSize];
174 
175  for (size_t i = 0; i < ArgvSize; i++)
176  {
177  // Subtract the length of the string and the null terminator
178  StackStrings -= strlen(argv[i]) + 1;
179  StackStringsVirtual -= strlen(argv[i]) + 1;
180  // Store the pointer to the string
181  ArgvStrings[i] = (uintptr_t)StackStringsVirtual;
182  // Copy the string to the stack
183  strcpy(StackStrings, argv[i]);
184  debug("argv[%d]: %s", i, argv[i]);
185  }
186 
187  for (size_t i = 0; i < EnvpSize; i++)
188  {
189  // Subtract the length of the string and the null terminator
190  StackStrings -= strlen(envp[i]) + 1;
191  StackStringsVirtual -= strlen(envp[i]) + 1;
192  // Store the pointer to the string
193  EnvpStrings[i] = (uintptr_t)StackStringsVirtual;
194  // Copy the string to the stack
195  strcpy(StackStrings, envp[i]);
196  debug("envp[%d]: %s", i, envp[i]);
197  }
198 
199  // Align the stack to 16 bytes
200  StackStrings -= (uintptr_t)StackStrings & 0xF;
201  // Set "Stack" to the new stack pointer
202  Stack = (char *)StackStrings;
203  // If argv and envp sizes are odd then we need to align the stack
204  Stack -= (ArgvSize + EnvpSize) % 2;
205 
206  // We need 8 bit pointers for the stack from here
207  uintptr_t *Stack64 = (uintptr_t *)Stack;
208 
209  // Store the null terminator
210  Stack64--;
211  *Stack64 = AT_NULL;
212 
213  // auxv_array is initialized with auxv elements. If the array is empty then we add a null terminator
214  std::vector<AuxiliaryVector> auxv_array = auxv;
215  if (auxv_array.size() == 0)
216  auxv_array.push_back({.archaux = {.a_type = AT_NULL, .a_un = {.a_val = 0}}});
217 
218  // Store auxillary vector
219  foreach (AuxiliaryVector var in auxv_array)
220  {
221  // Subtract the size of the auxillary vector
222  Stack64 -= sizeof(Elf_auxv_t) / sizeof(uintptr_t);
223  // Store the auxillary vector
224  POKE(Elf_auxv_t, Stack64) = var.archaux;
225  // TODO: Store strings to the stack
226  }
227 
228  // Store the null terminator
229  Stack64--;
230  *Stack64 = AT_NULL;
231 
232  // Store EnvpStrings[] to the stack
233  Stack64 -= EnvpSize; // (1 Stack64 = 8 bits; Stack64 = 8 * EnvpSize)
234  for (size_t i = 0; i < EnvpSize; i++)
235  {
236  *(Stack64 + i) = (uintptr_t)EnvpStrings[i];
237  debug("EnvpStrings[%d]: %#lx",
238  i, EnvpStrings[i]);
239  }
240 
241  // Store the null terminator
242  Stack64--;
243  *Stack64 = AT_NULL;
244 
245  // Store ArgvStrings[] to the stack
246  Stack64 -= ArgvSize; // (1 Stack64 = 8 bits; Stack64 = 8 * ArgvSize)
247  for (size_t i = 0; i < ArgvSize; i++)
248  {
249  *(Stack64 + i) = (uintptr_t)ArgvStrings[i];
250  debug("ArgvStrings[%d]: %#lx",
251  i, ArgvStrings[i]);
252  }
253 
254  // Store the argc
255  Stack64--;
256  *Stack64 = ArgvSize;
257 
258  // Set "Stack" to the new stack pointer
259  Stack = (char *)Stack64;
260 
261  /* We need the virtual address but because we are in the kernel we can't use the process page table.
262  So we modify the physical address and store how much we need to subtract to get the virtual address for RSP. */
263  uintptr_t SubtractStack = (uintptr_t)this->Stack->GetStackPhysicalTop() - (uintptr_t)Stack;
264  debug("SubtractStack: %#lx", SubtractStack);
265 
266  // Set the stack pointer to the new stack
267  uintptr_t StackPointerReg = ((uintptr_t)this->Stack->GetStackTop() - SubtractStack);
268 #if defined(a64)
269  this->Registers.rsp = StackPointerReg;
270 #elif defined(a32)
271  this->Registers.esp = StackPointerReg;
272 #elif defined(aa64)
273  this->Registers.sp = StackPointerReg;
274 #endif
275 
276  if (ArgvSize > 0)
277  delete[] ArgvStrings;
278  if (EnvpSize > 0)
279  delete[] EnvpStrings;
280 
281 #ifdef DEBUG
282  DumpData("Stack Data", (void *)((uintptr_t)this->Stack->GetStackPhysicalTop() - (uintptr_t)SubtractStack), SubtractStack);
283 #endif
284 
285 #if defined(a64)
286  this->Registers.rdi = (uintptr_t)ArgvSize; // argc
287  this->Registers.rsi = (uintptr_t)(this->Registers.rsp + 8); // argv
288  this->Registers.rcx = (uintptr_t)EnvpSize; // envc
289  this->Registers.rdx = (uintptr_t)(this->Registers.rsp + 8 + (8 * ArgvSize) + 8); // envp
290 #elif defined(a32)
291  this->Registers.eax = (uintptr_t)ArgvSize; // argc
292  this->Registers.ebx = (uintptr_t)(this->Registers.esp + 4); // argv
293  this->Registers.ecx = (uintptr_t)EnvpSize; // envc
294  this->Registers.edx = (uintptr_t)(this->Registers.esp + 4 + (4 * ArgvSize) + 4); // envp
295 #elif defined(aa64)
296  this->Registers.x0 = (uintptr_t)ArgvSize; // argc
297  this->Registers.x1 = (uintptr_t)(this->Registers.sp + 8); // argv
298  this->Registers.x2 = (uintptr_t)EnvpSize; // envc
299  this->Registers.x3 = (uintptr_t)(this->Registers.sp + 8 + (8 * ArgvSize) + 8); // envp
300 #endif
301  }

317  TCB::TCB(Task *ctx, PCB *Parent, IP EntryPoint,
318  const char **argv, const char **envp,
319  const std::vector<AuxiliaryVector> &auxv,
320  TaskArchitecture Architecture,
321  TaskCompatibility Compatibility,
322  bool ThreadNotReady)
323  {
324  assert(ctx != nullptr);
325  assert(Architecture >= _ArchitectureMin);
326  assert(Architecture <= _ArchitectureMax);
327  assert(Compatibility >= _CompatibilityMin);
328  assert(Compatibility <= _CompatibilityMax);
329 
330  if (Parent == nullptr)
331  {
332  this->Parent = ctx->GetCurrentProcess();
333  assert(this->Parent != nullptr);
334  }
335  else
336  this->Parent = Parent;
337 
338  this->ctx = ctx;
339  this->ID = ctx->NextTID++;
340 
341  if (this->Name)
342  delete[] this->Name;
343 
344  this->Name = new char[strlen(this->Parent->Name) + 1];
345  strcpy((char *)this->Name, this->Parent->Name);
346 
347  this->EntryPoint = EntryPoint;
348  this->ExitCode = KILL_CRASH;
349 
350  if (ThreadNotReady)
351  this->Status = TaskStatus::Zombie;
352  else
353  this->Status = TaskStatus::Ready;
354 
355  this->Memory = new Memory::MemMgr(this->Parent->PageTable,
356  this->Parent->memDirectory);
357 
358 #if defined(a64)
359  this->Registers.rip = EntryPoint;
360 #elif defined(a32)
361  this->Registers.eip = EntryPoint;
362 #elif defined(aa64)
363  this->Registers.pc = EntryPoint;
364 #endif
365 
366  switch (this->Parent->Security.ExecutionMode)
367  {
368  case TaskExecutionMode::System:
369  fixme("System mode not supported.");
370  [[fallthrough]];
371  case TaskExecutionMode::Kernel:
372  {
373  this->Security.IsCritical = true;
374  this->Stack = new Memory::StackGuard(false,
375  this->Parent->PageTable);
376 
377 #if defined(a64)
378  this->ShadowGSBase =
379  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_SHADOW_GS_BASE);
380  this->GSBase = CPU::x64::rdmsr(CPU::x64::MSRID::MSR_GS_BASE);
381  this->FSBase = CPU::x64::rdmsr(CPU::x64::MSRID::MSR_FS_BASE);
382  this->Registers.cs = GDT_KERNEL_CODE;
383  this->Registers.ss = GDT_KERNEL_DATA;
384  this->Registers.rflags.AlwaysOne = 1;
385  this->Registers.rflags.IF = 1;
386  this->Registers.rflags.ID = 1;
387  this->Registers.rsp = ((uintptr_t)this->Stack->GetStackTop());
388  POKE(uintptr_t, this->Registers.rsp) = (uintptr_t)ThreadDoExit;
389 #elif defined(a32)
390  this->Registers.cs = GDT_KERNEL_CODE;
391  this->Registers.r3_ss = GDT_KERNEL_DATA;
392  this->Registers.eflags.AlwaysOne = 1;
393  this->Registers.eflags.IF = 1;
394  this->Registers.eflags.ID = 1;
395  this->Registers.esp = ((uintptr_t)this->Stack->GetStackTop());
396  POKE(uintptr_t, this->Registers.esp) = (uintptr_t)ThreadDoExit;
397 #elif defined(aa64)
398  this->Registers.pc = EntryPoint;
399  this->Registers.sp = ((uintptr_t)this->Stack->GetStackTop());
400  POKE(uintptr_t, this->Registers.sp) = (uintptr_t)ThreadDoExit;
401 #endif
402  break;
403  }
404  case TaskExecutionMode::User:
405  {
406  this->Stack = new Memory::StackGuard(true,
407  this->Parent->PageTable);
408 
409  gsTCB *gsT = (gsTCB *)this->Memory->RequestPages(TO_PAGES(sizeof(gsTCB)));
410 
411  gsT->SyscallStack =
412  (uintptr_t)this->Memory->RequestPages(TO_PAGES(STACK_SIZE)) +
413  STACK_SIZE - 0x10;
414 
415  gsT->TempStack = 0x0;
416  gsT->t = this;
417 #if defined(a64)
418  this->ShadowGSBase = (uintptr_t)gsT;
419  this->GSBase = 0;
420  this->FSBase = 0;
421  this->Registers.cs = GDT_USER_CODE;
422  this->Registers.ss = GDT_USER_DATA;
423  this->Registers.rflags.AlwaysOne = 1;
424  this->Registers.rflags.IF = 1;
425  this->Registers.rflags.ID = 1;
426  /* We need to leave the libc's crt
427  to make a syscall when the Thread
428  is exited or we are going to get
429  an exception. */
430 
431  this->SetupUserStack_x86_64(argv, envp, auxv);
432 #elif defined(a32)
433  this->Registers.cs = GDT_USER_CODE;
434  this->Registers.r3_ss = GDT_USER_DATA;
435  this->Registers.eflags.AlwaysOne = 1;
436  this->Registers.eflags.IF = 1;
437  this->Registers.eflags.ID = 1;
438  /* We need to leave the libc's crt
439  to make a syscall when the Thread
440  is exited or we are going to get
441  an exception. */
442 
443  this->SetupUserStack_x86_32(argv, envp, auxv);
444 #elif defined(aa64)
445  this->SetupUserStack_aarch64(argv, envp, auxv);
446 #endif
447 #ifdef DEBUG_TASKING
448  DumpData(this->Name, this->Stack, STACK_SIZE);
449 #endif
450  break;
451  }
452  default:
453  assert(false);
454  }
455 
456  this->Info.Architecture = Architecture;
457  this->Info.Compatibility = Compatibility;
458  this->Security.ExecutionMode =
459  this->Parent->Security.ExecutionMode;
460 
461  // TODO: Is really a good idea to use the FPU in kernel mode?
462  this->FPU.mxcsr = 0b0001111110000000;
463  this->FPU.mxcsrmask = 0b1111111110111111;
464  this->FPU.fcw = 0b0000001100111111;
465 
466 #ifdef DEBUG
467 #ifdef a64
468  debug("Thread EntryPoint: %#lx => RIP: %#lx",
469  this->EntryPoint, this->Registers.rip);
470  if (this->Parent->Security.ExecutionMode == TaskExecutionMode::User)
471  debug("Thread stack region is %#lx-%#lx (U) and rsp is %#lx",
472  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
473  this->Registers.rsp);
474  else
475  debug("Thread stack region is %#lx-%#lx (K) and rsp is %#lx",
476  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
477  this->Registers.rsp);
478 #elif defined(a32)
479  debug("Thread EntryPoint: %#lx => RIP: %#lx",
480  this->EntryPoint, this->Registers.eip);
481  if (Parent->Security.ExecutionMode == TaskExecutionMode::User)
482  debug("Thread stack region is %#lx-%#lx (U) and rsp is %#lx",
483  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
484  this->Registers.esp);
485  else
486  debug("Thread stack region is %#lx-%#lx (K) and rsp is %#lx",
487  this->Stack->GetStackBottom(), this->Stack->GetStackTop(),
488  this->Registers.esp);
489 #elif defined(aa64)
490 #endif
491  debug("Created thread \"%s\"(%d) in process \"%s\"(%d)",
492  this->Name, this->ID, this->Parent->Name,
493  this->Parent->ID);
494 #endif
495 
496  this->Info.SpawnTime = TimeManager->GetCounter();
497 
498  this->Parent->Threads.push_back(this);
499 
500  if (this->Parent->Threads.size() == 1 &&
501  this->Parent->Status == Zombie &&
502  ThreadNotReady == false)
503  {
504  this->Parent->Status = Ready;
505  }
506  }
