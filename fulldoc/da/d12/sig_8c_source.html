225 /* Initialize signals that will terminate the shell to do some
226  unwind protection. For non-interactive shells, we only call
227  this when a trap is defined for EXIT (0) or when trap is run
228  to display signal dispositions. */
229 void
230 initialize_terminating_signals ()
231 {
232  register int i;
233 #if defined (HAVE_POSIX_SIGNALS)
234  struct sigaction act, oact;
235 #endif
236 
237  if (termsigs_initialized)
238  return;
239 
240  /* The following code is to avoid an expensive call to
241  set_signal_handler () for each terminating_signals. Fortunately,
242  this is possible in Posix. Unfortunately, we have to call signal ()
243  on non-Posix systems for each signal in terminating_signals. */
244 #if defined (HAVE_POSIX_SIGNALS)
245  act.sa_handler = termsig_sighandler;
246  act.sa_flags = 0;
247  sigemptyset (&act.sa_mask);
248  sigemptyset (&oact.sa_mask);
249  for (i = 0; i < TERMSIGS_LENGTH; i++)
250  sigaddset (&act.sa_mask, XSIG (i));
251  for (i = 0; i < TERMSIGS_LENGTH; i++)
252  {
253  /* If we've already trapped it, don't do anything. */
254  if (signal_is_trapped (XSIG (i)))
255  continue;
256 
257  sigaction (XSIG (i), &act, &oact);
258  XHANDLER(i) = oact.sa_handler;
259  XSAFLAGS(i) = oact.sa_flags;
260 
261 #if 0
262  set_original_signal (XSIG(i), XHANDLER(i)); /* optimization */
263 #else
264  set_original_signal (XSIG(i), act.sa_handler); /* optimization */
265 #endif
266 
267  /* Don't do anything with signals that are ignored at shell entry
268  if the shell is not interactive. */
269  /* XXX - should we do this for interactive shells, too? */
270  if (interactive_shell == 0 && XHANDLER (i) == SIG_IGN)
271  {
272  sigaction (XSIG (i), &oact, &act);
273  set_signal_hard_ignored (XSIG (i));
274  }
275 #if defined (SIGPROF) && !defined (_MINIX)
276  if (XSIG (i) == SIGPROF && XHANDLER (i) != SIG_DFL && XHANDLER (i) != SIG_IGN)
277  sigaction (XSIG (i), &oact, (struct sigaction *)NULL);
278 #endif /* SIGPROF && !_MINIX */
279  }
280 #else /* !HAVE_POSIX_SIGNALS */
281 
282  for (i = 0; i < TERMSIGS_LENGTH; i++)
283  {
284  /* If we've already trapped it, don't do anything. */
285  if (signal_is_trapped (XSIG (i)))
286  continue;
287 
288  XHANDLER(i) = signal (XSIG (i), termsig_sighandler);
289  XSAFLAGS(i) = 0;
290  /* Don't do anything with signals that are ignored at shell entry
291  if the shell is not interactive. */
292  /* XXX - should we do this for interactive shells, too? */
293  if (interactive_shell == 0 && XHANDLER (i) == SIG_IGN)
294  {
295  signal (XSIG (i), SIG_IGN);
296  set_signal_hard_ignored (XSIG (i));
297  }
298 #ifdef SIGPROF
299  if (XSIG (i) == SIGPROF && XHANDLER (i) != SIG_DFL && XHANDLER (i) != SIG_IGN)
300  signal (XSIG (i), XHANDLER (i));
301 #endif
302  }
303 
304 #endif /* !HAVE_POSIX_SIGNALS */
305 
306  termsigs_initialized = 1;
307 }

309 static void
310 initialize_shell_signals ()
311 {
312  if (interactive)
313  initialize_terminating_signals ();
314 
315 #if defined (JOB_CONTROL) || defined (HAVE_POSIX_SIGNALS)
316  /* All shells use the signal mask they inherit, and pass it along
317  to child processes. Children will never block SIGCHLD, though. */
318  sigemptyset (&top_level_mask);
319  sigprocmask (SIG_BLOCK, (sigset_t *)NULL, &top_level_mask);
320 # if defined (SIGCHLD)
321  if (sigismember (&top_level_mask, SIGCHLD))
322  {
323  sigdelset (&top_level_mask, SIGCHLD);
324  sigprocmask (SIG_SETMASK, &top_level_mask, (sigset_t *)NULL);
325  }
326 # endif
327 #endif /* JOB_CONTROL || HAVE_POSIX_SIGNALS */
328 
329  /* And, some signals that are specifically ignored by the shell. */
330  set_signal_handler (SIGQUIT, SIG_IGN);
331 
332  if (interactive)
333  {
334  set_signal_handler (SIGINT, sigint_sighandler);
335  get_original_signal (SIGTERM);
336  set_signal_handler (SIGTERM, SIG_IGN);
337  set_sigwinch_handler ();
338  }

341 void
342 reset_terminating_signals ()
343 {
344  register int i;
345 #if defined (HAVE_POSIX_SIGNALS)
346  struct sigaction act;
347 #endif
348 
349  if (termsigs_initialized == 0)
350  return;
351 
352 #if defined (HAVE_POSIX_SIGNALS)
353  act.sa_flags = 0;
354  sigemptyset (&act.sa_mask);
355  for (i = 0; i < TERMSIGS_LENGTH; i++)
356  {
357  /* Skip a signal if it's trapped or handled specially, because the
358  trap code will restore the correct value. */
359  if (signal_is_trapped (XSIG (i)) || signal_is_special (XSIG (i)))
360  continue;
361 
362  act.sa_handler = XHANDLER (i);
363  act.sa_flags = XSAFLAGS (i);
364  sigaction (XSIG (i), &act, (struct sigaction *) NULL);
365  }
366 #else /* !HAVE_POSIX_SIGNALS */
367  for (i = 0; i < TERMSIGS_LENGTH; i++)
368  {
369  if (signal_is_trapped (XSIG (i)) || signal_is_special (XSIG (i)))
370  continue;
371 
372  signal (XSIG (i), XHANDLER (i));
373  }
374 #endif /* !HAVE_POSIX_SIGNALS */
375 
376  termsigs_initialized = 0;
377 }

399 /* What to do when we've been interrupted, and it is safe to handle it. */
400 void
401 throw_to_top_level ()
402 {
403  int print_newline = 0;
404 
405  if (interrupt_state)
406  {
407  if (last_command_exit_value < 128)
408  last_command_exit_value = 128 + SIGINT;
409  set_pipestatus_from_exit (last_command_exit_value);
410  print_newline = 1;
411  DELINTERRUPT;
412  }
413 
414  if (interrupt_state)
415  return;
416 
417  last_command_exit_signal = (last_command_exit_value > 128) ?
418  (last_command_exit_value - 128) : 0;
419  last_command_exit_value |= 128;
420  set_pipestatus_from_exit (last_command_exit_value);
421 
422  /* Run any traps set on SIGINT, mostly for interactive shells */
423  if (signal_is_trapped (SIGINT) && signal_is_pending (SIGINT))
424  run_interrupt_trap (1);
425 
426  /* Clean up string parser environment. */
427  while (parse_and_execute_level)
428  parse_and_execute_cleanup (-1);
429 
430  if (running_trap > 0)
431  {
432  run_trap_cleanup (running_trap - 1);
433  running_trap = 0;
434  }
435 
436 #if defined (JOB_CONTROL)
437  give_terminal_to (shell_pgrp, 0);
438 #endif /* JOB_CONTROL */
439 
440  /* This needs to stay because jobs.c:make_child() uses it without resetting
441  the signal mask. */
442  restore_sigmask ();
443 
444  reset_parser ();
445 
446 #if defined (READLINE)
447  if (interactive)
448  bashline_reset ();
449 #endif /* READLINE */
450 
451 #if defined (PROCESS_SUBSTITUTION)
452  unlink_fifo_list ();
453 #endif /* PROCESS_SUBSTITUTION */
454 
455  run_unwind_protects ();
456  loop_level = continuing = breaking = funcnest = 0;
457  executing_list = comsub_ignore_return = return_catch_flag = wait_intr_flag = 0;
458 
459  if (interactive && print_newline)
460  {
461  fflush (stdout);
462  fprintf (stderr, "\n");
463  fflush (stderr);
464  }
465 
466  /* An interrupted `wait' command in a script does not exit the script. */
467  if (interactive || (interactive_shell && !shell_initialized) ||
468  (print_newline && signal_is_trapped (SIGINT)))
469  jump_to_top_level (DISCARD);
470  else
471  jump_to_top_level (EXITPROG);
472 }

492 sighandler
493 termsig_sighandler (sig)
494  int sig;
495 {
496  /* If we get called twice with the same signal before handling it,
497  terminate right away. */
498  if (
499 #ifdef SIGHUP
500  sig != SIGHUP &&
501 #endif
502 #ifdef SIGINT
503  sig != SIGINT &&
504 #endif
505 #ifdef SIGDANGER
506  sig != SIGDANGER &&
507 #endif
508 #ifdef SIGPIPE
509  sig != SIGPIPE &&
510 #endif
511 #ifdef SIGALRM
512  sig != SIGALRM &&
513 #endif
514 #ifdef SIGTERM
515  sig != SIGTERM &&
516 #endif
517 #ifdef SIGXCPU
518  sig != SIGXCPU &&
519 #endif
520 #ifdef SIGXFSZ
521  sig != SIGXFSZ &&
522 #endif
523 #ifdef SIGVTALRM
524  sig != SIGVTALRM &&
525 #endif
526 #ifdef SIGLOST
527  sig != SIGLOST &&
528 #endif
529 #ifdef SIGUSR1
530  sig != SIGUSR1 &&
531 #endif
532 #ifdef SIGUSR2
533  sig != SIGUSR2 &&
534 #endif
535  sig == terminating_signal)
536  terminate_immediately = 1;
537 
538  /* If we are currently handling a terminating signal, we have a couple of
539  choices here. We can ignore this second terminating signal and let the
540  shell exit from the first one, or we can exit immediately by killing
541  the shell with this signal. This code implements the latter; to implement
542  the former, replace the kill_shell(sig) with return. */
543  if (handling_termsig)
544  kill_shell (sig); /* just short-circuit now */
545 
546  terminating_signal = sig;
547 
548  if (terminate_immediately)
549  {
550 #if defined (HISTORY)
551  /* XXX - will inhibit history file being written */
552 # if defined (READLINE)
553  if (interactive_shell == 0 || interactive == 0 || (sig != SIGHUP && sig != SIGTERM) || no_line_editing || (RL_ISSTATE (RL_STATE_READCMD) == 0))
554 # endif
555  history_lines_this_session = 0;
556 #endif
557  terminate_immediately = 0;
558  termsig_handler (sig);
559  }
560 
561 #if defined (READLINE)
562  /* Set the event hook so readline will call it after the signal handlers
563  finish executing, so if this interrupted character input we can get
564  quick response. If readline is active or has modified the terminal we
565  need to set this no matter what the signal is, though the check for
566  RL_STATE_TERMPREPPED is possibly redundant. */
567  if (RL_ISSTATE (RL_STATE_SIGHANDLER) || RL_ISSTATE (RL_STATE_TERMPREPPED))
568  bashline_set_event_hook ();
569 #endif
570 
571  SIGRETURN (0);
572 }

574 void
575 termsig_handler (sig)
576  int sig;
577 {
578  /* Simple semaphore to keep this function from being executed multiple
579  times. Since we no longer are running as a signal handler, we don't
580  block multiple occurrences of the terminating signals while running. */
581  if (handling_termsig)
582  return;
583 
584  handling_termsig = terminating_signal; /* for termsig_sighandler */
585  terminating_signal = 0; /* keep macro from re-testing true. */
586 
587  /* I don't believe this condition ever tests true. */
588  if (sig == SIGINT && signal_is_trapped (SIGINT))
589  run_interrupt_trap (0);
590 
591 #if defined (HISTORY)
592  /* If we don't do something like this, the history will not be saved when
593  an interactive shell is running in a terminal window that gets closed
594  with the `close' button. We can't test for RL_STATE_READCMD because
595  readline no longer handles SIGTERM synchronously. */
596  if (interactive_shell && interactive && (sig == SIGHUP || sig == SIGTERM) && remember_on_history)
597  maybe_save_shell_history ();
598 #endif /* HISTORY */
599 
600  if (this_shell_builtin == read_builtin)
601  read_tty_cleanup ();
602 
603 #if defined (JOB_CONTROL)
604  if (sig == SIGHUP && (interactive || (subshell_environment & (SUBSHELL_COMSUB|SUBSHELL_PROCSUB))))
605  hangup_all_jobs ();
606 
607  if ((subshell_environment & (SUBSHELL_COMSUB|SUBSHELL_PROCSUB)) == 0)
608  end_job_control ();
609 #endif /* JOB_CONTROL */
610 
611 #if defined (PROCESS_SUBSTITUTION)
612  unlink_all_fifos ();
613 # if defined (JOB_CONTROL)
614  procsub_clear ();
615 # endif
616 #endif /* PROCESS_SUBSTITUTION */
617 
618  /* Reset execution context */
619  loop_level = continuing = breaking = funcnest = 0;
620  executing_list = comsub_ignore_return = return_catch_flag = wait_intr_flag = 0;
621 
622  run_exit_trap (); /* XXX - run exit trap possibly in signal context? */
623 
624  kill_shell (sig);
625 }

627 static void
628 kill_shell (sig)
629  int sig;
630 {
631  int i, core;
632  sigset_t mask;
633 
634  /* We don't change the set of blocked signals. If a user starts the shell
635  with a terminating signal blocked, we won't get here (and if by some
636  magic chance we do, we'll exit below). What we do is to restore the
637  top-level signal mask, in case this is called from a terminating signal
638  handler context, in which case the signal is blocked. */
639  restore_sigmask ();
640 
641  set_signal_handler (sig, SIG_DFL);
642 
643  kill (getpid (), sig);
644 
645  if (dollar_dollar_pid != 1)
646  exit (128+sig); /* just in case the kill fails? */
647 
648  /* We get here only under extraordinary circumstances. */
649 
650  /* We are PID 1, and the kill above failed to kill the process. We assume
651  this means that we are running as an init process in a pid namespace
652  on Linux. In this case, we can't send ourselves a fatal signal, so we
653  determine whether or not we should have generated a core dump with the
654  kill call and attempt to trick the kernel into generating one if
655  necessary. */
656  sigprocmask (SIG_SETMASK, (sigset_t *)NULL, &mask);
657  for (i = core = 0; i < TERMSIGS_LENGTH; i++)
658  {
659  set_signal_handler (XSIG (i), SIG_DFL);
660  sigdelset (&mask, XSIG (i));
661  if (sig == XSIG (i))
662  core = XCOREDUMP (i);
663  }
664  sigprocmask (SIG_SETMASK, &mask, (sigset_t *)NULL);
665 
666  if (core)
667  *((volatile unsigned long *) NULL) = 0xdead0000 + sig; /* SIGSEGV */
668 
669  exit (128+sig);
670 }

673 /* What we really do when SIGINT occurs. */
674 sighandler
675 sigint_sighandler (sig)
676  int sig;
677 {
678 #if defined (MUST_REINSTALL_SIGHANDLERS)
679  signal (sig, sigint_sighandler);
680 #endif
681 
682  /* interrupt_state needs to be set for the stack of interrupts to work
683  right. Should it be set unconditionally? */
684  if (interrupt_state == 0)
685  ADDINTERRUPT;
686 
687  /* We will get here in interactive shells with job control active; allow
688  an interactive wait to be interrupted. wait_intr_flag is only set during
689  the execution of the wait builtin and when wait_intr_buf is valid. */
690  if (wait_intr_flag)
691  {
692  last_command_exit_value = 128 + sig;
693  set_pipestatus_from_exit (last_command_exit_value);
694  wait_signal_received = sig;
695  SIGRETURN (0);
696  }
697 
698  /* In interactive shells, we will get here instead of trap_handler() so
699  note that we have a trap pending. */
700  if (signal_is_trapped (sig))
701  set_trap_state (sig);
702 
703  /* This is no longer used, but this code block remains as a reminder. */
704  if (interrupt_immediately)
705  {
706  interrupt_immediately = 0;
707  set_exit_status (128 + sig);
708  throw_to_top_level ();
709  }
