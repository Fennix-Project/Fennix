164 wipe_doMelt

191  if (y[i]+dy >= height) dy = height - y[i];
192  s = &((short *)wipe_scr_end)[i*height+y[i]];
193  d = &((short *)wipe_scr)[y[i]*width+i];
194  idx = 0;
195  for (j=dy;j;j--)
196  {
197  d[idx] = *(s++);

199  }
200  y[i] += dy;
201  s = &((short *)wipe_scr_start)[i*height];
202  d = &((short *)wipe_scr)[y[i]*width+i];
203  idx = 0;
204  for (j=height-y[i];j;j--)
205  {
206  d[idx] = *(s++);

208  }
209  done = false;
210  }
211  }
