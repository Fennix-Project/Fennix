64  SafeFunction bool Task::InvalidPCB(PCB *pcb)
65  {
66  if (!pcb)
67  return true;
68 
69  /* Uninitialized pointers may have uintptr_t max value instead of nullptr. */
70  if (pcb >= (PCB *)(UINTPTR_MAX - 0x1FFE))
71  return true;
72 
73  /* In this section of the memory is reserved by the kernel. */
74  if (pcb < (PCB *)(0xFFFFF))
75  return true;
76 
77  /* Check if it's mapped. */
78  if (!Memory::Virtual().Check((void *)pcb))
79  return true;
80 
81  return false;
82  }

119  SafeFunction bool Task::RemoveProcess(PCB *Process)
120  {
121  if (unlikely(InvalidPCB(Process)))
122  return false;
123 
124  if (Process->State == Terminated)
125  {
126  if (Process->KeepInMemory.load() == true)
127  return false;
128  if (Process->KeepTime > TimeManager->GetCounter())
129  return false;
130 
131  delete Process;
132  return true;
133  }
134 
135  foreach (TCB *Thread in Process->Threads)
136  {
137  if (Thread->State == Terminated)
138  RemoveThread(Thread);
139  }
140 
141  return true;
142  }

343 #elif defined(a32)

361  std::vector<AuxiliaryVector>(), Arch);
362  kthrd->Rename("Main Thread");

364  kproc->Name, kthrd->Name);

366  bool MONITORSupported = false;
367  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
368  {
369  CPU::x86::AMD::CPUID0x00000001 cpuid;

378  }
379

384

389  }
390

392  IdleProcess->ELFSymbolTable = KernelSymbolTable;

424  Task::~Task()
425  {
426  debug("Destructor called");
427  {
428  SmartLock(TaskingLock);
429  foreach (PCB *Process in ProcessList)
430  {
431  foreach (TCB *Thread in Process->Threads)
432  {
433  if (Thread == GetCurrentCPU()->CurrentThread.load() ||
434  Thread == CleanupThread)
435  continue;
436  this->KillThread(Thread, KILL_SCHEDULER_DESTRUCTION);
437  }
438 
439  if (Process == GetCurrentCPU()->CurrentProcess.load())
440  continue;
441  this->KillProcess(Process, KILL_SCHEDULER_DESTRUCTION);
442  }
443  }
444 
445  while (ProcessList.size() > 0)
446  {
447  trace("Waiting for %d processes to terminate", ProcessList.size());
448  int NotTerminated = 0;
449  foreach (PCB *Process in ProcessList)
450  {
451  debug("Process %s(%d) is still running (or waiting to be removed status %#lx)",
452  Process->Name, Process->ID, Process->State);
453  if (Process->State == TaskState::Terminated)
454  continue;
455  NotTerminated++;
456  }
457  if (NotTerminated == 1)
458  break;
459  TaskingScheduler_OneShot(100);
460  }
461 
462  debug("Tasking stopped");
463  }
