86  void *dequeueSingle() {
87  while(true) {
88  bool done;
89  _waitProgressFutex(&done);
90  if(done) {
91  // Reset and enqueue the chunk again.
92  _chunk->progressFutex = 0;

94  _queue->indexQueue[0] = 0;
95  _nextIndex = ((_nextIndex + 1) & kHelHeadMask);
96  _wakeHeadFutex();
97

99  continue;
100  }

103  auto ptr = (char *)_chunk + sizeof(HelChunk) + _lastProgress;
104  auto element = load<HelElement>(ptr);
105  _lastProgress += sizeof(HelElement) + element.length;
106  return ptr + sizeof(HelElement);
107  }

117  void _waitProgressFutex(bool *done) {
118  while(true) {
119  auto futex = __atomic_load_n(&_chunk->progressFutex, __ATOMIC_ACQUIRE);
120  __ensure(!(futex & ~(kHelProgressMask | kHelProgressWaiters | kHelProgressDone)));
121  do {
122  if(_lastProgress != (futex & kHelProgressMask)) {
123  *done = false;
124  return;
125  }else if(futex & kHelProgressDone) {
126  *done = true;
127  return;
128  }
129 
130  if(futex & kHelProgressWaiters)
131  break; // Waiters bit is already set (in a previous iteration).
132  } while(!__atomic_compare_exchange_n(&_chunk->progressFutex, &futex,
133  _lastProgress | kHelProgressWaiters,
134  false, __ATOMIC_ACQUIRE, __ATOMIC_ACQUIRE));
135 
136  HEL_CHECK(helFutexWait(&_chunk->progressFutex,
137  _lastProgress | kHelProgressWaiters, -1));
138  }
