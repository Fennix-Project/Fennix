139 //
140 // R_PointOnSide
141 // Traverse BSP (sub) tree,
142 // check point against partition plane.
143 // Returns side 0 (front) or 1 (back).
144 //
145 int
146 R_PointOnSide
147 ( fixed_t x,
148  fixed_t y,
149  node_t* node )
150 {
151  fixed_t dx;
152  fixed_t dy;
153  fixed_t left;
154  fixed_t right;
155 
156  if (!node->dx)
157  {
158  if (x <= node->x)
159  return node->dy > 0;
160 
161  return node->dy < 0;
162  }
163  if (!node->dy)
164  {
165  if (y <= node->y)
166  return node->dx < 0;
167 
168  return node->dx > 0;
169  }
170 
171  dx = (x - node->x);
172  dy = (y - node->y);
173 
174  // Try to quickly decide by looking at sign bits.
175  if ( (node->dy ^ node->dx ^ dx ^ dy)&0x80000000 )
176  {
177  if ( (node->dy ^ dx) & 0x80000000 )
178  {
179  // (left is negative)
180  return 1;
181  }
182  return 0;
183  }
184 
185  left = FixedMul ( node->dy>>FRACBITS , dx );
186  right = FixedMul ( dy , node->dx>>FRACBITS );
187 
188  if (right < left)
189  {
190  // front side
191  return 0;
192  }
193  // back side
194  return 1;
195 }

261 //
262 // R_PointToAngle
263 // To get a global angle from cartesian coordinates,
264 // the coordinates are flipped until they are in
265 // the first octant of the coordinate system, then
266 // the y (<=x) is scaled and divided by x to get a
267 // tangent (slope) value which is looked up in the
268 // tantoangle[] table.
269 
270 //
271 
272 
273 
274 
275 angle_t
276 R_PointToAngle
277 ( fixed_t x,
278  fixed_t y )
279 {
280  x -= viewx;
281  y -= viewy;
282 
283  if ( (!x) && (!y) )
284  return 0;
285 
286  if (x>= 0)
287  {
288  // x >=0
289  if (y>= 0)
290  {
291  // y>= 0
292 
293  if (x>y)
294  {
295  // octant 0
296  return tantoangle[ SlopeDiv(y,x)];
297  }
298  else
299  {
300  // octant 1
301  return ANG90-1-tantoangle[ SlopeDiv(x,y)];
302  }
303  }
304  else
305  {
306  // y<0
307  y = -y;
308 
309  if (x>y)
310  {
311  // octant 8
312  return -tantoangle[SlopeDiv(y,x)];
313  }
314  else
315  {
316  // octant 7
317  return ANG270+tantoangle[ SlopeDiv(x,y)];
318  }
319  }
320  }
321  else
322  {
323  // x<0
324  x = -x;
325 
326  if (y>= 0)
327  {
328  // y>= 0
329  if (x>y)
330  {
331  // octant 3
332  return ANG180-1-tantoangle[ SlopeDiv(y,x)];
333  }
334  else
335  {
336  // octant 2
337  return ANG90+ tantoangle[ SlopeDiv(x,y)];
338  }
339  }
340  else
341  {
342  // y<0
343  y = -y;
344 
345  if (x>y)
346  {
347  // octant 4
348  return ANG180+tantoangle[ SlopeDiv(y,x)];
349  }
350  else
351  {
352  // octant 5
353  return ANG270-1-tantoangle[ SlopeDiv(x,y)];
354  }
355  }
356  }
357  return 0;
358 }

537 //
538 // R_InitTextureMapping
539 //
540 void R_InitTextureMapping (void)
541 {
542  int i;
543  int x;
544  int t;
545  fixed_t focallength;
546 
547  // Use tangent table to generate viewangletox:
548  // viewangletox will give the next greatest x
549  // after the view angle.
550  //
551  // Calc focallength
552  // so FIELDOFVIEW angles covers SCREENWIDTH.
553  focallength = FixedDiv (centerxfrac,
554  finetangent[FINEANGLES/4+FIELDOFVIEW/2] );
555 
556  for (i=0 ; i<FINEANGLES/2 ; i++)
557  {
558  if (finetangent[i] > FRACUNIT*2)
559  t = -1;
560  else if (finetangent[i] < -FRACUNIT*2)
561  t = viewwidth+1;
562  else
563  {
564  t = FixedMul (finetangent[i], focallength);
565  t = (centerxfrac - t+FRACUNIT-1)>>FRACBITS;
566 
567  if (t < -1)
568  t = -1;
569  else if (t>viewwidth+1)
570  t = viewwidth+1;
571  }
572  viewangletox[i] = t;
573  }
574 
575  // Scan viewangletox[] to generate xtoviewangle[]:
576  // xtoviewangle will give the smallest view angle
577  // that maps to x.
578  for (x=0;x<=viewwidth;x++)
579  {
580  i = 0;
581  while (viewangletox[i]>x)
582  i++;
583  xtoviewangle[x] = (i<<ANGLETOFINESHIFT)-ANG90;
584  }
585 
586  // Take out the fencepost cases from viewangletox.
587  for (i=0 ; i<FINEANGLES/2 ; i++)
588  {
589  t = FixedMul (finetangent[i], focallength);
590  t = centerx - t;
591 
592  if (viewangletox[i] == -1)
593  viewangletox[i] = 0;
594  else if (viewangletox[i] == viewwidth+1)
595  viewangletox[i] = viewwidth;
596  }
597 
598  clipangle = xtoviewangle[0];
599 }
