40 static inline bool CheckTrust(int TrustLevel)
41 {
42  // SmartTimeoutLock(SyscallsLock, 10000); - This is already done in the caller
43  Token token = TaskManager->GetCurrentThread()->Security.UniqueToken;
44  if (TaskManager->GetSecurityManager()->IsTokenTrusted(token, TrustLevel))
45  return true;
46 
47  warn("Thread %s(%lld) tried to access a system call \"%s\" with insufficient trust level",
48  TaskManager->GetCurrentThread()->Name, TaskManager->GetCurrentThread()->ID,
49  KernelSymbolTable->GetSymbolFromAddress((uintptr_t)__builtin_extract_return_addr(__builtin_return_address(0))));
50  debug("Token: token=%#lx, trust=%d", token, TaskManager->GetSecurityManager()->GetTokenTrustLevel(token));
51  return false;
52 }

54 static int sys_exit(SyscallsFrame *Frame, int code)
55 {
56  /* Allow everyone to exit */
57  if (!CheckTrust(TrustedByKernel | Trusted | Untrusted | UnknownTrustLevel))
58  return SYSCALL_ACCESS_DENIED;
59 
60  trace("Userspace thread %s(%lld) exited with code %#llx", TaskManager->GetCurrentThread()->Name, TaskManager->GetCurrentThread()->ID, code);
61  TaskManager->GetCurrentThread()->ExitCode = code;
62  TaskManager->GetCurrentThread()->Status = Tasking::TaskStatus::Terminated;
63  UNUSED(Frame);
64  return SYSCALL_OK;
65 }

114 static uintptr_t sys_kernelctl(SyscallsFrame *Frame, enum KCtl Command, uint64_t Arg1, uint64_t Arg2, uint64_t Arg3, uint64_t Arg4)
115 {
116  /* Only trusted threads can use kernelctl */
117  if (!CheckTrust(TrustedByKernel | Trusted))
118  return SYSCALL_ACCESS_DENIED;
119 
120  switch (Command)
121  {
122  case KCTL_GET_PID:
123  return TaskManager->GetCurrentThread()->Parent->ID;
124  case KCTL_GET_TID:
125  return TaskManager->GetCurrentThread()->ID;
126  case KCTL_GET_PAGE_SIZE:
127  return PAGE_SIZE;
128  case KCTL_IS_CRITICAL:
129  return TaskManager->GetCurrentThread()->Security.IsCritical;
130  case KCTL_REGISTER_ELF_LIB:
131  {
132  char *Identifier = (char *)Arg1;
133  const char *Path = (const char *)Arg2;
134 
135  if (!Identifier || !Path)
136  return SYSCALL_INVALID_ARGUMENT;
137 
138  std::string FullPath = Path;
139  int retries = 0;
140  RetryReadPath:
141  debug("KCTL_REGISTER_ELF_LIB: Trying to open %s", FullPath.c_str());
142  VirtualFileSystem::File f = vfs->Open(FullPath.c_str());
143 
144  if (!f.IsOK())
145  {
146  FullPath.clear();
147  switch (retries)
148  {
149  case 0:
150  FullPath = "/system/lib/";
151  break;
152  case 1:
153  FullPath = "/system/lib64/";
154  break;
155  case 2:
156  FullPath = "/system/";
157  break;
158  case 3:
159  {
160  // TODO: Check process binary path
161  break;
162  }
163  default:
164  {
165  vfs->Close(f);
166  return SYSCALL_INVALID_ARGUMENT;
167  }
168  }
169  FullPath += Path;
170  vfs->Close(f);
171  retries++;
172  goto RetryReadPath;
173  }
174 
175  vfs->Close(f);
176  if (Execute::AddLibrary(Identifier, (void *)f.node->Address, f.node->Length))
177  return SYSCALL_OK;
178  else
179  return SYSCALL_INTERNAL_ERROR;
180  }
181  case KCTL_GET_ELF_LIB_FILE:
182  {
183  char *Identifier = (char *)Arg1;
184  if (!Identifier)
185  return 0;
186 
187  Execute::SharedLibraries lib = Execute::GetLibrary(Identifier);
188  if (!lib.Address)
189  debug("Failed to get library address %#lx", (uintptr_t)lib.Address);
190 
191  debug("Returning library address %#lx (%s)", (uintptr_t)lib.Address, Identifier);
192  return (uintptr_t)lib.Address;
193  }
194  case KCTL_GET_ELF_LIB_MEMORY_IMAGE:
195  {
196  char *Identifier = (char *)Arg1;
197  if (!Identifier)
198  return 0;
199 
200  Execute::SharedLibraries lib = Execute::GetLibrary(Identifier);
201 
202  if (!lib.MemoryImage)
203  debug("Failed to get library memory image %#lx", (uintptr_t)lib.MemoryImage);
204 
205  debug("Returning memory image %#lx (%s)", (uintptr_t)lib.MemoryImage, Identifier);
206  return (uintptr_t)lib.MemoryImage;
207  }
208  case KCTL_GET_FRAMEBUFFER_BUFFER:
209  return r_cst(uint64_t, Display->GetBuffer(0)->Buffer);
210  case KCTL_GET_FRAMEBUFFER_WIDTH:
211  return Display->GetBuffer(0)->Width;
212  case KCTL_GET_FRAMEBUFFER_HEIGHT:
213  return Display->GetBuffer(0)->Height;
214  case KCTL_GET_FRAMEBUFFER_SIZE:
215  return Display->GetBuffer(0)->Size;
216 
217  default:
218  {
219  warn("KernelCTL: Unknown command: %lld", Command);
220  return SYSCALL_INVALID_ARGUMENT;
221  }
222  }
223 
224  UNUSED(Arg1);
225  UNUSED(Arg2);
226  UNUSED(Arg3);
227  UNUSED(Arg4);
228  UNUSED(Frame);
229 }

415 uintptr_t HandleNativeSyscalls(SyscallsFrame *Frame)
416 {
417 #if defined(a64)
418  if (Frame->rax > sizeof(NativeSyscallsTable))
419  {
420  fixme("Syscall %ld not implemented", Frame->rax);
421  return SYSCALL_NOT_IMPLEMENTED;
422  }
423 
424  uintptr_t (*call)(uintptr_t, ...) = reinterpret_cast<uintptr_t (*)(uintptr_t, ...)>(NativeSyscallsTable[Frame->rax]);
425  if (!call)
426  {
427  error("Syscall %#lx failed.", Frame->rax);
428  return SYSCALL_INTERNAL_ERROR;
429  }
430 
431  // debug("[%#lx]->( %#lx %#lx %#lx %#lx %#lx %#lx )",
432  // Frame->rax,
433  // Frame->rdi, Frame->rsi, Frame->rdx, Frame->rcx, Frame->r8, Frame->r9);
434 
435  uintptr_t ret = call((uintptr_t)Frame, Frame->rdi, Frame->rsi, Frame->rdx, Frame->r10, Frame->r8, Frame->r9);
436  Frame->rax = ret;
437  return ret;
438 #elif defined(a32)
439  if (Frame->eax > sizeof(NativeSyscallsTable))
440  {
441  fixme("Syscall %ld not implemented", Frame->eax);
442  return SYSCALL_NOT_IMPLEMENTED;
443  }
444 
445  /* ... */
446 
447  return SYSCALL_INTERNAL_ERROR;
448 #elif defined(aa64)
449  return SYSCALL_INTERNAL_ERROR;
450 #endif
451 }
