134 // Given a time string (for LOOP_START/LOOP_END), parse it and return
135 // the time (in # samples since start of track) it represents.
136 static unsigned int ParseVorbisTime(unsigned int samplerate_hz, char *value)
137 {
138  char *num_start, *p;
139  unsigned int result = 0;
140  char c;
141 
142  if (strchr(value, ':') == NULL)
143  {
144  return atoi(value);
145  }
146 
147  result = 0;
148  num_start = value;
149 
150  for (p = value; *p != '\0'; ++p)
151  {
152  if (*p == '.' || *p == ':')
153  {
154  c = *p; *p = '\0';
155  result = result * 60 + atoi(num_start);
156  num_start = p + 1;
157  *p = c;
158  }
159 
160  if (*p == '.')
161  {
162  return result * samplerate_hz
163  + (unsigned int) (atof(p) * samplerate_hz);
164  }
165  }
166 
167  return (result * 60 + atoi(num_start)) * samplerate_hz;
168 }

170 // Given a vorbis comment string (eg. "LOOP_START=12345"), set fields
171 // in the metadata structure as appropriate.
172 static void ParseVorbisComment(file_metadata_t *metadata, char *comment)
173 {
174  char *eq, *key, *value;
175 
176  eq = strchr(comment, '=');
177 
178  if (eq == NULL)
179  {
180  return;
181  }
182 
183  key = comment;
184  *eq = '\0';
185  value = eq + 1;
186 
187  if (!strcmp(key, LOOP_START_TAG))
188  {
189  metadata->start_time = ParseVorbisTime(metadata->samplerate_hz, value);
190  }
191  else if (!strcmp(key, LOOP_END_TAG))
192  {
193  metadata->end_time = ParseVorbisTime(metadata->samplerate_hz, value);
194  }
195 }

197 // Parse a vorbis comments structure, reading from the given file.
198 static void ParseVorbisComments(file_metadata_t *metadata, FILE *fs)
199 {
200  uint32_t buf;
201  unsigned int num_comments, i, comment_len;
202  char *comment;
203 
204  // We must have read the sample rate already from an earlier header.
205  if (metadata->samplerate_hz == 0)
206  {
207  return;
208  }
209 
210  // Skip the starting part we don't care about.
211  if (fread(&buf, 4, 1, fs) < 1)
212  {
213  return;
214  }
215  if (fseek(fs, LONG(buf), SEEK_CUR) != 0)
216  {
217  return;
218  }
219 
220  // Read count field for number of comments.
221  if (fread(&buf, 4, 1, fs) < 1)
222  {
223  return;
224  }
225  num_comments = LONG(buf);
226 
227  // Read each individual comment.
228  for (i = 0; i < num_comments; ++i)
229  {
230  // Read length of comment.
231  if (fread(&buf, 4, 1, fs) < 1)
232  {
233  return;
234  }
235 
236  comment_len = LONG(buf);
237 
238  // Read actual comment data into string buffer.
239  comment = calloc(1, comment_len + 1);
240  if (comment == NULL
241  || fread(comment, 1, comment_len, fs) < comment_len)
242  {
243  free(comment);
244  break;
245  }
246 
247  // Parse comment string.
248  ParseVorbisComment(metadata, comment);
249  free(comment);
250  }
251 }

253 static void ParseFlacStreaminfo(file_metadata_t *metadata, FILE *fs)
254 {
255  byte buf[34];
256 
257  // Read block data.
258  if (fread(buf, sizeof(buf), 1, fs) < 1)
259  {
260  return;
261  }
262 
263  // We only care about sample rate and song length.
264  metadata->samplerate_hz = (buf[10] << 12) | (buf[11] << 4)
265  | (buf[12] >> 4);
266  // Song length is actually a 36 bit field, but 32 bits should be
267  // enough for everybody.
268  //metadata->song_length = (buf[14] << 24) | (buf[15] << 16)
269  // | (buf[16] << 8) | buf[17];
270 }

272 static void ParseFlacFile(file_metadata_t *metadata, FILE *fs)
273 {
274  byte header[4];
275  unsigned int block_type;
276  size_t block_len;
277  boolean last_block;
278 
279  for (;;)
280  {
281  long pos = -1;
282 
283  // Read METADATA_BLOCK_HEADER:
284  if (fread(header, 4, 1, fs) < 1)
285  {
286  return;
287  }
288 
289  block_type = header[0] & ~0x80;
290  last_block = (header[0] & 0x80) != 0;
291  block_len = (header[1] << 16) | (header[2] << 8) | header[3];
292 
293  pos = ftell(fs);
294  if (pos < 0)
295  {
296  return;
297  }
298 
299  if (block_type == FLAC_STREAMINFO)
300  {
301  ParseFlacStreaminfo(metadata, fs);
302  }
303  else if (block_type == FLAC_VORBIS_COMMENT)
304  {
305  ParseVorbisComments(metadata, fs);
306  }
307 
308  if (last_block)
309  {
310  break;
311  }
312 
313  // Seek to start of next block.
314  if (fseek(fs, pos + block_len, SEEK_SET) != 0)
315  {
316  return;
317  }
318  }
319 }

321 static void ParseOggIdHeader(file_metadata_t *metadata, FILE *fs)
322 {
323  byte buf[21];
324 
325  if (fread(buf, sizeof(buf), 1, fs) < 1)
326  {
327  return;
328  }
329 
330  metadata->samplerate_hz = (buf[8] << 24) | (buf[7] << 16)
331  | (buf[6] << 8) | buf[5];
332 }

334 static void ParseOggFile(file_metadata_t *metadata, FILE *fs)
335 {
336  byte buf[7];
337  unsigned int offset;
338 
339  // Scan through the start of the file looking for headers. They
340  // begin '[byte]vorbis' where the byte value indicates header type.
341  memset(buf, 0, sizeof(buf));
342 
343  for (offset = 0; offset < 100 * 1024; ++offset)
344  {
345  // buf[] is used as a sliding window. Each iteration, we
346  // move the buffer one byte to the left and read an extra
347  // byte onto the end.
348  memmove(buf, buf + 1, sizeof(buf) - 1);
349 
350  if (fread(&buf[6], 1, 1, fs) < 1)
351  {
352  return;
353  }
354 
355  if (!memcmp(buf + 1, "vorbis", 6))
356  {
357  switch (buf[0])
358  {
359  case OGG_ID_HEADER:
360  ParseOggIdHeader(metadata, fs);
361  break;
362  case OGG_COMMENT_HEADER:
363  ParseVorbisComments(metadata, fs);
364  break;
365  default:
366  break;
367  }
368  }
369  }
370 }

372 static void ReadLoopPoints(char *filename, file_metadata_t *metadata)
373 {
374  FILE *fs;
375  char header[4];
376 
377  metadata->valid = false;
378  metadata->samplerate_hz = 0;
379  metadata->start_time = 0;
380  metadata->end_time = -1;
381 
382  fs = fopen(filename, "r");
383 
384  if (fs == NULL)
385  {
386  return;
387  }
388 
389  // Check for a recognized file format; use the first four bytes
390  // of the file.
391 
392  if (fread(header, 4, 1, fs) < 1)
393  {
394  fclose(fs);
395  return;
396  }
397 
398  if (memcmp(header, FLAC_HEADER, 4) == 0)
399  {
400  ParseFlacFile(metadata, fs);
401  }
402  else if (memcmp(header, OGG_HEADER, 4) == 0)
403  {
404  ParseOggFile(metadata, fs);
405  }
406 
407  fclose(fs);
408 
409  // Only valid if at the very least we read the sample rate.
410  metadata->valid = metadata->samplerate_hz > 0;
411 }

413 // Given a MUS lump, look up a substitute MUS file to play instead
414 // (or NULL to just use normal MIDI playback).
415 
416 static char *GetSubstituteMusicFile(void *data, size_t data_len)
417 {
418  sha1_context_t context;
419  sha1_digest_t hash;
420  char *filename;
421  int i;
422 
423  // Don't bother doing a hash if we're never going to find anything.
424  if (subst_music_len == 0)
425  {
426  return NULL;
427  }
428 
429  SHA1_Init(&context);
430  SHA1_Update(&context, data, data_len);
431  SHA1_Final(hash, &context);
432 
433  // Look for a hash that matches.
434  // The substitute mapping list can (intentionally) contain multiple
435  // filename mappings for the same hash. This allows us to try
436  // different files and fall back if our first choice isn't found.
437 
438  filename = NULL;
439 
440  for (i = 0; i < subst_music_len; ++i)
441  {
442  if (memcmp(hash, subst_music[i].hash, sizeof(hash)) == 0)
443  {
444  filename = subst_music[i].filename;
445 
446  // If the file exists, then use this file in preference to
447  // any fallbacks. But we always return a filename if it's
448  // in the list, even if it's just so we can print an error
449  // message to the user saying it doesn't exist.
450  if (M_FileExists(filename))
451  {
452  break;
453  }
454  }
455  }
456 
457  return filename;
458 }

470 static int ParseHexDigit(char c)
471 {
472  c = tolower(c);
473 
474  if (c >= '0' && c <= '9')
475  {
476  return c - '0';
477  }
478  else if (c >= 'a' && c <= 'f')
479  {
480  return 10 + (c - 'a');
481  }
482  else
483  {
484  return -1;
485  }
486 }

488 static char *GetFullPath(char *base_filename, char *path)
489 {
490  char *basedir, *result;
491  char *p;
492 
493  // Starting with directory separator means we have an absolute path,
494  // so just return it.
495  if (path[0] == DIR_SEPARATOR)
496  {
497  return strdup(path);
498  }
499 
500 #ifdef _WIN32
501  // d:\path\...
502  if (isalpha(path[0]) && path[1] == ':' && path[2] == DIR_SEPARATOR)
503  {
504  return strdup(path);
505  }
506 #endif
507 
508  // Paths in the substitute filenames can contain Unix-style /
509  // path separators, but we should convert this to the separator
510  // for the native platform.
511  path = M_StringReplace(path, "/", DIR_SEPARATOR_S);
512 
513  // Copy config filename and cut off the filename to just get the
514  // parent dir.
515  basedir = strdup(base_filename);
516  p = strrchr(basedir, DIR_SEPARATOR);
517  if (p != NULL)
518  {
519  p[1] = '\0';
520  result = M_StringJoin(basedir, path, NULL);
521  }
522  else
523  {
524  result = strdup(path);
525  }
526  free(basedir);
527  free(path);
528 
529  return result;
530 }

532 // Parse a line from substitute music configuration file; returns error
533 // message or NULL for no error.
534 
535 static char *ParseSubstituteLine(char *filename, char *line)
536 {
537  subst_music_t subst;
538  char *p;
539  int hash_index;
540 
541  // Strip out comments if present.
542  p = strchr(line, '#');
543  if (p != NULL)
544  {
545  while (p > line && isspace(*(p - 1)))
546  {
547  --p;
548  }
549  *p = '\0';
550  }
551 
552  // Skip leading spaces.
553  for (p = line; *p != '\0' && isspace(*p); ++p);
554 
555  // Empty line? This includes comment lines now that comments have
556  // been stripped.
557  if (*p == '\0')
558  {
559  return NULL;
560  }
561 
562  // Read hash.
563  hash_index = 0;
564  while (*p != '\0' && *p != '=' && !isspace(*p))
565  {
566  int d1, d2;
567 
568  d1 = ParseHexDigit(p[0]);
569  d2 = ParseHexDigit(p[1]);
570 
571  if (d1 < 0 || d2 < 0)
572  {
573  return "Invalid hex digit in SHA1 hash";
574  }
575  else if (hash_index >= sizeof(sha1_digest_t))
576  {
577  return "SHA1 hash too long";
578  }
579 
580  subst.hash[hash_index] = (d1 << 4) | d2;
581  ++hash_index;
582 
583  p += 2;
584  }
585 
586  if (hash_index != sizeof(sha1_digest_t))
587  {
588  return "SHA1 hash too short";
589  }
590 
591  // Skip spaces.
592  for (; *p != '\0' && isspace(*p); ++p);
593 
594  if (*p != '=')
595  {
596  return "Expected '='";
597  }
598 
599  ++p;
600 
601  // Skip spaces.
602  for (; *p != '\0' && isspace(*p); ++p);
603 
604  // We're now at the filename. Cut off trailing space characters.
605  while (strlen(p) > 0 && isspace(p[strlen(p) - 1]))
606  {
607  p[strlen(p) - 1] = '\0';
608  }
609 
610  if (strlen(p) == 0)
611  {
612  return "No filename specified for music substitution";
613  }
614 
615  // Expand full path and add to our database of substitutes.
616  subst.filename = GetFullPath(filename, p);
617  AddSubstituteMusic(&subst);
618 
619  return NULL;
620 }

698 // Returns true if the given lump number is a music lump that should

703 static boolean IsMusicLump(int lumpnum)
704 {
705  byte *data;
706  boolean result;
707 
708  if (W_LumpLength(lumpnum) < 4)
709  {
710  return false;
711  }
712 
713  data = W_CacheLumpNum(lumpnum, PU_STATIC);
714 
715  result = memcmp(data, MUS_HEADER_MAGIC, 4) == 0

717 
718  W_ReleaseLumpNum(lumpnum);
719 
720  return result;

723 // Dump an example config file containing checksums for all MIDI music
724 // found in the WAD directory.
725 
726 static void DumpSubstituteConfig(char *filename)
727 {
728  sha1_context_t context;
729  sha1_digest_t digest;
730  char name[9];
731  byte *data;
732  FILE *fs;
733  int lumpnum, h;
734 
735  fs = fopen(filename, "w");
736 
737  if (fs == NULL)
738  {
739  I_Error("Failed to open %s for writing", filename);
740  return;
741  }
742 
743  fprintf(fs, "# Example %s substitute MIDI file.\n\n", PACKAGE_NAME);
744  fprintf(fs, "# SHA1 hash = filename\n");
745 
746  for (lumpnum = 0; lumpnum < numlumps; ++lumpnum)
747  {
748  strncpy(name, lumpinfo[lumpnum].name, 8);
749  name[8] = '\0';
750 
751  if (!IsMusicLump(lumpnum))
752  {
753  continue;
754  }
755 
756  // Calculate hash.
757  data = W_CacheLumpNum(lumpnum, PU_STATIC);
758  SHA1_Init(&context);
759  SHA1_Update(&context, data, W_LumpLength(lumpnum));
760  SHA1_Final(digest, &context);
761  W_ReleaseLumpNum(lumpnum);
762 
763  // Print line.
764  for (h = 0; h < sizeof(sha1_digest_t); ++h)
765  {
766  fprintf(fs, "%02x", digest[h]);
767  }
768 
769  fprintf(fs, " = %s.ogg\n", name);
770  }
771 
772  fprintf(fs, "\n");
773  fclose(fs);
774 
775  printf("Substitute MIDI config file written to %s.\n", filename);
776  I_Quit();
777 }
