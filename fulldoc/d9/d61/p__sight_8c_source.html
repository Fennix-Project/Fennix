43 //
44 // P_DivlineSide
45 // Returns side 0 (front), 1 (back), or 2 (on).
46 //
47 int
48 P_DivlineSide
49 ( fixed_t x,
50  fixed_t y,
51  divline_t* node )
52 {
53  fixed_t dx;
54  fixed_t dy;
55  fixed_t left;
56  fixed_t right;
57 
58  if (!node->dx)
59  {
60  if (x==node->x)
61  return 2;
62 
63  if (x <= node->x)
64  return node->dy > 0;
65 
66  return node->dy < 0;
67  }
68 
69  if (!node->dy)
70  {
71  if (x==node->y)
72  return 2;
73 
74  if (y <= node->y)
75  return node->dx < 0;
76 
77  return node->dx > 0;
78  }
79 
80  dx = (x - node->x);
81  dy = (y - node->y);
82 
83  left = (node->dy>>FRACBITS) * (dx>>FRACBITS);
84  right = (dy>>FRACBITS) * (node->dx>>FRACBITS);
85 
86  if (right < left)
87  return 0; // front side
88 
89  if (left == right)
90  return 2;
91  return 1; // back side
92 }

95 //
96 // P_InterceptVector2
97 // Returns the fractional intercept point
98 // along the first divline.
99 // This is only called by the addthings and addlines traversers.
100 //
101 fixed_t
102 P_InterceptVector2
103 ( divline_t* v2,
104  divline_t* v1 )
105 {
106  fixed_t frac;
107  fixed_t num;
108  fixed_t den;
109 
110  den = FixedMul (v1->dy>>8,v2->dx) - FixedMul(v1->dx>>8,v2->dy);
111 
112  if (den == 0)
113  return 0;
114  // I_Error ("P_InterceptVector: parallel");
115 
116  num = FixedMul ( (v1->x - v2->x)>>8 ,v1->dy) +
117  FixedMul ( (v2->y - v1->y)>>8 , v1->dx);
118  frac = FixedDiv (num , den);
119 
120  return frac;
121 }

123 //
124 // P_CrossSubsector
125 // Returns true
126 // if strace crosses the given subsector successfully.
127 //
128 boolean P_CrossSubsector (int num)
129 {
130  seg_t* seg;
131  line_t* line;
132  int s1;
133  int s2;
134  int count;
135  subsector_t* sub;
136  sector_t* front;
137  sector_t* back;
138  fixed_t opentop;
139  fixed_t openbottom;
140  divline_t divl;
141  vertex_t* v1;
142  vertex_t* v2;
143  fixed_t frac;
144  fixed_t slope;
145 
146 #ifdef RANGECHECK
147  if (num>=numsubsectors)
148  I_Error ("P_CrossSubsector: ss %i with numss = %i",
149  num,
150  numsubsectors);
151 #endif
152 
153  sub = &subsectors[num];
154 
155  // check lines
156  count = sub->numlines;
157  seg = &segs[sub->firstline];
158 
159  for ( ; count ; seg++, count--)
160  {
161  line = seg->linedef;
162 
163  // allready checked other side?
164  if (line->validcount == validcount)
165  continue;
166 
167  line->validcount = validcount;
168 
169  v1 = line->v1;
170  v2 = line->v2;
171  s1 = P_DivlineSide (v1->x,v1->y, &strace);
172  s2 = P_DivlineSide (v2->x, v2->y, &strace);
173 
174  // line isn't crossed?
175  if (s1 == s2)
176  continue;
177 
178  divl.x = v1->x;
179  divl.y = v1->y;
180  divl.dx = v2->x - v1->x;
181  divl.dy = v2->y - v1->y;
182  s1 = P_DivlineSide (strace.x, strace.y, &divl);
183  s2 = P_DivlineSide (t2x, t2y, &divl);
184 
185  // line isn't crossed?
186  if (s1 == s2)
187  continue;
188 
189  // Backsector may be NULL if this is an "impassible
190  // glass" hack line.
191 
192  if (line->backsector == NULL)
193  {
194  return false;
195  }
196 
197  // stop because it is not two sided anyway
198  // might do this after updating validcount?
199  if ( !(line->flags & ML_TWOSIDED) )
200  return false;
201 
202  // crosses a two sided line
203  front = seg->frontsector;
204  back = seg->backsector;
205 
206  // no wall to block sight with?
207  if (front->floorheight == back->floorheight
208  && front->ceilingheight == back->ceilingheight)
209  continue;
210 
211  // possible occluder
212  // because of ceiling height differences
213  if (front->ceilingheight < back->ceilingheight)
214  opentop = front->ceilingheight;
215  else
216  opentop = back->ceilingheight;
217 
218  // because of ceiling height differences
219  if (front->floorheight > back->floorheight)
220  openbottom = front->floorheight;
221  else
222  openbottom = back->floorheight;
223 
224  // quick test for totally closed doors
225  if (openbottom >= opentop)
226  return false; // stop
227 
228  frac = P_InterceptVector2 (&strace, &divl);
229 
230  if (front->floorheight != back->floorheight)
231  {
232  slope = FixedDiv (openbottom - sightzstart , frac);
233  if (slope > bottomslope)
234  bottomslope = slope;
235  }
236 
237  if (front->ceilingheight != back->ceilingheight)
238  {
239  slope = FixedDiv (opentop - sightzstart , frac);
240  if (slope < topslope)
241  topslope = slope;
242  }
243 
244  if (topslope <= bottomslope)
245  return false; // stop
246  }
247  // passed the subsector ok
248  return true;
249 }

253 //
254 // P_CrossBSPNode
255 // Returns true
256 // if strace crosses the given node successfully.
257 //
258 boolean P_CrossBSPNode (int bspnum)
259 {
260  node_t* bsp;
261  int side;
262 
263  if (bspnum & NF_SUBSECTOR)
264  {
265  if (bspnum == -1)
266  return P_CrossSubsector (0);
267  else
268  return P_CrossSubsector (bspnum&(~NF_SUBSECTOR));
269  }
270 
271  bsp = &nodes[bspnum];
272 
273  // decide which side the start point is on
274  side = P_DivlineSide (strace.x, strace.y, (divline_t *)bsp);
275  if (side == 2)
276  side = 0; // an "on" should cross both sides
277 
278  // cross the starting side
279  if (!P_CrossBSPNode (bsp->children[side]) )
280  return false;
281 
282  // the partition plane is crossed here
283  if (side == P_DivlineSide (t2x, t2y,(divline_t *)bsp))
284  {
285  // the line doesn't touch the other side
286  return true;
287  }
288 
289  // cross the ending side
290  return P_CrossBSPNode (bsp->children[side^1]);
291 }

294 //
295 // P_CheckSight
296 // Returns true
297 // if a straight line between t1 and t2 is unobstructed.
298 // Uses REJECT.
299 //
300 boolean
301 P_CheckSight
302 ( mobj_t* t1,
303  mobj_t* t2 )
304 {
305  int s1;
306  int s2;
307  int pnum;
308  int bytenum;
309  int bitnum;
310 
311  // First check for trivial rejection.
312 
313  // Determine subsector entries in REJECT table.
314  s1 = (t1->subsector->sector - sectors);
315  s2 = (t2->subsector->sector - sectors);
316  pnum = s1*numsectors + s2;
317  bytenum = pnum>>3;
318  bitnum = 1 << (pnum&7);
319 
320  // Check in REJECT table.
321  if (rejectmatrix[bytenum]&bitnum)
322  {
323  sightcounts[0]++;
324 
325  // can't possibly be connected
326  return false;
327  }
328 
329  // An unobstructed LOS is possible.
330  // Now look from eyes of t1 to any part of t2.
331  sightcounts[1]++;
332 
333  validcount++;
334 
335  sightzstart = t1->z + t1->height - (t1->height>>2);
336  topslope = (t2->z+t2->height) - sightzstart;
337  bottomslope = (t2->z) - sightzstart;
338 
339  strace.x = t1->x;
340  strace.y = t1->y;
341  t2x = t2->x;
342  t2y = t2->y;
343  strace.dx = t2->x - t1->x;
344  strace.dy = t2->y - t1->y;
345 
346  // the head node is the last node output
347  return P_CrossBSPNode (numnodes-1);
348 }
