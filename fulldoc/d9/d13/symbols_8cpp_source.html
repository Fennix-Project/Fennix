101  __no_sanitize("alignment") void Symbols::AddSymbolInfoFromGRUB(uint64_t Num,
102  uint64_t EntSize,
103  __unused uint64_t Shndx,
104  uintptr_t Sections)
105  {
106  char *sections = r_cst(char *, Sections);
107 
108  Elf_Sym *Symbols = nullptr;
109  uint8_t *StringAddress = nullptr;
110 
111 #if defined(a64) || defined(aa64)
112  Elf64_Xword SymbolSize = 0;
113  // Elf64_Xword StringSize = 0;
114 #elif defined(a32)
115  Elf32_Word SymbolSize = 0;
116  // Elf32_Word StringSize = 0;
117 #endif
118  size_t TotalEntries = 0;
119 
120  for (size_t i = 0; i < Num; ++i)
121  {
122  Elf_Shdr *sym = (Elf_Shdr *)&sections[EntSize * i];
123  Elf_Shdr *str = (Elf_Shdr *)&sections[EntSize * sym->sh_link];
124 
125  if (sym->sh_type == SHT_SYMTAB &&
126  str->sh_type == SHT_STRTAB)
127  {
128  Symbols = (Elf_Sym *)sym->sh_addr;
129  StringAddress = (uint8_t *)str->sh_addr;
130  SymbolSize = (int)sym->sh_size;
131  // StringSize = (int)str->sh_size;
132  // TotalEntries = Section.sh_size / sizeof(Elf64_Sym)
133  TotalEntries = sym->sh_size / sym->sh_entsize;
134  trace("Symbol table found, %d entries",
135  SymbolSize / sym->sh_entsize);
136  UNUSED(SymbolSize);
137  break;
138  }
139  }
140 
141  if (Symbols != nullptr && StringAddress != nullptr)
142  {
143  size_t Index, MinimumIndex;
144  for (size_t i = 0; i < TotalEntries - 1; i++)
145  {
146  MinimumIndex = i;
147  for (Index = i + 1; Index < TotalEntries; Index++)
148  if (Symbols[Index].st_value < Symbols[MinimumIndex].st_value)
149  MinimumIndex = Index;
150  Elf_Sym tmp = Symbols[MinimumIndex];
151  Symbols[MinimumIndex] = Symbols[i];
152  Symbols[i] = tmp;
153  }
154 
155  while (Symbols[0].st_value == 0)
156  {
157  if (TotalEntries <= 0)
158  break;
159  Symbols++;
160  TotalEntries--;
161  }
162 
163  if (TotalEntries <= 0)
164  {
165  error("Symbol table is empty");
166  return;
167  }
168 
169  trace("Symbol table loaded, %d entries (%ld KiB)",
170  TotalEntries, TO_KiB(TotalEntries * sizeof(SymbolTable)));
171  Elf_Sym *sym;
172  const char *name;
173  Memory::Virtual vmm;
174  for (size_t i = 0, g = TotalEntries; i < g; i++)
175  {
176  sym = &Symbols[i];
177  if (!vmm.Check(sym))
178  {
179  error("Symbol %d has invalid address %#lx!",
180  i, sym);
181  debug("Base: %#lx, Symbols[%d]: %#lx, Symbols[%d]: %#lx",
182  Symbols,
183  i - 1, &Symbols[i - 1],
184  i + 1, &Symbols[i + 1]);
185  continue;
186  }
187 
188  name = (const char *)&StringAddress[Symbols[i].st_name];
189  if (!vmm.Check((void *)name))
190  {
191  error("String %d has invalid address %#lx!",
192  i, name);
193  debug("st_name: %d, st_info: %d, st_other: %d, st_shndx: %d, st_value: %d, st_size: %d",
194  sym->st_name, sym->st_info, sym->st_other,
195  sym->st_shndx, sym->st_value, sym->st_size);
196  continue;
197  }
198 
199  if (strlen(name) == 0)
200  continue;
201  SymbolTable tbl{};
202  tbl.Address = sym->st_value;
203  tbl.FunctionName = new char[strlen(name) + 1];
204  strcpy(tbl.FunctionName, name);
205  this->SymTable.push_back(tbl);
206  this->SymbolTableExists = true;

234  Header->e_ident[1] != 'E' &&
