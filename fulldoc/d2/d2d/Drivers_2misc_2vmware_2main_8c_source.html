203 bool __CheckHypervisorBit()
204 {
205  uint32_t eax, ebx, ecx, edx;

214  const char hv[13] = {0};

216  __cpuid(0x40000000, &eax, &ebx, &ecx, &edx);

219  *(uint32_t *)(hv + 4) = ecx;
220  *(uint32_t *)(hv + 8) = edx;

282 static int OpenMessageChannel(ToolboxContext *ctx, uint32_t Protocol)
283 {
284  uintptr_t ax, bx, cx, dx, si = 0, di = 0;
285 
286  VM_PORT(MESSAGE_OPEN_CHANNEL,
287  (Protocol | FLAG_COOKIE), si, di,
288  0, VMWARE_MAGIC,
289  ax, bx, cx, dx, si, di);
290 
291  if ((HighWord(cx) & STATUS_SUCCESS) == 0)
292  {
293  KernelLog("Failed to open message channel %#lx", Protocol);
294  return -EINVAL;
295  }
296 
297  DebugLog("Opened message channel %d (Protocol: %#lx)",
298  HighWord(dx), Protocol);
299  ctx->ChannelID = (uint16_t)HighWord(dx);
300  ctx->CookieHigh = si;
301  ctx->CookieLow = di;
302  return 0;
303 }

320 static uintptr_t MessageSendHB(ToolboxContext *ctx,
321  const char *Message)
322 {
323  uintptr_t ax, bx, cx, dx,
324  si = (uintptr_t)Message,
325  di = ctx->CookieLow,
326  bp = ctx->CookieHigh;
327 
328  uint32_t ChannelID = ctx->ChannelID << 16;
329  size_t Size = StringLength(Message);
330 
331  VM_PORT_HB_OUT((STATUS_SUCCESS << 16) | MESSAGE_HB_MSG,
332  Size, si, di,
333  VMWARE_HYPERVISOR_HB | ChannelID | VMWARE_HYPERVISOR_OUT,
334  VMWARE_MAGIC, bp,
335  ax, bx, cx, dx, si, di);
336 
337  return bx;
338 }

340 static uintptr_t MessageSendLB(ToolboxContext *ctx,
341  const char *Message)
342 {
343  uintptr_t ax, bx,
344  cx = STATUS_SUCCESS << 16,
345  dx, si, di;
346 
347  size_t Size = StringLength(Message);
348  while (Size &&
349  (HighWord(cx) & STATUS_SUCCESS))
350  {
351  uint32_t TotalBytes = MIN((uint32_t)Size, (uint32_t)4);
352  uint32_t Word = 0;
353  MemoryCopy(&Word, Message, TotalBytes);
354  Message += TotalBytes;
355 
356  si = ctx->CookieHigh;
357  di = ctx->CookieLow;
358 
359  VM_PORT(MESSAGE_SEND_PAYLOAD,
360  Word, si, di,
361  ctx->ChannelID << 16,
362  VMWARE_MAGIC,
363  ax, bx, cx, dx, si, di);
364  }
365 
366  return cx;
367 }

369 static uintptr_t MessageReceiveHB(ToolboxContext *ctx,
370  char *Buffer,
371  size_t BufferSize)
372 {
373  uintptr_t ax, bx, cx, dx,
374  si = ctx->CookieHigh,
375  di = (uintptr_t)Buffer,
376  bp = ctx->CookieLow;
377 
378  uint32_t ChannelID = ctx->ChannelID << 16;
379 
380  VM_PORT_HB_IN((STATUS_SUCCESS << 16) | MESSAGE_HB_MSG,
381  BufferSize, si, di,
382  VMWARE_HYPERVISOR_HB | ChannelID | VMWARE_HYPERVISOR_OUT,
383  VMWARE_MAGIC, bp,
384  ax, bx, cx, dx, si, di);
385 
386  return bx;

389 static uintptr_t MessageReceiveLB(ToolboxContext *ctx,
390  char *Buffer,
391  size_t BufferSize)
392 {
393  uintptr_t ax, bx,
394  cx = STATUS_SUCCESS << 16,
395  dx, si, di;
396 
397  while (BufferSize)
398  {
399  uint32_t TotalBytes = MIN((uint32_t)BufferSize, (uint32_t)4);
400 
401  si = ctx->CookieHigh;
402  di = ctx->CookieLow;
403 
404  VM_PORT(MESSAGE_RECV_PAYLOAD,
405  STATUS_SUCCESS, si, di,
406  ctx->ChannelID << 16,
407  VMWARE_MAGIC,
408  ax, bx, cx, dx, si, di);
409 
410  if ((HighWord(cx) & STATUS_SUCCESS) == 0)
411  break;
412 
413  MemoryCopy(Buffer, &bx, TotalBytes);
414  Buffer += TotalBytes;
415  BufferSize -= TotalBytes;
416  }
417 
418  return cx;
419 }

421 static int MessageSend(ToolboxContext *ctx,
422  const char *Message)
423 {
424  uintptr_t ax, bx, cx, dx, si, di;
425  size_t Size = StringLength(Message);
426  int Retries = 0;
427 
428  while (Retries < 2)
429  {
430  Retries++;
431  si = ctx->CookieHigh;
432  di = ctx->CookieLow;
433 
434  VM_PORT(MESSAGE_SEND_SIZE,
435  Size, si, di,
436  ctx->ChannelID << 16,
437  VMWARE_MAGIC,
438  ax, bx, cx, dx, si, di);
439 
440  if ((HighWord(cx) & STATUS_SUCCESS) == 0)
441  {
442  KernelLog("Failed to send message size for \"%s\": %d",
443  Message, cx);
444  return -EINVAL;
445  }
446 
447  bool HighBand = (HighWord(cx) & STATUS_HB) != 0;
448  if (HighBand)
449  bx = MessageSendHB(ctx, Message);
450  else
451  bx = MessageSendLB(ctx, Message);
452 
453  int status = HighWord(bx);
454 
455  if ((status & STATUS_SUCCESS) != 0)
456  {
457  DebugLog("Message \"%s\" sent", Message);
458  return 0;
459  }
460  else if ((status & STATUS_CPT) == 0)
461  {
462  KernelLog("Checkpoint occurred for message \"%s\"", Message);
463  continue;
464  }
465  else
466  break;
467  }
468 
469  KernelLog("Failed to send message \"%s\": %#lx", Message, bx);
470  return -EINVAL;
471 }

473 static int MessageReceive(ToolboxContext *ctx,
474  char **Buffer,
475  size_t *BufferSize)
476 {
477  uintptr_t ax, bx, cx, dx, si, di;
478  int Retries = 0;
479 
480  *Buffer = NULL;
481  *BufferSize = 0;
482 
483  char *ReplyBuf = NULL;
484  size_t ReplyBufPages = 0;
485  size_t ReplySize = 0;
486  while (Retries < 2)
487  {
488  Retries++;
489  si = ctx->CookieHigh;
490  di = ctx->CookieLow;
491 
492  VM_PORT(MESSAGE_RECV_SIZE,
493  0, si, di,
494  ctx->ChannelID << 16,
495  VMWARE_MAGIC,
496  ax, bx, cx, dx, si, di);
497 
498  if ((HighWord(cx) & STATUS_SUCCESS) == 0)
499  {
500  KernelLog("Failed to receive message size: %d", cx);
501  return -EINVAL;
502  }
503  else if ((HighWord(cx) & STATUS_DORECV) == 0)
504  {
505  DebugLog("No message to receive");
506  return -EAGAIN;
507  }
508 
509  ReplySize = bx;
510 
511  if (ReplyBuf != NULL)
512  FreeMemory(ReplyBuf, ReplyBufPages);
513  ReplyBufPages = ReplySize / 0x1000 + 1;
514  ReplyBuf = AllocateMemory(ReplyBufPages);
515 
516  bool HighBand = (HighWord(cx) & STATUS_HB) != 0;
517  if (HighBand)
518  bx = MessageReceiveHB(ctx, ReplyBuf, ReplySize);
519  else
520  bx = MessageReceiveLB(ctx, ReplyBuf, ReplySize);
521 
522  if ((HighWord(bx) & STATUS_SUCCESS) == 0)
523  {
524  if ((HighWord(bx) & STATUS_CPT) == 0)
525  {
526  KernelLog("Checkpoint occurred for message payload");
527  continue;
528  }
529 
530  KernelLog("Failed to receive message payload: %d", HighWord(bx));
531  FreeMemory(ReplyBuf, ReplyBufPages);
532  return -EINVAL;
533  }
534 
535  ReplyBuf[ReplySize] = '\0';
536 
537  si = ctx->CookieHigh;
538  di = ctx->CookieLow;
539 
540  VM_PORT(MESSAGE_RECV_STATUS,
541  STATUS_SUCCESS, si, di,
542  ctx->ChannelID << 16,
543  VMWARE_MAGIC,
544  ax, bx, cx, dx, si, di);
545 
546  if ((HighWord(cx) & STATUS_SUCCESS) == 0)
547  {
548  if ((HighWord(cx) & STATUS_CPT) == 0)
549  {
550  KernelLog("Retrying message receive");
551  continue;
552  }
553 
554  KernelLog("Failed to receive message status: %d", HighWord(cx));
555  FreeMemory(ReplyBuf, ReplyBufPages);
556  return -EINVAL;
557  }
558 
559  break;
560  }
561 
562  if (ReplyBuf == NULL)
563  {
564  KernelLog("Failed to receive message");
565  return -EINVAL;
566  }
567 
568  *Buffer = ReplyBuf;
569  *BufferSize = ReplySize;
570  DebugLog("Received message \"%s\"", ReplyBuf);
571  return 0;
572 }

600 static int DisplayGetSize(ToolboxContext *ctx)
601 {
602  if (ctx->TCLOChannel != -1)
603  MessageClose(ctx);
604  OpenMessageChannel(ctx, MESSAGE_TCLO);
605 
606  char EmptyBuffer[256] = {'\0'};
607  MessageSend(ctx, EmptyBuffer);
608 
609  while (true)
610  {
611  /* FIXME: buf memory leak */
612  char *buf;
613  size_t len;
614 
615  int status = MessageReceive(ctx, &buf, &len);
616  if (status == -EAGAIN)
617  {
618  Sleep(1000);
619  continue;
620  }
621  else if (status < 0)
622  {
623  KernelLog("Failed to receive message");
624  return 1;
625  }
626 
627  buf[StringLength(buf)] = '\0';
628  if (MsgEqual(buf, "reset"))
629  {
630  if (MessageSend(ctx, "OK ATR toolbox") < 0)
631  return 1;
632  }
633  else if (MsgEqual(buf, "ping"))
634  {
635  if (MessageSend(ctx, "OK ") < 0)
636  return 1;
637  }
638  else if (MsgEqual(buf, "Capabilities_Register"))
639  {
640  SendRPCI(ctx, "tools.capability.resolution_set 1");
641  SendRPCI(ctx, "tools.capability.resolution_server toolbox 1");
642  SendRPCI(ctx, "tools.capability.display_topology_set 1");
643  SendRPCI(ctx, "tools.capability.color_depth_set 1");
644  SendRPCI(ctx, "tools.capability.resolution_min 0 0");
645  SendRPCI(ctx, "tools.capability.unity 1");
646 
647  if (MessageSend(ctx, "OK ") < 0)
648  return 1;
649  }
650  else if (MsgEqual(buf, "Resolution_Set"))
651  {
652  DebugLog("%s", buf);
653  if (MessageSend(ctx, "OK ") < 0)
654  return 1;
655  MessageClose(ctx);
656  return 0;
657  }
658  else
659  {
660  if (MessageSend(ctx, "ERROR Unknown command") < 0)
661  return 1;
662  }
663  }
664 }

726 InputReport ir = {0};
727 void InterruptHandler(TrapFrame *)
728 {
729  uint8_t Data = inb(0x60);
730  (void)Data;
731 
732  VMwareCommand cmd = {0};
733  cmd.bx = 0;
734  cmd.command = CMD_ABSPOINTER_STATUS;
735  CommandSend(&cmd);
736 
737  if (cmd.ax == 0xFFFF0000)
738  {
739  KernelLog("VMware mouse is not connected?");
740  Relative();
741  Absolute();
742  return;
743  }
744 
745  if ((cmd.ax & 0xFFFF) < 4)
746  return;
747 
748  cmd.bx = 4;
749  cmd.command = CMD_ABSPOINTER_DATA;
750  CommandSend(&cmd);
751 
752  int Buttons = (cmd.ax & 0xFFFF);
753 
759  uintptr_t AbsoluteX = cmd.bx;
760  uintptr_t AbsoluteY = cmd.cx;
761 
762  ir.Type = INPUT_TYPE_MOUSE;
763  ir.Device = MouseDevID;
764  ir.Mouse.X = AbsoluteX;
765  ir.Mouse.Y = AbsoluteY;
766  ir.Mouse.Z = (int8_t)cmd.dx;
767  ir.Mouse.Absolute = 1;
768  ir.Mouse.LeftButton = Buttons & 0x20;
769  ir.Mouse.RightButton = Buttons & 0x10;
770  ir.Mouse.MiddleButton = Buttons & 0x08;
771  // ir.Mouse.Button4 = 0x0;
772  // ir.Mouse.Button5 = 0x0;
773  // ir.Mouse.Button6 = 0x0;
774  // ir.Mouse.Button7 = 0x0;
