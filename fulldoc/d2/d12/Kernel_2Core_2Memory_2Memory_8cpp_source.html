102 NIF void MapFramebuffer(PageTable *PT, bool PSE, bool OneGB)
103 {
104  debug("Mapping Framebuffer");
105  Virtual va = Virtual(PT);
106  int itrfb = 0;
107  while (1)
108  {
109  if (!bInfo.Framebuffer[itrfb].BaseAddress)
110  break;
111 
112  size_t fbSize = bInfo.Framebuffer[itrfb].Pitch * bInfo.Framebuffer[itrfb].Height;
113 
114  if (PSE && OneGB)
115  {
116  va.OptimizedMap(bInfo.Framebuffer[itrfb].BaseAddress,
117  bInfo.Framebuffer[itrfb].BaseAddress,
118  fbSize, PTFlag::RW | PTFlag::US | PTFlag::G);
119  }
120  else
121  {
122  va.Map(bInfo.Framebuffer[itrfb].BaseAddress,
123  bInfo.Framebuffer[itrfb].BaseAddress,
124  fbSize, PTFlag::RW | PTFlag::US | PTFlag::G);
125  }
126  itrfb++;
127  }

130 NIF void MapKernel(PageTable *PT)
131 {
132  debug("Mapping Kernel");
133 
134  /* RWX */
135  uintptr_t BootstrapStart = (uintptr_t)&_bootstrap_start;
136  uintptr_t BootstrapEnd = (uintptr_t)&_bootstrap_end;
137 
138  /* RX */
139  uintptr_t KernelTextStart = (uintptr_t)&_kernel_text_start;
140  uintptr_t KernelTextEnd = (uintptr_t)&_kernel_text_end;
141 
142  /* RW */
143  uintptr_t KernelDataStart = (uintptr_t)&_kernel_data_start;
144  uintptr_t KernelDataEnd = (uintptr_t)&_kernel_data_end;
145 
146  /* R */
147  uintptr_t KernelRoDataStart = (uintptr_t)&_kernel_rodata_start;
148  uintptr_t KernelRoDataEnd = (uintptr_t)&_kernel_rodata_end;
149 
150  /* RW */
151  uintptr_t KernelBssStart = (uintptr_t)&_kernel_bss_start;
152  uintptr_t KernelBssEnd = (uintptr_t)&_kernel_bss_end;
153 
154 #ifdef DEBUG
155  uintptr_t KernelStart = (uintptr_t)&_kernel_start;
156  uintptr_t KernelEnd = (uintptr_t)&_kernel_end;
157 #endif
158  uintptr_t KernelFileStart = (uintptr_t)bInfo.Kernel.FileBase;
159  uintptr_t KernelFileEnd = KernelFileStart + bInfo.Kernel.Size;
160 
161  debug("Bootstrap: %#lx-%#lx", BootstrapStart, BootstrapEnd);
162  debug("Kernel text: %#lx-%#lx", KernelTextStart, KernelTextEnd);
163  debug("Kernel data: %#lx-%#lx", KernelDataStart, KernelDataEnd);
164  debug("Kernel rodata: %#lx-%#lx", KernelRoDataStart, KernelRoDataEnd);
165  debug("Kernel bss: %#lx-%#lx", KernelBssStart, KernelBssEnd);
166  debug("Kernel: %#lx-%#lx", KernelStart, KernelEnd);
167  debug("Kernel file: %#lx-%#lx", KernelFileStart, KernelFileEnd);
168 
169  debug("File size: %ld KiB", TO_KiB(bInfo.Kernel.Size));
170  debug(".bootstrap size: %ld KiB", TO_KiB(BootstrapEnd - BootstrapStart));
171  debug(".text size: %ld KiB", TO_KiB(KernelTextEnd - KernelTextStart));
172  debug(".data size: %ld KiB", TO_KiB(KernelDataEnd - KernelDataStart));
173  debug(".rodata size: %ld KiB", TO_KiB(KernelRoDataEnd - KernelRoDataStart));
174  debug(".bss size: %ld KiB", TO_KiB(KernelBssEnd - KernelBssStart));
175 
176  uintptr_t BaseKernelMapAddress = (uintptr_t)bInfo.Kernel.PhysicalBase;
177  debug("Base kernel map address: %#lx", BaseKernelMapAddress);
178  uintptr_t k;
179  Virtual va = Virtual(PT);
180 
181  /* Bootstrap section */
182  if (BaseKernelMapAddress == BootstrapStart)
183  {
184  for (k = BootstrapStart; k < BootstrapEnd; k += PAGE_SIZE)
185  {
186  va.Map((void *)k, (void *)BaseKernelMapAddress, PTFlag::RW | PTFlag::G);
187  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
188  BaseKernelMapAddress += PAGE_SIZE;
189  }
190  }
191  else
192  {
193  trace("Ignoring bootstrap section.");
194  /* Bootstrap section must be mapped at 0x100000. */
195  }
196 
197  /* Text section */
198  for (k = KernelTextStart; k < KernelTextEnd; k += PAGE_SIZE)
199  {
200  va.Map((void *)k, (void *)BaseKernelMapAddress, PTFlag::RW | PTFlag::G);
201  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
202  BaseKernelMapAddress += PAGE_SIZE;
203  }
204 
205  /* Data section */
206  for (k = KernelDataStart; k < KernelDataEnd; k += PAGE_SIZE)
207  {
208  va.Map((void *)k, (void *)BaseKernelMapAddress, PTFlag::RW | PTFlag::G);
209  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
210  BaseKernelMapAddress += PAGE_SIZE;
211  }
212 
213  /* Read only data section */
214  for (k = KernelRoDataStart; k < KernelRoDataEnd; k += PAGE_SIZE)
215  {
216  va.Map((void *)k, (void *)BaseKernelMapAddress, PTFlag::G);
217  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
218  BaseKernelMapAddress += PAGE_SIZE;
219  }
220 
221  /* Block starting symbol section */
222  for (k = KernelBssStart; k < KernelBssEnd; k += PAGE_SIZE)
223  {
224  va.Map((void *)k, (void *)BaseKernelMapAddress, PTFlag::RW | PTFlag::G);
225  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
226  BaseKernelMapAddress += PAGE_SIZE;
227  }
228 
229  debug("Base kernel map address: %#lx", BaseKernelMapAddress);
230 
231  /* Kernel file */
232  if (KernelFileStart != 0)
233  {
234  for (k = KernelFileStart; k < KernelFileEnd; k += PAGE_SIZE)
235  {
236  va.Map((void *)k, (void *)k, PTFlag::G);
237  KernelAllocator.ReservePage((void *)k);
238  }
239  }
240 }

242 NIF void InitializeMemoryManagement()
243 {
244 #ifdef DEBUG
245 #ifndef a32
246  for (uint64_t i = 0; i < bInfo.Memory.Entries; i++)
247  {
248  uintptr_t Base = r_cst(uintptr_t, bInfo.Memory.Entry[i].BaseAddress);
249  size_t Length = bInfo.Memory.Entry[i].Length;
250  uintptr_t End = Base + Length;
251  const char *Type = "Unknown";
252 
253  switch (bInfo.Memory.Entry[i].Type)
254  {
255  case likely(Usable):
256  Type = "Usable";
257  break;
258  case Reserved:
259  Type = "Reserved";
260  break;
261  case ACPIReclaimable:
262  Type = "ACPI Reclaimable";
263  break;
264  case ACPINVS:
265  Type = "ACPI NVS";
266  break;
267  case BadMemory:
268  Type = "Bad Memory";
269  break;
270  case BootloaderReclaimable:
271  Type = "Bootloader Reclaimable";
272  break;
273  case KernelAndModules:
274  Type = "Kernel and Modules";
275  break;
276  case Framebuffer:
277  Type = "Framebuffer";
278  break;
279  default:
280  break;
281  }
282 
283  debug("%ld: %p-%p %s",
284  i,
285  Base,
286  End,
287  Type);
288  }
289 #endif // a32
290 #endif // DEBUG
291  trace("Initializing Physical Memory Manager");
292  // KernelAllocator = Physical(); <- Already called in the constructor
293  KernelAllocator.Init();
294  debug("Memory Info:\n\n%lld MiB / %lld MiB (%lld MiB reserved)\n",
295  TO_MiB(KernelAllocator.GetUsedMemory()),
296  TO_MiB(KernelAllocator.GetTotalMemory()),
297  TO_MiB(KernelAllocator.GetReservedMemory()));
298 
299  /* -- Debugging --
300  size_t bmap_size = KernelAllocator.GetPageBitmap().Size;
301  for (size_t i = 0; i < bmap_size; i++)
302  {
303  bool idx = KernelAllocator.GetPageBitmap().Get(i);
304  if (idx == true)
305  debug("Page %04d: %#lx", i, i * PAGE_SIZE);
306  }
307 
308  inf_loop debug("Alloc.: %#lx", KernelAllocator.RequestPage());
309  */
310 
311  trace("Initializing Virtual Memory Manager");
312  KernelPageTable = (PageTable *)KernelAllocator.RequestPages(TO_PAGES(PAGE_SIZE + 1));
313  memset(KernelPageTable, 0, PAGE_SIZE);
314 
315  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
316  {
317  CPU::x86::AMD::CPUID0x80000001 cpuid;
318  cpuid.Get();
319  PSESupport = cpuid.EDX.PSE;
320  Page1GBSupport = cpuid.EDX.Page1GB;
321  }
322  else if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_INTEL) == 0)
323  {
324  CPU::x86::Intel::CPUID0x00000001 cpuid;
325  cpuid.Get();
326  PSESupport = cpuid.EDX.PSE;
327  }
328 
329  if (PSESupport)
330  {
331 #if defined(a64)
332  CPU::x64::CR4 cr4 = CPU::x64::readcr4();
333  cr4.PSE = 1;
334  CPU::x64::writecr4(cr4);
335 #elif defined(a32)
336  CPU::x32::CR4 cr4 = CPU::x32::readcr4();
337  cr4.PSE = 1;
338  CPU::x32::writecr4(cr4);
339 #elif defined(aa64)
340 #endif
341  trace("PSE Support Enabled");
342  }
343 
344 #ifdef DEBUG
345  if (Page1GBSupport)
346  debug("1GB Page Support Enabled");
347 #endif
348 
349  MapFromZero(KernelPageTable);
350  MapFramebuffer(KernelPageTable, PSESupport, Page1GBSupport);
351  MapKernel(KernelPageTable);
352 
353  trace("Applying new page table from address %#lx", KernelPageTable);
354 #ifdef DEBUG
355  tracepagetable(KernelPageTable);
356 #endif
357  CPU::PageTable(KernelPageTable);
358  debug("Page table updated.");
359 
360  XallocV1Allocator = new Xalloc::V1((void *)KERNEL_HEAP_BASE, false, false);
361  XallocV2Allocator = new Xalloc::V2((void *)KERNEL_HEAP_BASE);
362  trace("XallocV1 Allocator initialized at %#lx", XallocV1Allocator);
363  trace("XallocV2 Allocator initialized at %#lx", XallocV2Allocator);
364 
365  /* FIXME: Read kernel config */
366  AllocatorType = MemoryAllocatorType::liballoc11;
367 }

369 void *malloc(size_t Size)
370 {
371  assert(Size > 0);
372 
373  memdbg("malloc(%d)->[%s]", Size,
374  KernelSymbolTable ? KernelSymbolTable->GetSymbolFromAddress((uintptr_t)__builtin_return_address(0))
375  : "Unknown");
376 
377  void *ret = nullptr;
378  switch (AllocatorType)
379  {
380  case MemoryAllocatorType::Pages:
381  {
382  ret = KernelAllocator.RequestPages(TO_PAGES(Size + 1));
383  break;
384  }
385  case MemoryAllocatorType::XallocV1:
386  {
387  ret = XallocV1Allocator->malloc(Size);
388  break;
389  }
390  case MemoryAllocatorType::XallocV2:
391  {
392  ret = XallocV2Allocator->malloc(Size);
393  break;
394  }
395  case MemoryAllocatorType::liballoc11:
396  {
397  ret = PREFIX(malloc)(Size);
398  break;
399  }
400  default:
401  {
402  error("Unknown allocator type %d", AllocatorType);
403  CPU::Stop();
404  }
405  }
406 
407  memset(ret, 0, Size);
408  return ret;
