93 static int
94 octal(s)
95 char *s;
96 {
97  int r;
98 
99  r = *s - '0';
100  while (*++s >= '0' && *s <= '7')
101  r = (r * 8) + (*s - '0');
102  return r;
103 }

105 static int
106 finfo_main(argc, argv)
107 int argc;
108 char **argv;
109 {
110  register int i;
111  int mode, flags, opt;
112 
113  sh_optind = 0; /* XXX */
114  prog = base_pathname(argv[0]);
115  if (argc == 1) {
116  builtin_usage();
117  return(1);
118  }
119  flags = 0;
120  while ((opt = sh_getopt(argc, argv, OPTIONS)) != EOF) {
121  switch(opt) {
122  case 'a': flags |= OPT_ATIME; break;
123  case 'A': flags |= OPT_ATIME|OPT_ASCII; break;
124  case 'c': flags |= OPT_CTIME; break;
125  case 'C': flags |= OPT_CTIME|OPT_ASCII; break;
126  case 'd': flags |= OPT_DEV; break;
127  case 'i': flags |= OPT_INO; break;
128  case 'f': flags |= OPT_FID; break;
129  case 'g': flags |= OPT_GID; break;
130  case 'G': flags |= OPT_GID|OPT_ASCII; break;
131  case 'l': flags |= OPT_LNKNAM; break;
132  case 'm': flags |= OPT_MTIME; break;
133  case 'M': flags |= OPT_MTIME|OPT_ASCII; break;
134  case 'n': flags |= OPT_NLINK; break;
135  case 'o': flags |= OPT_OPERM; break;
136  case 'p': flags |= OPT_PERM; break;
137  case 'P': flags |= OPT_PMASK; pmask = octal(sh_optarg); break;
138  case 's': flags |= OPT_SIZE; break;
139  case 'u': flags |= OPT_UID; break;
140  case 'U': flags |= OPT_UID|OPT_ASCII; break;
141  default: builtin_usage (); return(1);
142  }
143  }
144 
145  argc -= sh_optind;
146  argv += sh_optind;
147 
148  if (argc == 0) {
149  builtin_usage();
150  return(1);
151  }
152 
153  for (i = 0; i < argc; i++)
154  opt = flags ? printsome (argv[i], flags) : printfinfo(argv[i]);
155 
156  return(opt);
157 }

159 static struct stat *
160 getstat(f)
161 char *f;
162 {
163  static struct stat st;
164  int fd, r;
165  intmax_t lfd;
166 
167  if (strncmp(f, "/dev/fd/", 8) == 0) {
168  if ((legal_number(f + 8, &lfd) == 0) || (int)lfd != lfd) {
169  builtin_error("%s: invalid fd", f + 8);
170  return ((struct stat *)0);
171  }
172  fd = lfd;
173  r = fstat(fd, &st);
174  } else
175 #ifdef HAVE_LSTAT
176  r = lstat(f, &st);
177 #else
178  r = stat(f, &st);
179 #endif
180  if (r < 0) {
181  builtin_error("%s: cannot stat: %s", f, strerror(errno));
182  return ((struct stat *)0);
183  }
184  return (&st);
185 }

204 static void
205 perms(m)
206 int m;
207 {
208  char ubits[4], gbits[4], obits[4]; /* u=rwx,g=rwx,o=rwx */
209  int i;
210 
211  i = 0;
212  if (m & S_IRUSR)
213  ubits[i++] = 'r';
214  if (m & S_IWUSR)
215  ubits[i++] = 'w';
216  if (m & S_IXUSR)
217  ubits[i++] = 'x';
218  ubits[i] = '\0';
219 
220  i = 0;
221  if (m & S_IRGRP)
222  gbits[i++] = 'r';
223  if (m & S_IWGRP)
224  gbits[i++] = 'w';
225  if (m & S_IXGRP)
226  gbits[i++] = 'x';
227  gbits[i] = '\0';
228 
229  i = 0;
230  if (m & S_IROTH)
231  obits[i++] = 'r';
232  if (m & S_IWOTH)
233  obits[i++] = 'w';
234  if (m & S_IXOTH)
235  obits[i++] = 'x';
236  obits[i] = '\0';
237 
238  if (m & S_ISUID)
239  ubits[2] = (m & S_IXUSR) ? 's' : 'S';
240  if (m & S_ISGID)
241  gbits[2] = (m & S_IXGRP) ? 's' : 'S';
242  if (m & S_ISVTX)
243  obits[2] = (m & S_IXOTH) ? 't' : 'T';
244 
245  printf ("u=%s,g=%s,o=%s", ubits, gbits, obits);
246 }

311 static int
312 printsome(f, flags)
313 char *f;
314 int flags;
315 {
316  struct stat *st;
317  struct passwd *pw;
318  struct group *gr;
319  int p;
320  char *b;
321 
322  st = getstat(f);
323  if (st == NULL)
324  return (1);
325 
326  /* Print requested info */
327  if (flags & OPT_ATIME) {
328  if (flags & OPT_ASCII)
329  printf("%s", ctime(&st->st_atime));
330  else
331  printf("%ld\n", st->st_atime);
332  } else if (flags & OPT_MTIME) {
333  if (flags & OPT_ASCII)
334  printf("%s", ctime(&st->st_mtime));
335  else
336  printf("%ld\n", st->st_mtime);
337  } else if (flags & OPT_CTIME) {
338  if (flags & OPT_ASCII)
339  printf("%s", ctime(&st->st_ctime));
340  else
341  printf("%ld\n", st->st_ctime);
342  } else if (flags & OPT_DEV)
343  printf("%lu\n", (unsigned long)st->st_dev);
344  else if (flags & OPT_INO)
345  printf("%lu\n", (unsigned long)st->st_ino);
346  else if (flags & OPT_FID)
347  printf("%lu:%lu\n", (unsigned long)st->st_dev, (unsigned long)st->st_ino);
348  else if (flags & OPT_NLINK)
349  printf("%lu\n", (unsigned long)st->st_nlink);
350  else if (flags & OPT_LNKNAM) {
351 #ifdef S_ISLNK
352  b = xmalloc(4096);
353  p = readlink(f, b, 4096);
354  if (p >= 0 && p < 4096)
355  b[p] = '\0';
356  else {
357  p = errno;
358  strcpy(b, prog);
359  strcat(b, ": ");
360  strcat(b, strerror(p));
361  }
362  printf("%s\n", b);
363  free(b);
364 #else
365  printf("%s\n", f);
366 #endif
367  } else if (flags & OPT_PERM) {
368  perms(st->st_mode);
369  printf("\n");
370  } else if (flags & OPT_OPERM)
371  printf("%o\n", getperm(st->st_mode));
372  else if (flags & OPT_PMASK)
373  printf("%o\n", getperm(st->st_mode) & pmask);
374  else if (flags & OPT_UID) {
375  pw = getpwuid(st->st_uid);
376  if (flags & OPT_ASCII)
377  printf("%s\n", pw ? pw->pw_name : "unknown");
378  else
379  printf("%d\n", st->st_uid);
380  } else if (flags & OPT_GID) {
381  gr = getgrgid(st->st_gid);
382  if (flags & OPT_ASCII)
383  printf("%s\n", gr ? gr->gr_name : "unknown");
384  else
385  printf("%d\n", st->st_gid);
386  } else if (flags & OPT_SIZE)
387  printf("%ld\n", (long) st->st_size);
388 
389  return (0);
390 }

516 int
517 legal_number (string, result)
518  char *string;
519  long *result;
520 {
521  int sign;
522  long value;
523 
524  sign = 1;
525  value = 0;
526 
527  if (result)
528  *result = 0;
529 
530  /* Skip leading whitespace characters. */
531  while (whitespace (*string))
532  string++;
533 
534  if (!*string)
535  return (0);
536 
537  /* We allow leading `-' or `+'. */
538  if (*string == '-' || *string == '+')
539  {
540  if (!digit (string[1]))
541  return (0);
542 
543  if (*string == '-')
544  sign = -1;
545 
546  string++;
547  }
548 
549  while (digit (*string))
550  {
551  if (result)
552  value = (value * 10) + digit_value (*string);
553  string++;
554  }
555 
556  /* Skip trailing whitespace, if any. */
557  while (whitespace (*string))
558  string++;
559 
560  /* Error if not at end of string. */
561  if (*string)
562  return (0);
563 
564  if (result)
565  *result = value * sign;
566 
567  return (1);
568 }
