238 static bool write_and_stats(const void *buf, size_t len, size_t obs,
239  const char *filename)
240 {
241  ssize_t n;
242 
243  IF_FEATURE_DD_IBS_OBS(write_again:)
244  n = full_write(ofd, buf, len);
245 #if ENABLE_FEATURE_DD_IBS_OBS
246 # ifdef O_DIRECT
247  if (n < 0 && (G.flags & FLAG_ODIRECT) && clear_O_DIRECT(ofd))
248  goto write_again;
249 # endif
250 #endif
251 
252 #if ENABLE_FEATURE_DD_THIRD_STATUS_LINE
253  if (n > 0)
254  G.total_bytes += n;
255 #endif
256  if ((size_t)n == obs) {
257  G.out_full++;
258  return 0;
259  }
260  if ((size_t)n == len) {
261  G.out_part++;
262  return 0;
263  }
264  /* n is < len (and possibly is -1).
265  * Even if n >= 0, errno is usually set correctly.
266  * For example, if writing to block device and getting ENOSPC,
267  * full_write() first sees a short write, then tries to write
268  * the remainder and gets errno set to ENOSPC.
269  * It returns n > 0 (the amount which it did write).
270  */
271  bb_perror_msg("error writing '%s'", filename);
272  return 1;
273 }

281 #if ENABLE_FEATURE_DD_IBS_OBS
282 static int parse_comma_flags(char *val, const char *words, const char *error_in)
283 {
284  int flags = 0;
285  while (1) {
286  int n;
287  char *arg;
288  /* find ',', replace them with NUL so we can use val for
289  * index_in_strings() without copying.
290  * We rely on val being non-null, else strchr would fault.
291  */
292  arg = strchr(val, ',');
293  if (arg)
294  *arg = '\0';
295  n = index_in_strings(words, val);
296  if (n < 0)
297  bb_error_msg_and_die(bb_msg_invalid_arg_to, val, error_in);
298  flags |= (1 << n);
299  if (!arg) /* no ',' left, so this was the last specifier */
300  break;
301  *arg = ','; /* to preserve ps listing */
302  val = arg + 1; /* skip this keyword and ',' */
303  }
304  return flags;
305 }
306 #endif
