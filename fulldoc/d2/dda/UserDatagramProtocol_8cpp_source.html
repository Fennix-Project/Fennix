100  bool UDP::OnIPv4PacketReceived(InternetProtocol SourceIP, InternetProtocol DestinationIP, uint8_t *Data, size_t Length)
101  {
102  netdbg("Received %d bytes from %s", Length, SourceIP.v4.ToStringLittleEndian());
103  if (Length < sizeof(UDPHeader))
104  return false;
105 
106  UDPHeader *udp = (UDPHeader *)Data;
107 
108  netdbg("SP:%d | DP:%d | L:%d | CHK:%#x", b16(udp->SourcePort), b16(udp->DestinationPort), b16(udp->Length), b16(udp->Checksum));
109 
110  Socket *GoodSocket = nullptr;
111 
112  foreach (auto &var in RegisteredEvents)
113  {
114  netdbg("UDP->SKT[]: LP:%d | LIP:%s | RP:%d | RIP:%s | LST:%d",
115  b16(var.UDPSocket->LocalPort),
116  var.UDPSocket->LocalIP.v4.ToStringLittleEndian(),
117  b16(var.UDPSocket->RemotePort),
118  var.UDPSocket->RemoteIP.v4.ToStringLittleEndian(),
119  b16(var.UDPSocket->Listening));
120  if (var.UDPSocket->LocalPort == udp->DestinationPort &&
121  var.UDPSocket->LocalIP.v4 == DestinationIP.v4 &&
122  var.UDPSocket->Listening == true)
123  {
124  var.UDPSocket->Listening = false;
125  var.UDPSocket->RemotePort = b16(udp->SourcePort);
126  var.UDPSocket->RemoteIP = SourceIP;
127  netdbg("E1");
128  return true;
129  }
130 
131  GoodSocket = var.UDPSocket;
132  }
133  if (GoodSocket)
134  GoodSocket->EventHandler->OnUDPPacketReceived(GoodSocket, ((UDPPacket *)Data)->Data, Length);
135 
136  netdbg("E0 (Success)");
137  return false;
138  }
