46  extern "C" int __cxa_atexit(void (*f)(void *), void *objptr, void *dso)
47  {
48  if (KernelSymbolTable)
49  {
50  debug("Registering atexit function for \"%s\" with destructor \"%s\"",
51  KernelSymbolTable->GetSymbolFromAddress((uintptr_t)objptr),
52  KernelSymbolTable->GetSymbolFromAddress((uintptr_t)f));
53  }
54  else
55  {
56  debug("Registering atexit function for %p with destructor %p",
57  objptr, f);
58  }
59 
60  if (__atexit_func_count >= ATEXIT_MAX_FUNCS)
61  return -1;
62  __atexit_funcs[__atexit_func_count].destructor_func = f;
63  __atexit_funcs[__atexit_func_count].obj_ptr = objptr;
64  __atexit_funcs[__atexit_func_count].dso_handle = dso;
65  __atexit_func_count++;

67  }

70  {
71  function("%p", f);

73  if (f == nullptr)
