42 //
43 // Move a plat up and down
44 //
45 void T_PlatRaise(plat_t* plat)
46 {
47  result_e res;
48 
49  switch(plat->status)
50  {
51  case up:
52  res = T_MovePlane(plat->sector,
53  plat->speed,
54  plat->high,
55  plat->crush,0,1);
56 
57  if (plat->type == raiseAndChange
58  || plat->type == raiseToNearestAndChange)
59  {
60  if (!(leveltime&7))
61  S_StartSound(&plat->sector->soundorg, sfx_stnmov);
62  }
63 
64 
65  if (res == crushed && (!plat->crush))
66  {
67  plat->count = plat->wait;
68  plat->status = down;
69  S_StartSound(&plat->sector->soundorg, sfx_pstart);
70  }
71  else
72  {
73  if (res == pastdest)
74  {
75  plat->count = plat->wait;
76  plat->status = waiting;
77  S_StartSound(&plat->sector->soundorg, sfx_pstop);
78 
79  switch(plat->type)
80  {
81  case blazeDWUS:
82  case downWaitUpStay:
83  P_RemoveActivePlat(plat);
84  break;
85 
86  case raiseAndChange:
87  case raiseToNearestAndChange:
88  P_RemoveActivePlat(plat);
89  break;
90 
91  default:
92  break;
93  }
94  }
95  }
96  break;
97 
98  case down:
99  res = T_MovePlane(plat->sector,plat->speed,plat->low,false,0,-1);
100 
101  if (res == pastdest)
102  {
103  plat->count = plat->wait;
104  plat->status = waiting;
105  S_StartSound(&plat->sector->soundorg,sfx_pstop);
106  }
107  break;
108 
109  case waiting:
110  if (!--plat->count)
111  {
112  if (plat->sector->floorheight == plat->low)
113  plat->status = up;
114  else
115  plat->status = down;
116  S_StartSound(&plat->sector->soundorg,sfx_pstart);
117  }
118  case in_stasis:
119  break;
120  }
121 }

125 // Do Platforms
126 // "amount" is only used for SOME platforms.
127 //
128 int
129 EV_DoPlat
130 ( line_t* line,
131  plattype_e type,
132  int amount )
133 {
134  plat_t* plat;
135  int secnum;
136  int rtn;
137  sector_t* sec;
138 
139  secnum = -1;
140  rtn = 0;
141 
142 
143  // Activate all <type> plats that are in_stasis
144  switch(type)
145  {
146  case perpetualRaise:
147  P_ActivateInStasis(line->tag);
148  break;
149 
150  default:
151  break;
152  }
153 
154  while ((secnum = P_FindSectorFromLineTag(line,secnum)) >= 0)
155  {
156  sec = &sectors[secnum];
157 
158  if (sec->specialdata)
159  continue;
160 
161  // Find lowest & highest floors around sector
162  rtn = 1;
163  plat = Z_Malloc( sizeof(*plat), PU_LEVSPEC, 0);
164  P_AddThinker(&plat->thinker);
165 
166  plat->type = type;
167  plat->sector = sec;
168  plat->sector->specialdata = plat;
169  plat->thinker.function.acp1 = (actionf_p1) T_PlatRaise;
170  plat->crush = false;
171  plat->tag = line->tag;
172 
173  switch(type)
174  {
175  case raiseToNearestAndChange:
176  plat->speed = PLATSPEED/2;
177  sec->floorpic = sides[line->sidenum[0]].sector->floorpic;
178  plat->high = P_FindNextHighestFloor(sec,sec->floorheight);
179  plat->wait = 0;
180  plat->status = up;
181  // NO MORE DAMAGE, IF APPLICABLE
182  sec->special = 0;
183 
184  S_StartSound(&sec->soundorg,sfx_stnmov);
185  break;
186 
187  case raiseAndChange:
188  plat->speed = PLATSPEED/2;
189  sec->floorpic = sides[line->sidenum[0]].sector->floorpic;
190  plat->high = sec->floorheight + amount*FRACUNIT;
191  plat->wait = 0;
192  plat->status = up;
193 
194  S_StartSound(&sec->soundorg,sfx_stnmov);
195  break;
196 
197  case downWaitUpStay:
198  plat->speed = PLATSPEED * 4;
199  plat->low = P_FindLowestFloorSurrounding(sec);
200 
201  if (plat->low > sec->floorheight)
202  plat->low = sec->floorheight;
203 
204  plat->high = sec->floorheight;
205  plat->wait = TICRATE*PLATWAIT;
206  plat->status = down;
207  S_StartSound(&sec->soundorg,sfx_pstart);
208  break;
209 
210  case blazeDWUS:
211  plat->speed = PLATSPEED * 8;
212  plat->low = P_FindLowestFloorSurrounding(sec);
213 
214  if (plat->low > sec->floorheight)
215  plat->low = sec->floorheight;
216 
217  plat->high = sec->floorheight;
218  plat->wait = TICRATE*PLATWAIT;
219  plat->status = down;
220  S_StartSound(&sec->soundorg,sfx_pstart);
221  break;
222 
223  case perpetualRaise:
224  plat->speed = PLATSPEED;
225  plat->low = P_FindLowestFloorSurrounding(sec);
226 
227  if (plat->low > sec->floorheight)
228  plat->low = sec->floorheight;
229 
230  plat->high = P_FindHighestFloorSurrounding(sec);
231 
232  if (plat->high < sec->floorheight)
233  plat->high = sec->floorheight;
234 
235  plat->wait = TICRATE*PLATWAIT;
236  plat->status = P_Random()&1;
237 
238  S_StartSound(&sec->soundorg,sfx_pstart);
239  break;
240  }
241  P_AddActivePlat(plat);
242  }
243  return rtn;
244 }
