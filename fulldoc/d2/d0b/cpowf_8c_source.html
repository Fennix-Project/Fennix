39 cpowf(float complex a, float complex z)
40 {
41  float complex w;
42  float x, y, r, theta, absa, arga;
43 
44  x = crealf(z);
45  y = cimagf(z);
46  absa = cabsf(a);
47  if (absa == 0.0f) {
48  return (0.0f + 0.0f * I);
49  }
50  arga = cargf(a);
51  r = powf(absa, x);
52  theta = x * arga;
53  if (y != 0.0f) {
54  r = r * expf(-y * arga);
55  theta = theta + y * logf(absa);
56  }
57  w = r * cosf(theta) + (r * sinf(theta)) * I;
58  return w;
59 }
