251 /* Get readline's idea of the screen size. TTY is a file descriptor open
252  to the terminal. If IGNORE_ENV is true, we do not pay attention to the
253  values of $LINES and $COLUMNS. The tests for TERM_STRING_BUFFER being
254  non-null serve to check whether or not we have initialized termcap. */
255 void
256 _rl_get_screen_size (int tty, int ignore_env)
257 {
258  char *ss;
259 #if defined (TIOCGWINSZ)
260  struct winsize window_size;
261 #endif /* TIOCGWINSZ */
262  int wr, wc;
263 
264  wr = wc = -1;
265 #if defined (TIOCGWINSZ)
266  if (ioctl (tty, TIOCGWINSZ, &window_size) == 0)
267  {
268  wc = (int) window_size.ws_col;
269  wr = (int) window_size.ws_row;
270  }
271 #endif /* TIOCGWINSZ */
272 
273 #if defined (__EMX__)
274  _emx_get_screensize (&wc, &wr);
275 #elif defined (__MINGW32__)
276  _win_get_screensize (&wc, &wr);
277 #endif
278 
279  if (ignore_env || rl_prefer_env_winsize == 0)
280  {
281  _rl_screenwidth = wc;
282  _rl_screenheight = wr;
283  }
284  else
285  _rl_screenwidth = _rl_screenheight = -1;
286 
287  /* Environment variable COLUMNS overrides setting of "co" if IGNORE_ENV
288  is unset. If we prefer the environment, check it first before
289  assigning the value returned by the kernel. */
290  if (_rl_screenwidth <= 0)
291  {
292  if (ignore_env == 0 && (ss = sh_get_env_value ("COLUMNS")))
293  _rl_screenwidth = atoi (ss);
294 
295  if (_rl_screenwidth <= 0)
296  _rl_screenwidth = wc;
297 
298 #if defined (__DJGPP__)
299  if (_rl_screenwidth <= 0)
300  _rl_screenwidth = ScreenCols ();
301 #else
302  if (_rl_screenwidth <= 0 && term_string_buffer)
303  _rl_screenwidth = tgetnum ("co");
304 #endif
305  }
306 
307  /* Environment variable LINES overrides setting of "li" if IGNORE_ENV
308  is unset. */
309  if (_rl_screenheight <= 0)
310  {
311  if (ignore_env == 0 && (ss = sh_get_env_value ("LINES")))
312  _rl_screenheight = atoi (ss);
313 
314  if (_rl_screenheight <= 0)
315  _rl_screenheight = wr;
316 
317 #if defined (__DJGPP__)
318  if (_rl_screenheight <= 0)
319  _rl_screenheight = ScreenRows ();
320 #else
321  if (_rl_screenheight <= 0 && term_string_buffer)
322  _rl_screenheight = tgetnum ("li");
323 #endif
324  }
325 
326  /* If all else fails, default to 80x24 terminal. */
327  if (_rl_screenwidth <= 1)
328  _rl_screenwidth = 80;
329 
330  if (_rl_screenheight <= 0)
331  _rl_screenheight = 24;
332 
333  /* If we're being compiled as part of bash, set the environment
334  variables $LINES and $COLUMNS to new values. Otherwise, just
335  do a pair of putenv () or setenv () calls. */
336  if (rl_change_environment)
337  sh_set_lines_and_columns (_rl_screenheight, _rl_screenwidth);
338 
339  if (_rl_term_autowrap == 0)
340  _rl_screenwidth--;
341 
342  _rl_screenchars = _rl_screenwidth * _rl_screenheight;
343 }

347 {

349  _rl_init_terminal_io (rl_terminal_name);
350

355  _rl_screenwidth = cols;
356  if (_rl_term_autowrap == 0)
357  _rl_screenwidth--;
358  }
359

469 int
470 _rl_init_terminal_io (const char *terminal_name)
471 {
472  const char *term;
473  char *buffer;
474  int tty, tgetent_ret, dumbterm, reset_region_colors;
475 
476  term = terminal_name ? terminal_name : sh_get_env_value ("TERM");
477  _rl_term_clrpag = _rl_term_cr = _rl_term_clreol = _rl_term_clrscroll = (char *)NULL;
478  tty = rl_instream ? fileno (rl_instream) : 0;
479 
480  if (term == 0)
481  term = "dumb";
482 
483  dumbterm = STREQ (term, "dumb");
484 
485  reset_region_colors = 1;
486 
487 #ifdef __MSDOS__
488  _rl_term_im = _rl_term_ei = _rl_term_ic = _rl_term_IC = (char *)NULL;
489  _rl_term_up = _rl_term_dc = _rl_term_DC = _rl_visible_bell = (char *)NULL;
490  _rl_term_ku = _rl_term_kd = _rl_term_kl = _rl_term_kr = (char *)NULL;
491  _rl_term_mm = _rl_term_mo = (char *)NULL;
492  _rl_terminal_can_insert = term_has_meta = _rl_term_autowrap = 0;
493  _rl_term_cr = "\r";
494  _rl_term_backspace = (char *)NULL;
495  _rl_term_goto = _rl_term_pc = _rl_term_ip = (char *)NULL;
496  _rl_term_ks = _rl_term_ke =_rl_term_vs = _rl_term_ve = (char *)NULL;
497  _rl_term_kh = _rl_term_kH = _rl_term_at7 = _rl_term_kI = (char *)NULL;
498  _rl_term_kN = _rl_term_kP = (char *)NULL;
499  _rl_term_so = _rl_term_se = (char *)NULL;
500 #if defined(HACK_TERMCAP_MOTION)
501  _rl_term_forward_char = (char *)NULL;
502 #endif
503 
504  _rl_get_screen_size (tty, 0);
505 #else /* !__MSDOS__ */
506  /* I've separated this out for later work on not calling tgetent at all
507  if the calling application has supplied a custom redisplay function,
508  (and possibly if the application has supplied a custom input function). */
509  if (CUSTOM_REDISPLAY_FUNC())
510  {
511  tgetent_ret = -1;
512  }
513  else
514  {
515  if (term_string_buffer == 0)
516  term_string_buffer = (char *)xmalloc(2032);
517 
518  if (term_buffer == 0)
519  term_buffer = (char *)xmalloc(4080);
520 
521  buffer = term_string_buffer;
522 
523  tgetent_ret = tgetent (term_buffer, term);
524  }
525 
526  if (tgetent_ret != TGETENT_SUCCESS)
527  {
528  FREE (term_string_buffer);
529  FREE (term_buffer);
530  buffer = term_buffer = term_string_buffer = (char *)NULL;
531 
532  _rl_term_autowrap = 0; /* used by _rl_get_screen_size */
533 
534  /* Allow calling application to set default height and width, using
535  rl_set_screen_size */
536  if (_rl_screenwidth <= 0 || _rl_screenheight <= 0)
537  {
538 #if defined (__EMX__)
539  _emx_get_screensize (&_rl_screenwidth, &_rl_screenheight);
540  _rl_screenwidth--;
541 #else /* !__EMX__ */
542  _rl_get_screen_size (tty, 0);
543 #endif /* !__EMX__ */
544  }
545 
546  /* Defaults. */
547  if (_rl_screenwidth <= 0 || _rl_screenheight <= 0)
548  {
549  _rl_screenwidth = 79;
550  _rl_screenheight = 24;
551  }
552 
553  /* Everything below here is used by the redisplay code (tputs). */
554  _rl_screenchars = _rl_screenwidth * _rl_screenheight;
555  _rl_term_cr = "\r";
556  _rl_term_im = _rl_term_ei = _rl_term_ic = _rl_term_IC = (char *)NULL;
557  _rl_term_up = _rl_term_dc = _rl_term_DC = _rl_visible_bell = (char *)NULL;
558  _rl_term_ku = _rl_term_kd = _rl_term_kl = _rl_term_kr = (char *)NULL;
559  _rl_term_kh = _rl_term_kH = _rl_term_kI = _rl_term_kD = (char *)NULL;
560  _rl_term_ks = _rl_term_ke = _rl_term_at7 = (char *)NULL;
561  _rl_term_kN = _rl_term_kP = (char *)NULL;
562  _rl_term_mm = _rl_term_mo = (char *)NULL;
563  _rl_term_ve = _rl_term_vs = (char *)NULL;
564  _rl_term_forward_char = (char *)NULL;
565  _rl_term_so = _rl_term_se = (char *)NULL;
566  _rl_terminal_can_insert = term_has_meta = 0;
567 
568  /* Assume generic unknown terminal can't handle the enable/disable
569  escape sequences */
570  _rl_enable_bracketed_paste = 0;
571 
572  /* No terminal so/se capabilities. */
573  _rl_enable_active_region = 0;
574  _rl_reset_region_color (0, NULL);
575  _rl_reset_region_color (1, NULL);
576 
577  /* Reasonable defaults for tgoto(). Readline currently only uses
578  tgoto if _rl_term_IC or _rl_term_DC is defined, but just in case we
579  change that later... */
580  PC = '\0';
581  BC = _rl_term_backspace = "\b";
582  UP = _rl_term_up;
583 
584  return 0;
585  }
586 
587  get_term_capabilities (&buffer);
588 
589  /* Set up the variables that the termcap library expects the application
590  to provide. */
591  PC = _rl_term_pc ? *_rl_term_pc : 0;
592  BC = _rl_term_backspace;
593  UP = _rl_term_up;
594 
595  if (_rl_term_cr == 0)
596  _rl_term_cr = "\r";
597 
598  _rl_term_autowrap = TGETFLAG ("am") && TGETFLAG ("xn");
599 
600  /* Allow calling application to set default height and width, using
601  rl_set_screen_size */
602  if (_rl_screenwidth <= 0 || _rl_screenheight <= 0)
603  _rl_get_screen_size (tty, 0);
604 
605  /* "An application program can assume that the terminal can do
606  character insertion if *any one of* the capabilities `IC',
607  `im', `ic' or `ip' is provided." But we can't do anything if
608  only `ip' is provided, so... */
609  _rl_terminal_can_insert = (_rl_term_IC || _rl_term_im || _rl_term_ic);
610 
611  /* Check to see if this terminal has a meta key and clear the capability
612  variables if there is none. */
613  term_has_meta = TGETFLAG ("km");
614  if (term_has_meta == 0)
615  _rl_term_mm = _rl_term_mo = (char *)NULL;
616 #endif /* !__MSDOS__ */
617 
618  /* Attempt to find and bind the arrow keys. Do not override already
619  bound keys in an overzealous attempt, however. */
620 
621  bind_termcap_arrow_keys (emacs_standard_keymap);
622 
623 #if defined (VI_MODE)
624  bind_termcap_arrow_keys (vi_movement_keymap);
625  bind_termcap_arrow_keys (vi_insertion_keymap);
626 #endif /* VI_MODE */
627 
628  /* There's no way to determine whether or not a given terminal supports
629  bracketed paste mode, so we assume a terminal named "dumb" does not. */
630  if (dumbterm)
631  _rl_enable_bracketed_paste = _rl_enable_active_region = 0;
632 
633  if (reset_region_colors)
634  {
635  _rl_reset_region_color (0, _rl_term_so);
636  _rl_reset_region_color (1, _rl_term_se);
637  }
638 
639  return 0;
640 }

643 static void
644 bind_termcap_arrow_keys (Keymap map)
645 {
646  Keymap xkeymap;
647 
648  xkeymap = _rl_keymap;
649  _rl_keymap = map;
