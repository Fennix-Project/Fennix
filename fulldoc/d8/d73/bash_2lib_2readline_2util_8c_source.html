178 /* A function for simple tilde expansion. */
179 int
180 rl_tilde_expand (int ignore, int key)
181 {
182  register int start, end;
183  char *homedir, *temp;
184  int len;
185 
186  end = rl_point;
187  start = end - 1;
188 
189  if (rl_point == rl_end && rl_line_buffer[rl_point] == '~')
190  {
191  homedir = tilde_expand ("~");
192  _rl_replace_text (homedir, start, end);
193  xfree (homedir);
194  return (0);
195  }
196  else if (start >= 0 && rl_line_buffer[start] != '~')
197  {
198  for (; start >= 0 && !whitespace (rl_line_buffer[start]); start--)
199  ;
200  start++;
201  }
202  else if (start < 0)
203  start = 0;
204 
205  end = start;
206  do
207  end++;
208  while (whitespace (rl_line_buffer[end]) == 0 && end < rl_end);
209 
210  if (whitespace (rl_line_buffer[end]) || end >= rl_end)
211  end--;
212 
213  /* If the first character of the current word is a tilde, perform
214  tilde expansion and insert the result. If not a tilde, do
215  nothing. */
216  if (rl_line_buffer[start] == '~')
217  {
218  len = end - start + 1;
219  temp = (char *)xmalloc (len + 1);
220  strncpy (temp, rl_line_buffer + start, len);
221  temp[len] = '\0';
222  homedir = tilde_expand (temp);
223  xfree (temp);
224 
225  _rl_replace_text (homedir, start, end);
226  xfree (homedir);
227  }
228 
229  return (0);
230 }

319 /* Determine if s2 occurs in s1. If so, return a pointer to the
320  match in s1. The compare is case insensitive. */
321 char *
322 _rl_strindex (const char *s1, const char *s2)
323 {
324  register int i, l, len;
325 
326  for (i = 0, l = strlen (s2), len = strlen (s1); (len - i) >= l; i++)
327  if (_rl_strnicmp (s1 + i, s2, l) == 0)
328  return ((char *) (s1 + i));
329  return ((char *)NULL);
330 }

366 #if !defined (HAVE_STRCASECMP)
367 /* Compare at most COUNT characters from string1 to string2. Case
368  doesn't matter (strncasecmp). */
369 int
370 _rl_strnicmp (const char *string1, const char *string2, int count)
371 {
372  register const char *s1;
373  register const char *s2;
374  register int d;
375 
376  if (count <= 0 || (string1 == string2))
377  return 0;
378 
379  s1 = string1;
380  s2 = string2;
381  do
382  {
383  d = _rl_to_lower (*s1) - _rl_to_lower (*s2); /* XXX - cast to unsigned char? */
384  if (d != 0)
385  return d;
386  if (*s1++ == '\0')
387  break;
388  s2++;
389  }
390  while (--count != 0);
391 
392  return (0);
393 }

402
