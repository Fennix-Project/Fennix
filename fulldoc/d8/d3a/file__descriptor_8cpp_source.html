100  {

128  return -EACCES;

259  {

262  {
263  bool Found = false;
264  foreach (auto fd in FileDescriptors)

268  Found = true;

271  }

367  int FileDescriptorTable::_close(int fd)
368  {
369  Fildes fdesc = this->GetFileDescriptor(fd);
370  DupFildes dfdesc = this->GetDupFildes(fd);
371 
372  if (fdesc.Descriptor < 0 &&
373  dfdesc.Descriptor < 0)
374  {
375  return -EBADF;
376  }
377 
378  if (RemoveFileDescriptor(fd) < 0)
379  return -EBADF;
380 
381  /* If the file descriptor is a duplicate,
382  we don't need to close the handle,
383  because it's a duplicate of another
384  file descriptor. */
385 
386  bool Found = false;
387  RefNode *hnd = nullptr;
388 
389  if (fdesc.Descriptor != -1)
390  hnd = fdesc.Handle;
391  else
392  hnd = dfdesc.Handle;
393 
394  foreach (auto dfd in FileDescriptors)
395  {
396  if (dfd.Handle == hnd)
397  {
398  Found = true;
399  break;
400  }
401  }
402 
403  foreach (auto dfd in FildesDuplicates)
404  {
405  if (dfd.Handle == hnd)
406  {
407  Found = true;
408  break;
409  }
410  }
411 
412  if (!Found)
413  delete hnd;
414  return 0;
415  }
