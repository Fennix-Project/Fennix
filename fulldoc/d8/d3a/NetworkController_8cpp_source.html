102  void NetworkInterface::StartNetworkStack()
103  {
104  TaskManager->GetCurrentThread()->SetPriority(Tasking::TaskPriority::Critical);
105  DeviceInterface *DefaultDevice = nullptr;
106  foreach (auto inf in Interfaces)

111  }
112 
113  if (!DefaultDevice)
114  error("No network device found!");
115  else
116  {
117  DbgWriteScreen("Using %s (%s) as the default network device", DefaultDevice->Name, DefaultDevice->MAC.ToString());
118  NetworkEthernet::Ethernet *eth = new NetworkEthernet::Ethernet(DefaultDevice); // Use the first device found as the ethernet device
119  NetworkARP::ARP *arp = new NetworkARP::ARP(eth);
120  NetworkIPv4::IPv4 *ipv4 = new NetworkIPv4::IPv4(arp, eth);
121  NetworkUDP::UDP *udp = new NetworkUDP::UDP(ipv4, DefaultDevice);
122  NetworkUDP::Socket *DHCP_Socket = udp->Connect(InternetProtocol() /* Default value is 255.255.255.255 */, 67);
123  NetworkDHCP::DHCP *dhcp = new NetworkDHCP::DHCP(DHCP_Socket, DefaultDevice);
124  debug("eth: %p; arp: %p; ipv4: %p; udp: %p; dhcp: %p", eth, arp, ipv4, udp, dhcp);
125  udp->Bind(DHCP_Socket, dhcp);
126  dhcp->Request();
127 
128  DefaultDevice->IP.v4.FromHex(dhcp->IP.v4.ToHex());
129  ipv4->SubNetworkMaskIP = dhcp->SubNetworkMask;
130  ipv4->GatewayIP = dhcp->Gateway;
131  arp->Broadcast(dhcp->Gateway);
132  TaskManager->Sleep(200);

206  {
