129 void cmap_to_rgb565(uint16_t * out, uint8_t * in, int in_pixels)
130 {
131  int i, j;
132  struct color c;
133  uint16_t r, g, b;
134 
135  for (i = 0; i < in_pixels; i++)
136  {
137  c = colors[*in];
138  r = ((uint16_t)(c.r >> 3)) << 11;
139  g = ((uint16_t)(c.g >> 2)) << 5;
140  b = ((uint16_t)(c.b >> 3)) << 0;
141  *out = (r | g | b);
142 
143  in++;
144  for (j = 0; j < fb_scaling; j++) {
145  out++;
146  }
147  }
148 }

150 void cmap_to_fb(uint8_t * out, uint8_t * in, int in_pixels)
151 {
152  int i, j, k;
153  struct color c;
154  uint32_t pix;
155  uint16_t r, g, b;
156 
157  for (i = 0; i < in_pixels; i++)
158  {
159  c = colors[*in]; /* R:8 G:8 B:8 format! */
160  r = (uint16_t)(c.r >> (8 - s_Fb.red.length));
161  g = (uint16_t)(c.g >> (8 - s_Fb.green.length));
162  b = (uint16_t)(c.b >> (8 - s_Fb.blue.length));
163  pix = r << s_Fb.red.offset;
164  pix |= g << s_Fb.green.offset;
165  pix |= b << s_Fb.blue.offset;
166 
167  for (k = 0; k < fb_scaling; k++) {
168  for (j = 0; j < s_Fb.bits_per_pixel/8; j++) {
169  *out = (pix >> (j*8));
170  out++;
171  }
172  }
173  in++;
174  }
175 }
