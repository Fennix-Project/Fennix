103 NIF void MapFramebuffer(PageTable *PT)
104 {
105  debug("Mapping Framebuffer");
106  Virtual vmm = Virtual(PT);
107  int itrfb = 0;
108  while (1)
109  {
110  if (!bInfo.Framebuffer[itrfb].BaseAddress)
111  break;
112 
113  size_t fbSize = bInfo.Framebuffer[itrfb].Pitch * bInfo.Framebuffer[itrfb].Height;
114 
115  if (PSESupport && Page1GBSupport)
116  {
117  vmm.OptimizedMap(bInfo.Framebuffer[itrfb].BaseAddress,
118  bInfo.Framebuffer[itrfb].BaseAddress,
119  fbSize, RW | US | G);
120  }
121  else
122  {
123  vmm.Map(bInfo.Framebuffer[itrfb].BaseAddress,
124  bInfo.Framebuffer[itrfb].BaseAddress,
125  fbSize, RW | US | G);
126  }
127  itrfb++;
128  }
129 }

131 NIF void MapKernel(PageTable *PT)
132 {
133  debug("Mapping Kernel");
134 
135  /* RWX */
136  uintptr_t BootstrapStart = (uintptr_t)&_bootstrap_start;
137  uintptr_t BootstrapEnd = (uintptr_t)&_bootstrap_end;
138 
139  /* RX */
140  uintptr_t KernelTextStart = (uintptr_t)&_kernel_text_start;
141  uintptr_t KernelTextEnd = (uintptr_t)&_kernel_text_end;
142 
143  /* RW */
144  uintptr_t KernelDataStart = (uintptr_t)&_kernel_data_start;
145  uintptr_t KernelDataEnd = (uintptr_t)&_kernel_data_end;
146 
147  /* R */
148  uintptr_t KernelRoDataStart = (uintptr_t)&_kernel_rodata_start;
149  uintptr_t KernelRoDataEnd = (uintptr_t)&_kernel_rodata_end;
150 
151  /* RW */
152  uintptr_t KernelBssStart = (uintptr_t)&_kernel_bss_start;
153  uintptr_t KernelBssEnd = (uintptr_t)&_kernel_bss_end;
154 
155 #ifdef DEBUG
156  uintptr_t KernelStart = (uintptr_t)&_kernel_start;
157  uintptr_t KernelEnd = (uintptr_t)&_kernel_end;
158 #endif
159  uintptr_t KernelFileStart = (uintptr_t)bInfo.Kernel.FileBase;
160  uintptr_t KernelFileEnd = KernelFileStart + bInfo.Kernel.Size;
161 
162  debug("Bootstrap: %#lx-%#lx", BootstrapStart, BootstrapEnd);
163  debug("Kernel text: %#lx-%#lx", KernelTextStart, KernelTextEnd);
164  debug("Kernel data: %#lx-%#lx", KernelDataStart, KernelDataEnd);
165  debug("Kernel rodata: %#lx-%#lx", KernelRoDataStart, KernelRoDataEnd);
166  debug("Kernel bss: %#lx-%#lx", KernelBssStart, KernelBssEnd);
167  debug("Kernel: %#lx-%#lx", KernelStart, KernelEnd);
168  debug("Kernel file: %#lx-%#lx", KernelFileStart, KernelFileEnd);
169 
170  debug("File size: %ld KiB", TO_KiB(bInfo.Kernel.Size));
171  debug(".bootstrap size: %ld KiB", TO_KiB(BootstrapEnd - BootstrapStart));
172  debug(".text size: %ld KiB", TO_KiB(KernelTextEnd - KernelTextStart));
173  debug(".data size: %ld KiB", TO_KiB(KernelDataEnd - KernelDataStart));
174  debug(".rodata size: %ld KiB", TO_KiB(KernelRoDataEnd - KernelRoDataStart));
175  debug(".bss size: %ld KiB", TO_KiB(KernelBssEnd - KernelBssStart));
176 
177  uintptr_t BaseKernelMapAddress = (uintptr_t)bInfo.Kernel.PhysicalBase;
178  debug("Base kernel map address: %#lx", BaseKernelMapAddress);
179  uintptr_t k;
180  Virtual vmm = Virtual(PT);
181 
182  /* Bootstrap section */
183  if (BaseKernelMapAddress == BootstrapStart)
184  {
185  for (k = BootstrapStart; k < BootstrapEnd; k += PAGE_SIZE)
186  {
187  vmm.Map((void *)k, (void *)BaseKernelMapAddress, RW | G);
188  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
189  BaseKernelMapAddress += PAGE_SIZE;
190  }
191  }
192  else
193  {
194  trace("Ignoring bootstrap section.");
195  /* Bootstrap section must be mapped at 0x100000. */
196  }
197 
198  /* Text section */
199  for (k = KernelTextStart; k < KernelTextEnd; k += PAGE_SIZE)
200  {
201  vmm.Map((void *)k, (void *)BaseKernelMapAddress, RW | G);
202  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
203  BaseKernelMapAddress += PAGE_SIZE;
204  }
205 
206  /* Data section */
207  for (k = KernelDataStart; k < KernelDataEnd; k += PAGE_SIZE)
208  {
209  vmm.Map((void *)k, (void *)BaseKernelMapAddress, RW | G);
210  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
211  BaseKernelMapAddress += PAGE_SIZE;
212  }
213 
214  /* Read only data section */
215  for (k = KernelRoDataStart; k < KernelRoDataEnd; k += PAGE_SIZE)
216  {
217  vmm.Map((void *)k, (void *)BaseKernelMapAddress, G);
218  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
219  BaseKernelMapAddress += PAGE_SIZE;
220  }
221 
222  /* Block starting symbol section */
223  for (k = KernelBssStart; k < KernelBssEnd; k += PAGE_SIZE)
224  {
225  vmm.Map((void *)k, (void *)BaseKernelMapAddress, RW | G);
226  KernelAllocator.ReservePage((void *)BaseKernelMapAddress);
227  BaseKernelMapAddress += PAGE_SIZE;
228  }
229 
230  debug("Base kernel map address: %#lx", BaseKernelMapAddress);
231 
232  /* Kernel file */
233  if (KernelFileStart != 0)
234  {
235  for (k = KernelFileStart; k < KernelFileEnd; k += PAGE_SIZE)
236  {
237  vmm.Map((void *)k, (void *)k, G);
238  KernelAllocator.ReservePage((void *)k);
239  }
240  }
241  else
242  info("Cannot determine kernel file address. Ignoring.");
243 }

245 NIF void CreatePageTable(PageTable *pt)
246 {
247  static int check_cpuid = 0;
248 
249  if (!check_cpuid++)
250  {
251  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
252  {
253  CPU::x86::AMD::CPUID0x80000001 cpuid;
254  cpuid.Get();
255  PSESupport = cpuid.EDX.PSE;
256  Page1GBSupport = cpuid.EDX.Page1GB;
257  }
258  else if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_INTEL) == 0)
259  {
260  CPU::x86::Intel::CPUID0x00000001 cpuid;
261  cpuid.Get();
262  PSESupport = cpuid.EDX.PSE;
263  }
264 
265  if (PSESupport)
266  {
267 #if defined(a64)
268  CPU::x64::CR4 cr4 = CPU::x64::readcr4();
269  cr4.PSE = 1;
270  CPU::x64::writecr4(cr4);
271 #elif defined(a32)
272  CPU::x32::CR4 cr4 = CPU::x32::readcr4();
273  cr4.PSE = 1;
274  CPU::x32::writecr4(cr4);
275 #elif defined(aa64)
276 #endif
277  trace("PSE Support Enabled");
278  }
279 
280 #ifdef DEBUG
281  if (Page1GBSupport)
282  debug("1GB Page Support Enabled");
283 #endif
284  }
285 
286  /* TODO: Map faster */
287  MapFromZero(pt);
288  MapFramebuffer(pt);
289  MapKernel(pt);
290 
291 #ifdef DEBUG
292  tracepagetable(pt);
293 #endif
294 }

296 NIF void InitializeMemoryManagement()
297 {
298 #ifdef DEBUG
299 #ifndef a32
300  for (uint64_t i = 0; i < bInfo.Memory.Entries; i++)
301  {
302  uintptr_t Base = r_cst(uintptr_t, bInfo.Memory.Entry[i].BaseAddress);
303  size_t Length = bInfo.Memory.Entry[i].Length;
304  uintptr_t End = Base + Length;
305  const char *Type = "Unknown";
306 
307  switch (bInfo.Memory.Entry[i].Type)
308  {
309  case likely(Usable):
310  Type = "Usable";
311  break;
312  case Reserved:
313  Type = "Reserved";
314  break;
315  case ACPIReclaimable:
316  Type = "ACPI Reclaimable";
317  break;
318  case ACPINVS:
319  Type = "ACPI NVS";
320  break;
321  case BadMemory:
322  Type = "Bad Memory";
323  break;
324  case BootloaderReclaimable:
325  Type = "Bootloader Reclaimable";
326  break;
327  case KernelAndModules:
328  Type = "Kernel and Modules";
329  break;
330  case Framebuffer:
331  Type = "Framebuffer";
332  break;
333  default:
334  break;
335  }
336 
337  debug("%ld: %p-%p %s",
338  i,
339  Base,
340  End,
341  Type);
342  }
343 #endif // a32
344 #endif // DEBUG
345  trace("Initializing Physical Memory Manager");
346  // KernelAllocator = Physical(); <- Already called in the constructor
347  KernelAllocator.Init();
348  debug("Memory Info:\n\n%lld MiB / %lld MiB (%lld MiB reserved)\n",
349  TO_MiB(KernelAllocator.GetUsedMemory()),
350  TO_MiB(KernelAllocator.GetTotalMemory()),
351  TO_MiB(KernelAllocator.GetReservedMemory()));
352 
353  /* -- Debugging --
354  size_t bmap_size = KernelAllocator.GetPageBitmap().Size;
355  for (size_t i = 0; i < bmap_size; i++)
356  {
357  bool idx = KernelAllocator.GetPageBitmap().Get(i);
358  if (idx == true)
359  debug("Page %04d: %#lx", i, i * PAGE_SIZE);
360  }
361 
362  inf_loop debug("Alloc.: %#lx", KernelAllocator.RequestPage());
363  */
364 
365  trace("Initializing Virtual Memory Manager");
366  KernelPageTable = (PageTable *)KernelAllocator.RequestPages(TO_PAGES(PAGE_SIZE + 1));
367  memset(KernelPageTable, 0, PAGE_SIZE);
368 
369  CreatePageTable(KernelPageTable);
370 
371  trace("Applying new page table from address %#lx",
372  KernelPageTable);
373  CPU::PageTable(KernelPageTable);
374  debug("Page table updated.");
375 
376  /* FIXME: Read kernel params */
377  AllocatorType = Config.AllocatorType;
378 
379  switch (AllocatorType)
380  {
381  case MemoryAllocatorType::Pages:
382  break;
383  case MemoryAllocatorType::XallocV1:
384  {
385  XallocV1Allocator = new Xalloc::V1((void *)KERNEL_HEAP_BASE, false, false);
386  trace("XallocV1 Allocator initialized at %#lx", XallocV1Allocator);
387  break;
388  }
389  case MemoryAllocatorType::XallocV2:
390  {
391  XallocV2Allocator = new Xalloc::V2((void *)KERNEL_HEAP_BASE);
392  trace("XallocV2 Allocator initialized at %#lx", XallocV2Allocator);
393  break;
394  }
395  case MemoryAllocatorType::liballoc11:
396  break;
397  case MemoryAllocatorType::rpmalloc_:
398  {
399  trace("Using rpmalloc allocator");
400  rpmalloc_initialize();
401  break;
402  rpmalloc_config_t config = {
403  .memory_map = nullptr,
404  .memory_unmap = nullptr,
405  .error_callback = nullptr,
406  .map_fail_callback = nullptr,
407  .page_size = PAGE_SIZE,
408  .span_size = 4 * 1024, /* 4 KiB */
409  .span_map_count = 1,
410  .enable_huge_pages = 0,
411  .page_name = nullptr,
412  .huge_page_name = nullptr};
413  rpmalloc_initialize_config(&config);
414  break;
415  }
416  default:
417  {
418  error("Unknown allocator type %d", AllocatorType);
419  CPU::Stop();
420  }
421  }
422 }
