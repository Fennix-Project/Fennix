62  Elf64_Sym *ELFLookupSymbol(Elf64_Ehdr *Header, const char *Name)
63  {
64  Elf64_Shdr *SymbolTable = nullptr;
65  Elf64_Shdr *StringTable = nullptr;
66 
67  for (Elf64_Half i = 0; i < Header->e_shnum; i++)
68  {
69  Elf64_Shdr *shdr = GetELFSection(Header, i);
70  switch (shdr->sh_type)
71  {
72  case SHT_SYMTAB:
73  SymbolTable = shdr;
74  StringTable = GetELFSection(Header, shdr->sh_link);
75  break;
76  default:
77  {
78  break;
79  }
80  }
81  }
82 
83  if (SymbolTable == nullptr || StringTable == nullptr)
84  return nullptr;
85 
86  for (size_t i = 0; i < (SymbolTable->sh_size / sizeof(Elf64_Sym)); i++)
87  {
88  Elf64_Sym *Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
89  char *String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
90  if (strcmp(String, Name) == 0)
91  return Symbol;
92  }
93  return nullptr;
94  }

96  Elf64_Sym ELFLookupSymbol(int fd, const char *Name)
97  {
98  off_t OldOffset = lseek(fd, 0, SEEK_CUR);
99 
100  Elf64_Ehdr Header;
101  lseek(fd, 0, SEEK_SET);
102  fread(fd, (uint8_t *)&Header, sizeof(Elf64_Ehdr));
103 
104  Elf64_Shdr SymbolTable;
105  Elf64_Shdr StringTable;
106 
107  for (Elf64_Half i = 0; i < Header.e_shnum; i++)
108  {
109  Elf64_Shdr shdr;
110  lseek(fd, Header.e_shoff + (i * sizeof(Elf64_Shdr)), SEEK_SET);
111  fread(fd, (uint8_t *)&shdr, sizeof(Elf64_Shdr));
112 
113  switch (shdr.sh_type)
114  {
115  case SHT_SYMTAB:
116  SymbolTable = shdr;
117  lseek(fd, Header.e_shoff + (shdr.sh_link * sizeof(Elf64_Shdr)), SEEK_SET);
118  fread(fd, (uint8_t *)&StringTable, sizeof(Elf64_Shdr));
119  break;
120  default:
121  {
122  break;
123  }
124  }
125  }
126 
127  if (SymbolTable.sh_name == 0 ||
128  StringTable.sh_name == 0)
129  {
130  error("Symbol table not found.");
131  lseek(fd, OldOffset, SEEK_SET);
132  return {};
133  }
134 
135  for (size_t i = 0; i < (SymbolTable.sh_size / sizeof(Elf64_Sym)); i++)
136  {
137  // Elf64_Sym *Symbol = (Elf64_Sym *)((uintptr_t)Header + SymbolTable->sh_offset + (i * sizeof(Elf64_Sym)));
138  Elf64_Sym Symbol;
139  lseek(fd, SymbolTable.sh_offset + (i * sizeof(Elf64_Sym)), SEEK_SET);
140  fread(fd, (uint8_t *)&Symbol, sizeof(Elf64_Sym));
141 
142  // char *String = (char *)((uintptr_t)Header + StringTable->sh_offset + Symbol->st_name);
143  char String[256];
144  lseek(fd, StringTable.sh_offset + Symbol.st_name, SEEK_SET);
145  fread(fd, (uint8_t *)&String, 256);
146 
147  if (strcmp(String, Name) == 0)
148  {
149  lseek(fd, OldOffset, SEEK_SET);
150  return Symbol;
151  }
152  }
153  error("Symbol not found.");
154  lseek(fd, OldOffset, SEEK_SET);
155  return {};
156  }
