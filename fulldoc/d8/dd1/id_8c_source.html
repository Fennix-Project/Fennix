76 int
77 id_builtin (list)
78  WORD_LIST *list;
79 {
80  int opt;
81  char *user;
82 
83  id_flags = 0;
84  reset_internal_getopt ();
85  while ((opt = internal_getopt (list, "Ggnru")) != -1)
86  {
87  switch (opt)
88  {
89  case 'G': id_flags |= ID_ALLGROUPS; break;
90  case 'g': id_flags |= ID_GIDONLY; break;
91  case 'n': id_flags |= ID_USENAME; break;
92  case 'r': id_flags |= ID_USEREAL; break;
93  case 'u': id_flags |= ID_USERONLY; break;
94  CASE_HELPOPT;
95  default:
96  builtin_usage ();
97  return (EX_USAGE);
98  }
99  }
100  list = loptend;
101 
102  user = list ? list->word->word : (char *)NULL;
103 
104  /* Check for some invalid option combinations */
105  opt = ID_FLAGSET (ID_ALLGROUPS) + ID_FLAGSET (ID_GIDONLY) + ID_FLAGSET (ID_USERONLY);
106  if (opt > 1 || (opt == 0 && ((id_flags & (ID_USEREAL|ID_USENAME)) != 0)))
107  {
108  builtin_usage ();
109  return (EX_USAGE);
110  }
111 
112  if (list && list->next)
113  {
114  builtin_usage ();
115  return (EX_USAGE);
116  }
117 
118  if (inituser (user) < 0)
119  return (EXECUTION_FAILURE);
120 
121  opt = 0;
122  if (id_flags & ID_USERONLY)
123  opt += id_pruser ((id_flags & ID_USEREAL) ? ruid : euid);
124  else if (id_flags & ID_GIDONLY)
125  opt += id_prgrp ((id_flags & ID_USEREAL) ? rgid : egid);
126  else if (id_flags & ID_ALLGROUPS)
127  opt += id_prgroups (user);
128  else
129  opt += id_prall (user);
130  putchar ('\n');
131  fflush (stdout);
132 
133  return (opt == 0 ? EXECUTION_SUCCESS : EXECUTION_FAILURE);
134 }
