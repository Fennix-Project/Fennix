113 int vconfig_main(int argc, char **argv)
114 {
115  struct vlan_ioctl_args ifr;
116  const char *p;
117  int fd;
118 
119  memset(&ifr, 0, sizeof(ifr));
120 
121  ++argv;
122  if (!argv[0])
123  bb_show_usage();
124  p = xfind_str(cmds + 2, argv[0]);
125  ifr.cmd = *p;
126  if (argc != p[-1])
127  bb_show_usage();
128 
129  if (ifr.cmd == SET_VLAN_NAME_TYPE_CMD) {
130  /* set_name_type */
131  ifr.u.name_type = *xfind_str(name_types + 1, argv[1]);
132  } else {
133  strncpy_IFNAMSIZ(ifr.device1, argv[1]);
134  p = argv[2];
135 
136  /* I suppose one could try to combine some of the function calls below,
137  * since ifr.u.flag, ifr.u.VID, and ifr.u.skb_priority are all same-sized
138  * (unsigned) int members of a unions. But because of the range checking,
139  * doing so wouldn't save that much space and would also make maintenance
140  * more of a pain.
141  */
142  if (ifr.cmd == SET_VLAN_FLAG_CMD) {
143  /* set_flag */
144  ifr.u.flag = xatou_range(p, 0, 1);
145  /* DM: in order to set reorder header, qos must be set */
146  ifr.vlan_qos = xatou_range(argv[3], 0, 7);
147  } else if (ifr.cmd == ADD_VLAN_CMD) {
148  /* add */
149  ifr.u.VID = xatou_range(p, 0, VLAN_GROUP_ARRAY_LEN - 1);
150  } else if (ifr.cmd != DEL_VLAN_CMD) {
151  /* set_{egress|ingress}_map */
152  ifr.u.skb_priority = xatou(p);
153  ifr.vlan_qos = xatou_range(argv[3], 0, 7);
154  }
155  }
156 
157  fd = xsocket(AF_INET, SOCK_STREAM, 0);
158  ioctl_or_perror_and_die(fd, SIOCSIFVLAN, &ifr,
159  "ioctl error for %s", argv[0]);
160 
161  return 0;
162 }
