31 void PS2MouseInterruptHandler(TrapFrame *)
32 {
33  uint8_t data = PS2ReadData();
34  if (data == PS2_MOUSE_RESP_ACK ||
35  data == PS2_MOUSE_RESP_RESEND)
36  return;
37 
38  if (!PacketReady)
39  {
40  switch (Cycle)
41  {
42  case 0:
43  {
44  if ((data & 0b00001000 /* Always 1 */) == 0)
45  return;
46 
47  Packet.Base.Raw = data;
48  Cycle++;
49  break;
50  }
51  case 1:
52  {
53  Packet.XMovement = data;
54  Cycle++;
55  break;
56  }
57  case 2:
58  {
59  Packet.YMovement = data;
60  if (FourPackets)
61  Cycle++;
62  else
63  {
64  Cycle = 0;
65  PacketReady = true;
66  }
67  break;
68  }
69  case 3:
70  {
71  Packet.ZMovement.Raw = data;
72  Cycle = 0;
73  PacketReady = true;
74  break;
75  }
76  default:
77  break;
78  }
79  return;
80  }

200 int DetectPS2Mouse()
201 {
202  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
203  PS2WriteData(PS2_MOUSE_CMD_DISABLE_DATA_REPORTING);
204  if (PS2ACKTimeout() != 0)
205  Log("PS/2 mouse failed to disable data reporting!");
206 
207  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
208  PS2WriteData(PS2_MOUSE_CMD_READ_ID);
209  if (PS2ACKTimeout() != 0)
210  Log("PS/2 mouse failed to read ID!");
211 
212  uint8_t recByte;
213  int timeout = 1000000;
214  while (timeout--)
215  {
216  recByte = PS2ReadData();
217  if (recByte != PS2_ACK)
218  break;
219  }
220  Device2ID[0] = recByte;
221 
222  timeout = 1000000;
223  while (timeout--)
224  {
225  recByte = PS2ReadData();
226  if (recByte != PS2_ACK)
227  break;
228  }
229  Device2ID[1] = recByte;
230 
231  Log("PS2 Mouse Device: 0x%X 0x%X", Device2ID[0], Device2ID[1]);
232  return 0;
