33 void PS2MouseInterruptHandler(TrapFrame *)
34 {
35  uint8_t data = PS2ReadData();
36  if (data == PS2_MOUSE_RESP_ACK ||
37  data == PS2_MOUSE_RESP_RESEND)
38  return;
39 
40  if (!PacketReady)
41  {
42  switch (Cycle)
43  {
44  case 0:
45  {
46  if ((data & 0b00001000 /* Always 1 */) == 0)
47  return;
48 
49  Packet.Base.Raw = data;
50  Cycle++;
51  break;
52  }
53  case 1:
54  {
55  Packet.XMovement = data;
56  Cycle++;
57  break;
58  }
59  case 2:
60  {
61  Packet.YMovement = data;
62  if (FourPackets)
63  Cycle++;
64  else
65  {
66  Cycle = 0;
67  PacketReady = true;
68  }
69  break;
70  }
71  case 3:
72  {
73  Packet.ZMovement.Raw = data;
74  Cycle = 0;
75  PacketReady = true;
76  break;
77  }
78  default:
79  break;
80  }
81  return;
82  }

153 int InitializeMouse()
154 {
155  PS2WriteData(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
156  PS2WriteData(PS2_MOUSE_CMD_RESET);
157  uint8_t test = PS2ReadData();
158  if (test != PS2_MOUSE_RESP_TEST_PASSED &&
159  test != PS2_MOUSE_RESP_ACK)
160  {
161  KernelLog("PS/2 mouse reset failed! (%#x)", test);
162  return -EFAULT;
163  }
164 
165  RegisterInterruptHandler(12, PS2MouseInterruptHandler);
166 
167  MouseDevID = RegisterDevice(INPUT_TYPE_MOUSE, &MouseOps);
168 
169  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
170  PS2WriteData(PS2_MOUSE_CMD_SET_DEFAULTS);
171  PS2ReadData();
172 
173  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
174  PS2WriteData(PS2_MOUSE_CMD_ENABLE_DATA_REPORTING);
175 
176  MouseSampleRate(200);

178  MouseSampleRate(80);
179 
180  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
181  PS2WriteData(PS2_MOUSE_CMD_READ_ID);
182  uint8_t Device2ID = PS2ReadData();
183  KernelLog("PS/2 Mouse ID: %#x", Device2ID);
184 
185  MouseSampleRate(200);
186  MouseSampleRate(200);
187  MouseSampleRate(80);
188 
189  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
190  PS2WriteData(PS2_MOUSE_CMD_READ_ID);
191  Device2ID = PS2ReadData();
192  KernelLog("PS/2 Mouse ID: %#x", Device2ID);

213  PS2WriteCommand(PS2_CMD_WRITE_NEXT_BYTE_TO_PS2_PORT_2_INPUT);
214  PS2WriteData(PS2_MOUSE_CMD_DISABLE_DATA_REPORTING);

219  PS2WriteData(PS2_MOUSE_CMD_READ_ID);
220  if (PS2ACKTimeout() != 0)
221  KernelLog("PS/2 mouse failed to read ID!");

223  uint8_t recByte;

226  {
227  recByte = PS2ReadData();

229  break;
230  }

232 
233  timeout = 1000000;
234  while (timeout--)
235  {

239  }
