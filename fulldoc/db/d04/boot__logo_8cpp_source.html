47 void BootLogoAnimationThread()
48 {
49  char BootAnimPath[16];
50  while (FrameCount < 27)

52  sprintf(BootAnimPath, "/etc/boot/%ld.tga", FrameCount);
53  RefNode *frame = fs->Open(BootAnimPath);
54  if (!frame)
55  {
56  debug("Failed to load boot animation frame %s", BootAnimPath);
