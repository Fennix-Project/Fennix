162 /* https://man7.org/linux/man-pages/man3/mmap.3p.html */
163 static void *sys_mmap(void *addr, size_t length, int prot,
164  int flags, int fildes, off_t offset)
165 {
166  UNUSED(offset); /* FIXME */
167  Tasking::PCB *pcb = thisProcess;
168  Memory::MemMgr *mgr = pcb->Memory;
169 
170  if (!addr)
171  {
172  void *newPages = mgr->RequestPages(TO_PAGES(length));
173  if (newPages == nullptr)
174  return MAP_FAILED;
175 
176  Memory::Virtual vma = Memory::Virtual(pcb->PageTable);
177 
178  if (prot & PROT_READ)
179  vma.Map(newPages, newPages, length, Memory::P | Memory::US, Memory::Virtual::FourKiB);
180  if (prot & PROT_WRITE)
181  vma.Map(newPages, newPages, length, Memory::RW, Memory::Virtual::FourKiB);
182  if (prot & PROT_EXEC)
183  fixme("PROT_EXEC not implemented");
184 
185  if (flags & MAP_FILE)
186  fixme("MAP_FILE not implemented");
187  if (flags & MAP_SHARED)
188  fixme("MAP_SHARED not implemented");
189  if (flags & MAP_PRIVATE)
190  fixme("MAP_PRIVATE not implemented");
191  if (flags & MAP_SHARED_VALIDATE)
192  fixme("MAP_SHARED_VALIDATE not implemented");
193  if (flags & MAP_TYPE)
194  fixme("MAP_TYPE not implemented");
195  if (flags & MAP_FIXED)
196  fixme("MAP_FIXED not implemented");
197  if (flags & MAP_ANONYMOUS)
198  fixme("MAP_ANONYMOUS not implemented");
199 
200  if (fildes != -1)
201  {
202  fixme("File mapping not implemented");
203  return MAP_FAILED;
204  }
205 
206  return newPages;
207  }
208 
209  stub;
210  return MAP_FAILED;
211 }

214 static int sys_mprotect(void *addr, size_t len, int prot)
215 {
216  Tasking::PCB *pcb = thisProcess;
217 
218  Memory::Virtual vma = Memory::Virtual(pcb->PageTable);
219  vma.Map(addr, addr, len, Memory::P, Memory::Virtual::FourKiB);
220 
221  if (prot & PROT_READ)
222  vma.Map(addr, addr, len, Memory::P | Memory::US, Memory::Virtual::FourKiB);
223  if (prot & PROT_WRITE)
224  vma.Map(addr, addr, len, Memory::RW, Memory::Virtual::FourKiB);
225  if (prot & PROT_EXEC)
226  fixme("PROT_EXEC not implemented");
227 
228  return 0;
229 }

251 /* https://man7.org/linux/man-pages/man3/readv.3p.html */
252 static ssize_t sys_readv(int fildes, const struct iovec *iov, int iovcnt)
253 {
254  ssize_t Total = 0;
255  for (int i = 0; i < iovcnt; i++)
256  {
257  ssize_t n = sys_read(fildes, iov[i].iov_base, iov[i].iov_len);
258  if (n < 0)
259  return n;
260 
261  Total += n;
262  if (n < (ssize_t)iov[i].iov_len)
263  break;
264  }
265  return Total;
266 }

307 /* https://man7.org/linux/man-pages/man2/arch_prctl.2.html */
308 static int sys_arch_prctl(int code, unsigned long addr)
309 {
310  switch (code)
311  {
312  case 0x1001: // ARCH_SET_GS
313  {
314 #if defined(a64)
315  CPU::x64::wrmsr(CPU::x64::MSRID::MSR_GS_BASE, addr);
316 #elif defined(a32)
317  CPU::x32::wrmsr(CPU::x32::MSRID::MSR_GS_BASE, addr);
318 #endif
319  return 0;
320  }
321  case 0x1002: // ARCH_SET_FS
322  {
323 #if defined(a64)
324  CPU::x64::wrmsr(CPU::x64::MSRID::MSR_FS_BASE, addr);
325 #elif defined(a32)
326  CPU::x32::wrmsr(CPU::x32::MSRID::MSR_FS_BASE, addr);
327 #endif
328  return 0;
329  }
330  case 0x1003: // ARCH_GET_FS
331  {
332 #if defined(a64)
333  *r_cst(uint64_t *, addr) =
334  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_FS_BASE);
335 #elif defined(a32)
336  *r_cst(uint64_t *, addr) =
337  CPU::x32::rdmsr(CPU::x32::MSRID::MSR_FS_BASE);
338 #endif
339  return 0;
340  }
341  case 0x1004: // ARCH_GET_GS
342  {
343 #if defined(a64)
344  *r_cst(uint64_t *, addr) =
345  CPU::x64::rdmsr(CPU::x64::MSRID::MSR_GS_BASE);
346 #elif defined(a32)
347  *r_cst(uint64_t *, addr) =
348  CPU::x32::rdmsr(CPU::x32::MSRID::MSR_GS_BASE);
349 #endif
350  return 0;
351  }
352  default:
353  fixme("code=%d", code);
354  return -ENOSYS;
355  }
356 }
