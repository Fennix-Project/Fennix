56  Tasking::PCB *pcb = thisProcess;

59 
60  for (uintptr_t i = uintptr_t(addr);
61  i < uintptr_t(addr) + len;
62  i += PAGE_SIZE)
63  {
64  if (likely(!vmm.Check((void *)i, Memory::G)))
65  vmm.Remap((void *)i, (void *)i, Memory::P | Memory::RW);
66  else

68  }
