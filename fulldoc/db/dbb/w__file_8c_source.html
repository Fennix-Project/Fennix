53 wad_file_t *W_OpenFile(char *path)
54 {
55  wad_file_t *result;
56  int i;
57 
59  // Use the OS's virtual memory subsystem to map WAD files
60  // directly into memory.
61  //
62 
63  if (!M_CheckParm("-mmap"))
64  {
65  return stdc_wad_file.OpenFile(path);
66  }
67 
68  // Try all classes in order until we find one that works
69 
70  result = NULL;
71 
72  for (i = 0; i < arrlen(wad_file_classes); ++i)
73  {
74  result = wad_file_classes[i]->OpenFile(path);
75 
76  if (result != NULL)
77  {
78  break;
79  }
80  }
81 
82  return result;
83 }
