59  struct rtattr * tb[RTA_MAX+1];
60 
61  if (n->nlmsg_type != RTM_NEWRULE)
62  return 0;
63 
64  len -= NLMSG_LENGTH(sizeof(*r));
65  if (len < 0)
66  return -1;
67 
68  //memset(tb, 0, sizeof(tb)); - parse_rtattr does this
69  parse_rtattr(tb, RTA_MAX, RTM_RTA(r), len);

71  if (r->rtm_family == AF_INET)

73  else if (r->rtm_family == AF_INET6)

75 /* else if (r->rtm_family == AF_DECnet)

77  else if (r->rtm_family == AF_IPX)
78  host_len = 80;

80  printf("%u:\t", tb[RTA_PRIORITY] ?
81  *(unsigned*)RTA_DATA(tb[RTA_PRIORITY])

98  } else {
99  printf("all");
100  }
101  bb_putchar(' ');
102 
103  if (tb[RTA_DST]) {
104  if (r->rtm_dst_len != host_len) {
105  printf("to %s/%u ", rt_addr_n2a(r->rtm_family,

211 static int iprule_modify(int cmd, char **argv)
212 {
213  bool table_ok = 0;
214  struct rtnl_handle rth;
215  struct {
216  struct nlmsghdr n;
217  struct rtmsg r;
218  char buf[1024];
219  } req;
220  smalluint key;
221 
222  memset(&req, 0, sizeof(req));
223 
224  req.n.nlmsg_type = cmd;
225  req.n.nlmsg_len = NLMSG_LENGTH(sizeof(struct rtmsg));
226  req.n.nlmsg_flags = NLM_F_REQUEST;
227  req.r.rtm_family = preferred_family;
228  req.r.rtm_protocol = RTPROT_BOOT;
229  if (RT_SCOPE_UNIVERSE != 0)
230  req.r.rtm_scope = RT_SCOPE_UNIVERSE;
231  /*req.r.rtm_table = 0; - already is */
232  if (RTN_UNSPEC != 0)
233  req.r.rtm_type = RTN_UNSPEC;
234 
235  if (cmd == RTM_NEWRULE) {
236  req.n.nlmsg_flags |= NLM_F_CREATE|NLM_F_EXCL;
237  req.r.rtm_type = RTN_UNICAST;
238  }
239 
240  while (*argv) {
241  key = index_in_substrings(keywords, *argv) + 1;
242  if (key == 0) /* no match found in keywords array, bail out. */
243  invarg_1_to_2(*argv, applet_name);
244  if (key == ARG_from) {
245  inet_prefix dst;
246  NEXT_ARG();
247  get_prefix(&dst, *argv, req.r.rtm_family);
248  req.r.rtm_src_len = dst.bitlen;
249  addattr_l(&req.n, sizeof(req), RTA_SRC, &dst.data, dst.bytelen);
250  } else if (key == ARG_to) {
251  inet_prefix dst;
252  NEXT_ARG();
253  get_prefix(&dst, *argv, req.r.rtm_family);
254  req.r.rtm_dst_len = dst.bitlen;
255  addattr_l(&req.n, sizeof(req), RTA_DST, &dst.data, dst.bytelen);
256  } else if (key == ARG_preference ||
257  key == ARG_order ||
258  key == ARG_priority
259  ) {
260  uint32_t pref;
261  NEXT_ARG();
262  pref = get_u32(*argv, keyword_preference);
263  addattr32(&req.n, sizeof(req), RTA_PRIORITY, pref);
264  } else if (key == ARG_tos) {
265  uint32_t tos;
266  NEXT_ARG();
267  if (rtnl_dsfield_a2n(&tos, *argv))
268  invarg_1_to_2(*argv, "TOS");
269  req.r.rtm_tos = tos;
270  } else if (key == ARG_fwmark) {
271  char *slash;
272  uint32_t fwmark, fwmask;
273  NEXT_ARG();
274  slash = strchr(*argv, '/');
275  if (slash)
276  *slash = '\0';
277  fwmark = get_u32(*argv, keyword_fwmark);
278  addattr32(&req.n, sizeof(req), FRA_FWMARK, fwmark);
279  if (slash) {
280  fwmask = get_u32(slash + 1, "fwmask");
281  addattr32(&req.n, sizeof(req), FRA_FWMASK, fwmask);
282  }
283  } else if (key == ARG_realms) {
284  uint32_t realm;
285  NEXT_ARG();
286  if (get_rt_realms(&realm, *argv))
287  invarg_1_to_2(*argv, keyword_realms);
288  addattr32(&req.n, sizeof(req), RTA_FLOW, realm);
289  } else if (key == ARG_table ||
290  key == ARG_lookup
291  ) {
292  uint32_t tid;
293  NEXT_ARG();
294  if (rtnl_rttable_a2n(&tid, *argv))
295  invarg_1_to_2(*argv, "table ID");
296 
297 #if HAVE_RTA_TABLE
298  if (tid > 255) {
299  req.r.rtm_table = RT_TABLE_UNSPEC;
300  addattr32(&req.n, sizeof(req), RTA_TABLE, tid);
301  } else
302 #endif
303  req.r.rtm_table = tid;
304 
305  table_ok = 1;
306  } else if (key == ARG_suppress_prefixlength) {
307  int prefix_length;
308  NEXT_ARG();
309  prefix_length = get_u32(*argv, keyword_suppress_prefixlength);
310  addattr32(&req.n, sizeof(req), FRA_SUPPRESS_PREFIXLEN, prefix_length);
311  } else if (key == ARG_suppress_ifgroup) {
312  int grp;
313  NEXT_ARG();
314  grp = get_u32(*argv, keyword_suppress_ifgroup);
315  addattr32(&req.n, sizeof(req), FRA_SUPPRESS_IFGROUP, grp);
316  } else if (key == ARG_dev ||
317  key == ARG_iif
318  ) {
319  NEXT_ARG();
320  addattr_l(&req.n, sizeof(req), RTA_IIF, *argv, strlen(*argv)+1);
321  } else if (key == ARG_nat ||
322  key == ARG_map_to
323  ) {
324  NEXT_ARG();
325  addattr32(&req.n, sizeof(req), RTA_GATEWAY, get_addr32(*argv));
326  req.r.rtm_type = RTN_NAT;
327  } else {
328  int type;
329 
330  if (key == ARG_type) {
331  NEXT_ARG();
332  }
333  if (key == ARG_help)
334  bb_show_usage();
335  if (rtnl_rtntype_a2n(&type, *argv))
336  invarg_1_to_2(*argv, "type");
337  req.r.rtm_type = type;
338  }
339  argv++;
340  }
341 
342  if (req.r.rtm_family == AF_UNSPEC)
343  req.r.rtm_family = AF_INET;
344 
345  if (!table_ok && cmd == RTM_NEWRULE)
346  req.r.rtm_table = RT_TABLE_MAIN;
347 
348  xrtnl_open(&rth);
349 
350  if (rtnl_talk(&rth, &req.n, 0, 0, NULL, NULL, NULL) < 0)
351  return 2;
352 
353  return 0;
354 }

358 {

368  }
