22 uint32_t* FAST_FUNC crc32_filltable(uint32_t *crc_table, int endian)
23 {
24  uint32_t polynomial = endian ? 0x04c11db7 : 0xedb88320;
25  uint32_t c;
26  unsigned i, j;
27 
28  if (!crc_table)
29  crc_table = xmalloc(256 * sizeof(uint32_t));
30 
31  for (i = 0; i < 256; i++) {
32  c = endian ? (i << 24) : i;
33  for (j = 8; j; j--) {
34  if (endian)
35  c = (c&0x80000000) ? ((c << 1) ^ polynomial) : (c << 1);
36  else
37  c = (c&1) ? ((c >> 1) ^ polynomial) : (c >> 1);
38  }
39  *crc_table++ = c;
40  }
41 
42  return crc_table - 256;
43 }
