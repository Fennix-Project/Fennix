92 
93  struct limine_framebuffer_response *FrameBufferResponse = FramebufferRequest.response;
94  struct limine_memmap_response *MemmapResponse = MemmapRequest.response;
95  struct limine_kernel_address_response *KernelAddressResponse = KernelAddressRequest.response;
96  struct limine_rsdp_response *RsdpResponse = RsdpRequest.response;
97  struct limine_kernel_file_response *KernelFileResponse = KernelFileRequest.response;
98  struct limine_module_response *ModuleResponse = ModuleRequest.response;
99  struct limine_smbios_response *SmbiosResponse = SmbiosRequest.response;
100 
101  if (FrameBufferResponse == NULL || FrameBufferResponse->framebuffer_count < 1)
102  {
103  error("No framebuffer available [%#lx;%ld]", FrameBufferResponse,
104  (FrameBufferResponse == NULL) ? 0 : FrameBufferResponse->framebuffer_count);
105  inf_loop asmv("hlt");
106  }
107 
108  if (MemmapResponse == NULL || MemmapResponse->entry_count < 1)
109  {
110  error("No memory map available [%#lx;%ld]", MemmapResponse,
111  (MemmapResponse == NULL) ? 0 : MemmapResponse->entry_count);
112  inf_loop asmv("hlt");
113  }
114 
115  if (KernelAddressResponse == NULL)
116  {
117  error("No kernel address available [%#lx]", KernelAddressResponse);
118  inf_loop asmv("hlt");
119  }
120 
121  if (RsdpResponse == NULL || RsdpResponse->address == 0)
122  {
123  error("No RSDP address available [%#lx;%#lx]", RsdpResponse,
124  (RsdpResponse == NULL) ? 0 : RsdpResponse->address);
125  inf_loop asmv("hlt");
126  }
127 
128  if (KernelFileResponse == NULL || KernelFileResponse->kernel_file == NULL)
129  {
130  error("No kernel file available [%#lx;%#lx]", KernelFileResponse,
131  (KernelFileResponse == NULL) ? 0 : KernelFileResponse->kernel_file);
132  inf_loop asmv("hlt");
133  }
134 
135  /* Actual parsing starts here */
136 
137  for (uint64_t i = 0; i < FrameBufferResponse->framebuffer_count; i++)
138  {
139  struct limine_framebuffer *framebuffer = FrameBufferResponse->framebuffers[i];
140  switch (framebuffer->memory_model)
141  {
142  case LIMINE_FRAMEBUFFER_RGB:
143  binfo.Framebuffer[i].Type = RGB;
144  break;
145  default:
146  {
147  error("Unsupported framebuffer memory model %d", framebuffer->memory_model);
148  inf_loop asmv("hlt");
149  }
150  }
151  binfo.Framebuffer[i].BaseAddress = (void *)((uintptr_t)framebuffer->address - 0xFFFF800000000000);
152  binfo.Framebuffer[i].Width = (uint32_t)framebuffer->width;
153  binfo.Framebuffer[i].Height = (uint32_t)framebuffer->height;
154  binfo.Framebuffer[i].Pitch = (uint32_t)framebuffer->pitch;
155  binfo.Framebuffer[i].BitsPerPixel = framebuffer->bpp;
156  binfo.Framebuffer[i].RedMaskSize = framebuffer->red_mask_size;
157  binfo.Framebuffer[i].RedMaskShift = framebuffer->red_mask_shift;
158  binfo.Framebuffer[i].GreenMaskSize = framebuffer->green_mask_size;
159  binfo.Framebuffer[i].GreenMaskShift = framebuffer->green_mask_shift;
160  binfo.Framebuffer[i].BlueMaskSize = framebuffer->blue_mask_size;
161  binfo.Framebuffer[i].BlueMaskShift = framebuffer->blue_mask_shift;
162  binfo.Framebuffer[i].ExtendedDisplayIdentificationData = framebuffer->edid;
163  binfo.Framebuffer[i].EDIDSize = framebuffer->edid_size;
164 
165  debug("Framebuffer %d: %dx%d %d bpp", i,
166  binfo.Framebuffer[i].Width,
167  binfo.Framebuffer[i].Height,
168  binfo.Framebuffer[i].BitsPerPixel);
169 
170  debug("More info:\nAddress: %#lx\nPitch: %ld\nType: %d\nRedMaskSize: %d\nRedMaskShift: %d\nGreenMaskSize: %d\nGreenMaskShift: %d\nBlueMaskSize: %d\nBlueMaskShift: %d\nEDID: %#lx\nEDIDSize: %d",
171  binfo.Framebuffer[i].BaseAddress,
172  binfo.Framebuffer[i].Pitch,
173  binfo.Framebuffer[i].Type,
174  binfo.Framebuffer[i].RedMaskSize,
175  binfo.Framebuffer[i].RedMaskShift,
176  binfo.Framebuffer[i].GreenMaskSize,
177  binfo.Framebuffer[i].GreenMaskShift,
178  binfo.Framebuffer[i].BlueMaskSize,
179  binfo.Framebuffer[i].BlueMaskShift,
180  binfo.Framebuffer[i].ExtendedDisplayIdentificationData,
181  binfo.Framebuffer[i].EDIDSize);
182  }
183 
184  binfo.Memory.Entries = MemmapResponse->entry_count;
185  for (uint64_t i = 0; i < MemmapResponse->entry_count; i++)
186  {
187  if (MemmapResponse->entry_count > MAX_MEMORY_ENTRIES)
188  {
189  warn("Too many memory entries, skipping the rest...");
190  break;
191  }
192 
193  struct limine_memmap_entry *entry = MemmapResponse->entries[i];
194  if (!entry)
195  {
196  warn("Null memory entry %ld (%#lx), skipping...", i, entry);
197  continue;
198  }
199 
200  binfo.Memory.Size += entry->length;
201  switch (entry->type)
202  {
203  case LIMINE_MEMMAP_USABLE:
204  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
205  binfo.Memory.Entry[i].Length = entry->length;
206  binfo.Memory.Entry[i].Type = Usable;
207  break;
208  case LIMINE_MEMMAP_RESERVED:
209  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
210  binfo.Memory.Entry[i].Length = entry->length;
211  binfo.Memory.Entry[i].Type = Reserved;
212  break;
213  case LIMINE_MEMMAP_ACPI_RECLAIMABLE:
214  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
215  binfo.Memory.Entry[i].Length = entry->length;
216  binfo.Memory.Entry[i].Type = ACPIReclaimable;
217  break;
218  case LIMINE_MEMMAP_ACPI_NVS:
219  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
220  binfo.Memory.Entry[i].Length = entry->length;
221  binfo.Memory.Entry[i].Type = ACPINVS;
222  break;
223  case LIMINE_MEMMAP_BAD_MEMORY:
224  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
225  binfo.Memory.Entry[i].Length = entry->length;
226  binfo.Memory.Entry[i].Type = BadMemory;
227  break;
228  case LIMINE_MEMMAP_BOOTLOADER_RECLAIMABLE:
229  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
230  binfo.Memory.Entry[i].Length = entry->length;
231  binfo.Memory.Entry[i].Type = BootloaderReclaimable;
232  break;
233  case LIMINE_MEMMAP_KERNEL_AND_MODULES:
234  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
235  binfo.Memory.Entry[i].Length = entry->length;
236  binfo.Memory.Entry[i].Type = KernelAndModules;
237  break;
238  case LIMINE_MEMMAP_FRAMEBUFFER:
239  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
240  binfo.Memory.Entry[i].Length = entry->length;
241  binfo.Memory.Entry[i].Type = Framebuffer;
242  break;
243  default:
244  binfo.Memory.Entry[i].BaseAddress = (void *)entry->base;
245  binfo.Memory.Entry[i].Length = entry->length;
246  binfo.Memory.Entry[i].Type = Unknown;
247  break;
248  }
249  }
250 
251  if (ModuleResponse != NULL && ModuleResponse->module_count > 0)
252  {
253  for (uint64_t i = 0; i < ModuleResponse->module_count; i++)
254  {
255  if (i > MAX_MODULES)
256  {
257  warn("Too many modules, skipping the rest...");
258  break;
259  }
260 
261  binfo.Modules[i].Address = (void *)((uint64_t)ModuleResponse->modules[i]->address - 0xFFFF800000000000);
262  binfo.Modules[i].Size = ModuleResponse->modules[i]->size;
263 
264  strncpy(binfo.Modules[i].Path,
265  ModuleResponse->modules[i]->path,
266  strlen(ModuleResponse->modules[i]->path) + 1);
267 
268  strncpy(binfo.Modules[i].CommandLine,
269  ModuleResponse->modules[i]->cmdline,
270  strlen(ModuleResponse->modules[i]->cmdline) + 1);
271 
272  debug("Module %d:\nAddress: %#lx\nPath: \"%s\"\nCommand Line: \"%s\"\nSize: %ld",
273  i,
274  binfo.Modules[i].Address,
275  binfo.Modules[i].Path,
276  binfo.Modules[i].CommandLine,
277  binfo.Modules[i].Size);
278  }
279  }
280 
281  binfo.RSDP = (struct RSDPInfo *)((uintptr_t)RsdpResponse->address - 0xFFFF800000000000);
282  debug("RSDP: %#lx [Signature: \"%.8s\"] [OEM: \"%.6s\"]",
283  binfo.RSDP, binfo.RSDP->Signature, binfo.RSDP->OEMID);
284 
285  if (SmbiosResponse->entry_64 != NULL)
286  binfo.SMBIOSPtr = (void *)((uintptr_t)SmbiosResponse->entry_64 - 0xFFFF800000000000);
287  else if (SmbiosResponse->entry_32 != NULL)
288  binfo.SMBIOSPtr = (void *)((uintptr_t)SmbiosResponse->entry_32 - 0xFFFF800000000000);
289  else
290  binfo.SMBIOSPtr = NULL;
291  debug("SMBIOS: %#lx %#lx (binfo: %#lx)",
292  SmbiosResponse->entry_32,
293  SmbiosResponse->entry_64,
294  binfo.SMBIOSPtr);
295 
296  binfo.Kernel.PhysicalBase = (void *)KernelAddressResponse->physical_base;
297  binfo.Kernel.VirtualBase = (void *)KernelAddressResponse->virtual_base;
298  binfo.Kernel.FileBase = (void *)((uintptr_t)KernelFileResponse->kernel_file->address - 0xFFFF800000000000);
299  binfo.Kernel.Size = KernelFileResponse->kernel_file->size;
300 
301  strncpy(binfo.Kernel.CommandLine,
302  KernelFileResponse->kernel_file->cmdline,
303  strlen(KernelFileResponse->kernel_file->cmdline) + 1);
304 
305  debug("Kernel physical address: %#lx", binfo.Kernel.PhysicalBase);
306  debug("Kernel virtual address: %#lx", binfo.Kernel.VirtualBase);
307 
308  strncpy(binfo.Bootloader.Name,
309  BootloaderInfoResponse->name,
310  strlen(BootloaderInfoResponse->name) + 1);
311 
312  strncpy(binfo.Bootloader.Version,
313  BootloaderInfoResponse->version,
314  strlen(BootloaderInfoResponse->version) + 1);
315 
316  Entry(&binfo);
