156 //
157 // EV_DoCeiling
158 // Move a ceiling up/down and all around!
159 //
160 int
161 EV_DoCeiling
162 ( line_t* line,
163  ceiling_e type )
164 {
165  int secnum;
166  int rtn;
167  sector_t* sec;
168  ceiling_t* ceiling;
169 
170  secnum = -1;
171  rtn = 0;
172 
173  // Reactivate in-stasis ceilings...for certain types.
174  switch(type)
175  {
176  case fastCrushAndRaise:
177  case silentCrushAndRaise:
178  case crushAndRaise:
179  P_ActivateInStasisCeiling(line);
180  default:
181  break;
182  }
183 
184  while ((secnum = P_FindSectorFromLineTag(line,secnum)) >= 0)
185  {
186  sec = &sectors[secnum];
187  if (sec->specialdata)
188  continue;
189 
190  // new door thinker
191  rtn = 1;
192  ceiling = Z_Malloc (sizeof(*ceiling), PU_LEVSPEC, 0);
193  P_AddThinker (&ceiling->thinker);
194  sec->specialdata = ceiling;
195  ceiling->thinker.function.acp1 = (actionf_p1)T_MoveCeiling;
196  ceiling->sector = sec;
197  ceiling->crush = false;
198 
199  switch(type)
200  {
201  case fastCrushAndRaise:
202  ceiling->crush = true;
203  ceiling->topheight = sec->ceilingheight;
204  ceiling->bottomheight = sec->floorheight + (8*FRACUNIT);
205  ceiling->direction = -1;
206  ceiling->speed = CEILSPEED * 2;
207  break;
208 
209  case silentCrushAndRaise:
210  case crushAndRaise:
211  ceiling->crush = true;
212  ceiling->topheight = sec->ceilingheight;
213  case lowerAndCrush:
214  case lowerToFloor:
215  ceiling->bottomheight = sec->floorheight;
216  if (type != lowerToFloor)
217  ceiling->bottomheight += 8*FRACUNIT;
218  ceiling->direction = -1;
219  ceiling->speed = CEILSPEED;
220  break;
221 
222  case raiseToHighest:
223  ceiling->topheight = P_FindHighestCeilingSurrounding(sec);
224  ceiling->direction = 1;
225  ceiling->speed = CEILSPEED;
226  break;
227  }
228 
229  ceiling->tag = sec->tag;
230  ceiling->type = type;
231  P_AddActiveCeiling(ceiling);
232  }
233  return rtn;
234 }
