125  bool Interrupts(InterruptsType Type)
126  {
127  switch (Type)
128  {
129  case Check:
130  {
131  uintptr_t Flags;
132 #if defined(a64)
133  asmv("pushfq");
134  asmv("popq %0"
135  : "=r"(Flags));
136  return Flags & (1 << 9);
137 #elif defined(a32)
138  asmv("pushfl");
139  asmv("popl %0"
140  : "=r"(Flags));
141  return Flags & (1 << 9);
142 #elif defined(aa64)
143  asmv("mrs %0, daif"
144  : "=r"(Flags));
145  return !(Flags & (1 << 2));
146 #endif
147  }
148  case Enable:
149  {
150 #if defined(a86)
151  asmv("sti");
152 #elif defined(aa64)
153  asmv("msr daifclr, #2");
154 #endif
155  return true;
156  }
157  case Disable:
158  {
159 #if defined(a86)
160  asmv("cli");
161 #elif defined(aa64)
162  asmv("msr daifset, #2");
163 #endif
164  return true;
165  }
166  default:
167  break;
168  }
169  return false;
170  }
