51  __no_sanitize("alignment") void Symbols::AddSymbolInfoFromGRUB(uint64_t Num,
52  uint64_t EntSize,
53  __unused uint64_t Shndx,
54  uintptr_t Sections)
55  {
56 #ifdef a32
57  fixme("Function not working on 32-bit");
58  return;
59 #endif
60  char *sections = reinterpret_cast<char *>(Sections);
61 
62  Elf_Sym *Symbols = nullptr;
63  uint8_t *StringAddress = nullptr;
64 
65 #if defined(a64) || defined(aa64)
66  Elf64_Xword SymbolSize = 0;
67  // Elf64_Xword StringSize = 0;
68 #elif defined(a32)
69  Elf32_Word SymbolSize = 0;
70  // Elf32_Word StringSize = 0;
71 #endif
72  int64_t TotalEntries = 0;
73 
74  for (size_t i = 0; i < Num; ++i)
75  {
76  Elf_Shdr *sym = (Elf_Shdr *)&sections[EntSize * i];
77  Elf_Shdr *str = (Elf_Shdr *)&sections[EntSize * sym->sh_link];
78 
79  if (sym->sh_type == SHT_SYMTAB &&
80  str->sh_type == SHT_STRTAB)
81  {
82  Symbols = (Elf_Sym *)sym->sh_addr;
83  StringAddress = (uint8_t *)str->sh_addr;
84  SymbolSize = (int)sym->sh_size;
85  // StringSize = (int)str->sh_size;
86  // TotalEntries = Section.sh_size / sizeof(Elf64_Sym)
87  TotalEntries = sym->sh_size / sym->sh_entsize;
88  debug("Symbol table found, %d entries",
89  SymbolSize / sym->sh_entsize);
90  UNUSED(SymbolSize);
91  break;
92  }
93  }
94 
95  if (Symbols != nullptr && StringAddress != nullptr)
96  {
97  int64_t Index, MinimumIndex;
98  for (int64_t i = 0; i < TotalEntries - 1; i++)
99  {
100  MinimumIndex = i;
101  for (Index = i + 1; Index < TotalEntries; Index++)
102  if (Symbols[Index].st_value < Symbols[MinimumIndex].st_value)
103  MinimumIndex = Index;
104  Elf_Sym tmp = Symbols[MinimumIndex];
105  Symbols[MinimumIndex] = Symbols[i];
106  Symbols[i] = tmp;
107  }
108 
109  while (Symbols[0].st_value == 0)
110  {
111  if (TotalEntries <= 0)
112  break;
113  Symbols++;
114  TotalEntries--;
115  }
116 
117  if (TotalEntries <= 0)
118  {
119  error("Symbol table is empty");
120  return;
121  }
122 
123  debug("Symbol table loaded, %d entries (%ld KiB)",
124  TotalEntries, TO_KiB(TotalEntries * sizeof(SymbolTable)));
125  Elf_Sym *sym = nullptr;
126  const char *name = nullptr;
127  for (int64_t i = 0, g = TotalEntries; i < g; i++)
128  {
129  sym = &Symbols[i];
130  name = (const char *)&StringAddress[Symbols[i].st_name];
131  if (strlen(name) == 0)
132  continue;
133  SymbolTable tbl{};
134  tbl.Address = sym->st_value;
135  tbl.FunctionName = (char *)name;
136  this->SymTable.push_back(tbl);
137  this->SymbolTableExists = true;
138

147  void Symbols::AppendSymbols(uintptr_t ImageAddress, uintptr_t BaseAddress)
148  {
149  /* FIXME: Get only the required headers instead of the whole file */
150 
151  if (ImageAddress == 0 || Memory::Virtual().Check((void *)ImageAddress) == false)
152  {
153  error("Invalid image address %#lx", ImageAddress);
154  return;
155  }
156  debug("Solving symbols for address: %#llx", ImageAddress);
157 
158 #if defined(a64) || defined(aa64)
159  Elf64_Ehdr *Header = (Elf64_Ehdr *)ImageAddress;
160 #elif defined(a32)
161  Elf32_Ehdr *Header = (Elf32_Ehdr *)ImageAddress;
162 #endif
163  if (Header->e_ident[0] != 0x7F &&
164  Header->e_ident[1] != 'E' &&
165  Header->e_ident[2] != 'L' &&
166  Header->e_ident[3] != 'F')
167  {
168  error("Invalid ELF header");
169  return;
170  }
171 
172  Elf_Shdr *ElfSections = (Elf_Shdr *)(ImageAddress + Header->e_shoff);
173  Elf_Sym *ElfSymbols = nullptr;
174  char *strtab = nullptr;
175  int64_t TotalEntries = 0;
176 
177  for (uint16_t i = 0; i < Header->e_shnum; i++)
178  {
179  switch (ElfSections[i].sh_type)
180  {
181  case SHT_SYMTAB:
182  ElfSymbols = (Elf_Sym *)(ImageAddress + ElfSections[i].sh_offset);
183  TotalEntries = ElfSections[i].sh_size / sizeof(Elf_Sym);
184  debug("Symbol table found, %d entries", TotalEntries);
185  break;
186  case SHT_STRTAB:
187  if (Header->e_shstrndx == i)
188  {
189  debug("String table found, %d entries", ElfSections[i].sh_size);
190  }
191  else
192  {
193  strtab = (char *)(ImageAddress + ElfSections[i].sh_offset);
194  debug("String table found, %d entries", ElfSections[i].sh_size);
195  }
196  break;
197  default:
198  break;
199  }
200  }
201 
202  if (ElfSymbols != nullptr && strtab != nullptr)
203  {
204  int64_t Index, MinimumIndex;
205  for (int64_t i = 0; i < TotalEntries - 1; i++)
206  {
207  MinimumIndex = i;
208  for (Index = i + 1; Index < TotalEntries; Index++)
209  if (ElfSymbols[Index].st_value < ElfSymbols[MinimumIndex].st_value)
210  MinimumIndex = Index;
211  Elf_Sym tmp = ElfSymbols[MinimumIndex];
212  ElfSymbols[MinimumIndex] = ElfSymbols[i];
213  ElfSymbols[i] = tmp;
214  }
215 
216  while (ElfSymbols[0].st_value == 0)
217  {
218  ElfSymbols++;
219  TotalEntries--;
220  }
221 
222  trace("Symbol table loaded, %d entries (%ld KiB)",
223  TotalEntries, TO_KiB(TotalEntries * sizeof(SymbolTable)));
224 
225  /* TODO: maybe a checker for duplicated addresses? */
226  Elf_Sym *sym = nullptr;
227  const char *name = nullptr;
228  for (int64_t i = 0, g = TotalEntries; i < g; i++)
229  {
230  sym = &ElfSymbols[i];
231  name = &strtab[ElfSymbols[i].st_name];
232  SymbolTable tbl{};
233  tbl.Address = sym->st_value + BaseAddress;
234  tbl.FunctionName = (char *)name;
235  this->SymTable.push_back(tbl);
236  this->SymbolTableExists = true;
237 
238  // debug("Symbol %d: %#llx %s", i,
239  // this->SymTable[i].Address,
240  // this->SymTable[i].FunctionName);
241  }
242  }
243  }
