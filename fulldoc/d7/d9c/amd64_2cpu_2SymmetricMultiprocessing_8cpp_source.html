51 SafeFunction CPUData *GetCurrentCPU()
52 {
53  if (unlikely(!Interrupts::apic[0]))
54  return &CPUs[0]; /* No APIC means we are on the BSP. */
55 
56  int CoreID = ((APIC::APIC *)Interrupts::apic[0])->Read(APIC::APIC_ID) >> 24;
57 
58  if (unlikely((&CPUs[CoreID])->IsActive != true))
59  {
60  error("CPU %d is not active!", CoreID);
61  assert((&CPUs[0])->IsActive == true); /* We can't continue without the BSP. */
62  return &CPUs[0];
63  }
64 
65  assert((&CPUs[CoreID])->Checksum == CPU_DATA_CHECKSUM); /* This should never happen. */
66  return &CPUs[CoreID];
67 }

94  if (Config.Cores > ((ACPI::MADT *)madt)->CPUCores + 1)

96  else if (Config.Cores != 0)
97  Cores = Config.Cores;

99  CPUCores = Cores;
100 
101  uint64_t TrampolineLength = (uintptr_t)&_trampoline_end - (uintptr_t)&_trampoline_start;
102  Memory::Virtual().Map(0x0, 0x0, Memory::PTFlag::RW);
103  /* We reserved the TRAMPOLINE_START address inside Physical class. */
104  Memory::Virtual().Map((void *)TRAMPOLINE_START, (void *)TRAMPOLINE_START, TrampolineLength, Memory::PTFlag::RW);
105  memcpy((void *)TRAMPOLINE_START, &_trampoline_start, TrampolineLength);
106  debug("Trampoline address: %#lx-%#lx", TRAMPOLINE_START, TRAMPOLINE_START + TrampolineLength);
107 
108  void *CPUTmpStack = KernelAllocator.RequestPages(TO_PAGES(STACK_SIZE + 1));
109  asmv("sgdt [0x580]\n"
110  "sidt [0x590]\n");
111  VPOKE(uintptr_t, STACK) = (uintptr_t)CPUTmpStack + STACK_SIZE;
112  VPOKE(uintptr_t, PAGE_TABLE) = (uintptr_t)KernelPageTable;
113  VPOKE(uint64_t, START_ADDR) = (uintptr_t)&StartCPU;
114 
115  for (int i = 0; i < Cores; i++)
116  {
117  debug("Initializing CPU %d", i);
118  if ((((APIC::APIC *)Interrupts::apic[0])->Read(APIC::APIC_ID) >> 24) != ((ACPI::MADT *)madt)->lapic[i]->ACPIProcessorId)
119  {
120  VPOKE(int, CORE) = i;
121 
122  ((APIC::APIC *)Interrupts::apic[0])->Write(APIC::APIC_ICRHI, (((ACPI::MADT *)madt)->lapic[i]->APICId << 24));
123  ((APIC::APIC *)Interrupts::apic[0])->Write(APIC::APIC_ICRLO, 0x500);
124 
125  ((APIC::APIC *)Interrupts::apic[0])->SendInitIPI(((ACPI::MADT *)madt)->lapic[i]->APICId);
126  ((APIC::APIC *)Interrupts::apic[0])->SendStartupIPI(((ACPI::MADT *)madt)->lapic[i]->APICId, TRAMPOLINE_START);
127 
128  while (!CPUEnabled.load(std::memory_order_acquire))
129  CPU::Pause();
130  CPUEnabled.store(false, std::memory_order_release);
131  trace("CPU %d loaded.", ((ACPI::MADT *)madt)->lapic[i]->APICId);
132  }
133  else
134  KPrint("\e058C19CPU \e8888FF%d \e058C19is the BSP", ((ACPI::MADT *)madt)->lapic[i]->APICId);
135  }
136 
137  KernelAllocator.FreePages(CPUTmpStack, TO_PAGES(STACK_SIZE + 1));
138  /* We are going to unmap the page after we are done with it. */
139  Memory::Virtual().Unmap(0x0);
140  }
