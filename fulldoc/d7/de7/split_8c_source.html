56 /* Increment the suffix part of the filename.
57  * Returns NULL if we are out of filenames.
58  */
59 static char *next_file(char *old, unsigned suffix_len)
60 {
61  size_t end = strlen(old);
62  unsigned i = 1;
63  char *curr;
64 
65  while (1) {
66  curr = old + end - i;
67  if (*curr < 'z') {
68  *curr += 1;
69  break;
70  }
71  i++;
72  if (i > suffix_len) {
73  return NULL;
74  }
75  *curr = 'a';
76  }

79 }
