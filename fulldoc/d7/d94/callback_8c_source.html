53 /* Applications can set this to non-zero to have readline's signal handlers
54  installed during the entire duration of reading a complete line, as in

56  readline receiving signals and not handling them until it's called again
57  via rl_callback_read_char, thereby stealing them from the application.

67 /* Allow using readline in situations where a program may have multiple
68  things to handle at once, and dispatches them via select(). Call
69  rl_callback_handler_install() with the prompt and a function to call
70  whenever a complete line of input is ready. The user must then
71  call rl_callback_read_char() every time some input is available, and
72  rl_callback_read_char() will call the user's function with the complete
73  text read in at each end of line. The terminal is kept prepped
74  all the time, except during calls to the user's function. Signal
75  handlers are only installed when the application calls back into
76  readline, so readline doesn't `steal' signals from the application. */

125 /* Read one character, and dispatch to the handler if it ends the line. */
126 void
127 rl_callback_read_char (void)
128 {
129  char *line;
130  int eof, jcode;
131  static procenv_t olevel;
132 
133  if (rl_linefunc == NULL)
134  {
135  _rl_errmsg ("readline_callback_read_char() called with no handler!");
136  abort ();
137  }
138 
139  eof = 0;
140 
141  memcpy ((void *)olevel, (void *)_rl_top_level, sizeof (procenv_t));
142 #if defined (HAVE_POSIX_SIGSETJMP)
143  jcode = sigsetjmp (_rl_top_level, 0);
144 #else
145  jcode = setjmp (_rl_top_level);
146 #endif
147  if (jcode)
148  {
149  (*rl_redisplay_function) ();
150  _rl_want_redisplay = 0;
151  memcpy ((void *)_rl_top_level, (void *)olevel, sizeof (procenv_t));
152 
153  /* If we longjmped because of a timeout, handle it here. */
154  if (RL_ISSTATE (RL_STATE_TIMEOUT))
155  {
156  RL_SETSTATE (RL_STATE_DONE);
157  rl_done = 1;
158  }
159 
160  CALLBACK_READ_RETURN ();
161  }
162 
163 #if defined (HANDLE_SIGNALS)
164  /* Install signal handlers only when readline has control. */
165  if (rl_persistent_signal_handlers == 0)
166  rl_set_signals ();
167 #endif
168 
169  do
170  {
171  RL_CHECK_SIGNALS ();
172  if (RL_ISSTATE (RL_STATE_ISEARCH))
173  {
174  eof = _rl_isearch_callback (_rl_iscxt);
175  if (eof == 0 && (RL_ISSTATE (RL_STATE_ISEARCH) == 0) && RL_ISSTATE (RL_STATE_INPUTPENDING))
176  rl_callback_read_char ();
177 
178  CALLBACK_READ_RETURN ();
179  }
180  else if (RL_ISSTATE (RL_STATE_NSEARCH))
181  {
182  eof = _rl_nsearch_callback (_rl_nscxt);
183 
184  CALLBACK_READ_RETURN ();
185  }
186 #if defined (VI_MODE)
187  /* States that can occur while in state VIMOTION have to be checked
188  before RL_STATE_VIMOTION */
189  else if (RL_ISSTATE (RL_STATE_CHARSEARCH))
190  {
191  int k;
192 
193  k = _rl_callback_data->i2;
194 
195  eof = (*_rl_callback_func) (_rl_callback_data);
196  /* If the function `deregisters' itself, make sure the data is
197  cleaned up. */
198  if (_rl_callback_func == 0) /* XXX - just sanity check */
199  {
200  if (_rl_callback_data)
201  {
202  _rl_callback_data_dispose (_rl_callback_data);
203  _rl_callback_data = 0;
204  }
205  }
206 
207  /* Messy case where vi motion command can be char search */
208  if (RL_ISSTATE (RL_STATE_VIMOTION))
209  {
210  _rl_vi_domove_motion_cleanup (k, _rl_vimvcxt);
211  _rl_internal_char_cleanup ();
212  CALLBACK_READ_RETURN ();
213  }
214 
215  _rl_internal_char_cleanup ();
216  }
217  else if (RL_ISSTATE (RL_STATE_VIMOTION))
218  {
219  eof = _rl_vi_domove_callback (_rl_vimvcxt);
220  /* Should handle everything, including cleanup, numeric arguments,
221  and turning off RL_STATE_VIMOTION */
222  if (RL_ISSTATE (RL_STATE_NUMERICARG) == 0)
223  _rl_internal_char_cleanup ();
224 
225  CALLBACK_READ_RETURN ();
226  }
227 #endif
228  else if (RL_ISSTATE (RL_STATE_NUMERICARG))
229  {
230  eof = _rl_arg_callback (_rl_argcxt);
231  if (eof == 0 && (RL_ISSTATE (RL_STATE_NUMERICARG) == 0) && RL_ISSTATE (RL_STATE_INPUTPENDING))
232  rl_callback_read_char ();
233  /* XXX - this should handle _rl_last_command_was_kill better */
234  else if (RL_ISSTATE (RL_STATE_NUMERICARG) == 0)
235  _rl_internal_char_cleanup ();
236 
237  CALLBACK_READ_RETURN ();
238  }
239  else if (RL_ISSTATE (RL_STATE_MULTIKEY))
240  {
241  eof = _rl_dispatch_callback (_rl_kscxt); /* For now */
242  while ((eof == -1 || eof == -2) && RL_ISSTATE (RL_STATE_MULTIKEY) && _rl_kscxt && (_rl_kscxt->flags & KSEQ_DISPATCHED))
243  eof = _rl_dispatch_callback (_rl_kscxt);
244  if (RL_ISSTATE (RL_STATE_MULTIKEY) == 0)
245  {
246  _rl_internal_char_cleanup ();
247  _rl_want_redisplay = 1;
248  }
249  }
250  else if (_rl_callback_func)
251  {
252  /* This allows functions that simply need to read an additional
253  character (like quoted-insert) to register a function to be
254  called when input is available. _rl_callback_data is a
255  pointer to a struct that has the argument count originally
256  passed to the registering function and space for any additional
257  parameters. */
258  eof = (*_rl_callback_func) (_rl_callback_data);
259  /* If the function `deregisters' itself, make sure the data is
260  cleaned up. */
261  if (_rl_callback_func == 0)
262  {
263  if (_rl_callback_data)
264  {
265  _rl_callback_data_dispose (_rl_callback_data);
266  _rl_callback_data = 0;
267  }
268  _rl_internal_char_cleanup ();
269  }
270  }
271  else
272  eof = readline_internal_char ();
273 
274  RL_CHECK_SIGNALS ();
275  if (rl_done == 0 && _rl_want_redisplay)
276  {
277  (*rl_redisplay_function) ();
278  _rl_want_redisplay = 0;
279  }
280 
281  /* Make sure application hooks can see whether we saw EOF. */
282  if (eof > 0)
283  {
284  rl_eof_found = eof;
285  RL_SETSTATE(RL_STATE_EOF);
286  }
287 
288  if (rl_done)
289  {
290  line = readline_internal_teardown (eof);
291 
292  if (rl_deprep_term_function)
293  (*rl_deprep_term_function) ();
294 #if defined (HANDLE_SIGNALS)
295  rl_clear_signals ();
296 #endif
297  in_handler = 0;
298  if (rl_linefunc) /* just in case */
299  (*rl_linefunc) (line);
300 
301  /* If the user did not clear out the line, do it for him. */
302  if (rl_line_buffer[0])
303  _rl_init_line_state ();
304 
305  /* Redisplay the prompt if readline_handler_{install,remove}
306  not called. */
307  if (in_handler == 0 && rl_linefunc)
308  _rl_callback_newline ();
309  }
310  }
311  while (rl_pending_input || _rl_pushed_input_available () || RL_ISSTATE (RL_STATE_MACROINPUT));
312 
313  CALLBACK_READ_RETURN ();
314 }
