45 nsa bool DetectUSBKeyboard()
46 {
47  if (!PCIManager)
48  return false;
49  std::list<PCIDevice> uhci = PCIManager->FindPCIDevice(0xC, 0x3, 0x00); /* UHCI */
50  debug("There are %lu UHCI devices", uhci.size());
51 
52  std::list<PCIDevice> ohci = PCIManager->FindPCIDevice(0xC, 0x3, 0x10); /* OHCI */
53  debug("There are %lu OHCI devices", ohci.size());
54 
55  std::list<PCIDevice> ehci = PCIManager->FindPCIDevice(0xC, 0x3, 0x20); /* EHCI */
56  debug("There are %lu EHCI devices", uhci.size());
57 
58  std::list<PCIDevice> xhci = PCIManager->FindPCIDevice(0xC, 0x3, 0x30); /* XHCI */
59  debug("There are %lu XHCI devices", xhci.size());
60 
61  debug("Initializing UHCI devices");
62  for (const auto &dev : uhci)
63  {
64  CrashUHCIKeyboardDriver *usb = new CrashUHCIKeyboardDriver(dev);
65  if (!usb->Initialize())
66  {
67  error("Failed to initialize UHCI keyboard driver");
68  uhci.pop_front();
69  }
70  }
71 
72  debug("Initializing EHCI devices");
73  for (const auto &dev : ehci)
74  {
75  CrashEHCIKeyboardDriver *usb = new CrashEHCIKeyboardDriver(dev);
76  if (!usb->Initialize())
77  {
78  error("Failed to initialize UHCI keyboard driver");
79  ehci.pop_front();
80  }
81  }
82 
83  debug("Initializing XHCI devices");
84  for (const auto &dev : xhci)
85  {
86  CrashXHCIKeyboardDriver *usb = new CrashXHCIKeyboardDriver(dev);
87  if (!usb->Initialize())
88  {
89  error("Failed to initialize XHCI keyboard driver");
90  xhci.pop_front();
91  }
92  }
93 
94  return xhci.size() > 0 || uhci.size() > 0 || ohci.size() > 0;
95 }
