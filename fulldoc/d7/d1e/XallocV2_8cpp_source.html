81  Xuint8_t *V2::AllocateHeap(Xsize_t Size)
82  {
83  Size = this->Align(Size);
84  Xsize_t Pages = XStoP(Size);
85 
86  Xuint8_t *FinalAddress = 0x0;
87  if (this->HeapUsed + Size >= this->HeapSize)
88  {
89  void *Address = Xalloc_REQUEST_PAGES(Pages);
90  void *VirtualAddress = (void *)(this->BaseVirtualAddress + this->HeapSize);
91  if (Xalloc_MapPages)
92  {
93  for (Xsize_t i = 0; i < Pages; i++)
94  {
95  Xuintptr_t Page = i * Xalloc_PAGE_SIZE;
96  void *vAddress = (void *)((Xuintptr_t)VirtualAddress + Page);
97  Xalloc_MAP_MEMORY(vAddress, (void *)((Xuintptr_t)Address + Page), 0x3);
98  }
99  }
100 
101  this->HeapSize += XPtoS(Pages);
102  FinalAddress = (Xuint8_t *)VirtualAddress;
103  }
104  else
105  FinalAddress = (Xuint8_t *)(this->BaseVirtualAddress + this->HeapUsed);
106 
107  this->HeapUsed += Size;
108  return (uint8_t *)FinalAddress;
109  }

193  void V2::free(void *Address)
194  {
195  if (Address == nullptr)
196  {
197  Xalloc_warn("Attempted to free a null pointer!");
198  return;
199  }
200 
201  XallocV2_lock;
202 
203  Block *CurrentBlock = ((Block *)this->FirstBlock);
204  while (CurrentBlock != nullptr)
205  {
206  CurrentBlock->Check();
207 
208  if (CurrentBlock->Data == Address)
209  {
210  if (CurrentBlock->IsFree)
211  Xalloc_warn("Attempted to free an already freed block! %#lx", Address);
212 
213  CurrentBlock->IsFree = true;
214  XallocV2_unlock;
215  return;
216  }
217  CurrentBlock = CurrentBlock->Next;
218  }
219 
220  Xalloc_err("Invalid address %#lx.", Address);
221  XallocV2_unlock;
222  }
