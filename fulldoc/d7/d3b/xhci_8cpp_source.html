83 nsa CrashXHCIKeyboardDriver::CrashXHCIKeyboardDriver(PCIDevice xhci)
84  : Interrupts::Handler(xhci)
85 {
86  Header = (PCIDeviceHeader *)xhci.Header;
87 
88  switch (Header->HeaderType)
89  {
90  case 128:
91  {
92  ExPrint(ERROR_COLOR "Unknown header type %d! Guessing PCI Header 0\n" DEFAULT_COLOR,
93  Header->HeaderType);
94  [[fallthrough]];
95  }
96  case 0: /* PCI Header 0 */
97  {
98  PCI::PCIHeader0 *hdr = (PCI::PCIHeader0 *)Header;
99 
100  uint32_t BAR[6];
101  size_t BARsSize[6];
102 
103  BAR[0] = hdr->BAR0;
104  BAR[1] = hdr->BAR1;
105  BAR[2] = hdr->BAR2;
106  BAR[3] = hdr->BAR3;
107  BAR[4] = hdr->BAR4;
108  BAR[5] = hdr->BAR5;
109 
110  /* BARs Size */
111  for (short i = 0; i < 6; i++)
112  {
113  if (BAR[i] == 0)
114  continue;
115 
116  size_t size;
117  if ((BAR[i] & 1) == 0) /* Memory Base */
118  {
119  hdr->BAR0 = 0xFFFFFFFF;
120  size = hdr->BAR0;
121  hdr->BAR0 = BAR[i];
122  BARsSize[i] = size & (~15);
123  BARsSize[i] = ~BARsSize[i] + 1;
124  BARsSize[i] = BARsSize[i] & 0xFFFFFFFF;
125  debug("MEM BAR%d %#lx size: %d",
126  i, BAR[i], BARsSize[i]);
127  }
128  else if ((BAR[i] & 1) == 1) /* I/O Base */
129  {
130  hdr->BAR1 = 0xFFFFFFFF;
131  size = hdr->BAR1;
132  hdr->BAR1 = BAR[i];
133  BARsSize[i] = size & (~3);
134  BARsSize[i] = ~BARsSize[i] + 1;
135  BARsSize[i] = BARsSize[i] & 0xFFFF;
136  debug("IO BAR%d %#lx size: %d",
137  i, BAR[i], BARsSize[i]);
138  }
139  }
140 
141  debug("IO %d 64-BIT %d", BAR[0] & 0x1, BAR[0] & 0x4);
142 
143  uintptr_t baseAddress = BAR[0];
144  if (BAR[0] & 0x4)
145  baseAddress |= (uintptr_t)BAR[1] << 32;
146 
147  if (baseAddress & 0x1)
148  baseAddress &= 0xFFFFFFFFFFFFFFFC;
149  else
150  baseAddress &= 0xFFFFFFFFFFFFFFF0;
151 
152  debug("baseAddress: %#lx", baseAddress);
153  Memory::Virtual vmm;
154  vmm.Map((void *)baseAddress, (void *)baseAddress, BARsSize[0], Memory::RW);
155  caps = (XHCIcap *)baseAddress;
156  ops = (XHCIop *)(baseAddress + caps->CAPLENGTH);
157  port = (XHCIport *)(baseAddress + caps->CAPLENGTH + 0x400);
158  runtime = (XHCIruntime *)(baseAddress + (caps->RTSOFF & ~0x1F));
159  doorbell = (XHCIdoorbell *)(baseAddress + (caps->DBOFF & ~0x3));
160  uint16_t exCapOffset = caps->HCCPARAMS1.xHCIExtendedCapacitiesPointer << 2;
161  ExtendedCaps = (uintptr_t)caps + exCapOffset;
162  debug("ExtendedCaps: %#lx (%#lx + %#lx)", ExtendedCaps, caps, exCapOffset);
163 
164 #pragma GCC diagnostic push
165 #pragma GCC diagnostic ignored "-Waddress-of-packed-member"
166  Interrupter = &runtime->Interrupter[0];
167 #pragma GCC diagnostic pop
168  break;
169  }
170  case 1: /* PCI Header 1 (PCI-to-PCI Bridge) */
171  {
172  ExPrint(ERROR_COLOR "PCI-to-PCI Bridge not supported\n" DEFAULT_COLOR);
173  break;
174  }
175  case 2: /* PCI Header 2 (PCI-to-CardBus Bridge) */
176  {
177  ExPrint(ERROR_COLOR "PCI-to-CardBus Bridge not supported\n" DEFAULT_COLOR);
178  break;
179  }
180  default:
181  {
182  ExPrint(ERROR_COLOR "Invalid PCI header type\n" DEFAULT_COLOR);
183  break;
184  }
185  }
186 }
