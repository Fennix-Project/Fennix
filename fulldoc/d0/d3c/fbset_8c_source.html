270 static NOINLINE int read_mode_db(struct fb_var_screeninfo *base, const char *fn,
271  const char *mode)
272 {
273  char *token[2], *p, *s;
274  parser_t *parser = config_open(fn);
275 
276  while (config_read(parser, token, 2, 1, "# \t\r", PARSE_NORMAL)) {
277  if (strcmp(token[0], "mode") != 0 || !token[1])
278  continue;
279  p = strstr(token[1], mode);
280  if (!p)
281  continue;
282  s = p + strlen(mode);
283  //bb_error_msg("CHECK[%s][%s][%d]", mode, p-1, *s);
284  /* exact match? */
285  if (((!*s || isspace(*s)) && '"' != s[-1]) /* end-of-token */
286  || ('"' == *s && '"' == p[-1]) /* ends with " but starts with " too! */
287  ) {
288  //bb_error_msg("FOUND[%s][%s][%s][%d]", token[1], p, mode, isspace(*s));
289  break;
290  }
291  }
292 
293  if (!token[0])
294  return 0;
295 
296  while (config_read(parser, token, 2, 1, "# \t", PARSE_NORMAL)) {
297  int i;
298 
299 //bb_error_msg("???[%s][%s]", token[0], token[1]);
300  if (strcmp(token[0], "endmode") == 0) {
301 //bb_error_msg("OK[%s]", mode);
302  return 1;
303  }
304  p = token[1];
305  i = index_in_strings(
306  "geometry\0timings\0interlaced\0double\0vsync\0hsync\0csync\0extsync\0rgba\0",
307  token[0]);
308  switch (i) {
309  case 0:
310  if (sizeof(int) == sizeof(base->xres)) {
311  sscanf(p, "%d %d %d %d %d",
312  &base->xres, &base->yres,
313  &base->xres_virtual, &base->yres_virtual,
314  &base->bits_per_pixel);
315  } else {
316  int base_xres, base_yres;
317  int base_xres_virtual, base_yres_virtual;
318  int base_bits_per_pixel;
319  sscanf(p, "%d %d %d %d %d",
320  &base_xres, &base_yres,
321  &base_xres_virtual, &base_yres_virtual,
322  &base_bits_per_pixel);
323  base->xres = base_xres;
324  base->yres = base_yres;
325  base->xres_virtual = base_xres_virtual;
326  base->yres_virtual = base_yres_virtual;
327  base->bits_per_pixel = base_bits_per_pixel;
328  }
329 //bb_error_msg("GEO[%s]", p);
330  break;
331  case 1:
332  if (sizeof(int) == sizeof(base->xres)) {
333  sscanf(p, "%d %d %d %d %d %d %d",
334  &base->pixclock,
335  &base->left_margin, &base->right_margin,
336  &base->upper_margin, &base->lower_margin,
337  &base->hsync_len, &base->vsync_len);
338  } else {
339  int base_pixclock;
340  int base_left_margin, base_right_margin;
341  int base_upper_margin, base_lower_margin;
342  int base_hsync_len, base_vsync_len;
343  sscanf(p, "%d %d %d %d %d %d %d",
344  &base_pixclock,
345  &base_left_margin, &base_right_margin,
346  &base_upper_margin, &base_lower_margin,
347  &base_hsync_len, &base_vsync_len);
348  base->pixclock = base_pixclock;
349  base->left_margin = base_left_margin;
350  base->right_margin = base_right_margin;
351  base->upper_margin = base_upper_margin;
352  base->lower_margin = base_lower_margin;
353  base->hsync_len = base_hsync_len;
354  base->vsync_len = base_vsync_len;
355  }
356 //bb_error_msg("TIM[%s]", p);
357  break;
358  case 2:
359  case 3: {
360  static const uint32_t syncs[] = {FB_VMODE_INTERLACED, FB_VMODE_DOUBLE};
361  ss(&base->vmode, syncs[i-2], p, "false");
362 //bb_error_msg("VMODE[%s]", p);
363  break;
364  }
365  case 4:
366  case 5:
367  case 6: {
368  static const uint32_t syncs[] = {FB_SYNC_VERT_HIGH_ACT, FB_SYNC_HOR_HIGH_ACT, FB_SYNC_COMP_HIGH_ACT};
369  ss(&base->sync, syncs[i-4], p, "low");
370 //bb_error_msg("SYNC[%s]", p);
371  break;
372  }
373  case 7:

375 //bb_error_msg("EXTSYNC[%s]", p);
376  break;
377  case 8: {
378  int red_offset, red_length;
379  int green_offset, green_length;
380  int blue_offset, blue_length;
381  int transp_offset, transp_length;
382 
383  sscanf(p, "%d/%d,%d/%d,%d/%d,%d/%d",
384  &red_length, &red_offset,

386  &blue_length, &blue_offset,
387  &transp_length, &transp_offset);

397  base->transp.offset = transp_offset;

407 static NOINLINE void showmode(struct fb_var_screeninfo *v)
408 {
409  double drate = 0, hrate = 0, vrate = 0;
410 
411  if (v->pixclock) {
412  drate = 1e12 / v->pixclock;
413  hrate = drate / (v->left_margin + v->xres + v->right_margin + v->hsync_len);
414  vrate = hrate / (v->upper_margin + v->yres + v->lower_margin + v->vsync_len);
415  }
416  printf("\nmode \"%ux%u-%u\"\n"
417 #if ENABLE_FEATURE_FBSET_FANCY
418  "\t# D: %.3f MHz, H: %.3f kHz, V: %.3f Hz\n"
419 #endif
420  "\tgeometry %u %u %u %u %u\n"
421  "\ttimings %u %u %u %u %u %u %u\n"
422  "\taccel %s\n"
423  "\trgba %u/%u,%u/%u,%u/%u,%u/%u\n"
424  "endmode\n\n",
425  v->xres, v->yres, (int) (vrate + 0.5),
426 #if ENABLE_FEATURE_FBSET_FANCY
427  drate / 1e6, hrate / 1e3, vrate,
428 #endif
429  v->xres, v->yres, v->xres_virtual, v->yres_virtual, v->bits_per_pixel,
430  v->pixclock, v->left_margin, v->right_margin, v->upper_margin, v->lower_margin,
431  v->hsync_len, v->vsync_len,
432  (v->accel_flags > 0 ? "true" : "false"),
433  v->red.length, v->red.offset, v->green.length, v->green.offset,
434  v->blue.length, v->blue.offset, v->transp.length, v->transp.offset);
435 }

437 int fbset_main(int argc, char **argv) MAIN_EXTERNALLY_VISIBLE;
438 int fbset_main(int argc, char **argv)
439 {
440  enum {
441  OPT_CHANGE = (1 << 0),
442  OPT_SHOW = (1 << 1),
443  OPT_READMODE = (1 << 2),
444  OPT_ALL = (1 << 3),
445  };
446  struct fb_var_screeninfo var_old, var_set;
447  int fh, i;
448  unsigned options = 0;
449 
450  const char *fbdev = DEFAULTFBDEV;
451  IF_FEATURE_FBSET_READMODE(const char *modefile = DEFAULTFBMODE;)
452  char *thisarg;
453  char *mode = mode; /* for compiler */
454 
455  memset(&var_set, 0xff, sizeof(var_set)); /* set all to -1 */
456 
457  /* parse cmd args.... why do they have to make things so difficult? */
458  argv++;
459  argc--;
460  for (; argc > 0 && (thisarg = *argv) != NULL; argc--, argv++) {
461  if (thisarg[0] != '-') {
462  if (!ENABLE_FEATURE_FBSET_READMODE || argc != 1)
463  bb_show_usage();
464  mode = thisarg;
465  options |= OPT_READMODE;
466  goto contin;
467  }
468  for (i = 0; i < ARRAY_SIZE(g_cmdoptions); i++) {
469  if (strcmp(thisarg + 1, g_cmdoptions[i].name) != 0)
470  continue;
471  if (argc <= g_cmdoptions[i].param_count)
472  bb_show_usage();
473 
474  switch (g_cmdoptions[i].code) {
475  case CMD_FB:
476  fbdev = argv[1];
477  break;
478  case CMD_DB:
479  IF_FEATURE_FBSET_READMODE(modefile = argv[1];)
480  break;
481  case CMD_ALL:
482  options |= OPT_ALL;
483  break;
484  case CMD_SHOW:
485  options |= OPT_SHOW;
486  break;
487  case CMD_GEOMETRY:
488  var_set.xres = xatou32(argv[1]);
489  var_set.yres = xatou32(argv[2]);
490  var_set.xres_virtual = xatou32(argv[3]);
491  var_set.yres_virtual = xatou32(argv[4]);
492  var_set.bits_per_pixel = xatou32(argv[5]);
493  break;
494  case CMD_TIMING:
495  var_set.pixclock = xatou32(argv[1]);
496  var_set.left_margin = xatou32(argv[2]);
497  var_set.right_margin = xatou32(argv[3]);
498  var_set.upper_margin = xatou32(argv[4]);
499  var_set.lower_margin = xatou32(argv[5]);
500  var_set.hsync_len = xatou32(argv[6]);
501  var_set.vsync_len = xatou32(argv[7]);
502  break;
503  case CMD_ACCEL:
504  break;
505  case CMD_HSYNC:
506  var_set.sync |= FB_SYNC_HOR_HIGH_ACT;
507  break;
508  case CMD_VSYNC:
509  var_set.sync |= FB_SYNC_VERT_HIGH_ACT;
510  break;
511 #if ENABLE_FEATURE_FBSET_FANCY
512  case CMD_XRES:
513  var_set.xres = xatou32(argv[1]);
514  break;
515  case CMD_YRES:
516  var_set.yres = xatou32(argv[1]);
517  break;
518  case CMD_DEPTH:
519  var_set.bits_per_pixel = xatou32(argv[1]);
520  break;
521  case CMD_PIXCLOCK:
522  var_set.pixclock = xatou32(argv[1]);
523  break;
524 #endif
525  default:
526  bb_perror_msg_and_die("option '%s' not handled",
527  g_cmdoptions[i].name);
528  }
529  switch (g_cmdoptions[i].code) {
530  case CMD_FB:
531  case CMD_DB:
532  case CMD_ALL:
533  case CMD_SHOW:
534  break;
535  default:
536  /* other commands imply changes */
537  options |= OPT_CHANGE;
538  }
539  argc -= g_cmdoptions[i].param_count;
540  argv += g_cmdoptions[i].param_count;
541  goto contin;
542  }
543  bb_show_usage();
544  contin: ;
545  }
546 
547  fh = xopen(fbdev, O_RDONLY);
548  xioctl(fh, FBIOGET_VSCREENINFO, &var_old);
549 
550  if (options & OPT_READMODE) {
551 #if ENABLE_FEATURE_FBSET_READMODE
552  if (!read_mode_db(&var_old, modefile, mode)) {
553  bb_error_msg_and_die("unknown video mode '%s'", mode);
554  }
555  options |= OPT_CHANGE;
556 #endif
557  }
558 
559  if (options & OPT_CHANGE) {
560  copy_changed_values(&var_old, &var_set);
561  if (options & OPT_ALL)
562  var_old.activate = FB_ACTIVATE_ALL;
563  xioctl(fh, FBIOPUT_VSCREENINFO, &var_old);
564  }
565 
566  if (options == 0 || (options & OPT_SHOW))
567  showmode(&var_old);
568 
569  if (ENABLE_FEATURE_CLEAN_UP)
570  close(fh);
571 
572  return EXIT_SUCCESS;
573 }
