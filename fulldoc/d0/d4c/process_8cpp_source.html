138  PCB::PCB(Task *ctx, PCB *Parent, const char *Name,
139  TaskExecutionMode ExecutionMode,
140  bool UseKernelPageTable,
141  uint16_t UserID, uint16_t GroupID)
142  : Signals(this)
143  {
144  debug("+ %#lx", this);
145 
146  assert(ctx != nullptr);
147  assert(Name != nullptr);
148  assert(strlen(Name) > 0);
149  assert(ExecutionMode >= _ExecuteModeMin);
150  assert(ExecutionMode <= _ExecuteModeMax);
151 
152  FileNode *procDir = fs->GetByPath("/proc", nullptr);
153  assert(procDir != nullptr);
154 
155  /* d r-x r-x r-x */
156  mode_t mode = S_IROTH | S_IXOTH |
157  S_IRGRP | S_IXGRP |
158  S_IRUSR | S_IXUSR |
159  S_IFDIR;
160 
161  ProcDirectory = fs->Create(procDir, std::to_string(ctx->NextPID).c_str(), mode);
162  assert(ProcDirectory != nullptr);
163 
164  this->ctx = ctx;
165  this->ID = ctx->NextPID++;
166 
167  if (this->Name) /* Prevent memory leak */
168  delete[] this->Name;
169 
170  this->Name = new char[strlen(Name) + 1];
171  strcpy((char *)this->Name, Name);
172  this->ExitCode = KILL_CRASH;
173 
174  /* Check parent */
175  if (Parent == nullptr)
176  this->Parent = ctx->GetCurrentProcess();
177  else
178  this->Parent = Parent;
179 
180  /* Set uid & gid */
181  if (this->Parent &&
182  UserID == UINT16_MAX &&
183  GroupID == UINT16_MAX)
184  {
185  UserID = this->Parent->Security.Real.UserID;
186  GroupID = this->Parent->Security.Real.GroupID;
187  debug("Inherited uid & gid from parent process %s(%d) with uid %d and gid %d",
188  this->Parent->Name, this->Parent->ID, UserID, GroupID);
189  }
190 
191  this->Security.Real.UserID = UserID;
192  this->Security.Real.GroupID = GroupID;
193  this->Security.Effective.UserID = UserID;
194  this->Security.Effective.GroupID = GroupID;
195  this->Security.ExecutionMode = ExecutionMode;
196 
197  switch (ExecutionMode)
198  {
199  case TaskExecutionMode::System:
200  fixme("Mode not supported.");
201  [[fallthrough]];
202  case TaskExecutionMode::Kernel:
203  {
204  this->Security.IsCritical = true;
205  break;
206  }
207  case TaskExecutionMode::User:
208  {
209  break;
210  }
211  default:
212  assert(false);
213  }
214 
215  this->FileDescriptors = new FileDescriptorTable(this);
216  this->tty = nullptr;
217 
218  /* If create page table */
219  if (UseKernelPageTable == false)
220  {
221  OwnPageTable = true;
222  this->PageTable = KernelPageTable->Fork();
223  debug("Process %s(%d) has page table at %#lx",
224  this->Name, this->ID, this->PageTable);
225  }
226  else
227  this->PageTable = KernelPageTable;
228 
229  this->vma = new Memory::VirtualMemoryArea(this->PageTable);
230  this->ProgramBreak = new Memory::ProgramBreak(this->PageTable, this->vma);
231 
232  debug("Process page table: %#lx", this->PageTable);
233  debug("Created %s process \"%s\"(%d). Parent \"%s\"(%d)",
234  ExecutionMode == TaskExecutionMode::User ? "user" : "kernel",
235  this->Name, this->ID,
236  Parent ? this->Parent->Name : "None",
237  Parent ? this->Parent->ID : 0);
238 
239  this->AllocatedMemory += strlen(Name) + 1;
240  this->AllocatedMemory += sizeof(PCB);
241  this->AllocatedMemory += sizeof(FileDescriptorTable);
242  this->AllocatedMemory += FROM_PAGES(TO_PAGES(sizeof(Memory::PageTable) + 1));
243  this->AllocatedMemory += sizeof(Memory::VirtualMemoryArea);
244  this->AllocatedMemory += sizeof(Memory::ProgramBreak);
245  this->AllocatedMemory += sizeof(SymbolResolver::Symbols);
246 
247  this->Info.SpawnTime = TimeManager->GetCounter();
248 
249  if (Parent)
250  Parent->Children.push_back(this);
251  ctx->PushProcess(this);
252  }

330  this->Name, this->ID);
331  }
