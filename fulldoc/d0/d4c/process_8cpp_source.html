129  PCB::PCB(Task *ctx, PCB *Parent, const char *Name,
130  TaskExecutionMode ExecutionMode,
131  bool UseKernelPageTable,
132  uint16_t UserID, uint16_t GroupID)
133  : Node(ProcFS, std::to_string(ctx->NextPID), NodeType::DIRECTORY)
134  {
135  debug("+ %#lx", this);
136 
137  assert(ctx != nullptr);
138  assert(Name != nullptr);
139  assert(strlen(Name) > 0);
140  assert(ExecutionMode >= _ExecuteModeMin);
141  assert(ExecutionMode <= _ExecuteModeMax);
142 
143  this->ctx = ctx;
144  this->ID = ctx->NextPID++;
145 
146  if (this->Name) /* Prevent memory leak */
147  delete[] this->Name;
148 
149  this->Name = new char[strlen(Name) + 1];
150  strcpy((char *)this->Name, Name);
151  this->ExitCode = KILL_CRASH;
152 
153  /* Check parent */
154  if (Parent == nullptr)
155  this->Parent = ctx->GetCurrentProcess();
156  else
157  this->Parent = Parent;
158 
159  /* Set uid & gid */
160  if (this->Parent &&
161  UserID == UINT16_MAX &&
162  GroupID == UINT16_MAX)
163  {
164  UserID = this->Parent->Security.Real.UserID;
165  GroupID = this->Parent->Security.Real.GroupID;
166  debug("Inherited uid & gid from parent process %s(%d) with uid %d and gid %d",
167  this->Parent->Name, this->Parent->ID, UserID, GroupID);
168  }
169 
170  this->Security.Real.UserID = UserID;
171  this->Security.Real.GroupID = GroupID;
172  this->Security.Effective.UserID = UserID;
173  this->Security.Effective.GroupID = GroupID;
174  this->Security.ExecutionMode = ExecutionMode;
175 
176  switch (ExecutionMode)
177  {
178  case TaskExecutionMode::System:
179  fixme("Mode not supported.");
180  [[fallthrough]];
181  case TaskExecutionMode::Kernel:
182  {
183  this->Security.IsCritical = true;
184  break;
185  }
186  case TaskExecutionMode::User:
187  {
188  break;
189  }
190  default:
191  assert(false);
192  }
193 
194  this->FileDescriptors = new FileDescriptorTable(this);
195 
196  /* If create page table */
197  if (UseKernelPageTable == false)
198  {
199  OwnPageTable = true;
200  this->PageTable = KernelPageTable->Fork();
201  debug("Process %s(%d) has page table at %#lx",
202  this->Name, this->ID, this->PageTable);
203  }
204  else
205  this->PageTable = KernelPageTable;
206 
207  this->vma = new Memory::VirtualMemoryArea(this->PageTable);
208  this->ProgramBreak = new Memory::ProgramBreak(this->PageTable, this->vma);
209  this->Signals = new Signal(this);
210 
211  debug("Process page table: %#lx", this->PageTable);
212  debug("Created %s process \"%s\"(%d). Parent \"%s\"(%d)",
213  ExecutionMode == TaskExecutionMode::User ? "user" : "kernel",
214  this->Name, this->ID,
215  Parent ? this->Parent->Name : "None",
216  Parent ? this->Parent->ID : 0);
217 
218  this->AllocatedMemory += strlen(Name) + 1;
219  this->AllocatedMemory += sizeof(PCB);
220  this->AllocatedMemory += sizeof(FileDescriptorTable);
221  this->AllocatedMemory += FROM_PAGES(TO_PAGES(sizeof(Memory::PageTable) + 1));
222  this->AllocatedMemory += sizeof(Memory::VirtualMemoryArea);
223  this->AllocatedMemory += sizeof(Memory::ProgramBreak);
224  this->AllocatedMemory += sizeof(SymbolResolver::Symbols);
225  this->AllocatedMemory += sizeof(Signal);
226 
227  this->Info.SpawnTime = TimeManager->GetCounter();
228 
229  if (Parent)
230  Parent->Children.push_back(this);
231  ctx->ProcessList.push_back(this);
232  }

301 
302  pChild.erase(std::find(pChild.begin(),
303  pChild.end(),
304  this));
305  }

308  this->Name, this->ID);
