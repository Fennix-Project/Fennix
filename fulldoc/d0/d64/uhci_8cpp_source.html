93 nsa CrashUHCIKeyboardDriver::CrashUHCIKeyboardDriver(PCI::PCIDevice dev)
94  : Interrupts::Handler(dev)
95 {
96  Header = (PCIDeviceHeader *)dev.Header;
97 
98  switch (Header->HeaderType)
99  {
100  case 128:
101  {
102  ExPrint(ERROR_COLOR "Unknown header type %d! Guessing PCI Header 0\n" DEFAULT_COLOR,
103  Header->HeaderType);
104  [[fallthrough]];
105  }
106  case 0: /* PCI Header 0 */
107  {
108  PCI::PCIHeader0 *hdr = (PCI::PCIHeader0 *)Header;
109 
110  uint32_t BAR[6];
111  size_t BARsSize[6];
112 
113  BAR[0] = hdr->BAR0;
114  BAR[1] = hdr->BAR1;
115  BAR[2] = hdr->BAR2;
116  BAR[3] = hdr->BAR3;
117  BAR[4] = hdr->BAR4;
118  BAR[5] = hdr->BAR5;
119 
120  /* BARs Size */
121  for (short i = 0; i < 6; i++)
122  {
123  if (BAR[i] == 0)
124  continue;
125 
126  size_t size;
127  if ((BAR[i] & 1) == 0) /* Memory Base */
128  {
129  hdr->BAR0 = 0xFFFFFFFF;
130  size = hdr->BAR0;
131  hdr->BAR0 = BAR[i];
132  BARsSize[i] = size & (~15);
133  BARsSize[i] = ~BARsSize[i] + 1;
134  BARsSize[i] = BARsSize[i] & 0xFFFFFFFF;
135  debug("MEM BAR%d %#lx size: %d",
136  i, BAR[i], BARsSize[i]);
137  }
138  else if ((BAR[i] & 1) == 1) /* I/O Base */
139  {
140  hdr->BAR1 = 0xFFFFFFFF;
141  size = hdr->BAR1;
142  hdr->BAR1 = BAR[i];
143  BARsSize[i] = size & (~3);
144  BARsSize[i] = ~BARsSize[i] + 1;
145  BARsSize[i] = BARsSize[i] & 0xFFFF;
146  debug("I/O BAR%d %#lx size: %d",
147  i, BAR[i], BARsSize[i]);
148  }
149  }
150 
151  uintptr_t baseAddress = BAR[4];
152 
153  assert(baseAddress & 0x1); /* must be I/O */
154 
155  debug("baseAddress: %#lx size %#lx", baseAddress, BARsSize[4]);
156  Memory::Virtual vmm;
157  vmm.Map((void *)baseAddress, (void *)baseAddress, BARsSize[4], Memory::RW);
158  io = (IORegisters *)baseAddress;
159 
160  break;
161  }
162  case 1: /* PCI Header 1 (PCI-to-PCI Bridge) */
163  {
164  ExPrint(ERROR_COLOR "PCI-to-PCI Bridge not supported\n" DEFAULT_COLOR);
165  break;
166  }
167  case 2: /* PCI Header 2 (PCI-to-CardBus Bridge) */
168  {
169  ExPrint(ERROR_COLOR "PCI-to-CardBus Bridge not supported\n" DEFAULT_COLOR);
170  break;
171  }
172  default:
173  {
174  ExPrint(ERROR_COLOR "Invalid PCI header type\n" DEFAULT_COLOR);
175  break;
176  }
177  }
