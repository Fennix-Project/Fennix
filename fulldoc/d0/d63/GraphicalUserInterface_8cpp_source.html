38  void GUI::FetchInputs()
39  {
40  MouseData Mouse;
41  bool FoundMouseDriver = false;
42  if (likely(DriverManager->GetDrivers().size() > 0))
43  {
44  foreach (auto Driver in DriverManager->GetDrivers())
45  {
46  if (((FexExtended *)((uintptr_t)Driver.Address + EXTENDED_SECTION_ADDRESS))->Driver.Type == FexDriverType::FexDriverType_Input &&
47  ((FexExtended *)((uintptr_t)Driver.Address + EXTENDED_SECTION_ADDRESS))->Driver.TypeFlags & FexDriverInputTypes::FexDriverInputTypes_Mouse)
48  {
49 #ifdef DEBUG
50  static int once = 0;
51  if (!once++)
52  debug("Found mouse driver %ld", Driver.DriverUID);
53 #endif
54  KernelCallback callback{};
55  callback.Reason = FetchReason;
56  DriverManager->IOCB(Driver.DriverUID, &callback);
57  Mouse.X = callback.InputCallback.Mouse.X;
58  Mouse.Y = callback.InputCallback.Mouse.Y;
59  Mouse.Z = callback.InputCallback.Mouse.Z;
60  Mouse.Left = callback.InputCallback.Mouse.Buttons.Left;
61  Mouse.Right = callback.InputCallback.Mouse.Buttons.Right;
62  Mouse.Middle = callback.InputCallback.Mouse.Buttons.Middle;
63  FoundMouseDriver = true;
64  break;
65  }
66  }
67  }
68 
69  if (unlikely(!FoundMouseDriver))
70  {
71  debug("No mouse driver found.");
72  Mouse.X = Display->GetBuffer(200)->Width / 2;
73  Mouse.Y = Display->GetBuffer(200)->Height / 2;
74  Mouse.Z = 0;
75  Mouse.Left = false;
76  Mouse.Right = false;
77  Mouse.Middle = false;
78  }
79 
80  Event eTemplate;
81  memset(&eTemplate, 0, sizeof(Event));
82  foreach (auto wnd in this->Windows)
83  {
84  /* On mouse down event */
85  if (unlikely((!MouseArray[1].Left || !MouseArray[1].Right || !MouseArray[1].Middle) &&
86  (Mouse.Left || Mouse.Right || Mouse.Middle)))
87  {
88  eTemplate.MouseDown.X = Mouse.X;
89  eTemplate.MouseDown.Y = Mouse.Y;
90  eTemplate.MouseDown.Left = Mouse.Left;
91  eTemplate.MouseDown.Right = Mouse.Right;
92  eTemplate.MouseDown.Middle = Mouse.Middle;
93  wnd->OnMouseDown(&eTemplate);
94  }
95 
96  /* On mouse up event */
97  if (unlikely((MouseArray[1].Left || MouseArray[1].Right || MouseArray[1].Middle) &&
98  (!Mouse.Left || !Mouse.Right || !Mouse.Middle)))
99  {
100  eTemplate.MouseUp.X = Mouse.X;
101  eTemplate.MouseUp.Y = Mouse.Y;
102  eTemplate.MouseUp.Left = Mouse.Left;
103  eTemplate.MouseUp.Right = Mouse.Right;
104  eTemplate.MouseUp.Middle = Mouse.Middle;
105  wnd->OnMouseUp(&eTemplate);
106  }
107 
108  static int Idle = 0;
109 
110  if (likely(Mouse.X != MouseArray[1].X || Mouse.Y != MouseArray[1].Y))
111  {
112  Idle = 0;
113  Rect TopBarPos = wnd->GetPosition();
114  TopBarPos.Top -= 20;
115  TopBarPos.Height = 20;
116  TopBarPos.Width -= 60; /* buttons */
117  if (unlikely(TopBarPos.Top < 0))
118  {
119  TopBarPos.Top = 0;
120  wnd->GetPositionPtr()->Top = 20;
121  }
122 
123  Rect ResizeHintPos = wnd->GetPosition();
124  ResizeHintPos.Left += ResizeHintPos.Width - 20;
125  ResizeHintPos.Top += ResizeHintPos.Height - 20;
126  ResizeHintPos.Width = 20;
127  ResizeHintPos.Height = 20;
128 
129  if (unlikely(TopBarPos.Contains(Mouse.X, Mouse.Y) ||
130  TopBarPos.Contains(MouseArray[0].X, MouseArray[0].Y) ||
131  TopBarPos.Contains(MouseArray[1].X, MouseArray[1].Y)))
132  {
133  if (likely(Mouse.Left))
134  {
135  if (likely(MouseArray[1].Left))
136  {
137  wnd->GetPositionPtr()->Left += Mouse.X - MouseArray[0].X;
138  wnd->GetPositionPtr()->Top += Mouse.Y - MouseArray[0].Y;
139  OverlayBufferRepaint = true;
140  OverlayFullRepaint = true;
141  }
142  }
143  }
144 
145  if (ResizeHintPos.Contains(Mouse.X, Mouse.Y) ||
146  ResizeHintPos.Contains(MouseArray[0].X, MouseArray[0].Y) ||
147  ResizeHintPos.Contains(MouseArray[1].X, MouseArray[1].Y))
148  {
149  if (Mouse.Left)
150  {
151  if (MouseArray[1].Left)
152  {
153  wnd->GetPositionPtr()->Width += Mouse.X - MouseArray[0].X;
154  wnd->GetPositionPtr()->Height += Mouse.Y - MouseArray[0].Y;
155 
156  if (wnd->GetPositionPtr()->Width < 200)
157  {
158  wnd->GetPositionPtr()->Width = 200;
159  Mouse.X = MouseArray[0].X;
160  }
161 
162  if (wnd->GetPositionPtr()->Height < 100)
163  {
164  wnd->GetPositionPtr()->Height = 100;
165  Mouse.Y = MouseArray[0].Y;
166  }
167 
168  OverlayBufferRepaint = true;
169  OverlayFullRepaint = true;
170  eTemplate.Resize.Width = wnd->GetPosition().Width;
171  eTemplate.Resize.Height = wnd->GetPosition().Height;
172 
173  wnd->OnResize(&eTemplate);
174  }
175  }
176 
177  if (unlikely(Cursor != CursorType::ResizeAll))
178  {
179  Cursor = CursorType::ResizeAll;
180  CursorBufferRepaint = true;
181  }
182  }
183  else
184  {
185  if (unlikely(Cursor != CursorType::Arrow))
186  {
187  Cursor = CursorType::Arrow;
188  CursorBufferRepaint = true;
189  }
190  }
191 
192  eTemplate.MouseMove.X = Mouse.X;
193  eTemplate.MouseMove.Y = Mouse.Y;
194  eTemplate.MouseMove.Left = Mouse.Left;
195  eTemplate.MouseMove.Right = Mouse.Right;
196  eTemplate.MouseMove.Middle = Mouse.Middle;
197  wnd->OnMouseMove(&eTemplate);
198  }
199  else
200  {
201  if (unlikely(Idle > 1000))
202  CPU::Pause();
203  else
204  Idle++;
205  }
206  }
207 
208  foreach (auto wdg in this->Widgets)
209  {
210  // TODO: Implement mouse events for widgets
211  UNUSED(wdg);
212  }
213 
214  memmove(MouseArray + 1, MouseArray, sizeof(MouseArray) - sizeof(MouseArray[1]));
215  MouseArray[0] = Mouse;
216 
217  LastCursor = Cursor;
218  }

551  void GUI::Loop()
552  {
553  /*
554  Because we do not use a gpu to do the rendering, we need to do it manually.
555  This is why the mouse is slow when we have to draw a bunch of things.
556  */
557  while (IsRunning)
558  {
559 #ifdef DEBUG
560  FIi = CPU::Counter();
561 #endif
562  FetchInputs();
563 #ifdef DEBUG
564  FIi = CPU::Counter() - FIi;
565  PDi = CPU::Counter();
566 #endif
567  PaintDesktop();
568 #ifdef DEBUG
569  PDi = CPU::Counter() - PDi;
570  PWi = CPU::Counter();
571 #endif
572  PaintWidgets();
573 #ifdef DEBUG
574  PWi = CPU::Counter() - PWi;
575  PWWi = CPU::Counter();
576 #endif
577  PaintWindows();
578 #ifdef DEBUG
579  PWWi = CPU::Counter() - PWWi;
580  PCi = CPU::Counter();
581 #endif
582  PaintCursor();
583 #ifdef DEBUG
584  PCi = CPU::Counter() - PCi;
585  mmi = CPU::Counter();
586 #endif
587  memcpy(Display->GetBuffer(200)->Buffer, this->BackBuffer->Data, this->BackBuffer->Size);
588  Display->SetBuffer(200);
589 #ifdef DEBUG
590  mmi = CPU::Counter() - mmi;
591 #endif
592  }
593  }
