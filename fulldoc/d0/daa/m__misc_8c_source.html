90 long M_FileLength(FILE *handle)
91 {
92  long savedpos;
93  long length;
94 
95  // save the current position in the file
96  savedpos = ftell(handle);
97 
98  // jump to the end and find the length
99  fseek(handle, 0, SEEK_END);
100  length = ftell(handle);
101 
102  // go back to the old location
103  fseek(handle, savedpos, SEEK_SET);
104 
105  return length;

145

147  // reading the current position
148 
149  length = M_FileLength(handle);

152  count = fread(buf, 1, length, handle);

158  *buffer = buf;

198 void M_ExtractFileBase(char *path, char *dest)
199 {
200  char *src;
201  char *filename;
202  int length;
203 
204  src = path + strlen(path) - 1;
205 
206  // back up until a \ or the start
207  while (src != path && *(src - 1) != DIR_SEPARATOR)
208  {
209  src--;
210  }
211 
212  filename = src;
213 
214  // Copy up to eight characters
215  // Note: Vanilla Doom exits with an error if a filename is specified
216  // with a base of more than eight characters. To remove the 8.3
217  // filename limit, instead we simply truncate the name.
218 
219  length = 0;
220  memset(dest, 0, 8);
221 
222  while (*src != '\0' && *src != '.')
223  {
224  if (length >= 8)
225  {
226  printf("Warning: Truncated '%s' lump name to '%.8s'.\n",
227  filename, dest);
228  break;
229  }
230 
231  dest[length++] = toupper((int)*src++);
232  }
233 }

259 char *M_StrCaseStr(char *haystack, char *needle)
260 {

262  unsigned int needle_len;
263  unsigned int len;
264  unsigned int i;
265 
266  haystack_len = strlen(haystack);
267  needle_len = strlen(needle);
268 
269  if (haystack_len < needle_len)
270  {
271  return NULL;
272  }
273 
274  len = haystack_len - needle_len;
275 
276  for (i = 0; i <= len; ++i)
277  {
278  if (!strncasecmp(haystack + i, needle, needle_len))
279  {
280  return haystack + i;
281  }
282  }
283 
284  return NULL;

311 char *M_StringReplace(const char *haystack, const char *needle,
312  const char *replacement)
313 {
314  char *result, *dst;
315  const char *p;
316  size_t needle_len = strlen(needle);
317  size_t result_len, dst_len;
318 
319  // Iterate through occurrences of 'needle' and calculate the size of
320  // the new string.
321  result_len = strlen(haystack) + 1;
322  p = haystack;
323 
324  for (;;)
325  {
326  p = strstr(p, needle);
327  if (p == NULL)
328  {
329  break;
330  }
331 
332  p += needle_len;
333  result_len += strlen(replacement) - needle_len;
334  }
335 
336  // Construct new string.
337 
338  result = malloc(result_len);
339  if (result == NULL)
340  {
341  I_Error("M_StringReplace: Failed to allocate new string");
342  return NULL;
343  }
344 
345  dst = result; dst_len = result_len;
346  p = haystack;
347 
348  while (*p != '\0')
349  {
350  if (!strncmp(p, needle, needle_len))
351  {
352  M_StringCopy(dst, replacement, dst_len);
353  p += needle_len;
354  dst += strlen(replacement);
355  dst_len -= strlen(replacement);
356  }
357  else
358  {
359  *dst = *p;
360  ++dst; --dst_len;
361  ++p;
362  }
363  }
364 
365  *dst = '\0';
366 
367  return result;
368 }

481 // Safe, portable vsnprintf().
482 int M_vsnprintf(char *buf, size_t buf_len, const char *s, va_list args)
483 {
484  int result;
485 
486  if (buf_len < 1)
487  {
488  return 0;
489  }
490 
491  // Windows (and other OSes?) has a vsnprintf() that doesn't always
492  // append a trailing \0. So we must do it, and write into a buffer
493  // that is one byte shorter; otherwise this function is unsafe.
494  result = vsnprintf(buf, buf_len, s, args);
495 
496  // If truncated, change the final char in the buffer to a \0.
497  // A negative result indicates a truncated buffer on Windows.
498  if (result < 0 || result >= buf_len)
499  {
500  buf[buf_len - 1] = '\0';
501  result = buf_len - 1;
502  }
503 
504  return result;
505 }
