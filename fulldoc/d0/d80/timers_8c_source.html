97 /* We keep the timer as an offset into the future from the time it's set. */
98 void
99 shtimer_set (sh_timer *t, time_t sec, long usec)
100 {
101  struct timeval now;
102 
103  if (t->flags & SHTIMER_ALARM)
104  {
105  t->alrmflag = 0; /* just paranoia */
106  t->old_handler = set_signal_handler (SIGALRM, t->alrm_handler);
107  t->flags |= SHTIMER_SIGSET;
108  falarm (t->tmout.tv_sec = sec, t->tmout.tv_usec = usec);
109  t->flags |= SHTIMER_ALRMSET;
110  return;
111  }
112 
113  if (gettimeofday (&now, 0) < 0)
114  timerclear (&now);
115 
116  t->tmout.tv_sec = now.tv_sec + sec;
117  t->tmout.tv_usec = now.tv_usec + usec;
118  if (t->tmout.tv_usec > USEC_PER_SEC)
119  {
120  t->tmout.tv_sec++;
121  t->tmout.tv_usec -= USEC_PER_SEC;
122  }
123 }

125 void
126 shtimer_unset (sh_timer *t)
127 {
128  t->tmout.tv_sec = 0;
129  t->tmout.tv_usec = 0;
130 
131  if (t->flags & SHTIMER_ALARM)
132  {
133  t->alrmflag = 0;
134  if (t->flags & SHTIMER_ALRMSET)
135  falarm (0, 0);
136  if (t->old_handler && (t->flags & SHTIMER_SIGSET))
137  {
138  set_signal_handler (SIGALRM, t->old_handler);
139  t->flags &= ~SHTIMER_SIGSET;
140  t->old_handler = 0;
141  }
142  }
143 }

158 int
159 shtimer_chktimeout (sh_timer *t)
160 {
161  struct timeval now;
162  int r;
163 
164  /* Use the flag to avoid returning sigalrm_seen here */
165  if (t->flags & SHTIMER_ALARM)
166  return t->alrmflag;
167 
168  /* Could check a flag for this */
169  if (t->tmout.tv_sec == 0 && t->tmout.tv_usec == 0)
170  return 0;
171 
172  if (gettimeofday (&now, 0) < 0)
173  return 0;
174  r = ((now.tv_sec > t->tmout.tv_sec) ||
175  (now.tv_sec == t->tmout.tv_sec && now.tv_usec >= t->tmout.tv_usec));
176 
177  return r;
178 }

180 #if defined (HAVE_SELECT) || defined (HAVE_PSELECT)
181 int
182 shtimer_select (sh_timer *t)
183 {
184  int r, nfd;
185  sigset_t blocked_sigs, prevmask;
186  struct timeval now, tv;
187  fd_set readfds;
188 #if defined (HAVE_PSELECT)
189  struct timespec ts;
190 #endif
191 
192  /* We don't want a SIGCHLD to interrupt this */
193  sigemptyset (&blocked_sigs);
194 # if defined (SIGCHLD)
195  sigaddset (&blocked_sigs, SIGCHLD);
196 # endif
197 
198  if (gettimeofday (&now, 0) < 0)
199  {
200  if (t->flags & SHTIMER_LONGJMP)
201  sh_longjmp (t->jmpenv, 1);
202  else
203  return -1;
204  }
205 
206  /* If the timer has already expired, return immediately */
207  if ((now.tv_sec > t->tmout.tv_sec) ||
208  (now.tv_sec == t->tmout.tv_sec && now.tv_usec >= t->tmout.tv_usec))
209  {
210  if (t->flags & SHTIMER_LONGJMP)
211  sh_longjmp (t->jmpenv, 1);
212  else if (t->tm_handler)
213  return ((*t->tm_handler) (t));
214  else
215  return 0;
216  }
217 
218  /* compute timeout */
219  tv.tv_sec = t->tmout.tv_sec - now.tv_sec;
220  tv.tv_usec = t->tmout.tv_usec - now.tv_usec;
221  if (tv.tv_usec < 0)
222  {
223  tv.tv_sec--;
224  tv.tv_usec += USEC_PER_SEC;
225  }
226 
227 #if defined (HAVE_PSELECT)
228  ts.tv_sec = tv.tv_sec;
229  ts.tv_nsec = tv.tv_usec * 1000;
230 #else
231  sigemptyset (&prevmask);
232 #endif /* !HAVE_PSELECT */
233 
234  nfd = (t->fd >= 0) ? t->fd + 1 : 0;
235  FD_ZERO (&readfds);
236  if (t->fd >= 0)
237  FD_SET (t->fd, &readfds);
238 
239 #if defined (HAVE_PSELECT)
240  r = pselect(nfd, &readfds, (fd_set *)0, (fd_set *)0, &ts, &blocked_sigs);
241 #else
242  sigprocmask (SIG_SETMASK, &blocked_sigs, &prevmask);
243  r = select(nfd, &readfds, (fd_set *)0, (fd_set *)0, &tv);
244  sigprocmask (SIG_SETMASK, &prevmask, NULL);
245 #endif
246 
247  if (r < 0)
248  return r; /* caller will handle */
249  else if (r == 0 && (t->flags & SHTIMER_LONGJMP))
250  sh_longjmp (t->jmpenv, 1);
251  else if (r == 0 && t->tm_handler)
252  return ((*t->tm_handler) (t));
253  else
254  return r;
255 }
256 #endif /* !HAVE_TIMEVAL || !HAVE_SELECT */
