354 private:

360

362  char Channels = 2;
363  uint8_t Volume = AV_Maximum;
364  bool Mute = false;
365  int SampleRate = 48000;
366  char SampleSize = 2;
367 
368 public:
369  size_t write(uint8_t *Buffer, size_t Size)
370  {
371  if (Buffer == nullptr)
372  {
373  Log("Invalid buffer.");
374  return -EINVAL;
375  }
376 
377  if ((Size == 0) || (Size % (SampleSize * Channels)))
378  {
379  Log("Invalid buffer length.");
380  return -EINVAL;
381  }
382 
383  int TotalBDLToFill = (int)((Size + PAGE_SIZE - 1) >> 12);
384 
385  while (Size > 0)
386  {
387  bool ActiveDMA = !(inw(BusMasterAddress + PCMOUT_Status) & TC_DMAControllerControl);
388 
389  if (ActiveDMA)
390  {
391  int RemainingBDL = 0;
392 
393  do
394  {
395  int CurrentBDL = inb(BusMasterAddress + PCMOUT_BufferDescriptorEntry);
396  int LastBDL = inb(BusMasterAddress + PCMOUT_DescriptorEntries);
397 
398  RemainingBDL = LastBDL - CurrentBDL;
399  if (RemainingBDL < 0)
400  RemainingBDL += DescriptorListLength;
401 
402  RemainingBDL += 1;
403 
404  if (RemainingBDL >= DescriptorListLength - 1)
405  {
406  long SampleCount = DescriptorList[(CurrentBDL + 1) % DescriptorListLength].SampleCount / Channels;
407  if (SampleCount > 0)
408  Sleep(SampleCount * 1000 / SampleRate);
409  }
410 
411  } while (RemainingBDL >= DescriptorListLength - 1 &&
412  !(inw(BusMasterAddress + PCMOUT_Status) & TC_DMAControllerControl));
413  }
414 
415  uint8_t CurrentBDL = inb(BusMasterAddress + PCMOUT_BufferDescriptorEntry);
416  uint8_t LastBDL = inb(BusMasterAddress + PCMOUT_DescriptorEntries);
417  uint8_t NextBDL = LastBDL % DescriptorListLength;
418 
419  ActiveDMA = !(inw(BusMasterAddress + PCMOUT_Status) & TC_DMAControllerControl);
420  if (ActiveDMA)
421  {
422  NextBDL = (uint8_t)((LastBDL + 1) % DescriptorListLength);
423  if (NextBDL == CurrentBDL)
424  continue;
425  }
426 
427  do
428  {
429  size_t Wrote = (PAGE_SIZE > Size) ? size_t(Size)
430  : size_t(PAGE_SIZE);
431 
432  if (Wrote == 0)
433  {
434  Log("Wrote 0 bytes.");
435  break;
436  }
437 
438  memcpy((void *)((uint64_t)DescriptorList[NextBDL].Address), Buffer, Wrote);
439  DescriptorList[NextBDL].Flags = 0;
440 
441  Buffer += Wrote;
442  Size -= (unsigned int)Wrote;
443 
444  DescriptorList[NextBDL].SampleCount = uint16_t(Wrote / SampleSize);
445  TotalBDLToFill--;
446  NextBDL = (uint8_t)((NextBDL + 1) % DescriptorListLength);
447  } while (TotalBDLToFill-- && NextBDL != CurrentBDL);
448 
449  outb(BusMasterAddress + PCMOUT_DescriptorEntries, NextBDL - 1);
450 
451  ActiveDMA = !(inw(BusMasterAddress + PCMOUT_Status) & TC_DMAControllerControl);
452  if (!ActiveDMA)
453  {
454  // Start DMA
455  outb(BusMasterAddress + PCMOUT_TransferControl,
456  inb(BusMasterAddress + PCMOUT_TransferControl) | TC_DMAControllerControl);
457  }
458  }
459  return Size;
460  }

556  void OnInterruptReceived(TrapFrame *)
557  {
558  uint16_t Status = inw(MixerAddress + PCMOUT_Status);
559  if (Status & TC_IOCInterruptEnable)
560  {
561  DebugLog("IOC");
562  outw(MixerAddress + PCMOUT_Status, TC_IOCInterruptEnable);
563  uint16_t CurrentBDL = inb(BusMasterAddress + PCMOUT_BufferDescriptorEntry);
564  uint16_t LastBDL = (CurrentBDL + 2) & (DescriptorListLength - 1);
565  outb(BusMasterAddress + PCMOUT_DescriptorEntries, LastBDL);
566  Log("FIXME: CurrentBDL: %d, LastBDL: %d", CurrentBDL, LastBDL);
567  }
568  else if (Status & TC_LastBufferEntryInterruptEnable)
569  {
570  DebugLog("Last buffer entry");
571  // Stop DMA
572  uint8_t TransferControl = inb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl));
573  TransferControl &= ~TC_DMAControllerControl;
574  outb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl), TransferControl);
575 
576  outw(MixerAddress + PCMOUT_Status, TC_LastBufferEntryInterruptEnable);
577  }
578  else if (Status & TC_FifoERRORInterruptEnable)
579  {
580  Log("FIFO error");
581  outw(MixerAddress + PCMOUT_Status, TC_FifoERRORInterruptEnable);
582  }
583  else
584  {
585  DebugLog("Unknown interrupt status %#x", Status);
586  outw(MixerAddress + PCMOUT_Status, 0xFFFF);
587  }
588  }

590  void Panic()
591  {
592  uint8_t TransferControl = inb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl));
593  TransferControl &= ~(TC_LastBufferEntryInterruptEnable |
594  TC_IOCInterruptEnable |
595  TC_FifoERRORInterruptEnable);
596  outb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl), TransferControl);
597 
598  uint32_t GlobalControl = inl((uint16_t)(BusMasterAddress + NABM_GlobalControl));
599  GlobalControl &= ~GC_GlobalInterruptEnable;
600  GlobalControl |= GC_ShutDown;
601  outl((uint16_t)(BusMasterAddress + NABM_GlobalControl), GlobalControl);
602  }

604  AC97Device(PCIHeader0 *_Header)
605  : Header(_Header)
606  {
607  /* Native Audio Mixer Base Address */
608  uint32_t PCIBAR0 = Header->BAR0;
609 
610  /* Native Audio Bus Master Base Address */
611  uint32_t PCIBAR1 = Header->BAR1;
612 
613  // uint8_t Type = PCIBAR0 & 1;
614  MixerAddress = (uint16_t)(PCIBAR0 & (~3));
615  BusMasterAddress = PCIBAR1 & (~15);
616 
617  uint16_t OutputPCMTransferControl = BusMasterAddress + PCMOUT_TransferControl;
618 
619  /* DescriptorList address MUST be physical. */
620  DescriptorList = (BufferDescriptorList *)AllocateMemory(TO_PAGES(sizeof(BufferDescriptorList) * DescriptorListLength));
621  memset(DescriptorList, 0, sizeof(BufferDescriptorList) * DescriptorListLength);
622 
623  uint16_t DLSampleCount = (uint16_t)(PAGE_SIZE / SampleSize);
624  for (int i = 0; i < DescriptorListLength; i++)
625  {
626  DescriptorList[i].Address = (uint32_t)(uintptr_t)AllocateMemory(TO_PAGES(sizeof(uint16_t *)));
627  DescriptorList[i].SampleCount = DLSampleCount;
628  DescriptorList[i].Flags = 0;
629  DebugLog("DescriptorList[%d] = { Address: %#lx, SampleCount: %d, Flags: %#lx }",
630  i,
631  DescriptorList[i].Address,
632  DescriptorList[i].SampleCount,
633  DescriptorList[i].Flags);
634  }
635 
636  outw(MixerAddress + NAM_MasterVolume, MixerVolume(Volume, Volume, Mute));
637  outw(MixerAddress + NAM_PCMOutVolume, MixerVolume(Volume, Volume, Mute));
638 
639  Volume = 0x3F - (0x3F * /* VOL 50% */ 50 / 100);
640  outw(MixerAddress + NAM_MasterVolume, MixerVolume(Volume, Volume, Mute));
641 
642  outb(OutputPCMTransferControl, inb(OutputPCMTransferControl) | TC_TransferReset);
643  while (inb(OutputPCMTransferControl) & TC_TransferReset)
644  ;
645 
646  uint32_t GlobalControl = inl(BusMasterAddress + NABM_GlobalControl);
647  GlobalControl = (GlobalControl & ~((0x3U) << 0x16)); /* PCM 16-bit mode */
648  GlobalControl = (GlobalControl & ~((0x3U) << 20)); /* 2 channels */
649  GlobalControl |= GC_GlobalInterruptEnable;
650  GlobalControl &= ~GC_ShutDown;
651 
652  outl(BusMasterAddress + PCMOUT_BufferDescriptorList,
653  (uint32_t)(uint64_t)DescriptorList);
654 
655  outl(BusMasterAddress + NABM_GlobalControl, GlobalControl);
656 
657  uint8_t TransferControl = inb(OutputPCMTransferControl);
658  TransferControl |= TC_IOCInterruptEnable |
659  TC_FifoERRORInterruptEnable;
660  outb(OutputPCMTransferControl, TransferControl);
661 
662  // Stop DMA
663  outb(OutputPCMTransferControl, inb(OutputPCMTransferControl) & ~TC_DMAControllerControl);
664  }
665 
666  ~AC97Device()
667  {
668  outw(MixerAddress + NAM_MasterVolume, MixerVolume(AV_Maximum, AV_Maximum, true));
669  outw(MixerAddress + NAM_PCMOutVolume, MixerVolume(AV_Maximum, AV_Maximum, true));
670 
671  // Stop DMA
672  outb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl),
673  inb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl)) & ~TC_DMAControllerControl);
674 
675  // Disable interrupts
676  uint8_t TransferControl = inb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl));
677 
678  TransferControl &= ~(TC_LastBufferEntryInterruptEnable |
679  TC_IOCInterruptEnable |
680  TC_FifoERRORInterruptEnable);
681  outb((uint16_t)(BusMasterAddress + PCMOUT_TransferControl), TransferControl);
682 
683  // Disable global control
684  uint32_t GlobalControl = inl((uint16_t)(BusMasterAddress + NABM_GlobalControl));
685  GlobalControl &= ~GC_GlobalInterruptEnable;
686  GlobalControl |= GC_ShutDown;
687  outl((uint16_t)(BusMasterAddress + NABM_GlobalControl), GlobalControl);
688  }
