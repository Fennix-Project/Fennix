45  off_t RefNode::seek(off_t Offset, int Whence)
46  {
47  if (this->SymlinkTo)
48  return this->SymlinkTo->seek(Offset, Whence);
49 
50  // debug("Current offset is %d", this->Offset.load());
51  switch (Whence)
52  {
53  case SEEK_SET:
54  {
55  if (Offset > this->FileSize)
56  return -EINVAL;
57 
58  if (Offset < 0)
59  {
60  fixme("Negative offset %d is not implemented", Offset);
61  Offset = 0;
62  }
63 
64  if (Offset > this->FileSize)
65  {
66  fixme("Offset %d is bigger than file size %d",
67  Offset, this->FileSize);
68  Offset = this->FileSize;
69  }
70 
71  this->FileOffset.store(Offset);
72  break;
73  }
74  case SEEK_CUR:
75  {
76  off_t NewOffset = off_t(this->FileOffset.load()) + Offset;
77  if (NewOffset > this->FileSize ||
78  NewOffset < 0)
79  {
80  return -EINVAL;
81  }
82 
83  this->FileOffset.store(NewOffset);
84  break;
85  }
86  case SEEK_END:
87  {
88  off_t NewOffset = this->FileSize + Offset;
89  if (NewOffset > this->FileSize ||
90  NewOffset < 0)
91  {
92  return -EINVAL;
93  }
94 
95  this->FileOffset.store(NewOffset);
96  break;
97  }
98  default:
99  {
100  error("Invalid whence!");
101  return -EINVAL;
102  }
103  }
104 
105  off_t RetOffset = off_t(this->FileOffset.load());
106  // debug("( %d %ld %s[%d] ) -> %d",
107  // Offset, this->Offset.load(),
108  // SeekStrings[Whence], Whence,
109  // RetOffset);
