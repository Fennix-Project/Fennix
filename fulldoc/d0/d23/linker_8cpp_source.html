340 void ObjectRepository::_fetchFromPhdrs(SharedObject *object, void *phdr_pointer,
341  size_t phdr_entry_size, size_t phdr_count, void *entry_pointer) {
342  __ensure(object->isMainObject);
343  object->phdrPointer = phdr_pointer;
344  object->phdrEntrySize = phdr_entry_size;
345  object->phdrCount = phdr_count;
346  if(verbose)
347  mlibc::infoLogger() << "rtdl: Loading " << object->name << frg::endlog;
348 
349  // Note: the entry pointer is absolute and not relative to the base address.
350  object->entry = entry_pointer;
351 
352  frg::optional<ptrdiff_t> dynamic_offset;
353  frg::optional<ptrdiff_t> tls_offset;
354 
355  // segments are already mapped, so we just have to find the dynamic section
356  for(size_t i = 0; i < phdr_count; i++) {
357  auto phdr = (Elf64_Phdr *)((uintptr_t)phdr_pointer + i * phdr_entry_size);
358  switch(phdr->p_type) {
359  case PT_PHDR:
360  // Determine the executable's base address (in the PIE case) by comparing
361  // the PHDR segment's load address against it's address in the ELF file.
362  object->baseAddress = reinterpret_cast<uintptr_t>(phdr_pointer) - phdr->p_vaddr;
363  if(verbose)
364  mlibc::infoLogger() << "rtdl: Executable is loaded at "
365  << (void *)object->baseAddress << frg::endlog;
366  break;
367  case PT_DYNAMIC:
368  dynamic_offset = phdr->p_vaddr;
369  break;
370  case PT_TLS: {
371  object->tlsSegmentSize = phdr->p_memsz;
372  object->tlsAlignment = phdr->p_align;
373  object->tlsImageSize = phdr->p_filesz;
374  tls_offset = phdr->p_vaddr;
375  break;
376  case PT_INTERP:
377  object->interpreterPath = frg::string<MemoryAllocator>{
378  (char*)(object->baseAddress + phdr->p_vaddr),
379  getAllocator()
380  };
381  } break;
382  default:
383  //FIXME warn about unknown phdrs
384  break;
385  }
386  }
387 
388  if(dynamic_offset)
389  object->dynamic = (Elf64_Dyn *)(object->baseAddress + *dynamic_offset);
390  if(tls_offset)
391  object->tlsImagePtr = (void *)(object->baseAddress + *tls_offset);
392 }

395 void ObjectRepository::_fetchFromFile(SharedObject *object, int fd) {
396  __ensure(!object->isMainObject);
397 
398  // read the elf file header
399  Elf64_Ehdr ehdr;
400  readExactlyOrDie(fd, &ehdr, sizeof(Elf64_Ehdr));
401 
402  __ensure(ehdr.e_ident[0] == 0x7F
403  && ehdr.e_ident[1] == 'E'
404  && ehdr.e_ident[2] == 'L'
405  && ehdr.e_ident[3] == 'F');
406  __ensure(ehdr.e_type == ET_EXEC || ehdr.e_type == ET_DYN);
407 
408  // read the elf program headers
409  auto phdr_buffer = (char *)getAllocator().allocate(ehdr.e_phnum * ehdr.e_phentsize);
410  seekOrDie(fd, ehdr.e_phoff);
411  readExactlyOrDie(fd, phdr_buffer, ehdr.e_phnum * ehdr.e_phentsize);
412 
413  object->phdrPointer = phdr_buffer;
414  object->phdrCount = ehdr.e_phnum;
415  object->phdrEntrySize = ehdr.e_phentsize;
416 
417  // Allocate virtual address space for the DSO.
418  constexpr size_t hugeSize = 0x200000;
419 
420  uintptr_t highest_address = 0;
421  for(int i = 0; i < ehdr.e_phnum; i++) {
422  auto phdr = (Elf64_Phdr *)(phdr_buffer + i * ehdr.e_phentsize);
423 
424  if(phdr->p_type != PT_LOAD)
425  continue;
426 
427  auto limit = phdr->p_vaddr + phdr->p_memsz;
428  if(limit > highest_address)
429  highest_address = limit;
430  }
431 
432  __ensure(!(object->baseAddress & (hugeSize - 1)));
433 
434 #if MLIBC_MMAP_ALLOCATE_DSO
435  void *mappedAddr = nullptr;
436 
437  if (mlibc::sys_vm_map(nullptr,
438  highest_address - object->baseAddress, PROT_NONE,
439  MAP_PRIVATE | MAP_ANONYMOUS, -1, 0, &mappedAddr)) {
440  mlibc::panicLogger() << "sys_vm_map failed when allocating address space for DSO \""
441  << object->name << "\""
442  << ", base " << (void *)object->baseAddress
443  << ", requested " << (highest_address - object->baseAddress) << " bytes"
444  << frg::endlog;
445  }
446 
447  object->baseAddress = reinterpret_cast<uintptr_t>(mappedAddr);
448 #else
449  object->baseAddress = libraryBase;
450  libraryBase += (highest_address + (hugeSize - 1)) & ~(hugeSize - 1);
451 #endif
452 
453  if(verbose || logBaseAddresses)
454  mlibc::infoLogger() << "rtdl: Loading " << object->name
455  << " at " << (void *)object->baseAddress << frg::endlog;
456 
457  // Load all segments.
458  constexpr size_t pageSize = 0x1000;
459  for(int i = 0; i < ehdr.e_phnum; i++) {
460  auto phdr = (Elf64_Phdr *)(phdr_buffer + i * ehdr.e_phentsize);
461 
462  if(phdr->p_type == PT_LOAD) {
463  size_t misalign = phdr->p_vaddr & (pageSize - 1);
464  __ensure(phdr->p_memsz > 0);
465  __ensure(phdr->p_memsz >= phdr->p_filesz);
466 
467  // If the following condition is violated, we cannot use mmap() the segment;
468  // however, GCC only generates ELF files that satisfy this.
469  __ensure(misalign == (phdr->p_offset & (pageSize - 1)));
470 
471  auto map_address = object->baseAddress + phdr->p_vaddr - misalign;
472  auto backed_map_size = (phdr->p_filesz + misalign + pageSize - 1) & ~(pageSize - 1);
473  auto total_map_size = (phdr->p_memsz + misalign + pageSize - 1) & ~(pageSize - 1);
474 
475  int prot = 0;
476  if(phdr->p_flags & PF_R)
477  prot |= PROT_READ;
478  if(phdr->p_flags & PF_W)
479  prot |= PROT_WRITE;
480  if(phdr->p_flags & PF_X)
481  prot |= PROT_EXEC;
482 
483  #if MLIBC_MAP_DSO_SEGMENTS
484  void *map_pointer;
485  if(mlibc::sys_vm_map(reinterpret_cast<void *>(map_address),
486  backed_map_size, prot | PROT_WRITE,
487  MAP_PRIVATE | MAP_FIXED, fd, phdr->p_offset - misalign, &map_pointer))
488  __ensure(!"sys_vm_map failed");
489  if(total_map_size > backed_map_size)
490  if(mlibc::sys_vm_map(reinterpret_cast<void *>(map_address + backed_map_size),
491  total_map_size - backed_map_size, prot | PROT_WRITE,
492  MAP_PRIVATE | MAP_FIXED | MAP_ANONYMOUS, -1, 0, &map_pointer))
493  __ensure(!"sys_vm_map failed");
494 
495  if(mlibc::sys_vm_readahead)
496  if(mlibc::sys_vm_readahead(reinterpret_cast<void *>(map_address),
497  backed_map_size))
498  mlibc::infoLogger() << "mlibc: sys_vm_readahead() failed in ld.so"
499  << frg::endlog;
500 
501  // Clear the trailing area at the end of the backed mapping.
502  // We do not clear the leading area; programs are not supposed to access it.
503  memset(reinterpret_cast<void *>(map_address + misalign + phdr->p_filesz),
504  0, phdr->p_memsz - phdr->p_filesz);
505  #else
506  (void)backed_map_size;
507 
508  void *map_pointer;
509  if(mlibc::sys_vm_map(reinterpret_cast<void *>(map_address),
510  total_map_size, prot | PROT_WRITE,
511  MAP_PRIVATE | MAP_FIXED | MAP_ANONYMOUS, -1, 0, &map_pointer))
512  __ensure(!"sys_vm_map failed");
513 
514  seekOrDie(fd, phdr->p_offset);
515  readExactlyOrDie(fd, reinterpret_cast<char *>(map_address) + misalign,
516  phdr->p_filesz);
517  #endif
518  // Take care of removing superfluous permissions.
519  if(mlibc::sys_vm_protect && ((prot & PROT_WRITE) == 0))
520  if(mlibc::sys_vm_protect(map_pointer, total_map_size, prot))
521  mlibc::infoLogger() << "mlibc: sys_vm_protect() failed in ld.so" << frg::endlog;
522  }else if(phdr->p_type == PT_TLS) {
523  object->tlsSegmentSize = phdr->p_memsz;
524  object->tlsAlignment = phdr->p_align;
525  object->tlsImageSize = phdr->p_filesz;
526  object->tlsImagePtr = (void *)(object->baseAddress + phdr->p_vaddr);
527  }else if(phdr->p_type == PT_DYNAMIC) {
528  object->dynamic = (Elf64_Dyn *)(object->baseAddress + phdr->p_vaddr);
529  }else if(phdr->p_type == PT_INTERP
530  || phdr->p_type == PT_PHDR
531  || phdr->p_type == PT_NOTE
532  || phdr->p_type == PT_RISCV_ATTRIBUTES
533  || phdr->p_type == PT_GNU_EH_FRAME
534  || phdr->p_type == PT_GNU_RELRO
535  || phdr->p_type == PT_GNU_STACK
536  || phdr->p_type == PT_GNU_PROPERTY) {
537  // ignore the phdr
538  }else{
539  mlibc::panicLogger() << "Unexpected PHDR type 0x"
540  << frg::hex_fmt(phdr->p_type) << " in DSO " << object->name << frg::endlog;
541  }
542  }

549 void ObjectRepository::_parseDynamic(SharedObject *object) {
550  if(!object->dynamic)
551  mlibc::infoLogger() << "ldso: Object '" << object->name
552  << "' does not have a dynamic section" << frg::endlog;
553  __ensure(object->dynamic);
554 
555  // Fix up these offsets to addresses after the loop, since the
556  // addresses depend on the value of DT_STRTAB.
557  frg::optional<ptrdiff_t> runpath_offset;
558  frg::optional<ptrdiff_t> soname_offset;
559 
560  for(size_t i = 0; object->dynamic[i].d_tag != DT_NULL; i++) {
561  Elf64_Dyn *dynamic = &object->dynamic[i];
562  switch(dynamic->d_tag) {
563  // handle hash table, symbol table and string table
564  case DT_HASH:
565  object->hashStyle = HashStyle::systemV;
566  object->hashTableOffset = dynamic->d_un.d_ptr;
567  break;
568  case DT_GNU_HASH:
569  object->hashStyle = HashStyle::gnu;
570  object->hashTableOffset = dynamic->d_un.d_ptr;
571  break;
572  case DT_STRTAB:
573  object->stringTableOffset = dynamic->d_un.d_ptr;
574  break;
575  case DT_STRSZ:
576  break; // we don't need the size of the string table
577  case DT_SYMTAB:
578  object->symbolTableOffset = dynamic->d_un.d_ptr;
579  break;
580  case DT_SYMENT:
581  __ensure(dynamic->d_un.d_val == sizeof(Elf64_Sym));
582  break;
583  // handle lazy relocation table
584  case DT_PLTGOT:
585  object->globalOffsetTable = (void **)(object->baseAddress
586  + dynamic->d_un.d_ptr);
587  break;
588  case DT_JMPREL:
589  object->lazyRelocTableOffset = dynamic->d_un.d_ptr;
590  break;
591  case DT_PLTRELSZ:
592  object->lazyTableSize = dynamic->d_un.d_val;
593  break;
594  case DT_PLTREL:
595  if(dynamic->d_un.d_val == DT_RELA) {
596  object->lazyExplicitAddend = true;
597  }else{

864 Tcb *allocateTcb() {
865  size_t tlsInitialSize = runtimeTlsMap->initialLimit;
866 
867  // To make sure that both the TCB and TLS data are sufficiently aligned, allocate
868  // slightly more than necessary and adjust alignment afterwards.
869  size_t alignOverhead = frg::min(alignof(Tcb), tlsMaxAlignment);
870  size_t allocSize = tlsInitialSize + sizeof(Tcb) + alignOverhead;
871  auto allocation = reinterpret_cast<uintptr_t>(getAllocator().allocate(allocSize));
872  memset(reinterpret_cast<void *>(allocation), 0, allocSize);
873 
874  uintptr_t tlsAddress, tcbAddress;
875  if constexpr (tlsAboveTp) {
876  // Here we must satisfy two requirements of the TCB and the TLS data:
877  // 1. One should follow the other immediately in memory. We do this so that
878  // we can simply add or subtract sizeof(Tcb) to obtain the address of the other.
879  // 2. Both should be sufficiently aligned.
880  // To do this, we will fix whichever address has stricter alignment requirements, and
881  // derive the other from it.
882  if (tlsMaxAlignment > alignof(Tcb)) {
883  tlsAddress = alignUp(allocation + sizeof(Tcb), tlsMaxAlignment);
884  tcbAddress = tlsAddress - sizeof(Tcb);
885  } else {
886  tcbAddress = alignUp(allocation, alignof(Tcb));
887  tlsAddress = tcbAddress + sizeof(Tcb);
888  }
889  __ensure((tlsAddress & (tlsMaxAlignment - 1)) == 0);
890  __ensure(tlsAddress == tcbAddress + sizeof(Tcb));
891  } else {
892  // The TCB should be aligned such that the preceding blocks are aligned too.
893  tcbAddress = alignUp(allocation + tlsInitialSize, alignOverhead);
894  tlsAddress = tcbAddress - tlsInitialSize;
895  }
896  __ensure((tcbAddress & (alignof(Tcb) - 1)) == 0);
897 
898  if (verbose) {
899  mlibc::infoLogger() << "rtdl: tcb allocated at " << (void *)tcbAddress
900  << ", size = 0x" << frg::hex_fmt{sizeof(Tcb)} << frg::endlog;
901  mlibc::infoLogger() << "rtdl: tls allocated at " << (void *)tlsAddress
902  << ", size = 0x" << frg::hex_fmt{tlsInitialSize} << frg::endlog;
903  }
904 
905  Tcb *tcb_ptr = new ((char *)tcbAddress) Tcb;
906  tcb_ptr->selfPointer = tcb_ptr;
907 
908  tcb_ptr->stackCanary = __stack_chk_guard;
909  tcb_ptr->cancelBits = tcbCancelEnableBit;
910  tcb_ptr->didExit = 0;
911  tcb_ptr->isJoinable = 1;
912  tcb_ptr->returnValue = nullptr;
913  tcb_ptr->localKeys = frg::construct<frg::array<Tcb::LocalKey, PTHREAD_KEYS_MAX>>(getAllocator());
914  tcb_ptr->dtvSize = runtimeTlsMap->indices.size();
915  tcb_ptr->dtvPointers = frg::construct_n<void *>(getAllocator(), runtimeTlsMap->indices.size());
916  memset(tcb_ptr->dtvPointers, 0, sizeof(void *) * runtimeTlsMap->indices.size());
917  for(size_t i = 0; i < runtimeTlsMap->indices.size(); ++i) {
918  auto object = runtimeTlsMap->indices[i];
919  if(object->tlsModel != TlsModel::initial)
920  continue;
921  tcb_ptr->dtvPointers[i] = reinterpret_cast<char *>(tcb_ptr) + object->tlsOffset;
922  }
923 
924  return tcb_ptr;
925 }

1017 frg::optional<ObjectSymbol> resolveInObject(SharedObject *object, frg::string_view string) {
1018  // Checks if the symbol can be used to satisfy the dependency.
1019  auto eligible = [&] (ObjectSymbol cand) {
1020  if(cand.symbol()->st_shndx == SHN_UNDEF)
1021  return false;
1022 
1023  auto bind = ELF64_ST_BIND(cand.symbol()->st_info);
1024  if(bind != STB_GLOBAL && bind != STB_WEAK && bind != STB_GNU_UNIQUE)
1025  return false;
1026 
1027  return true;
1028  };
1029 
1030  if (object->hashStyle == HashStyle::systemV) {
1031  auto hash_table = (Elf64_Word *)(object->baseAddress + object->hashTableOffset);
1032  Elf64_Word num_buckets = hash_table[0];
1033  auto bucket = elf64Hash(string) % num_buckets;
1034 
1035  auto index = hash_table[2 + bucket];
1036  while(index != 0) {
1037  ObjectSymbol cand{object, (Elf64_Sym *)(object->baseAddress
1038  + object->symbolTableOffset + index * sizeof(Elf64_Sym))};
1039  if(eligible(cand) && frg::string_view{cand.getString()} == string)
1040  return cand;
1041 
1042  index = hash_table[2 + num_buckets + index];
1043  }
1044 
1045  return frg::optional<ObjectSymbol>{};
1046  }else{
1047  __ensure(object->hashStyle == HashStyle::gnu);
1048 
1049  struct GnuTable {
1050  Elf64_Word nBuckets;
1051  Elf64_Word symbolOffset;
1052  Elf64_Word bloomSize;
1053  Elf64_Word bloomShift;
1054  };
1055 
1056  auto hash_table = reinterpret_cast<const GnuTable *>(object->baseAddress
1057  + object->hashTableOffset);
1058  auto buckets = reinterpret_cast<const Elf64_Word *>(object->baseAddress
1059  + object->hashTableOffset + sizeof(GnuTable)
1060  + hash_table->bloomSize * sizeof(Elf64_Addr));
1061  auto chains = reinterpret_cast<const Elf64_Word *>(object->baseAddress
1062  + object->hashTableOffset + sizeof(GnuTable)
1063  + hash_table->bloomSize * sizeof(Elf64_Addr)
1064  + hash_table->nBuckets * sizeof(Elf64_Word));
1065 
1066  // TODO: Use the bloom filter.
1067 
1068  // The symbols of a given bucket are contiguous in the table.
1069  auto hash = gnuHash(string);
1070  auto index = buckets[hash % hash_table->nBuckets];
1071 
1072  if(!index)
1073  return frg::optional<ObjectSymbol>{};
1074 
1075  while(true) {
1076  // chains[] contains an array of hashes, parallel to the symbol table.
1077  auto chash = chains[index - hash_table->symbolOffset];
1078  if ((chash & ~1) == (hash & ~1)) {
1079  ObjectSymbol cand{object, (Elf64_Sym *)(object->baseAddress
1080  + object->symbolTableOffset + index * sizeof(Elf64_Sym))};
1081  if(eligible(cand) && frg::string_view{cand.getString()} == string)
1082  return cand;
1083  }
1084 
1085  // If we hit the end of the chain, the symbol is not present.
1086  if(chash & 1)
1087  return frg::optional<ObjectSymbol>{};
1088  index++;
1089  }
1090  }
1091 }

1282 void Loader::_buildTlsMaps() {
1283  if(_isInitialLink) {
1284  __ensure(runtimeTlsMap->initialPtr == 0);
1285  __ensure(runtimeTlsMap->initialLimit == 0);
1286 
1287  __ensure(!_linkBfs.empty());

1289 
1290  for(auto object : _linkBfs) {
1291  __ensure(object->tlsModel == TlsModel::null);
1292 
1293  if(object->tlsSegmentSize == 0)
1294  continue;
1295 
1296  // Allocate an index for the object.
1297  object->tlsIndex = runtimeTlsMap->indices.size();
1298  runtimeTlsMap->indices.push_back(object);
1299 
1300  object->tlsModel = TlsModel::initial;
1301 
1302  if constexpr (tlsAboveTp) {
1303  // As per the comment in allocateTcb(), we may simply add sizeof(Tcb) to
1304  // reach the TLS data.
1305  object->tlsOffset = runtimeTlsMap->initialPtr + sizeof(Tcb);
1306  runtimeTlsMap->initialPtr += object->tlsSegmentSize;
1307 
1308  size_t misalign = runtimeTlsMap->initialPtr & (object->tlsAlignment - 1);
1309  if(misalign)
1310  runtimeTlsMap->initialPtr += object->tlsAlignment - misalign;
1311  } else {
1312  runtimeTlsMap->initialPtr += object->tlsSegmentSize;
1313 
1314  size_t misalign = runtimeTlsMap->initialPtr & (object->tlsAlignment - 1);
1315  if(misalign)
1316  runtimeTlsMap->initialPtr += object->tlsAlignment - misalign;
1317 
1318  object->tlsOffset = -runtimeTlsMap->initialPtr;
1319  }
1320 
1321  if(verbose)
1322  mlibc::infoLogger() << "rtdl: TLS of " << object->name
1323  << " mapped to 0x" << frg::hex_fmt{object->tlsOffset}
1324  << ", size: " << object->tlsSegmentSize
1325  << ", alignment: " << object->tlsAlignment << frg::endlog;
1326  }
1327 
1328  // Reserve some additional space for future libraries.
1329  runtimeTlsMap->initialLimit = runtimeTlsMap->initialPtr + 64;
1330  }else{
1331  for(auto object : _linkBfs) {
1332  if(object->tlsModel != TlsModel::null)
1333  continue;
1334  if(object->tlsSegmentSize == 0)
1335  continue;
1336 
1337  // Allocate an index for the object.
1338  object->tlsIndex = runtimeTlsMap->indices.size();
1339  runtimeTlsMap->indices.push_back(object);
1340

1342  // they expect to be dynamically loaded.
1343  if(object->haveStaticTls) {

1346  if(misalign)
1347  ptr += object->tlsAlignment - misalign;

1351  " allocating TLS for " << object->name << frg::endlog;

1353  object->tlsModel = TlsModel::initial;
1354 
1355  if constexpr (tlsAboveTp) {
1356  size_t tcbSize = ((sizeof(Tcb) + tlsMaxAlignment - 1) & ~(tlsMaxAlignment - 1));
1357 
1358  object->tlsOffset = runtimeTlsMap->initialPtr + tcbSize;
1359  runtimeTlsMap->initialPtr = ptr;
1360  } else {
1361  runtimeTlsMap->initialPtr = ptr;
1362  object->tlsOffset = -runtimeTlsMap->initialPtr;
1363  }
1364 
1365  if(verbose)
1366  mlibc::infoLogger() << "rtdl: TLS of " << object->name
1367  << " mapped to 0x" << frg::hex_fmt{object->tlsOffset}
1368  << ", size: " << object->tlsSegmentSize
1369  << ", alignment: " << object->tlsAlignment << frg::endlog;
1370  }else{

1372  }
1373  }
