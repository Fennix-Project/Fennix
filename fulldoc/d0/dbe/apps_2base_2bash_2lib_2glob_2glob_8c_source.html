195 /* Return 1 if all subpatterns in the extended globbing pattern PAT indicate
196  that the name should be skipped. XXX - doesn't handle pattern negation,
197  not sure if it should */
198 static int
199 extglob_skipname (pat, dname, flags)
200  char *pat, *dname;
201  int flags;
202 {
203  char *pp, *pe, *t, *se;
204  int n, r, negate, wild, nullpat, xflags;
205 
206  negate = *pat == '!';
207  wild = *pat == '*' || *pat == '?';
208  pp = pat + 2;
209  se = pp + strlen (pp); /* end of pattern string */
210  pe = glob_patscan (pp, se, 0); /* end of extglob pattern */
211 
212  /* if pe == 0, this is an invalid extglob pattern */
213  if (pe == 0)
214  return 0;
215 
216  xflags = flags | ( negate ? GX_NEGATE : 0);
217 
218  /* if pe != se we have more of the pattern at the end of the extglob
219  pattern. Check the easy case first ( */
220  if (pe == se && *pe == 0 && pe[-1] == ')' && (t = strchr (pp, '|')) == 0)
221  {
222  pe[-1] = '\0';
223  /* This is where we check whether the pattern is being negated and
224  match all files beginning with `.' if the pattern begins with a
225  literal `.'. */
226  r = XSKIPNAME (pp, dname, xflags); /*(*/
227  pe[-1] = ')';
228  return r;
229  }
230 
231  /* Is the extglob pattern between the parens the null pattern? The null
232  pattern can match nothing, so should we check any remaining portion of
233  the pattern? */
234  nullpat = pe >= (pat + 2) && pe[-2] == '(' && pe[-1] == ')';
235 
236  /* check every subpattern */
237  while (t = glob_patscan (pp, pe, '|'))
238  {
239  /* If T == PE and *T == 0 (&& PE[-1] == RPAREN), we have hit the end
240  of a pattern with no trailing characters. */
241  n = t[-1]; /* ( */
242  if (extglob_pattern_p (pp) && n == ')') /* nested extglob? */
243  t[-1] = n; /* no-op for now */
244  else
245  t[-1] = '\0';
246  r = XSKIPNAME (pp, dname, xflags);
247  t[-1] = n;
248  if (r == 0) /* if any pattern says not skip, we don't skip */
249  return r;
250  pp = t;
251  if (pp == pe)
252  break;
253  }
254 
255  /* glob_patscan might find end of string */
256  if (pp == se)
257  return r;
258 
259  /* but if it doesn't then we didn't match a leading dot */
260  if (wild && *pe) /* if we can match zero instances, check further */
261  return (XSKIPNAME (pe, dname, flags));
262 
263  return 1;
264 }

592 /* Recursively scan SDIR for directories matching PAT (PAT is always `**').
593  FLAGS is simply passed down to the recursive call to glob_vector. Returns
594  a list of matching directory names. EP, if non-null, is set to the last
595  element of the returned list. NP, if non-null, is set to the number of
596  directories in the returned list. These two variables exist for the
597  convenience of the caller (always glob_vector). */
598 static struct globval *
599 finddirs (pat, sdir, flags, ep, np)
600  char *pat;
601  char *sdir;
602  int flags;
603  struct globval **ep;
604  int *np;
605 {
606  char **r, *n;
607  int ndirs;
608  struct globval *ret, *e, *g;
609 
610 /*itrace("finddirs: pat = `%s' sdir = `%s' flags = 0x%x", pat, sdir, flags);*/
611  e = ret = 0;
612  r = glob_vector (pat, sdir, flags);
613  if (r == 0 || r[0] == 0)
614  {
615  if (np)
616  *np = 0;
617  if (ep)
618  *ep = 0;
619  if (r && r != &glob_error_return)
620  free (r);
621  return (struct globval *)0;
622  }
623  for (ndirs = 0; r[ndirs] != 0; ndirs++)
624  {
625  g = (struct globval *) malloc (sizeof (struct globval));
626  if (g == 0)
627  {
628  while (ret) /* free list built so far */
629  {
630  g = ret->next;
631  free (ret);
632  ret = g;
633  }
634 
635  free (r);
636  if (np)
637  *np = 0;
638  if (ep)
639  *ep = 0;
640  return (&finddirs_error_return);
641  }
642  if (e == 0)
643  e = g;
644 
645  g->next = ret;
646  ret = g;
647 
648  g->name = r[ndirs];
649  }
650 
651  free (r);
652  if (ep)
653  *ep = e;
654  if (np)
655  *np = ndirs;
656 
657  return ret;
658 }
