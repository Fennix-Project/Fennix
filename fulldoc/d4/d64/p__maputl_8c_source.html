43 fixed_t
44 P_AproxDistance
45 ( fixed_t dx,
46  fixed_t dy )

49  dy = abs(dy);
50  if (dx < dy)
51  return dx+dy-(dx>>1);
52  return dx+dy-(dy>>1);

56 //
57 // P_PointOnLineSide
58 // Returns 0 or 1
59 //
60 int
61 P_PointOnLineSide
62 ( fixed_t x,
63  fixed_t y,
64  line_t* line )
65 {
66  fixed_t dx;
67  fixed_t dy;
68  fixed_t left;
69  fixed_t right;
70 
71  if (!line->dx)
72  {
73  if (x <= line->v1->x)
74  return line->dy > 0;
75 
76  return line->dy < 0;
77  }
78  if (!line->dy)
79  {
80  if (y <= line->v1->y)
81  return line->dx < 0;
82 
83  return line->dx > 0;
84  }
85 
86  dx = (x - line->v1->x);
87  dy = (y - line->v1->y);
88 
89  left = FixedMul ( line->dy>>FRACBITS , dx );
90  right = FixedMul ( dy , line->dx>>FRACBITS );
91 
92  if (right < left)
93  return 0; // front side
94  return 1; // back side
95 }

99 //
100 // P_BoxOnLineSide
101 // Considers the line to be infinite
102 // Returns side 0 or 1, -1 if box crosses the line.
103 //
104 int
105 P_BoxOnLineSide
106 ( fixed_t* tmbox,
107  line_t* ld )
108 {
109  int p1 = 0;
110  int p2 = 0;
111 
112  switch (ld->slopetype)
113  {
114  case ST_HORIZONTAL:
115  p1 = tmbox[BOXTOP] > ld->v1->y;
116  p2 = tmbox[BOXBOTTOM] > ld->v1->y;
117  if (ld->dx < 0)
118  {
119  p1 ^= 1;
120  p2 ^= 1;
121  }
122  break;
123 
124  case ST_VERTICAL:
125  p1 = tmbox[BOXRIGHT] < ld->v1->x;
126  p2 = tmbox[BOXLEFT] < ld->v1->x;
127  if (ld->dy < 0)
128  {
129  p1 ^= 1;
130  p2 ^= 1;
131  }
132  break;
133 
134  case ST_POSITIVE:
135  p1 = P_PointOnLineSide (tmbox[BOXLEFT], tmbox[BOXTOP], ld);
136  p2 = P_PointOnLineSide (tmbox[BOXRIGHT], tmbox[BOXBOTTOM], ld);
137  break;
138 
139  case ST_NEGATIVE:
140  p1 = P_PointOnLineSide (tmbox[BOXRIGHT], tmbox[BOXTOP], ld);
141  p2 = P_PointOnLineSide (tmbox[BOXLEFT], tmbox[BOXBOTTOM], ld);
142  break;
143  }
144 
145  if (p1 == p2)
146  return p1;
147  return -1;
148 }

151 //
152 // P_PointOnDivlineSide
153 // Returns 0 or 1.
154 //
155 int
156 P_PointOnDivlineSide
157 ( fixed_t x,
158  fixed_t y,
159  divline_t* line )
160 {
161  fixed_t dx;
162  fixed_t dy;
163  fixed_t left;
164  fixed_t right;
165 
166  if (!line->dx)
167  {
168  if (x <= line->x)
169  return line->dy > 0;
170 
171  return line->dy < 0;
172  }
173  if (!line->dy)
174  {
175  if (y <= line->y)
176  return line->dx < 0;
177 
178  return line->dx > 0;
179  }
180 
181  dx = (x - line->x);
182  dy = (y - line->y);
183 
184  // try to quickly decide by looking at sign bits
185  if ( (line->dy ^ line->dx ^ dx ^ dy)&0x80000000 )
186  {
187  if ( (line->dy ^ dx) & 0x80000000 )
188  return 1; // (left is negative)
189  return 0;
190  }
191 
192  left = FixedMul ( line->dy>>8, dx>>8 );
193  right = FixedMul ( dy>>8 , line->dx>>8 );
194 
195  if (right < left)
196  return 0; // front side
197  return 1; // back side
198 }

218 //
219 // P_InterceptVector
220 // Returns the fractional intercept point
221 // along the first divline.
222 // This is only called by the addthings
223 // and addlines traversers.
224 //
225 fixed_t
226 P_InterceptVector
227 ( divline_t* v2,
228  divline_t* v1 )
229 {
230 #if 1
231  fixed_t frac;
232  fixed_t num;
233  fixed_t den;
234 
235  den = FixedMul (v1->dy>>8,v2->dx) - FixedMul(v1->dx>>8,v2->dy);
236 
237  if (den == 0)
238  return 0;
239  // I_Error ("P_InterceptVector: parallel");
240 
241  num =
242  FixedMul ( (v1->x - v2->x)>>8 ,v1->dy )
243  +FixedMul ( (v2->y - v1->y)>>8, v1->dx );
244 
245  frac = FixedDiv (num , den);
246 
247  return frac;
248 #else // UNUSED, float debug.
249  float frac;
250  float num;
251  float den;
252  float v1x;
253  float v1y;
254  float v1dx;
255  float v1dy;
256  float v2x;
257  float v2y;
258  float v2dx;
259  float v2dy;
260 
261  v1x = (float)v1->x/FRACUNIT;
262  v1y = (float)v1->y/FRACUNIT;
263  v1dx = (float)v1->dx/FRACUNIT;
264  v1dy = (float)v1->dy/FRACUNIT;
265  v2x = (float)v2->x/FRACUNIT;
266  v2y = (float)v2->y/FRACUNIT;
267  v2dx = (float)v2->dx/FRACUNIT;
268  v2dy = (float)v2->dy/FRACUNIT;
269 
270  den = v1dy*v2dx - v1dx*v2dy;
271 
272  if (den == 0)
273  return 0; // parallel
274 
275  num = (v1x - v2x)*v1dy + (v2y - v1y)*v1dx;
276  frac = num / den;
277 
278  return frac*FRACUNIT;
279 #endif
280 }

296 {

298  sector_t* back;
299 
300  if (linedef->sidenum[1] == -1)
301  {
302  // single sided line
303  openrange = 0;

306 
307  front = linedef->frontsector;

309 
310  if (front->ceilingheight < back->ceilingheight)

312  else

314 
315  if (front->floorheight > back->floorheight)

322  openbottom = back->floorheight;
323  lowfloor = front->floorheight;
324  }
325 
326  openrange = opentop - openbottom;
327 }

386 // Links a thing into both a block and a subsector
387 // based on it's x y.
388 // Sets thing->subsector properly
389 //
390 void
391 P_SetThingPosition (mobj_t* thing)
392 {
393  subsector_t* ss;
394  sector_t* sec;
395  int blockx;
396  int blocky;
397  mobj_t** link;
398 
399 
400  // link into subsector
401  ss = R_PointInSubsector (thing->x,thing->y);
402  thing->subsector = ss;
403 
404  if ( ! (thing->flags & MF_NOSECTOR) )
405  {
406  // invisible things don't go into the sector links
407  sec = ss->sector;
408 
409  thing->sprev = NULL;
410  thing->snext = sec->thinglist;
411 
412  if (sec->thinglist)
413  sec->thinglist->sprev = thing;
414 
415  sec->thinglist = thing;
416  }
417 
418 
419  // link into blockmap
420  if ( ! (thing->flags & MF_NOBLOCKMAP) )
421  {
422  // inert things don't need to be in blockmap
423  blockx = (thing->x - bmaporgx)>>MAPBLOCKSHIFT;
424  blocky = (thing->y - bmaporgy)>>MAPBLOCKSHIFT;

433  thing->bnext = *link;

435  (*link)->bprev = thing;

438  }

442  thing->bnext = thing->bprev = NULL;

468 ( int x,

471 {

485

548 //
549 // PIT_AddLineIntercepts.
550 // Looks for lines in the given block
551 // that intercept the given trace
552 // to add to the intercepts list.
553 //
554 // A line is crossed if its endpoints
555 // are on opposite sides of the trace.
556 // Returns true if earlyout and a solid line hit.
557 //
558 boolean
559 PIT_AddLineIntercepts (line_t* ld)
560 {
561  int s1;
562  int s2;
563  fixed_t frac;
564  divline_t dl;
565 
566  // avoid precision problems with two routines
567  if ( trace.dx > FRACUNIT*16
568  || trace.dy > FRACUNIT*16
569  || trace.dx < -FRACUNIT*16
570  || trace.dy < -FRACUNIT*16)
571  {
572  s1 = P_PointOnDivlineSide (ld->v1->x, ld->v1->y, &trace);
573  s2 = P_PointOnDivlineSide (ld->v2->x, ld->v2->y, &trace);
574  }
575  else
576  {
577  s1 = P_PointOnLineSide (trace.x, trace.y, ld);
578  s2 = P_PointOnLineSide (trace.x+trace.dx, trace.y+trace.dy, ld);
579  }
580 
581  if (s1 == s2)
582  return true; // line isn't crossed
583 
584  // hit the line
585  P_MakeDivline (ld, &dl);
586  frac = P_InterceptVector (&trace, &dl);
587 
588  if (frac < 0)
589  return true; // behind source
590 
591  // try to early out the check
592  if (earlyout
593  && frac < FRACUNIT
594  && !ld->backsector)
595  {
596  return false; // stop checking
597  }
598 
599 
600  intercept_p->frac = frac;
601  intercept_p->isaline = true;
602  intercept_p->d.line = ld;
603  InterceptsOverrun(intercept_p - intercepts, intercept_p);
604  intercept_p++;
605 
606  return true; // continue
607 }

611 //
612 // PIT_AddThingIntercepts
613 //
614 boolean PIT_AddThingIntercepts (mobj_t* thing)
615 {
616  fixed_t x1;
617  fixed_t y1;
618  fixed_t x2;
619  fixed_t y2;
620 
621  int s1;
622  int s2;
623 
624  boolean tracepositive;
625 
626  divline_t dl;
627 
628  fixed_t frac;
629 
630  tracepositive = (trace.dx ^ trace.dy)>0;
631 
632  // check a corner to corner crossection for hit
633  if (tracepositive)
634  {
635  x1 = thing->x - thing->radius;
636  y1 = thing->y + thing->radius;
637 
638  x2 = thing->x + thing->radius;
639  y2 = thing->y - thing->radius;
640  }
641  else
642  {
643  x1 = thing->x - thing->radius;
644  y1 = thing->y - thing->radius;
645 
646  x2 = thing->x + thing->radius;
647  y2 = thing->y + thing->radius;
648  }
649 
650  s1 = P_PointOnDivlineSide (x1, y1, &trace);
651  s2 = P_PointOnDivlineSide (x2, y2, &trace);
652 
653  if (s1 == s2)
654  return true; // line isn't crossed
655 
656  dl.x = x1;
657  dl.y = y1;
658  dl.dx = x2-x1;
659  dl.dy = y2-y1;
660 
661  frac = P_InterceptVector (&trace, &dl);
662 
663  if (frac < 0)
664  return true; // behind source
665 
666  intercept_p->frac = frac;
667  intercept_p->isaline = false;
668  intercept_p->d.thing = thing;
669  InterceptsOverrun(intercept_p - intercepts, intercept_p);
670  intercept_p++;
671 
672  return true; // keep going
673 }

738 typedef struct

745 // Intercepts memory table. This is where various variables are located
746 // in memory in Vanilla Doom. When the intercepts table overflows, we
747 // need to write to them.
748 //
749 // Almost all of the values to overwrite are 32-bit integers, except for
750 // playerstarts, which is effectively an array of 16-bit integers and
751 // must be treated differently.
752 
753 static intercepts_overrun_t intercepts_overrun[] =
754 {
755  {4, NULL, false},
756  {4, NULL, /* &earlyout, */ false},
757  {4, NULL, /* &intercept_p, */ false},
758  {4, &lowfloor, false},
759  {4, &openbottom, false},
760  {4, &opentop, false},
761  {4, &openrange, false},
762  {4, NULL, false},
763  {120, NULL, /* &activeplats, */ false},
764  {8, NULL, false},
765  {4, &bulletslope, false},
766  {4, NULL, /* &swingx, */ false},
767  {4, NULL, /* &swingy, */ false},
768  {4, NULL, false},
769  {40, &playerstarts, true},
770  {4, NULL, /* &blocklinks, */ false},
771  {4, &bmapwidth, false},
772  {4, NULL, /* &blockmap, */ false},
773  {4, &bmaporgx, false},
774  {4, &bmaporgy, false},
775  {4, NULL, /* &blockmaplump, */ false},

780 // Overwrite a specific memory location with a value.
781 
782 static void InterceptsMemoryOverrun(int location, int value)
783 {
784  int i, offset;
785  int index;
786  void *addr;
787 
788  i = 0;
789  offset = 0;
790 
791  // Search down the array until we find the right entry
792 
793  while (intercepts_overrun[i].len != 0)
794  {
795  if (offset + intercepts_overrun[i].len > location)
796  {
797  addr = intercepts_overrun[i].addr;
798 
799  // Write the value to the memory location.
800  // 16-bit and 32-bit values are written differently.
801 
802  if (addr != NULL)
803  {
804  if (intercepts_overrun[i].int16_array)
805  {
806  index = (location - offset) / 2;
807  ((short *) addr)[index] = value & 0xffff;
808  ((short *) addr)[index + 1] = (value >> 16) & 0xffff;
809  }
810  else
811  {
812  index = (location - offset) / 4;
813  ((int *) addr)[index] = value;
814  }
815  }
816 
817  break;
818  }
819 
820  offset += intercepts_overrun[i].len;
821  ++i;
822  }
823 }

825 // Emulate overruns of the intercepts[] array.
826 
827 static void InterceptsOverrun(int num_intercepts, intercept_t *intercept)
828 {
829  int location;
830 
831  if (num_intercepts <= MAXINTERCEPTS_ORIGINAL)
832  {
833  // No overrun
834 
835  return;
836  }
837 
838  location = (num_intercepts - MAXINTERCEPTS_ORIGINAL - 1) * 12;
839 
840  // Overwrite memory that is overwritten in Vanilla Doom, using
841  // the values from the intercept structure.
842  //
843  // Note: the ->d.{thing,line} member should really have its
844  // address translated into the correct address value for

914  yt2 = y2>>MAPBLOCKSHIFT;

923  {
924  mapxstep = -1;
925  partial = (x1>>MAPBTOFRAC)&(FRACUNIT-1);
926  ystep = FixedDiv (y2-y1,abs(x2-x1));
927  }

932  ystep = 256*FRACUNIT;

943  }

949  }

953  partial = FRACUNIT;
954  xstep = 256*FRACUNIT;
955  }

957 
958  // Step through map blocks.
959  // Count is present to prevent a round off error
960  // from skipping the break.
961  mapx = xt1;
962  mapy = yt1;

975  return false; // early out
976  }

979  && mapy == yt2)
980  {
981  break;
982  }

987  mapx += mapxstep;
988  }
989  else if ( (xintercept >> FRACBITS) == mapx)
990  {
991  xintercept += xstep;
992  mapy += mapystep;
993  }
994 
995  }
996  // go through the sorted list
997  return P_TraverseIntercepts ( trav, FRACUNIT );
