50  {

199  void APIC::RawRedirectIRQ(uint16_t Vector, uint32_t GSI, uint16_t Flags, int CPU, int Status)
200  {
201  uint64_t Value = Vector;
202 
203  int64_t IOAPICTarget = -1;
204  for (uint64_t i = 0; ((ACPI::MADT *)PowerManager->GetMADT())->ioapic[i] != 0; i++)
205  if (((ACPI::MADT *)PowerManager->GetMADT())->ioapic[i]->GSIBase <= GSI)
206  if (((ACPI::MADT *)PowerManager->GetMADT())->ioapic[i]->GSIBase + IOGetMaxRedirect(s_cst(uint32_t, i)) > GSI)
207  {
208  IOAPICTarget = i;
209  break;
210  }
211 
212  if (IOAPICTarget == -1)
213  {
214  error("No ISO table found for I/O APIC");
215  return;
216  }
217 
218  // TODO: IOAPICRedirectEntry Entry = {.raw = 0};
219 
220  if (Flags & ActiveHighLow)
221  Value |= (1 << 13);
222 
223  if (Flags & EdgeLevel)
224  Value |= (1 << 15);
225 
226  if (!Status)
227  Value |= (1 << 16);
228 
229  Value |= (((uintptr_t)CPU) << 56);
230  uint32_t IORegister = (GSI - ((ACPI::MADT *)PowerManager->GetMADT())->ioapic[IOAPICTarget]->GSIBase) * 2 + 16;
231 
232  this->IOWrite(((ACPI::MADT *)PowerManager->GetMADT())->ioapic[IOAPICTarget]->Address, IORegister, (uint32_t)Value);
233  this->IOWrite(((ACPI::MADT *)PowerManager->GetMADT())->ioapic[IOAPICTarget]->Address, IORegister + 1, (uint32_t)(Value >> 32));
234  }

241  debug("[ISO %d] Mapping to source IRQ%#d GSI:%#lx on CPU %d",
242  i, ((ACPI::MADT *)PowerManager->GetMADT())->iso[i]->IRQSource, ((ACPI::MADT *)PowerManager->GetMADT())->iso[i]->GSI, CPU);

260  APIC::APIC(int Core)
261  {
262  SmartCriticalSection(APICLock);
263  APIC_BASE BaseStruct = {.raw = rdmsr(MSR_APIC_BASE)};
264  uint64_t BaseLow = BaseStruct.ApicBaseLo;
265  uint64_t BaseHigh = BaseStruct.ApicBaseHi;
266  this->APICBaseAddress = BaseLow << 12u | BaseHigh << 32u;
267  trace("APIC Address: %#lx", this->APICBaseAddress);
268  Memory::Virtual().Map((void *)this->APICBaseAddress, (void *)this->APICBaseAddress, Memory::PTFlag::RW | Memory::PTFlag::PCD);
269 
270  bool x2APICSupported = false;
271  if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_AMD) == 0)
272  {
273  CPU::x86::AMD::CPUID0x00000001 cpuid;
274  cpuid.Get();
275  if (cpuid.ECX.x2APIC)
276  {
277  // x2APICSupported = cpuid.ECX.x2APIC;
278  fixme("x2APIC is supported");
279  }
280  }
281  else if (strcmp(CPU::Vendor(), x86_CPUID_VENDOR_INTEL) == 0)
282  {
283  CPU::x86::Intel::CPUID0x00000001 cpuid;
284  cpuid.Get();
285  if (cpuid.ECX.x2APIC)
286  {
287  // x2APICSupported = cpuid.ECX.x2APIC;
288  fixme("x2APIC is supported");
289  }
290  }
291 
292  if (x2APICSupported)
293  {
294  this->x2APICSupported = true;
295  wrmsr(MSR_APIC_BASE, (rdmsr(MSR_APIC_BASE) | (1 << 11)) & ~(1 << 10));
296  BaseStruct.EN = 1;
297  wrmsr(MSR_APIC_BASE, BaseStruct.raw);
298  }
299  else
300  {
301  BaseStruct.EN = 1;
302  wrmsr(MSR_APIC_BASE, BaseStruct.raw);
303  }
304 
305  this->Write(APIC_TPR, 0x0);
306  // this->Write(APIC_SVR, this->Read(APIC_SVR) | 0x100); // 0x1FF or 0x100 ? on https://wiki.osdev.org/APIC is 0x100
307 
308  if (!this->x2APICSupported)
309  {
310  this->Write(APIC_DFR, 0xF0000000);
311  this->Write(APIC_LDR, this->Read(APIC_ID));
312  }
313 
314  ACPI::MADT *madt = (ACPI::MADT *)PowerManager->GetMADT();
315 
316  for (size_t i = 0; i < madt->nmi.size(); i++)
317  {
318  if (madt->nmi[i]->processor != 0xFF && Core != madt->nmi[i]->processor)
319  return;
320 
321  uint32_t nmi = 0x402;
322  if (madt->nmi[i]->flags & 2)
323  nmi |= 1 << 13;
324  if (madt->nmi[i]->flags & 8)
325  nmi |= 1 << 15;
326  if (madt->nmi[i]->lint == 0)
327  this->Write(APIC_LINT0, nmi);
328  else if (madt->nmi[i]->lint == 1)
329  this->Write(APIC_LINT1, nmi);
330  }
331 
332  // Setup the spurrious interrupt vector
333  Spurious Spurious = {.raw = this->Read(APIC_SVR)};
334  Spurious.Vector = IRQ223; // TODO: Should I map the IRQ to something?
335  Spurious.Software = 1;
336  this->Write(APIC_SVR, s_cst(uint32_t, Spurious.raw));
337 
338  static int once = 0;
339  if (!once++)
340  {
341  // Disable PIT
342  outb(0x43, 0x28);
343  outb(0x40, 0x0);
344 
345  // Disable PIC
346  outb(0x21, 0xFF);
347  outb(0xA1, 0xFF);
348  }
349  }
