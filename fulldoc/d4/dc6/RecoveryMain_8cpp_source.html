53  void PlayAudio()
54  {
55  TaskManager->GetCurrentThread()->Rename("PlayAudio Test");
56  SmartLock(PlayAudioLock);
57  Driver::DriverFile AudioDrv;
58 
59  foreach (auto Driver in DriverManager->GetDrivers())
60  {
61  if (((FexExtended *)Driver.ExtendedHeaderAddress)->Driver.Type == FexDriverType::FexDriverType_Audio)
62  {
63  AudioDrv = Driver;
64  break;
65  }
66  }
67 
68  if (!AudioDrv.Enabled)
69  {
70  error("No audio drivers found! Cannot play audio!");
71  return;
72  }
73 
74  VirtualFileSystem::File pcm = vfs->Open(AudioFile);
75 
76  if (!pcm.IsOK())
77  {
78  error("Cannot open audio file! Cannot play audio!");
79  return;
80  }
81 
82  void *PCMRaw = KernelAllocator.RequestPages(TO_PAGES(pcm.GetLength() + 1));
83  vfs->Read(pcm, (uint8_t *)PCMRaw, pcm.GetLength());
84 
85  KernelCallback callback{};
86  callback.Reason = SendReason;
87  callback.AudioCallback.Send.Data = (uint8_t *)PCMRaw;
88  callback.AudioCallback.Send.Length = pcm.GetLength();
89  debug("Playing audio...");
90  int status = DriverManager->IOCB(AudioDrv.DriverUID, &callback);
91  UNUSED(status);
92  debug("Audio played! %d", status);
93  KernelAllocator.FreePages((void *)PCMRaw, TO_PAGES(pcm.GetLength() + 1));
94  vfs->Close(pcm);
95  TEXIT(0);
96  }
