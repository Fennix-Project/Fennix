209 char *realpath(const char *path, char *out) {
210  if(debugPathResolution)
211  mlibc::infoLogger() << "mlibc realpath(): Called on '" << path << "'" << frg::endlog;
212  frg::string_view path_view{path};
213 
214  // In case of the root, the string only contains the null-terminator.
215  frg::small_vector<char, PATH_MAX, MemoryAllocator> resolv{getAllocator()};
216  size_t ps;
217 
218  // If the path is relative, we have to preprend the working directory.
219  if(path[0] == '/') {
220  resolv.push_back(0);
221  ps = 1;
222  }else{
223  MLIBC_CHECK_OR_ENOSYS(mlibc::sys_getcwd, nullptr);
224 
225  // Try to getcwd() until the buffer is large enough.
226  resolv.resize(128);
227  while(true) {
228  int e = mlibc::sys_getcwd(resolv.data(), resolv.size());
229  if(e == ERANGE) {
230  resolv.resize(2 * resolv.size());
231  }else if(!e) {
232  break;
233  }else{
234  errno = e;
235  return nullptr;
236  }
237  }
238  frg::string_view cwd_view{resolv.data()};
239  if(cwd_view == "/") {
240  // Restore our invariant that we only store the null-terminator for the root.
241  resolv.resize(1);
242  resolv[0] = 0;
243  }else{
244  resolv.resize(cwd_view.size() + 1);
245  }
246  ps = 0;
247  }
248 
249  // Contains unresolved links as a relative path compared to resolv.
250  frg::small_vector<char, PATH_MAX, MemoryAllocator> lnk{getAllocator()};
251  size_t ls = 0;
252 
253  auto process_segment = [&] (frg::string_view s_view) -> int {
254  if(debugPathResolution)
255  mlibc::infoLogger() << "mlibc realpath(): resolv is '" << resolv.data() << "'"
256  << ", segment is " << s_view.data()
257  << ", size: " << s_view.size() << frg::endlog;
258 
259  if(!s_view.size() || s_view == ".") {
260  // Keep resolv invariant.
261  return 0;
262  }else if(s_view == "..") {
263  // Remove a single segment from resolv.
264  if(resolv.size() > 1) {
265  auto slash = strrchr(resolv.data(), '/');
266  __ensure(slash); // We never remove the leading sla.
267  resolv.resize((slash - resolv.data()) + 1);
268  *slash = 0; // Replace the slash by a null-terminator.
269  }
270  return 0;
271  }
272 
273  // Append the segment to resolv.
274  auto rsz = resolv.size();
275  resolv[rsz - 1] = '/'; // Replace null-terminator by a slash.
276  resolv.resize(rsz + s_view.size() + 1);
277  memcpy(resolv.data() + rsz, s_view.data(), s_view.size());
278  resolv[rsz + s_view.size()] = 0;
279 
280  // stat() the path to (1) see if it exists and (2) see if it is a link.
281  if(!mlibc::sys_stat) {
282  MLIBC_MISSING_SYSDEP();
283  return ENOSYS;
284  }
285  if(debugPathResolution)
286  mlibc::infoLogger() << "mlibc realpath(): stat()ing '"
287  << resolv.data() << "'" << frg::endlog;
288  struct stat st;
289  if(int e = mlibc::sys_stat(mlibc::fsfd_target::path,
290  -1, resolv.data(), AT_SYMLINK_NOFOLLOW, &st); e)
291  return e;
292 
293  if(S_ISLNK(st.st_mode)) {
294  if(debugPathResolution) {
295  mlibc::infoLogger() << "mlibc realpath(): Encountered symlink '"
296  << resolv.data() << "'" << frg::endlog;
297  }
298 
299  if(!mlibc::sys_readlink) {

301  return ENOSYS;
302  }
303

305  char path[512];
306 
307  if (int e = mlibc::sys_readlink(resolv.data(), path, 512, &sz); e)
308  return e;
309

312  << frg::string_view{path, static_cast<size_t>(sz)} << "'" << frg::endlog;
313  }
314 
315  if (path[0] == '/') {
316  // Absolute path, replace resolv
317  resolv.resize(sz);
318  strncpy(resolv.data(), path, sz - 1);
319  resolv.data()[sz - 1] = 0;
320 
321  if(debugPathResolution) {
322  mlibc::infoLogger() << "mlibc realpath(): Symlink is absolute, resolv: '"
323  << resolv.data() << "'" << frg::endlog;
324  }
325  } else {

327  resolv.resize(rsz);
328  resolv[rsz - 1] = 0;
329

331  lnk.resize((lsz - ls) + sz);
332  memmove(lnk.data() + sz, lnk.data() + ls, lsz - ls);
333  memcpy(lnk.data(), path, sz);
334 
335  ls = 0;
336 
337  if(debugPathResolution) {
338  mlibc::infoLogger() << "mlibc realpath(): Symlink is relative, resolv: '"
339  << resolv.data() << "' lnk: '"

341  }
342  }
343  }
344 
345  return 0;
346  };
347

349  // This design avoids copying the input path into lnk;
350  // the latter could often involve additional allocations.
351  while(ps < path_view.size()) {
352  frg::string_view ps_view;
353  if(auto slash = strchr(path + ps, '/'); slash) {
354  ps_view = frg::string_view{path + ps, static_cast<size_t>(slash - (path + ps))};
355  }else{

357  }
358  ps += ps_view.size() + 1;
359 
360  // Handle one segment from the input path.
361  if(int e = process_segment(ps_view); e) {
362  errno = e;
363  return nullptr;
364  }
365 
366  // This inner loop consumes segments of lnk.
367  while(ls < lnk.size()) {
368  frg::string_view ls_view;
369  if(auto slash = strchr(lnk.data() + ls, '/'); slash) {
370  ls_view = frg::string_view{lnk.data() + ls, static_cast<size_t>(slash - (lnk.data() + ls))};
371  }else{
372  ls_view = frg::string_view{lnk.data() + ls, strlen(lnk.data()) - ls};
373  }
374  ls += ls_view.size() + 1;
375 
376  // Handle one segment from the link
377  if(int e = process_segment(ls_view); e) {
378  errno = e;
379  return nullptr;
380  }
381  }
382 
383  // All of lnk was consumed, reset it
384  lnk.resize(0);
385  ls = 0;
386  }
387 
388  if(resolv.size() == 1) {
389  resolv.resize(0);
390  resolv.push_back('/');
391  resolv.push_back(0);
392  }
393 
394  if(debugPathResolution)
395  mlibc::infoLogger() << "mlibc realpath(): Returns '" << resolv.data() << "'" << frg::endlog;
396 
397  if(resolv.size() > PATH_MAX) {
398  errno = ENAMETOOLONG;
399  return nullptr;
400  }
401 
402  if(!out)
403  out = reinterpret_cast<char *>(getAllocator().allocate(resolv.size()));
404  strcpy(out, resolv.data());
405  return out;
406 }
