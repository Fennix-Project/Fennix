25 {

28  KPrint("Process %s(%d) has no children", pcb->Name, pcb->ID);
29  return;
30  }
31 
32  std::list<Tasking::PCB *> children = pcb->Children;
33 
34  foreach (auto child in children)
35  {
36  if (child->State.load() == Tasking::Terminated)
37  {
38  KPrint("Process %s(%d) is already dead", child->Name, child->ID);

41
