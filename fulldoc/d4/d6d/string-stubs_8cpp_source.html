478 int strverscmp(const char *l0, const char *r0) {
479  const unsigned char *l = (const unsigned char *)l0;
480  const unsigned char *r = (const unsigned char *)r0;
481  size_t i, dp, j;
482  int z = 1;
483 
484  /* Find maximal matching prefix and track its maximal digit
485  * suffix and whether those digits are all zeros. */
486  for(dp = i = 0; l[i] == r[i]; i++) {
487  int c = l[i];
488  if(!c)
489  return 0;
490  if(!isdigit(c))
491  dp = i + 1, z = 1;
492  else if(c != '0')
493  z = 0;
494  }
495 
496  if(l[dp] != '0' && r[dp] != '0') {
497  /* If we're not looking at a digit sequence that began
498  * with a zero, longest digit string is greater. */
499  for(j = i; isdigit(l[j]); j++) {
500  if(!isdigit(r[j]))
501  return 1;
502  }
503  if(isdigit(r[j]))
504  return -1;
505  } else if(z && dp < i && (isdigit(l[i]) || isdigit(r[i]))) {
506  /* Otherwise, if common prefix of digit sequence is
507  * all zeros, digits order less than non-digits. */
508  return (unsigned char)(l[i] - '0') - (unsigned char)(r[i] - '0');
509  }
510 
511  return l[i] - r[i];
512 }
