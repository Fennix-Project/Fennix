50 mkfifo_builtin (list)
51  WORD_LIST *list;
52 {
53  int opt, mflag, omode, rval, nmode, basemode;
54  char *mode;
55  WORD_LIST *l;
56 
57  mflag = 0;
58  mode = (char *)NULL;
59 
60  reset_internal_getopt ();
61  while ((opt = internal_getopt(list, "m:")) != -1)
62  switch (opt)
63  {
64  case 'm':
65  mflag = 1;
66  mode = list_optarg;
67  break;
68  CASE_HELPOPT;
69  default:
70  builtin_usage();
71  return (EX_USAGE);
72  }
73  list = loptend;
74 
75  if (list == 0)
76  {
77  builtin_usage ();
78  return (EX_USAGE);
79  }
80 
81  basemode = S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH;
82  if (mode == NULL)
83  omode = basemode;
84  else if (ISOCTAL (*mode)) /* octal number */
85  {
86  omode = read_octal (mode);
87  if (omode < 0)
88  {
89  builtin_error ("invalid file mode: %s", mode);
90  return (EXECUTION_FAILURE);
91  }
92  }
93  else /* symbolic mode */
94  {
95  /* initial bits are a=rwx; the mode argument modifies them */
96  omode = parse_symbolic_mode (mode, basemode);
97  if (omode < 0)
98  {
99  builtin_error ("invalid file mode: %s", mode);
100  return (EXECUTION_FAILURE);
101  }
102  }
103 
104  /* Make the new mode */
105  original_umask = umask (0);
106  umask (original_umask);
107 
108  nmode = basemode & ~original_umask;
109  /* Adjust new mode based on mode argument */
110  nmode &= omode;
111 
112  for (rval = EXECUTION_SUCCESS, l = list; l; l = l->next)
113  {
114  if (mkfifo (l->word->word, nmode) < 0)
115  {
116  builtin_error ("cannot create FIFO `%s': %s", l->word->word, strerror (errno));
117  rval = EXECUTION_FAILURE;
118  }
119  }
120  return rval;
121 }
