71 static char *from_mask(const ul *mask, unsigned sz_in_bytes)
72 {
73  char *str = xzalloc((sz_in_bytes+1) * 2); /* we will leak it */
74  char *p = str;
75  for (;;) {

77  if (SZOF_UL == 4)

84  if ((int)sz_in_bytes <= 0)

86  }

89  return str;
90 }

99 static unsigned long *get_aff(int pid, unsigned *sz)
100 {
101  int r;
102  unsigned long *mask = NULL;
103  unsigned sz_in_bytes = *sz;
104 
105  for (;;) {
106  mask = xrealloc(mask, sz_in_bytes);
107  r = sched_getaffinity(pid, sz_in_bytes, (void*)mask);
108  if (r == 0)
109  break;
110  sz_in_bytes *= 2;
111  if (errno == EINVAL && (int)sz_in_bytes > 0)
112  continue;
113  bb_perror_msg_and_die("can't %cet pid %d's affinity", 'g', pid);
114  }
115  //bb_error_msg("get mask[0]:%lx sz_in_bytes:%d", mask[0], sz_in_bytes);
116  *sz = sz_in_bytes;

121 /*
122  * Parse the CPU list and set the mask accordingly.
123  *
124  * The list element can be either a CPU index or a range of CPU indices.
125  * Example: "1,3,5-7". Stride can be specified: "0-7:2" is "0,2,4,6".
126  * Note: leading and trailing whitespace is not allowed.
127  * util-linux 2.31 allows leading and sometimes trailing whitespace:
128  * ok: taskset -c ' 1, 2'
129  * ok: taskset -c ' 1 , 2'
130  * ok: taskset -c ' 1-7: 2 ,8'
131  * not ok: taskset -c ' 1 '
132  * not ok: taskset -c ' 1-7: 2 '
133  */
134 static void parse_cpulist(ul *mask, unsigned max, char *s)
135 {
136  char *aff = s;
137  for (;;) {
138  unsigned bit, end;
139  unsigned stride = 1;
140 
141  bit = end = bb_strtou(s, &s, 10);
142  if (*s == '-') {
143  s++;
144  end = bb_strtou(s, &s, 10);
145  if (*s == ':') {
146  s++;
147  stride = bb_strtou(s, &s, 10);
148  }
149  }
150  if ((*s != ',' && *s != '\0')
151  || bit > end
152  || end == UINT_MAX /* bb_strtou returns this on malformed / ERANGE numbers */
153  || (stride - 1) > (UINT_MAX / 4)
154  /* disallow 0, malformed input, and too large stride prone to overflows */
155  ) {
156  bb_error_msg_and_die("bad affinity '%s'", aff);
157  }
158  while (bit <= end && bit < max) {
159  mask[bit / BITS_UL] |= (1UL << (bit & MASK_UL));
160  bit += stride;
161  }
162  if (*s == '\0')
163  break;
164  s++;
165  }
166 }
167 static void print_cpulist(const ul *mask, unsigned mask_size_in_bytes)
168 {
169  const ul *mask_end;
170  const char *delim;
171  unsigned pos;
172  ul bit;
173 
174  mask_end = mask + mask_size_in_bytes / sizeof(mask[0]);
175  delim = "";
176  pos = 0;
177  bit = 1;
178  for (;;) {
179  if (*mask & bit) {
180  unsigned onebit = pos + 1;
181  printf("%s%u", delim, pos);
182  do {
183  pos++;
184  bit <<= 1;
185  if (bit == 0) {
186  mask++;
187  if (mask >= mask_end)
188  break;
189  bit = 1;
190  }
191  } while (*mask & bit);
192  if (onebit != pos)
193  printf("-%u", pos - 1);
194  delim = ",";
195  }
196  pos++;
197  bit <<= 1;
198  if (bit == 0) {
199  mask++;
200  if (mask >= mask_end)
201  break;
202  bit = 1;
203  }
204  }
205  bb_putchar('\n');
206 }

215 static int process_pid_str(const char *pid_str, unsigned opts, char *aff)
216 {
217  ul *mask;
218  unsigned mask_size_in_bytes;
219  const char *current_new;
220  pid_t pid = !pid_str ? 0 : xatou_range(pid_str, 1, INT_MAX); /* disallow "0": "taskset -p 0" should fail */
221 
222  mask_size_in_bytes = SZOF_UL;
223  current_new = "current";
224  print_aff:
225  mask = get_aff(pid, &mask_size_in_bytes);
226  if (opts & OPT_p) {
227 #if ENABLE_FEATURE_TASKSET_CPULIST
228  if (opts & OPT_c) {
229  printf("pid %d's %s affinity list: ", pid, current_new);
230  print_cpulist(mask, mask_size_in_bytes);
231  } else
232 #endif
233  printf("pid %d's %s affinity mask: "TASKSET_PRINTF_MASK"\n",
234  pid, current_new, from_mask(mask, mask_size_in_bytes));
235  if (!aff) {
236  /* Either it was just "-p <pid>",
237  * or it was "-p <aff> <pid>" and we came here
238  * for the second time (see goto below) */
239  return 0;
240  }
241  current_new = "new";
242  }
243  memset(mask, 0, mask_size_in_bytes);
244 
245  if (!ENABLE_FEATURE_TASKSET_FANCY) {
246  /* Affinity was specified, translate it into mask */
247  /* it is always in hex, skip "0x" if it exists */
248  if (aff[0] == '0' && (aff[1]|0x20) == 'x')
249  aff += 2;
250  mask[0] = xstrtoul(aff, 16);
251  }
252 #if ENABLE_FEATURE_TASKSET_CPULIST
253  else if (opts & OPT_c) {
254  parse_cpulist(mask, mask_size_in_bytes * 8, aff);
255  }
256 #endif
257  else {
258  unsigned i;
259  char *last_char;
260 
261  /* Affinity was specified, translate it into mask */
262  /* it is always in hex, skip "0x" if it exists */
263  if (aff[0] == '0' && (aff[1]|0x20) == 'x')
264  aff += 2;
265 
266  i = 0; /* bit pos in mask[] */
267 
268  /* aff is ASCII hex string, accept very long masks in this form.
269  * Process hex string AABBCCDD... to ulong mask[]
270  * from the rightmost nibble, which is least-significant.
271  * Bits not fitting into mask[] are ignored: (example: 1234
272  * in 12340000000000000000000000000000000000000ff)
273  */
274  last_char = strchrnul(aff, '\0');
275  while (last_char > aff) {
276  char c;
277  ul val;
278 
279  last_char--;
280  c = *last_char;
281  if (isdigit(c))
282  val = c - '0';
283  else if ((c|0x20) >= 'a' && (c|0x20) <= 'f')
284  val = (c|0x20) - ('a' - 10);
285  else
286  bb_error_msg_and_die("bad affinity '%s'", aff);
287 
288  if (i < mask_size_in_bytes * 8) {
289  mask[i / BITS_UL] |= val << (i & MASK_UL);
290  //bb_error_msg("bit %d set", i);
291  }
292  /* else:
293  * We can error out here, but we don't.
294  * For one, kernel itself ignores bits in mask[]
295  * which do not map to any CPUs:
296  * if mask[] has one 32-bit long element,
297  * but you have only 8 CPUs, all bits beyond first 8
298  * are ignored, silently.
299  * No point in making bits past 31th to be errors.
300  */
301  i += 4;
302  }
303  }
304 
305  /* Set pid's or our own (pid==0) affinity */
306  if (sched_setaffinity(pid, mask_size_in_bytes, (void*)mask))
307  bb_perror_msg_and_die("can't %cet pid %d's affinity", 's', pid);
308  //bb_error_msg("set mask[0]:%lx", mask[0]);
309 
310  if ((opts & OPT_p) && aff) { /* "-p <aff> <pid> [...ignored...]" */
311  aff = NULL;
312  goto print_aff; /* print new affinity and exit */
313  }
314  return 0;
315 }
