121 static void InitSfxModule(boolean use_sfx_prefix)
122 {
123  int i;
124 
125  sound_module = NULL;
126 
127  for (i=0; sound_modules[i] != NULL; ++i)
128  {
129  // Is the sfx device in the list of devices supported by
130  // this module?
131 
132  if (SndDeviceInList(snd_sfxdevice,
133  sound_modules[i]->sound_devices,
134  sound_modules[i]->num_sound_devices))
135  {
136  // Initialize the module
137 
138  if (sound_modules[i]->Init(use_sfx_prefix))
139  {
140  sound_module = sound_modules[i];
141  return;
142  }
143  }
144  }
145 }

162 void I_InitSound(boolean use_sfx_prefix)
163 {
164  boolean nosound, nosfx, nomusic;
165 
167  // @vanilla
168  //
169  // Disable all sound output.
170  //
171 
172  nosound = M_CheckParm("-nosound") > 0;
173 
175  // @vanilla
176  //
177  // Disable sound effects.
178  //
179 
180  nosfx = M_CheckParm("-nosfx") > 0;
181 
183  // @vanilla
184  //
185  // Disable music.
186  //
187 
188  nomusic = M_CheckParm("-nomusic") > 0;
189 
190  // Initialize the sound and music subsystems.
191 
192  if (!nosound && !screensaver_mode)
193  {
194  // This is kind of a hack. If native MIDI is enabled, set up
195  // the TIMIDITY_CFG environment variable here before SDL_mixer
196  // is opened.
197 
198  if (!nomusic
199  && (snd_musicdevice == SNDDEVICE_GENMIDI
200  || snd_musicdevice == SNDDEVICE_GUS))
201  {
202  I_InitTimidityConfig();
203  }
204 
205  if (!nosfx)
206  {
207  InitSfxModule(use_sfx_prefix);
208  }
209 
210  if (!nomusic)
211  {
212  InitMusicModule();
213  }
214  }
215 
216 }

256 static void CheckVolumeSeparation(int *vol, int *sep)
257 {
258  if (*sep < 0)
259  {
260  *sep = 0;
261  }
262  else if (*sep > 254)
263  {
264  *sep = 254;

267  if (*vol < 0)

274  }
