476 extern "C" [[ gnu::visibility("default") ]]
477 void *__dlapi_resolve(void *handle, const char *string, void *returnAddress) {
478  if (logDlCalls) {
479  const char *name;
480  bool quote = false;
481  if (handle == RTLD_DEFAULT) {
482  name = "RTLD_DEFAULT";
483  } else if (handle == RTLD_NEXT) {
484  name = "RTLD_NEXT";
485  } else {
486  name = ((SharedObject *)handle)->name.data();
487  quote = true;
488  }
489 
490  mlibc::infoLogger() << "rtdl: __dlapi_resolve(" << (quote ? "\"" : "") << name
491  << (quote ? "\"" : "") << ", \"" << string << "\")" << frg::endlog;
492  }
493 
494  frg::optional<ObjectSymbol> target;
495 
496  if (handle == RTLD_DEFAULT) {
497  target = globalScope->resolveSymbol(string, 0, 0);
498  } else if (handle == RTLD_NEXT) {
499  SharedObject *origin = initialRepository->findCaller(returnAddress);
500  if (!origin) {
501  mlibc::panicLogger() << "rtdl: unable to determine calling object of dlsym "
502  << "(ra = " << returnAddress << ")" << frg::endlog;
503  }
504 
505  target = Scope::resolveGlobalOrLocalNext(*globalScope, origin->localScope, string, origin);
506  } else {
507  // POSIX does not unambiguously state how dlsym() is supposed to work; it just
508  // states that "The symbol resolution algorithm used shall be dependency order
509  // as described in dlopen()".
510  //
511  // Linux libc's lookup the symbol in the given DSO and all of its dependencies
512  // in breadth-first order. That is also what we implement here.
513  //
514  // Note that this *differs* from the algorithm that is used for relocations
515  // (since the algorithm used for relocations takes (i) the global scope,
516  // and (ii) the local scope of the DSO into account (which can contain more objects
517  // than just the dependencies of the DSO, if the DSO was loaded as a dependency
518  // of a dlopen()ed DSO).
519 
520  frg::vector<SharedObject *, MemoryAllocator> queue{getAllocator()};
521 
522  struct Token { };
523  frg::hash_map<
524  SharedObject *, Token,
525  frg::hash<SharedObject *>, MemoryAllocator
526  > visited{frg::hash<SharedObject *>{}, getAllocator()};
527 
528  auto root = reinterpret_cast<SharedObject *>(handle);
529  visited.insert(root, Token{});
530  queue.push_back(root);
531 
532  for(size_t i = 0; i < queue.size(); i++) {
533  auto current = queue[i];
534 
535  target = resolveInObject(current, string);
536  if(target)
537  break;
538 
539  for(auto dep : current->dependencies) {
540  if(visited.get(dep))
541  continue;
542  visited.insert(dep, Token{});
543  queue.push_back(dep);
544  }
545  }
546  }
547 
548  if (!target) {
549  if (logDlCalls)
550  mlibc::infoLogger() << "rtdl: could not resolve \"" << string << "\"" << frg::endlog;
551 
552  lastError = "Cannot resolve requested symbol";
553  return nullptr;
554  }
555  return reinterpret_cast<void *>(target->virtualAddress());
556 }
