131  bool Custom::RemoveProcess(PCB *Process)
132  {
133  if (Process->State == Terminated)
134  {
135  delete Process;
136  return true;
137  }
138 
139  foreach (TCB *Thread in Process->Threads)
140  {
141  if (Thread->State == Terminated)
142  RemoveThread(Thread);
143  }
144 
145  return true;
146  }

171  void Custom::StartIdleProcess()
172  {
173  IdleProcess = ctx->CreateProcess(nullptr, (char *)"Idle",
174  TaskExecutionMode::Kernel, true);
175  for (int i = 0; i < SMP::CPUCores; i++)
176  {
177  TCB *thd = ctx->CreateThread(IdleProcess, IP(__custom_sched_idle_loop));
178  char IdleName[16];
179  sprintf(IdleName, "Idle Thread %d", i);
180  thd->Rename(IdleName);
181  thd->SetPriority(Idle);
182  for (int j = 0; j < MAX_CPU; j++)
183  thd->Info.Affinity[j] = false;
184  thd->Info.Affinity[i] = true;
185 
186  if (unlikely(i == 0))
187  IdleThread = thd;
188  }
189  }

196  void Custom::StartScheduler()
197  {
198 #if defined(a86)
199  if (Interrupts::apicTimer[0])
200  {
201  ((APIC::Timer *)Interrupts::apicTimer[0])->OneShot(CPU::x86::IRQ16, 100);
202 
203  /* FIXME: The kernel is not ready for multi-core tasking. */
204  return;
205 
206  APIC::InterruptCommandRegister icr{};
207  bool x2APIC = ((APIC::APIC *)Interrupts::apic[0])->x2APIC;
208 
209  if (likely(x2APIC))
210  {
211  icr.x2.VEC = s_cst(uint8_t, CPU::x86::IRQ16);
212  icr.x2.MT = APIC::Fixed;
213  icr.x2.L = APIC::Assert;
214  icr.x2.DES = 0xFFFFFFFF; /* Broadcast IPI to all local APICs. */
215  ((APIC::APIC *)Interrupts::apic[0])->ICR(icr);
216  }
217  else
218  {
219  icr.VEC = s_cst(uint8_t, CPU::x86::IRQ16);
220  icr.MT = APIC::Fixed;
221  icr.L = APIC::Assert;
222 
223  for (int i = 0; i < SMP::CPUCores; i++)
224  {
225  icr.DES = uint8_t(i);
226  ((APIC::APIC *)Interrupts::apic[i])->ICR(icr);
227  }
228  }
229  }
230 #endif
231  }

288  {

300  nsa NIF bool Custom::FindNewProcess(void *CPUDataPointer)
301  {
302  CPUData *CurrentCPU = (CPUData *)CPUDataPointer;
303  fnp_schedbg("%d processes", ProcessList.size());
304 #ifdef DEBUG_FIND_NEW_PROCESS
305  foreach (auto process in ProcessList)
306  fnp_schedbg("Process %d %s", process->ID,
307  process->Name);
308 #endif
309  foreach (auto process in ProcessList)
310  {
311  switch (process->State.load())
312  {
313  case TaskState::Ready:
314  fnp_schedbg("Ready process (%s)%d",
315  process->Name, process->ID);
316  break;
317  default:
318  fnp_schedbg("Process \"%s\"(%d) status %d",
319  process->Name, process->ID,
320  process->State);
321 
322  /* We don't actually remove the process. RemoveProcess
323  firstly checks if it's terminated, if not, it will
324  loop through Threads and call RemoveThread on
325  terminated threads. */
326  RemoveProcess(process);
327  continue;
328  }
329 
330  foreach (auto thread in process->Threads)
331  {
332  if (thread->State.load() != TaskState::Ready)
333  continue;
334 
335  if (thread->Info.Affinity[CurrentCPU->ID] == false)
336  continue;
337 
338  CurrentCPU->CurrentProcess = process;
339  CurrentCPU->CurrentThread = thread;
340  return true;
341  }
342  }
343  fnp_schedbg("No process to run.");
344  return false;
345  }

347  nsa NIF bool Custom::GetNextAvailableThread(void *CPUDataPointer)
348  {
349  CPUData *CurrentCPU = (CPUData *)CPUDataPointer;
350 
351  size_t ThreadsSize = CurrentCPU->CurrentProcess->Threads.size();
352 
353  for (size_t i = 0; i < ThreadsSize; i++)
354  {
355  if (CurrentCPU->CurrentProcess->Threads[i] == CurrentCPU->CurrentThread.load())
356  {
357  size_t TempIndex = i;
358  RetryAnotherThread:
359  if (TempIndex + 1 >= ThreadsSize)
360  break;
361 
362  TCB *nextThread = CurrentCPU->CurrentProcess->Threads[TempIndex + 1];
363 
364  gnat_schedbg("\"%s\"(%d) and next thread is \"%s\"(%d)",
365  CurrentCPU->CurrentProcess->Threads[i]->Name,
366  CurrentCPU->CurrentProcess->Threads[i]->ID,
367  nextThread->Name, nextThread->ID);
368 
369  if (nextThread->State.load() != TaskState::Ready)
370  {
371  gnat_schedbg("Thread %d is not ready", nextThread->ID);
372  TempIndex++;
373  goto RetryAnotherThread;
374  }
375 
376  if (nextThread->Info.Affinity[CurrentCPU->ID] == false)
377  continue;
378 
379  CurrentCPU->CurrentThread = nextThread;
380  gnat_schedbg("[thd 0 -> end] Scheduling thread %d parent of %s->%d Procs %d",
381  nextThread->ID, nextThread->Parent->Name,
382  ThreadsSize, ProcessList.size());
383  return true;
384  }
385 #ifdef DEBUG
386  else
387  {
388  gnat_schedbg("Thread %d is not the current one",
389  CurrentCPU->CurrentProcess->Threads[i]->ID);
390  }
391 #endif
392  }
393  return false;
394  }

396  nsa NIF bool Custom::GetNextAvailableProcess(void *CPUDataPointer)
397  {
398  CPUData *CurrentCPU = (CPUData *)CPUDataPointer;
399 
400  bool Skip = true;
401  foreach (auto process in ProcessList)
402  {
403  if (process == CurrentCPU->CurrentProcess.load())
404  {
405  Skip = false;
406  gnap_schedbg("Found current process %#lx", process);
407  continue;
408  }
409 
410  if (Skip)
411  {
412  gnap_schedbg("Skipping process %#lx", process);
413  continue;
414  }
415 
416  if (process->State.load() != TaskState::Ready)
417  {
418  gnap_schedbg("Process %d is not ready", process->ID);
419  continue;
420  }
421 
422  foreach (auto thread in process->Threads)
423  {
424  if (thread->State.load() != TaskState::Ready)
425  {
426  gnap_schedbg("Thread %d is not ready", thread->ID);
427  continue;
428  }
429 
430  if (thread->Info.Affinity[CurrentCPU->ID] == false)
431  continue;
432 
433  CurrentCPU->CurrentProcess = process;
434  CurrentCPU->CurrentThread = thread;
435  gnap_schedbg("[cur proc+1 -> first thd] Scheduling thread %d %s->%d (Total Procs %d)",
436  thread->ID, thread->Name, process->Threads.size(), ProcessList.size());
437  return true;
438  }
439  }
440  gnap_schedbg("No process to run.");
441  return false;
442  }

444  nsa NIF bool Custom::SchedulerSearchProcessThread(void *CPUDataPointer)
445  {
446  CPUData *CurrentCPU = (CPUData *)CPUDataPointer;
447 
448  foreach (auto process in ProcessList)
449  {
450  if (process->State.load() != TaskState::Ready)
451  {
452  sspt_schedbg("Process %d is not ready", process->ID);
453  continue;
454  }
455 
456  foreach (auto thread in process->Threads)
457  {
458  if (thread->State.load() != TaskState::Ready)
459  {
460  sspt_schedbg("Thread %d is not ready", thread->ID);
461  continue;
462  }
463 
464  if (thread->Info.Affinity[CurrentCPU->ID] == false)
465  continue;
466 
467  CurrentCPU->CurrentProcess = process;
468  CurrentCPU->CurrentThread = thread;
469  sspt_schedbg("[proc 0 -> end -> first thd] Scheduling thread %d parent of %s->%d (Procs %d)",
470  thread->ID, thread->Parent->Name, process->Threads.size(), ProcessList.size());
471  return true;
472  }
473  }
474  return false;

477  nsa NIF void Custom::UpdateProcessState()
478  {
479  foreach (auto process in ProcessList)
480  {
481  if (process->State.load() == TaskState::Terminated)
482  continue;
483 
484  if (process->Threads.size() == 1)
485  {
486  process->State.exchange(process->Threads.front()->State.load());
487  continue;
488  }
489 
490  bool AllThreadsSleeping = true;
491  foreach (auto thread in process->Threads)
492  {
493  if (thread->State.load() == TaskState::Terminated)
494  continue;
495 
496  if (thread->State.load() != TaskState::Sleeping)
497  {
498  AllThreadsSleeping = false;
499  break;
500  }
501  }
502 
503  if (AllThreadsSleeping)
504  process->State.store(TaskState::Sleeping);
505  else if (process->State.load() == TaskState::Sleeping)
506  process->State.store(TaskState::Ready);
507  }
508  }

510  nsa NIF void Custom::WakeUpThreads()
511  {
512  foreach (auto process in ProcessList)
513  {
514  Tasking::TaskState pState = process->State.load();
515  if (pState != TaskState::Ready &&
516  pState != TaskState::Sleeping &&
517  pState != TaskState::Blocked)
518  continue;
519 
520  foreach (auto thread in process->Threads)
521  {
522  if (likely(thread->State.load() != TaskState::Sleeping))
523  continue;
524 
525  /* Check if the thread is ready to wake up. */
526  if (unlikely(thread->Info.SleepUntil < TimeManager->GetCounter()))
527  {
528  if (pState == TaskState::Sleeping)
529  process->State.store(TaskState::Ready);
530  thread->State.store(TaskState::Ready);
531 
532  thread->Info.SleepUntil = 0;
533  wut_schedbg("Thread \"%s\"(%d) woke up.", thread->Name, thread->ID);
534  }
535  else
536  {
537  wut_schedbg("Thread \"%s\"(%d) is not ready to wake up. (SleepUntil: %d, Counter: %d)",
538  thread->Name, thread->ID, thread->Info.SleepUntil, TimeManager->GetCounter());
539  }
540  }
541  }
542  }

563  nsa NIF void Custom::Schedule(CPU::SchedulerFrame *Frame)
564  {
565  if (unlikely(StopScheduler))
566  {
567  warn("Scheduler stopped.");
568  return;
569  }
570  bool ProcessNotChanged = false;
571  uint64_t SchedTmpTicks = TimeManager->GetCounter();
572  this->LastTaskTicks.store(size_t(SchedTmpTicks - this->SchedulerTicks.load()));
573  CPUData *CurrentCPU = GetCurrentCPU();
574  this->LastCore.store(CurrentCPU->ID);
575  schedbg("Scheduler called on CPU %d.", CurrentCPU->ID);
576 
577  if (unlikely(!CurrentCPU->CurrentProcess.load() ||
578  !CurrentCPU->CurrentThread.load()))
579  {
580  schedbg("Invalid process or thread. Finding a new one.");
581  ProcessNotChanged = true;
582  if (this->FindNewProcess(CurrentCPU))
583  goto Success;
584  else
585  goto Idle;
586  }
587  else
588  {
589  CurrentCPU->CurrentThread->Registers = *Frame;
590  CPU::x64::fxsave(&CurrentCPU->CurrentThread->FPU);
591 #ifdef a64
592  CurrentCPU->CurrentThread->ShadowGSBase = CPU::x64::rdmsr(CPU::x64::MSR_SHADOW_GS_BASE);
593  CurrentCPU->CurrentThread->GSBase = CPU::x64::rdmsr(CPU::x64::MSR_GS_BASE);
594  CurrentCPU->CurrentThread->FSBase = CPU::x64::rdmsr(CPU::x64::MSR_FS_BASE);
595 #else
596  CurrentCPU->CurrentThread->ShadowGSBase = uintptr_t(CPU::x32::rdmsr(CPU::x32::MSR_SHADOW_GS_BASE));
597  CurrentCPU->CurrentThread->GSBase = uintptr_t(CPU::x32::rdmsr(CPU::x32::MSR_GS_BASE));
598  CurrentCPU->CurrentThread->FSBase = uintptr_t(CPU::x32::rdmsr(CPU::x32::MSR_FS_BASE));
599 #endif
600 
601  if (CurrentCPU->CurrentProcess->State.load() == TaskState::Running)
602  CurrentCPU->CurrentProcess->State.store(TaskState::Ready);
603  if (CurrentCPU->CurrentThread->State.load() == TaskState::Running)
604  CurrentCPU->CurrentThread->State.store(TaskState::Ready);
605 
606  this->CleanupTerminated();
607  schedbg("Passed CleanupTerminated");
608 
609  this->UpdateProcessState();
610  schedbg("Passed UpdateProcessState");
611 
612  this->WakeUpThreads();
613  schedbg("Passed WakeUpThreads");
614 
615  if (this->SchedulerUpdateTrapFrame)
616  {
617  debug("Updating trap frame");
618  this->SchedulerUpdateTrapFrame = false;
619  CurrentCPU->CurrentProcess->State.store(TaskState::Running);
620  CurrentCPU->CurrentThread->State.store(TaskState::Running);
621  *Frame = CurrentCPU->CurrentThread->Registers;
622  this->SchedulerTicks.store(size_t(TimeManager->GetCounter() - SchedTmpTicks));
623  return;
624  }
625 
626  if (this->GetNextAvailableThread(CurrentCPU))
627  {
628  ProcessNotChanged = true;
629  goto Success;
630  }
631  schedbg("Passed GetNextAvailableThread");
632 
633  if (this->GetNextAvailableProcess(CurrentCPU))
634  {
635  goto Success;
636  }
637  schedbg("Passed GetNextAvailableProcess");
638 
639  if (SchedulerSearchProcessThread(CurrentCPU))
640  {
641  schedbg("Passed SchedulerSearchProcessThread");
642  goto Success;
643  }
644  else
645  {
646  schedbg("SchedulerSearchProcessThread failed. Going idle.");
647  goto Idle;
648  }
649  }
650 
651  assert(!"Unwanted code execution");
652 
653  Idle:
654  ProcessNotChanged = true;
655  CurrentCPU->CurrentProcess = IdleProcess;
656  CurrentCPU->CurrentThread = IdleThread;
657 
658  Success:
659  schedbg("Process \"%s\"(%d) Thread \"%s\"(%d) is now running on CPU %d",
660  CurrentCPU->CurrentProcess->Name, CurrentCPU->CurrentProcess->ID,
661  CurrentCPU->CurrentThread->Name, CurrentCPU->CurrentThread->ID, CurrentCPU->ID);
662 
663  if (!ProcessNotChanged)
664  UpdateUsage(&CurrentCPU->CurrentProcess->Info,
665  CurrentCPU->CurrentProcess->Security.ExecutionMode,
666  CurrentCPU->ID);
667 
668  UpdateUsage(&CurrentCPU->CurrentThread->Info,
669  CurrentCPU->CurrentThread->Security.ExecutionMode,
670  CurrentCPU->ID);
671 
672  CurrentCPU->CurrentProcess->State.store(TaskState::Running);
673  CurrentCPU->CurrentThread->State.store(TaskState::Running);
674 
675  if (CurrentCPU->CurrentThread->Registers.cs != GDT_KERNEL_CODE)
676  CurrentCPU->CurrentThread->Registers.ppt = (uint64_t)(void *)CurrentCPU->CurrentProcess->PageTable;
677  else
678  CurrentCPU->CurrentThread->Registers.ppt = (uint64_t)(void *)KernelPageTable;
679 
680  // if (!SchedulerUpdateTrapFrame) {} // TODO
681 
682  *Frame = CurrentCPU->CurrentThread->Registers;
683 
684 #ifdef a64
685  GlobalDescriptorTable::SetKernelStack((void *)((uintptr_t)CurrentCPU->CurrentThread->Stack->GetStackTop()));
686  CPU::x64::fxrstor(&CurrentCPU->CurrentThread->FPU);
687  CPU::x64::wrmsr(CPU::x64::MSR_SHADOW_GS_BASE, CurrentCPU->CurrentThread->ShadowGSBase);
688  CPU::x64::wrmsr(CPU::x64::MSR_GS_BASE, CurrentCPU->CurrentThread->GSBase);
689  CPU::x64::wrmsr(CPU::x64::MSR_FS_BASE, CurrentCPU->CurrentThread->FSBase);
690 #else
691  GlobalDescriptorTable::SetKernelStack((void *)((uintptr_t)CurrentCPU->CurrentThread->Stack->GetStackTop()));
692  CPU::x32::fxrstor(&CurrentCPU->CurrentThread->FPU);
693  CPU::x32::wrmsr(CPU::x32::MSR_SHADOW_GS_BASE, CurrentCPU->CurrentThread->ShadowGSBase);
694  CPU::x32::wrmsr(CPU::x32::MSR_GS_BASE, CurrentCPU->CurrentThread->GSBase);
695  CPU::x32::wrmsr(CPU::x32::MSR_FS_BASE, CurrentCPU->CurrentThread->FSBase);
696 #endif
697 
698  CurrentCPU->CurrentProcess->Signals.HandleSignal(Frame, CurrentCPU->CurrentThread.load());
699 
700  if (!ProcessNotChanged)
701  (&CurrentCPU->CurrentProcess->Info)->LastUpdateTime = TimeManager->GetCounter();
702  (&CurrentCPU->CurrentThread->Info)->LastUpdateTime = TimeManager->GetCounter();
703  this->OneShot(CurrentCPU->CurrentThread->Info.Priority);
704 
705  if (CurrentCPU->CurrentThread->Security.IsDebugEnabled &&
706  CurrentCPU->CurrentThread->Security.IsKernelDebugEnabled)
707  {
708 #ifdef a64
709  trace("%s[%ld]: RIP=%#lx RBP=%#lx RSP=%#lx",
710  CurrentCPU->CurrentThread->Name, CurrentCPU->CurrentThread->ID,
711  CurrentCPU->CurrentThread->Registers.rip,
712  CurrentCPU->CurrentThread->Registers.rbp,
713  CurrentCPU->CurrentThread->Registers.rsp);
714 #else
715  trace("%s[%ld]: EIP=%#lx EBP=%#lx ESP=%#lx",
716  CurrentCPU->CurrentThread->Name, CurrentCPU->CurrentThread->ID,
717  CurrentCPU->CurrentThread->Registers.eip,
718  CurrentCPU->CurrentThread->Registers.ebp,
