533  void sigcancel_handler(int signal, siginfo_t *info, void *ucontext) {
534  ucontext_t *uctx = static_cast<ucontext_t*>(ucontext);
535  // The function could be called from other signals, or from another
536  // process, in which case we should do nothing.
537  if (signal != SIGCANCEL || info->si_pid != getpid() ||
538  info->si_code != SI_TKILL)
539  return;
540 
541  auto tcb = reinterpret_cast<Tcb*>(mlibc::get_current_tcb());
542  int old_value = tcb->cancelBits;
543 
544  /*
545  * When a thread is marked with deferred cancellation and performs a blocking syscall,
546  * the spec mandates that the syscall can get interrupted before it has caused any side
547  * effects (e.g. before a read() has read any bytes from disk). If the syscall has
548  * already caused side effects it should return its partial work, and set the program
549  * counter just after the syscall. If the syscall hasn't caused any side effects, it
550  * should fail with EINTR and set the program counter to the syscall instruction.
551  *
552  * cancellable_syscall:
553  * test whether_a_cancel_is_queued
554  * je cancel
555  * syscall
556  * end_cancellable_syscall
557  *
558  * The mlibc::sys_before_cancellable_syscall sysdep should return 1 when the
559  * program counter is between the 'canellable_syscall' and 'end_cancellable_syscall' label.
560  */
561  if (!(old_value & tcbCancelAsyncBit) &&
562  mlibc::sys_before_cancellable_syscall && !mlibc::sys_before_cancellable_syscall(uctx))
563  return;
564 
565  int bitmask = tcbCancelTriggerBit | tcbCancelingBit;
566  while (1) {
567  int new_value = old_value | bitmask;
568 
569  // Check if we are already cancelled or exiting
570  if (old_value == new_value || old_value & tcbExitingBit)
571  return;
572 
573  int current_value = old_value;
574  if (__atomic_compare_exchange_n(&tcb->cancelBits, &current_value,
575  new_value, true,__ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
576  tcb->returnValue = PTHREAD_CANCELED;
577 
578  // Perform cancellation
579  __mlibc_do_cancel();
580 
581  break;
582  }
583 
584  old_value = current_value;
585  }
586  }

651  auto self = reinterpret_cast<Tcb *>(mlibc::get_current_tcb());
652  int old_value = self->cancelBits;
653  while (1) {
654  int new_value = old_value & ~tcbCancelEnableBit;
655  if (state == PTHREAD_CANCEL_ENABLE)
656  new_value |= tcbCancelEnableBit;
657 
658  if (oldstate)
659  *oldstate = ((old_value & tcbCancelEnableBit)
660  ? PTHREAD_CANCEL_ENABLE
661  : PTHREAD_CANCEL_DISABLE);
662 
663  // Avoid unecessary atomic op.
664  if (old_value == new_value)
665  break;
666 
667  int current_value = old_value;
668  if (__atomic_compare_exchange_n(&self->cancelBits, &current_value,
669  new_value, true, __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
670 
671  if (mlibc::tcb_async_cancelled(new_value))
672  __mlibc_do_cancel();
673 
674  sigset_t set = {};
675  sigaddset(&set, SIGCANCEL);
676  if (new_value & PTHREAD_CANCEL_ENABLE)
677  sigprocmask(SIG_UNBLOCK, &set, NULL);
678  else
679  sigprocmask(SIG_BLOCK, &set, NULL);
680  break;
681  }
682 
683  old_value = current_value;
684  }
685 
686  return 0;
687 }
688 void pthread_testcancel(void) {

690  int value = self->cancelBits;

692  __mlibc_do_cancel();

696 int pthread_cancel(pthread_t thread) {
697  if (!mlibc::sys_tgkill) {
698  MLIBC_MISSING_SYSDEP();
699  return ENOSYS;
700  }
701 
702  auto tcb = reinterpret_cast<Tcb *>(thread);
703  // Check if the TCB is valid, somewhat..
704  if (tcb->selfPointer != tcb)
705  return ESRCH;
706 
707  int old_value = __atomic_load_n(&tcb->cancelBits, __ATOMIC_RELAXED);
708  while (1) {
709  int bitmask = tcbCancelTriggerBit;
710 
711  int new_value = old_value | bitmask;
712  if (old_value == new_value)
713  break;
714 
715  int current_value = old_value;
716  if (__atomic_compare_exchange_n(&tcb->cancelBits, &current_value,
717  new_value, true, __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
718  if (mlibc::tcb_cancel_enabled(new_value)) {
719  pid_t pid = getpid();
720 
721  int res = mlibc::sys_tgkill(pid, tcb->tid, SIGCANCEL);
722 
723  current_value = __atomic_load_n(&tcb->cancelBits, __ATOMIC_RELAXED);
724 
725  // If we can't find the thread anymore, it's possible that it exited between
726  // us setting the cancel trigger bit, and us sending the signal. Check the
727  // cancelBits for tcbExitingBit to confirm that.
728  // XXX(qookie): This will be an use-after-free once we start freeing TCBs on
729  // exit. Perhaps the TCB should be refcounted.
730  if (!(res == ESRCH && (current_value & tcbExitingBit)))
731  return res;
732  }
733 
734  break;
735  }
736 
737  old_value = current_value;
738  }
739 
740  return 0;
741 }

847 static constexpr unsigned int onceLocked = 2;

849 int pthread_once(pthread_once_t *once, void (*function) (void)) {
850  SCOPE_TRACE();
851 
852  auto expected = __atomic_load_n(&once->__mlibc_done, __ATOMIC_ACQUIRE);
853 
854  // fast path: the function was already run.
855  while(!(expected & onceComplete)) {
856  if(!expected) {
857  // try to acquire the mutex.
858  if(!__atomic_compare_exchange_n(&once->__mlibc_done,
859  &expected, onceLocked, false, __ATOMIC_ACQUIRE, __ATOMIC_ACQUIRE))
860  continue;
861 
862  function();
863 
864  // unlock the mutex.
865  __atomic_exchange_n(&once->__mlibc_done, onceComplete, __ATOMIC_RELEASE);
866  if(int e = mlibc::sys_futex_wake((int *)&once->__mlibc_done); e)
867  __ensure(!"sys_futex_wake() failed");
868  return 0;
869  }else{
870  // a different thread is currently running the initializer.
871  __ensure(expected == onceLocked);
872  if(int e = mlibc::sys_futex_wait((int *)&once->__mlibc_done, onceLocked, nullptr); e)
873  __ensure(!"sys_futex_wait() failed");
874  expected = __atomic_load_n(&once->__mlibc_done, __ATOMIC_ACQUIRE);
875  }
876  }
877 
878  return 0;
879 }

1004 int pthread_mutex_lock(pthread_mutex_t *mutex) {
1005  SCOPE_TRACE();
1006 
1007  unsigned int this_tid = mlibc::this_tid();
1008  unsigned int expected = 0;
1009  while(true) {
1010  if(!expected) {
1011  // Try to take the mutex here.
1012  if(__atomic_compare_exchange_n(&mutex->__mlibc_state,
1013  &expected, this_tid, false, __ATOMIC_ACQUIRE, __ATOMIC_ACQUIRE)) {
1014  __ensure(!mutex->__mlibc_recursion);
1015  mutex->__mlibc_recursion = 1;
1016  return 0;
1017  }
1018  }else{
1019  // If this (recursive) mutex is already owned by us, increment the recursion level.
1020  if((expected & mutex_owner_mask) == this_tid) {
1021  if(!(mutex->__mlibc_flags & mutexRecursive)) {
1022  if (mutex->__mlibc_flags & mutexErrorCheck)
1023  return EDEADLK;
1024  else
1025  mlibc::panicLogger() << "mlibc: pthread_mutex deadlock detected!"
1026  << frg::endlog;
1027  }
1028  ++mutex->__mlibc_recursion;
1029  return 0;
1030  }
1031 
1032  // Wait on the futex if the waiters flag is set.
1033  if(expected & mutex_waiters_bit) {
1034  int e = mlibc::sys_futex_wait((int *)&mutex->__mlibc_state, expected, nullptr);
1035 
1036  // If the wait returns EAGAIN, that means that the mutex_waiters_bit was just unset by
1037  // some other thread. In this case, we should loop back around.
1038  if (e && e != EAGAIN)
1039  mlibc::panicLogger() << "sys_futex_wait() failed with error code " << e << frg::endlog;
1040 
1041  // Opportunistically try to take the lock after we wake up.
1042  expected = 0;
1043  }else{
1044  // Otherwise we have to set the waiters flag first.
1045  unsigned int desired = expected | mutex_waiters_bit;
1046  if(__atomic_compare_exchange_n((int *)&mutex->__mlibc_state,
1047  reinterpret_cast<int*>(&expected), desired, false, __ATOMIC_RELAXED, __ATOMIC_RELAXED))
1048  expected = desired;
1049  }
1050  }
1051  }
1052 }

1087 int pthread_mutex_unlock(pthread_mutex_t *mutex) {
1088  SCOPE_TRACE();
1089 
1090  // Decrement the recursion level and unlock if we hit zero.
1091  __ensure(mutex->__mlibc_recursion);
1092  if(--mutex->__mlibc_recursion)
1093  return 0;
1094 
1095  auto flags = mutex->__mlibc_flags;
1096 
1097  // Reset the mutex to the unlocked state.
1098  auto state = __atomic_exchange_n(&mutex->__mlibc_state, 0, __ATOMIC_RELEASE);
1099 
1100  // After this point the mutex is unlocked, and therefore we cannot access its contents as it
1101  // may have been destroyed by another thread.
1102 
1103  unsigned int this_tid = mlibc::this_tid();
1104  if ((flags & mutexErrorCheck) && (state & mutex_owner_mask) != this_tid)
1105  return EPERM;
1106 
1107  if ((flags & mutexErrorCheck) && !(state & mutex_owner_mask))
1108  return EINVAL;
1109 
1110  __ensure((state & mutex_owner_mask) == this_tid);
1111 
1112  if(state & mutex_waiters_bit) {
1113  // Wake the futex if there were waiters. Since the mutex might not exist at this location
1114  // anymore, we must conservatively ignore EACCES and EINVAL which may occur as a result.
1115  int e = mlibc::sys_futex_wake((int *)&mutex->__mlibc_state);
1116  __ensure(e >= 0 || e == EACCES || e == EINVAL);
1117  }
1118 
1119  return 0;
1120 }

1196 int pthread_cond_timedwait(pthread_cond_t *__restrict cond, pthread_mutex_t *__restrict mutex,
1197  const struct timespec *__restrict abstime) {
1198  // TODO: pshared isn't supported yet.
1199  __ensure(cond->__mlibc_flags == 0);
1200 
1201  constexpr long nanos_per_second = 1'000'000'000;
1202  if (abstime && (abstime->tv_nsec < 0 || abstime->tv_nsec >= nanos_per_second))
1203  return EINVAL;
1204 
1205  auto seq = __atomic_load_n(&cond->__mlibc_seq, __ATOMIC_ACQUIRE);
1206 
1207  // TODO: handle locking errors and cancellation properly.
1208  while (true) {
1209  if (pthread_mutex_unlock(mutex))
1210  __ensure(!"Failed to unlock the mutex");
1211 
1212  int e;
1213  if (abstime) {
1214  // Adjust for the fact that sys_futex_wait accepts a *timeout*, but
1215  // pthread_cond_timedwait accepts an *absolute time*.
1216  // Note: mlibc::sys_clock_get is available unconditionally.
1217  struct timespec now;
1218  if (mlibc::sys_clock_get(cond->__mlibc_clock, &now.tv_sec, &now.tv_nsec))
1219  __ensure(!"sys_clock_get() failed");
1220 
1221  struct timespec timeout;
1222  timeout.tv_sec = abstime->tv_sec - now.tv_sec;
1223  timeout.tv_nsec = abstime->tv_nsec - now.tv_nsec;
1224 
1225  // Check if abstime has already passed.
1226  if (timeout.tv_sec < 0 || (timeout.tv_sec == 0 && timeout.tv_nsec < 0)) {
1227  if (pthread_mutex_lock(mutex))
1228  __ensure(!"Failed to lock the mutex");
1229  return ETIMEDOUT;
1230  } else if (timeout.tv_nsec >= nanos_per_second) {
1231  timeout.tv_nsec -= nanos_per_second;
1232  timeout.tv_sec++;
1233  __ensure(timeout.tv_nsec < nanos_per_second);
1234  } else if (timeout.tv_nsec < 0) {
1235  timeout.tv_nsec += nanos_per_second;
1236  timeout.tv_sec--;
1237  __ensure(timeout.tv_nsec >= 0);
1238  }
1239 
1240  e = mlibc::sys_futex_wait((int *)&cond->__mlibc_seq, seq, &timeout);
1241  } else {
1242  e = mlibc::sys_futex_wait((int *)&cond->__mlibc_seq, seq, nullptr);
1243  }
1244 
1245  if (pthread_mutex_lock(mutex))
1246  __ensure(!"Failed to lock the mutex");
1247 
1248  // There are four cases to handle:
1249  // 1. e == 0: this indicates a (potentially spurious) wakeup. The value of
1250  // seq *must* be checked to distinguish these two cases.
1251  // 2. e == EAGAIN: this indicates that the value of seq changed before we
1252  // went to sleep. We don't need to check seq in this case.
1253  // 3. e == EINTR: a signal was delivered. The man page allows us to choose

1255  // to match other libcs.

1359  // the barrier and immediately destroy its state while other threads still rely on it.
1360
