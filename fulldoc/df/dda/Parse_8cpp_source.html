27  BinaryType GetBinaryType(void *Image)
28  {
29  Fex *FexHdr = (Fex *)Image;
30 
31  /* Elf64_Ehdr and Elf32_Ehdr are very similar (Elf64_Half and
32  Elf32_Half are the same size type) so we can use directly Elf64_Ehdr. */
33  Elf64_Ehdr *ELFHeader = (Elf64_Ehdr *)Image;
34 
35  IMAGE_DOS_HEADER *MZHeader = (IMAGE_DOS_HEADER *)Image;
36 
37  /* Check Fex magic */
38  if (FexHdr->Magic[0] == 'F' && FexHdr->Magic[1] == 'E' && FexHdr->Magic[2] == 'X' && FexHdr->Magic[3] == '\0')
39  {
40  /* If the fex type is driver, we shouldn't return as Fex. */
41  if (FexHdr->Type == FexFormatType::FexFormatType_Executable)
42  {
43  debug("Image - Fex");
44  return BinaryType::BinTypeFex;
45  }
46  else if (FexHdr->Type == FexFormatType::FexFormatType_Driver)
47  debug("Fex Driver is not supposed to be executed.");
48  }
49  /* Check ELF magic. */
50  else if (ELFHeader->e_ident[EI_MAG0] == ELFMAG0 &&
51  ELFHeader->e_ident[EI_MAG1] == ELFMAG1 &&
52  ELFHeader->e_ident[EI_MAG2] == ELFMAG2 &&
53  ELFHeader->e_ident[EI_MAG3] == ELFMAG3)
54  {
55  debug("Image - ELF");
56  return BinaryType::BinTypeELF;
57  }
58  /* Every Windows executable starts with MZ header. */
59  else if (MZHeader->e_magic == IMAGE_DOS_SIGNATURE)
60  {
61  IMAGE_NT_HEADERS *PEHeader = (IMAGE_NT_HEADERS *)(((char *)Image) + MZHeader->e_lfanew);
62  IMAGE_OS2_HEADER *NEHeader = (IMAGE_OS2_HEADER *)(((char *)Image) + MZHeader->e_lfanew);
63 
64  /* TODO: LE, EDOS */
65  if (PEHeader->Signature == IMAGE_NT_SIGNATURE)
66  {
67  debug("Image - PE");
68  return BinaryType::BinTypePE;
69  }
70  else if (NEHeader->ne_magic == IMAGE_OS2_SIGNATURE)
71  {
72  debug("Image - NE");
73  return BinaryType::BinTypeNE;
74  }
75  else
76  {
77  debug("Image - MZ");
78  return BinaryType::BinTypeMZ;
79  }
80  }
81 
82  /* ... */
83  return BinaryType::BinTypeUnknown;
84  }
