104 
105  addtimeval (&t1, ut, st);

108 
109  for (i = 0; i < 6; i++)
110  {

112  break;
113  t1.tv_sec *= 10;
114  t1.tv_sec += t1.tv_usec / 100000;
115  t1.tv_usec *= 10;
116  t1.tv_usec %= 1000000;

118  t2.tv_sec += t2.tv_usec / 100000;
119  t2.tv_usec *= 10;

121  }
122  for (i = 0; i < 4; i++)

133 /* Convert a pointer to a struct timeval to seconds and thousandths of a
134  second, returning the values in *SP and *SFP, respectively. This does
135  rounding on the fractional part, not just truncation to three places. */
136 void
137 timeval_to_secs (tvp, sp, sfp)
138  struct timeval *tvp;

140  int *sfp;
141 {

143
