67 nsa bool WriteDiagDataToNode(FileNode *node)
68 {
69  uintptr_t KStart = (uintptr_t)&_kernel_start;
70  uintptr_t kEnd = (uintptr_t)&_kernel_end;
71  size_t kSize = kEnd - KStart;
72 
73  size_t fileSize = sizeof(DiagnosticFile) + kSize;
74  uint8_t *buf = (uint8_t *)KernelAllocator.RequestPages(TO_PAGES(fileSize));
75  if (!buf)
76  {
77  ExPrint("\x1b[0;30;41mFailed to allocate memory for diagnostic data\x1b[0m\n");
78  Display->UpdateBuffer();
79  return false;
80  }
81 
82  DiagnosticFile *file = (DiagnosticFile *)buf;
83  file->Header = {};
84  file->Header.Length = uint32_t(fileSize);
85  file->Header.Is64Bit = sizeof(void *) == 8;
86 
87  file->Data.Frame = *ExFrame;
88  file->Data.KernelMemoryLength = uint32_t(kSize);
89  memcpy(file->Data.KernelMemory, (void *)KStart, kSize);
90 
91  ExPrint("Writing to %s\n", node->Path.c_str());
92  size_t w = node->Write(buf, fileSize, 0);
93  if (w != fileSize)
94  {
95  debug("%d out of %d bytes written", w, fileSize);
96  ExPrint("\x1b[0;30;41mFailed to write diagnostic data to file: %s\x1b[0m\n",
97  strerror((int)w));
98  Display->UpdateBuffer();
99  return false;
100  }
101 
102  return true;
103 }
