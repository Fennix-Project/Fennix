51  void ELFInterpreterIPCThread(PCB *Process, char *Path, void *MemoryImage, void *ElfFile, std::vector<const char *> NeededLibraries)
52  {
53  debug("Interpreter thread started for %s", Path);
54  // Interpreter will create an IPC with token "LOAD".
55  char UniqueToken[16] = {'L', 'O', 'A', 'D', '\0'};
56  InterProcessCommunication::IPCHandle *Handle = nullptr;
57  while (Handle == nullptr)
58  {
59  debug("Searching for IPC with token %s", UniqueToken);
60  Handle = Process->IPC->SearchByToken(UniqueToken);
61  if (Handle == nullptr)
62  {
63  debug("Failed");
64  }
65 
66  TaskManager->Sleep(200);
67  if (Handle == nullptr)
68  {
69  debug("Retrying...");
70  }
71  }
72  debug("IPC found, sending data...");
73  InterpreterIPCData *TmpBuffer = new InterpreterIPCData;
74  strcpy(TmpBuffer->Path, Path);
75  TmpBuffer->ElfFile = ElfFile;
76  TmpBuffer->MemoryImage = MemoryImage;
77  if (NeededLibraries.size() > 256)
78  warn("Too many libraries! (max 256)");
79  for (size_t i = 0; i < NeededLibraries.size(); i++)
80  {
81  strcpy(TmpBuffer->Libraries[i].Name, NeededLibraries[i]);
82  }
83 
84 #ifdef DEBUG
85  debug("OUTSIDE DATA");
86  debug("Path: %s", Path);
87  debug("ElfFile: %p", ElfFile);
88  debug("MemoryImage: %p", MemoryImage);
89  for (size_t i = 0; i < NeededLibraries.size(); i++)
90  {
91  debug("Library: %s", NeededLibraries[i]);
92  }
93  debug("INSIDE DATA");
94  debug("Path: %s", TmpBuffer->Path);
95  debug("ElfFile: %p", TmpBuffer->ElfFile);
96  debug("MemoryImage: %p", TmpBuffer->MemoryImage);
97  for (size_t i = 0; i < NeededLibraries.size(); i++)
98  {
99  debug("Library: %s", TmpBuffer->Libraries[i].Name);
100  }
101 #endif
102 
103  RetryIPCWrite:
104  InterProcessCommunication::IPCErrorCode ret = Process->IPC->Write(Handle->ID, TmpBuffer, sizeof(InterpreterIPCData));
105  debug("Write returned %d", ret);
106  if (ret == InterProcessCommunication::IPCErrorCode::IPCNotListening)
107  {
108  debug("IPC not listening, retrying...");
109  TaskManager->Sleep(100);
110  goto RetryIPCWrite;
111  }
112  delete TmpBuffer;
113  /* Prevent race condition, maybe a
114  better idea is to watch when the
115  IPC is destroyed. */
116  TaskManager->Sleep(5000);
117  TEXIT(0);
118  }

166  /* Copy the file to the allocated memory */
167  memcpy(ElfFile, (void *)ExFile.node->Address, ExFileSize);
168  debug("Elf file: %#lx - %#lx (length: %ld)", ElfFile, (uintptr_t)ElfFile + ExFileSize, ExFileSize);
