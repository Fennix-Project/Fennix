33  int Spawn(char *Path, const char **argv, const char **envp,
34  Tasking::PCB *Parent, bool Fork,
35  Tasking::TaskCompatibility Compatibility,
36  bool Critical)
37  {
38  int fd = fopen(Path, "r");
39  if (fd < 0)
40  return fd;
41 
42  struct stat statbuf;
43  fstat(fd, &statbuf);
44  if (!S_ISREG(statbuf.st_mode))
45  {
46  fclose(fd);
47  return -EISDIR;
48  }
49 
50  switch (GetBinaryType(Path))
51  {
52  case BinaryType::BinTypeELF:
53  {
54  TaskArchitecture Arch = TaskArchitecture::UnknownArchitecture;
55  const char *BaseName;
56  cwk_path_get_basename(Path, &BaseName, nullptr);
57  Elf32_Ehdr ELFHeader;
58  fread(fd, (uint8_t *)&ELFHeader, sizeof(Elf32_Ehdr));
59 
60  switch (ELFHeader.e_machine)
61  {
62  case EM_386:
63  Arch = TaskArchitecture::x32;
64  break;
65  case EM_X86_64:
66  Arch = TaskArchitecture::x64;
67  break;
68  case EM_ARM:
69  Arch = TaskArchitecture::ARM32;
70  break;
71  case EM_AARCH64:
72  Arch = TaskArchitecture::ARM64;
73  break;
74  default:
75  error("Unknown ELF architecture %d",
76  ELFHeader.e_machine);
77  break;
78  }
79 
80  // TODO: This shouldn't be ignored
81  if (ELFHeader.e_ident[EI_CLASS] == ELFCLASS32)
82  fixme("32-bit ELF");
83  else if (ELFHeader.e_ident[EI_CLASS] == ELFCLASS64)
84  fixme("64-bit ELF");
85  else
86  fixme("Unknown class %d", ELFHeader.e_ident[EI_CLASS]);
87 
88 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
89  if (ELFHeader.e_ident[EI_DATA] != ELFDATA2LSB)
90  {
91  fixme("ELF32 LSB expected, got %d", ELFHeader.e_ident[EI_DATA]);
92  }
93 #else
94  if (ELFHeader.e_ident[EI_DATA] != ELFDATA2MSB)
95  {
96  fixme("ELF32 MSB expected, got %d", ELFHeader.e_ident[EI_DATA]);
97  }
98 #endif
99 
100  /* ------------------------------------------------------------------------------------------------------------------------------ */
101 
102  PCB *Process;
103  if (Fork)
104  {
105  assert(Parent != nullptr);
106  CriticalSection cs;
107 
108  Process = Parent;
109  foreach (auto tcb in Process->Threads)
110  {
111  debug("Deleting thread %d", tcb->ID);
112  // delete tcb;
113  tcb->SetState(Tasking::Terminated);
114  }
115 
116  fixme("free allocated memory");
117  // Process->vma->FreeAllPages();
118  }
119  else
120  {
121  if (Parent == nullptr)
122  Parent = thisProcess;
123 
124  Process = TaskManager->CreateProcess(Parent,
125  BaseName,
126  TaskExecutionMode::User,
127  false, 0, 0);
128  Process->Info.Compatibility = Compatibility;
129  Process->Info.Architecture = Arch;
130  }
131 
132  Process->SetWorkingDirectory(fs->GetNodeFromPath(Path)->Parent);
133  Process->SetExe(Path);
134 
135  ELFObject *obj = new ELFObject(Path, Process, argv, envp);
136  if (!obj->IsValid)
137  {
138  error("Failed to load ELF object");
139  fclose(fd);
140  delete Process;
141  return -ENOEXEC;
142  }
143 
144  vfs::FileDescriptorTable *pfdt = Parent->FileDescriptors;
145  vfs::FileDescriptorTable *fdt = Process->FileDescriptors;
146 
147  auto ForkStdio = [pfdt, fdt](Node *SearchNode)
148  {
149  if (unlikely(SearchNode == nullptr))
150  return false;
151 
152  std::vector<FileDescriptorTable::Fildes>
153  pfds = pfdt->GetFileDescriptors();
154 
155  foreach (auto ffd in pfds)
156  {
157  if (ffd.Flags & O_CLOEXEC)
158  continue;
159 
160  if (ffd.Handle->node == SearchNode)
161  {
162  fdt->_open(ffd.Handle->node->FullPath,
163  ffd.Flags, ffd.Mode);
164  return true;
165  }
166  }
167  return false;
168  };
169 
170  if (!ForkStdio(Parent->stdin))
171  fdt->_open("/dev/kcon", O_RDWR, 0666);
172 
173  if (!ForkStdio(Parent->stdout))
174  fdt->_open("/dev/kcon", O_RDWR, 0666);
175 
176  if (!ForkStdio(Parent->stderr))
177  fdt->_open("/dev/kcon", O_RDWR, 0666);
178 
179  TCB *Thread = nullptr;
180  {
181  CriticalSection cs;
182  Thread = TaskManager->CreateThread(Process,
183  obj->InstructionPointer,
184  obj->argv, obj->envp, obj->auxv,
185  Arch,
186  Compatibility);
187  Thread->SetCritical(Critical);
188  }
189  fclose(fd);
190  return Thread->ID;
191  }
192  default:
193  {
194  debug("Unknown binary type: %d",
195  GetBinaryType(Path));
196  fclose(fd);
197  return -ENOEXEC;
198  }
199  }
200 
201  fclose(fd);
202  return -ENOEXEC;
203  }
204 }
