39 /*
40  * cprojf(float complex z)
41  *
42  * These functions return the value of the projection (not stereographic!)
43  * onto the Riemann sphere.
44  *
45  * z projects to z, except that all complex infinities (even those with one
46  * infinite part and one NaN part) project to positive infinity on the real axis.
47  * If z has an infinite part, then cproj(z) shall be equivalent to:
48  *
49  * INFINITY + I * copysign(0.0, cimag(z))
50  */
51 
52 float complex
53 cprojf(float complex z)
54 {
55  float_complex w = { .z = z };
56 
57  if (isinf(crealf(z)) || isinf(cimagf(z))) {
58 #ifdef __INFINITY
59  REAL_PART(w) = __INFINITY;
60 #else
61  REAL_PART(w) = INFINITY;
62 #endif
63  IMAG_PART(w) = copysignf(0.0, cimagf(z));
64  }
65 
66  return (w.z);
67 }
