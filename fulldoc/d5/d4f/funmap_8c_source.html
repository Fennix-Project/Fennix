212 int
213 rl_add_funmap_entry (const char *name, rl_command_func_t *function)
214 {
215  if (funmap_entry + 2 >= funmap_size)
216  {
217  funmap_size += 64;
218  funmap = (FUNMAP **)xrealloc (funmap, funmap_size * sizeof (FUNMAP *));
219  }
220 
221  funmap[funmap_entry] = (FUNMAP *)xmalloc (sizeof (FUNMAP));
222  funmap[funmap_entry]->name = name;
223  funmap[funmap_entry]->function = function;
224 
225  funmap[++funmap_entry] = (FUNMAP *)NULL;
226  return funmap_entry;
227 }
228 
229 static int funmap_initialized;
230 
231 /* Make the funmap contain all of the default entries. */
232 void
233 rl_initialize_funmap (void)
234 {
235  register int i;
236 
237  if (funmap_initialized)
238  return;
239 
240  for (i = 0; default_funmap[i].name; i++)
241  rl_add_funmap_entry (default_funmap[i].name, default_funmap[i].function);
242 
243  funmap_initialized = 1;
244  funmap_program_specific_entry_start = i;
245 }

247 /* Produce a NULL terminated array of known function names. The array
248  is sorted. The array itself is allocated, but not the strings inside.
249  You should free () the array when you done, but not the pointers. */
250 const char **
251 rl_funmap_names (void)
252 {
253  const char **result;
254  int result_size, result_index;
255 
256  /* Make sure that the function map has been initialized. */
257  rl_initialize_funmap ();
258 
259  for (result_index = result_size = 0, result = (const char **)NULL; funmap[result_index]; result_index++)
260  {
261  if (result_index + 2 > result_size)
262  {
263  result_size += 20;
264  result = (const char **)xrealloc (result, result_size * sizeof (char *));
265  }
266 
267  result[result_index] = funmap[result_index]->name;
268  result[result_index + 1] = (char *)NULL;
269  }
270 
271  qsort (result, result_index, sizeof (char *), (QSFUNC *)_rl_qsort_string_compare);
272  return (result);
273 }
