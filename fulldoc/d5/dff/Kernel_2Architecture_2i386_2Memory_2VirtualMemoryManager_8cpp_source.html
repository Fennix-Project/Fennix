25  bool Virtual::Check(void *VirtualAddress, PTFlag Flag, MapType Type)
26  {
27  // 0x1000 aligned
28  uintptr_t Address = (uintptr_t)VirtualAddress;
29  Address &= 0xFFFFFFFFFFFFF000;
30 
31  PageMapIndexer Index = PageMapIndexer(Address);
32  PageDirectoryEntry PDE = this->Table->Entries[Index.PDEIndex];
33  PageTableEntryPtr *PTE = nullptr;
34 
35  if ((PDE.raw & Flag) > 0)
36  {
37  if (Type == MapType::FourMB && PDE.PageSize)
38  return true;
39 
40  PTE = (PageTableEntryPtr *)((uintptr_t)PDE.GetAddress() << 12);
41  if (PTE)
42  {
43  if ((PTE->Entries[Index.PTEIndex].Present))
44  return true;
45  }
46  }
47  return false;

75  void Virtual::Map(void *VirtualAddress, void *PhysicalAddress, uint64_t Flags, MapType Type)
76  {
77  SmartLock(this->MemoryLock);
78  if (unlikely(!this->Table))
79  {
80  error("No page table");
81  return;
82  }
83 
84  Flags |= PTFlag::P;
85 
86  PageMapIndexer Index = PageMapIndexer((uintptr_t)VirtualAddress);
87  // Clear any flags that are not 1 << 0 (Present) - 1 << 5 (Accessed) because rest are for page table entries only
88  uint64_t DirectoryFlags = Flags & 0x3F;
89 
90  PageDirectoryEntry *PDE = &this->Table->Entries[Index.PDEIndex];
91  if (Type == MapType::FourMB)
92  {
93  PDE->raw |= Flags;
94  PDE->PageSize = true;
95  PDE->SetAddress((uintptr_t)PhysicalAddress >> 12);
96  debug("Mapped 4MB page at %p to %p", VirtualAddress, PhysicalAddress);
97  return;
98  }
99 
100  PageTableEntryPtr *PTEPtr = nullptr;
101  if (!PDE->Present)
102  {
103  PTEPtr = (PageTableEntryPtr *)KernelAllocator.RequestPages(TO_PAGES(sizeof(PageTableEntryPtr) + 1));
104  memset(PTEPtr, 0, sizeof(PageTableEntryPtr));
105  PDE->Present = true;
106  PDE->SetAddress((uintptr_t)PTEPtr >> 12);
107  }
108  else
109  PTEPtr = (PageTableEntryPtr *)(PDE->GetAddress() << 12);
110  PDE->raw |= DirectoryFlags;
111 
112  PageTableEntry *PTE = &PTEPtr->Entries[Index.PTEIndex];
113  PTE->Present = true;
114  PTE->raw |= Flags;
115  PTE->SetAddress((uintptr_t)PhysicalAddress >> 12);
116 
117 #if defined(a64)
118  CPU::x64::invlpg(VirtualAddress);
119 #elif defined(a32)
120  CPU::x32::invlpg(VirtualAddress);
121 #elif defined(aa64)
122  asmv("dsb sy");
123  asmv("tlbi vae1is, %0"
124  :
125  : "r"(VirtualAddress)
126  : "memory");
127  asmv("dsb sy");
128  asmv("isb");
129 #endif

131 #ifdef DEBUG

133 #define BYTE_TO_BINARY_PATTERN "%c%c%c%c%c%c%c%c"
134 #define BYTE_TO_BINARY(byte) \

150  {
