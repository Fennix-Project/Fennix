39  int RegisterFunction(dev_t MajorID, void *Function, __driverRegFunc Type)
40  {
41  dbg_api("%d, %#lx, %d", MajorID, (uintptr_t)Function, Type);
42 
43  std::unordered_map<dev_t, DriverObject> &Drivers =
44  DriverManager->GetDrivers();
45 
46  auto itr = Drivers.find(MajorID);
47  if (itr == Drivers.end())
48  return -EINVAL;
49 
50  DriverObject *drv = &itr->second;
51 
52  switch (Type)
53  {
54  case _drf_Entry:
55  drv->Entry = (int (*)())Function;
56  debug("Entry %#lx for %s", (uintptr_t)Function, drv->Path);
57  break;
58  case _drf_Final:
59  drv->Final = (int (*)())Function;
60  debug("Finalize %#lx for %s", (uintptr_t)Function, drv->Path);
61  break;
62  case _drf_Panic:
63  drv->Panic = (int (*)())Function;
64  debug("Panic %#lx for %s", (uintptr_t)Function, drv->Path);
65  break;
66  case _drf_Probe:
67  drv->Probe = (int (*)())Function;
68  debug("Probe %#lx for %s", (uintptr_t)Function, drv->Path);
69  break;
70  default:
71  assert(!"Invalid driver function type");
72  }
73  return 0;
74  }

120  int OverrideInterruptHandler(dev_t MajorID, uint8_t IRQ, void *Handler)
121  {
122  dbg_api("%d, %d, %#lx", MajorID, IRQ, Handler);
123 
124  debug("Overriding IRQ %d with %#lx", IRQ, Handler);
125 
126  std::unordered_map<dev_t, DriverObject> &Drivers =
127  DriverManager->GetDrivers();
128 
129  foreach (auto &var in Drivers)
130  {
131  DriverObject *drv = &var.second;
132 
133  foreach (auto &ih in * drv->InterruptHandlers)
134  {
135  if (ih.first == IRQ)
136  {
137  debug("Removing IRQ %d: %#lx for %s", IRQ, (uintptr_t)ih.second, drv->Path);
138  Interrupts::RemoveHandler((void (*)(CPU::TrapFrame *))ih.second, IRQ);
139  drv->InterruptHandlers->erase(IRQ);
140  break;
141  }
142  }
143  }
144 
145  return RegisterInterruptHandler(MajorID, IRQ, Handler);
146  }

600  void InitializePCI(dev_t MajorID, void *_Header)
601  {
602  dbg_api("%d, %#lx", MajorID, _Header);
603 
604  PCI::PCIDeviceHeader *Header = (PCI::PCIDeviceHeader *)_Header;
605 
606  debug("Header Type: %d", Header->HeaderType);
607  switch (Header->HeaderType)
608  {
609  case 128:
610  warn("Unknown header type %d! Guessing PCI Header 0",
611  Header->HeaderType);
612  [[fallthrough]];
613  case 0: /* PCI Header 0 */
614  {
615  PCI::PCIHeader0 *hdr0 = (PCI::PCIHeader0 *)Header;
616 
617  uint32_t BAR[6];
618  size_t BARsSize[6];
619 
620  BAR[0] = hdr0->BAR0;
621  BAR[1] = hdr0->BAR1;
622  BAR[2] = hdr0->BAR2;
623  BAR[3] = hdr0->BAR3;
624  BAR[4] = hdr0->BAR4;
625  BAR[5] = hdr0->BAR5;
626 
627  debug("Type: %d; IOBase: %#lx; MemoryBase: %#lx",
628  BAR[0] & 1, BAR[1] & (~3), BAR[0] & (~15));
629 
630  /* BARs Size */
631  for (short i = 0; i < 6; i++)
632  {
633  if (BAR[i] == 0)
634  continue;
635 
636  size_t size;
637  if ((BAR[i] & 1) == 0) /* Memory Base */
638  {
639  hdr0->BAR0 = 0xFFFFFFFF;
640  size = hdr0->BAR0;
641  hdr0->BAR0 = BAR[i];
642  BARsSize[i] = size & (~15);
643  BARsSize[i] = ~BARsSize[i] + 1;
644  BARsSize[i] = BARsSize[i] & 0xFFFFFFFF;
645  debug("BAR%d %#lx size: %d",
646  i, BAR[i], BARsSize[i]);
647  }
648  else if ((BAR[i] & 1) == 1) /* I/O Base */
649  {
650  hdr0->BAR1 = 0xFFFFFFFF;
651  size = hdr0->BAR1;
652  hdr0->BAR1 = BAR[i];
653  BARsSize[i] = size & (~3);
654  BARsSize[i] = ~BARsSize[i] + 1;
655  BARsSize[i] = BARsSize[i] & 0xFFFF;
656  debug("BAR%d %#lx size: %d",
657  i, BAR[i], BARsSize[i]);
658  }
659  }
660 
661  Memory::Virtual vmm(KernelPageTable);
662 
663  /* Mapping the BARs */
664  for (short i = 0; i < 6; i++)
665  {
666  if (BAR[i] == 0)
667  continue;
668
