58  CriticalState EnterCriticalSection(dev_t DriverID)
59  {
60  dbg_api("%d", DriverID);
61 
62  CriticalState cs;
63 
64 #if defined(__i386__) || defined(__x86_64__)
65 
66  uintptr_t Flags;
67 #if defined(__x86_64__)
68  asmv("pushfq");
69  asmv("popq %0"
70  : "=r"(Flags));
71 #else
72  asmv("pushfl");
73  asmv("popl %0"
74  : "=r"(Flags));
75 #endif
76  cs = Flags & (1 << 9);
77  asmv("cli");
78 
79 #elif defined(__arm__) || defined(__aarch64__)
80 
81  uintptr_t Flags;
82  asmv("mrs %0, cpsr"
83  : "=r"(Flags));
84  cs = Flags & (1 << 7);
85  asmv("cpsid i");
86 
87 #endif
88 
89  return cs;
90  }

109  int RegisterInterruptHandler(dev_t DriverID, uint8_t IRQ, void *Handler)
110  {
111  dbg_api("%d, %d, %#lx", DriverID, IRQ, Handler);
112 
113  std::unordered_map<dev_t, Driver::DriverObject> &drivers =
114  DriverManager->GetDrivers();
115  const auto it = drivers.find(DriverID);
116  if (it == drivers.end())
117  ReturnLogError(-EINVAL, "Driver %d not found", DriverID);
118  const Driver::DriverObject *drv = &it->second;
119 
120  if (drv->InterruptHandlers->contains(IRQ))
121  return -EEXIST;
122 
123  Interrupts::AddHandler((void (*)(CPU::TrapFrame *))Handler, IRQ);
124  auto ih = drv->InterruptHandlers;
125  ih->insert(std::pair<uint8_t, void *>(IRQ, Handler));
126  return 0;
127  }

129  int OverrideInterruptHandler(dev_t DriverID, uint8_t IRQ, void *Handler)
130  {
131  dbg_api("%d, %d, %#lx", DriverID, IRQ, Handler);
132 
133  debug("Overriding IRQ %d with %#lx", IRQ, Handler);
134 
135  std::unordered_map<dev_t, Driver::DriverObject> &drivers =
136  DriverManager->GetDrivers();
137 
138  for (auto &var : drivers)
139  {
140  Driver::DriverObject *drv = &var.second;
141  for (const auto &ih : *drv->InterruptHandlers)
142  {
143  if (ih.first != IRQ)
144  continue;
145 
146  debug("Removing IRQ %d: %#lx for %s", IRQ, (uintptr_t)ih.second, drv->Path.c_str());
147  Interrupts::RemoveHandler((void (*)(CPU::TrapFrame *))ih.second, IRQ);
148  drv->InterruptHandlers->erase(IRQ);
149  break;
150  }
151  }
152 
153  return RegisterInterruptHandler(DriverID, IRQ, Handler);
154  }

598  void InitializePCI(dev_t DriverID, void *_Header)
599  {
600  dbg_api("%d, %#lx", DriverID, _Header);
601 
602  PCI::PCIDevice *__device = (PCI::PCIDevice *)_Header;
603  PCI::PCIDeviceHeader *Header = (PCI::PCIDeviceHeader *)__device->Header;
604 
605  debug("Header Type: %d", Header->HeaderType);
606  switch (Header->HeaderType)
607  {
608  case 128:
609  warn("Unknown header type %d! Guessing PCI Header 0",
610  Header->HeaderType);
611  [[fallthrough]];
612  case 0: /* PCI Header 0 */
613  {
614  PCI::PCIHeader0 *hdr0 = (PCI::PCIHeader0 *)Header;
615 
616  uint32_t BAR[6];
617  size_t BARsSize[6];
618 
619  BAR[0] = hdr0->BAR0;
620  BAR[1] = hdr0->BAR1;
621  BAR[2] = hdr0->BAR2;
622  BAR[3] = hdr0->BAR3;
623  BAR[4] = hdr0->BAR4;
624  BAR[5] = hdr0->BAR5;
625 
626  debug("Type: %d; IOBase: %#lx; MemoryBase: %#lx",
627  BAR[0] & 1, BAR[1] & (~3), BAR[0] & (~15));
628 
629  /* BARs Size */
630  for (short i = 0; i < 6; i++)
631  {
632  if (BAR[i] == 0)
633  continue;
634 
635  size_t size;
636  if ((BAR[i] & 1) == 0) /* Memory Base */
637  {
638  hdr0->BAR0 = 0xFFFFFFFF;
639  size = hdr0->BAR0;
640  hdr0->BAR0 = BAR[i];
641  BARsSize[i] = size & (~15);
642  BARsSize[i] = ~BARsSize[i] + 1;
643  BARsSize[i] = BARsSize[i] & 0xFFFFFFFF;
644  debug("BAR%d %#lx size: %d",
645  i, BAR[i], BARsSize[i]);
646  }
647  else if ((BAR[i] & 1) == 1) /* I/O Base */
648  {
649  hdr0->BAR1 = 0xFFFFFFFF;
650  size = hdr0->BAR1;
651  hdr0->BAR1 = BAR[i];
652  BARsSize[i] = size & (~3);
653  BARsSize[i] = ~BARsSize[i] + 1;
654  BARsSize[i] = BARsSize[i] & 0xFFFF;
655  debug("BAR%d %#lx size: %d",
656  i, BAR[i], BARsSize[i]);
657  }
658  }
659 
660  Memory::Virtual vmm(KernelPageTable);
661 
662  /* Mapping the BARs */
663  for (short i = 0; i < 6; i++)
664  {
665  if (BAR[i] == 0)
666  continue;
667 
668  if ((BAR[i] & 1) == 0) /* Memory Base */
669  {
670  uintptr_t BARBase = BAR[i] & (~15);
671  size_t BARSize = BARsSize[i];
672 
673  debug("Mapping BAR%d %#lx-%#lx",
674  i, BARBase, BARBase + BARSize);
675 
676  if (BARSize == 0)
677  {
678  warn("BAR%d size is zero!", i);
679  BARSize++;
680  }
681 
682  vmm.Map((void *)BARBase, (void *)BARBase,
683  BARSize, Memory::RW | Memory::PWT | Memory::PCD);
684  }
685  else if ((BAR[i] & 1) == 1) /* I/O Base */
686  {
687  uintptr_t BARBase = BAR[i] & (~3);
688  size_t BARSize = BARsSize[i];
689 
690  debug("Mapping BAR%d %#x-%#x",
691  i, BARBase, BARBase + BARSize);
692 
693  if (BARSize == 0)
694  {
695  warn("BAR%d size is zero!", i);
696  BARSize++;
697  }
698 
699  vmm.Map((void *)BARBase, (void *)BARBase,
700  BARSize, Memory::RW | Memory::PWT | Memory::PCD);
701  }
702  }
703  break;
704  }
705  case 1: /* PCI Header 1 (PCI-to-PCI Bridge) */
706  {
707  fixme("PCI Header 1 (PCI-to-PCI Bridge) not implemented yet");
708  break;
709  }
710  case 2: /* PCI Header 2 (PCI-to-CardBus Bridge) */
711  {
712  fixme("PCI Header 2 (PCI-to-CardBus Bridge) not implemented yet");
713  break;
714  }
715  default:
716  {
717  error("Unknown header type %d", Header->HeaderType);
718  break;
719  }
720  }
721 
722  Header->Command |= PCI_COMMAND_MASTER |
723  PCI_COMMAND_IO |
724  PCI_COMMAND_MEMORY;
725  Header->Command &= ~PCI_COMMAND_INTX_DISABLE;
726  }
