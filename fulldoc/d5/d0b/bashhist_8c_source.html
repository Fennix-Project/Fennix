115  > echo $i
116  > done

118  bash$ history

120  11 do
121  12 echo $i

208 #if defined (BANG_HISTORY)
209 /* Is the history expansion starting at string[i] one that should not
210  be expanded? */
211 static int
212 bash_history_inhibit_expansion (string, i)
213  char *string;
214  int i;
215 {
216  int t, si;
217  char hx[2];
218 
219  hx[0] = history_expansion_char;
220  hx[1] = '\0';
221 
222  /* The shell uses ! as a pattern negation character in globbing [...]
223  expressions, so let those pass without expansion. */
224  if (i > 0 && (string[i - 1] == '[') && member (']', string + i + 1))
225  return (1);
226  /* The shell uses ! as the indirect expansion character, so let those
227  expansions pass as well. */
228  else if (i > 1 && string[i - 1] == '{' && string[i - 2] == '$' &&
229  member ('}', string + i + 1))
230  return (1);
231  /* The shell uses $! as a defined parameter expansion. */
232  else if (i > 1 && string[i - 1] == '$' && string[i] == '!')
233  return (1);
234 #if defined (EXTENDED_GLOB)
235  else if (extended_glob && i > 1 && string[i+1] == '(' && member (')', string + i + 2))
236  return (1);
237 #endif
238 
239  si = 0;
240  /* If we're supposed to be in single-quoted string, skip over the
241  single-quoted part and then look at what's left. */
242  if (history_quoting_state == '\'')
243  {
244  si = skip_to_delim (string, 0, "'", SD_NOJMP|SD_HISTEXP);
245  if (string[si] == 0 || si >= i)
246  return (1);
247  si++;
248  }
249 
250  /* Make sure the history expansion should not be skipped by quoting or
251  command/process substitution. */
252  if ((t = skip_to_histexp (string, si, hx, SD_NOJMP|SD_HISTEXP)) > 0)
253  {
254  /* Skip instances of history expansion appearing on the line before
255  this one. */
256  while (t < i)
257  {
258  t = skip_to_histexp (string, t+1, hx, SD_NOJMP|SD_HISTEXP);
259  if (t <= 0)
260  return 0;
261  }
262  return (t > i);
263  }
264  else
265  return (0);
266 }
267 #endif

439 int
440 maybe_append_history (filename)
441  char *filename;
442 {
443  int fd, result, histlen;
444  struct stat buf;
445 
446  result = EXECUTION_SUCCESS;
447  if (history_lines_this_session > 0)
448  {
449  /* If the filename was supplied, then create it if necessary. */
450  if (stat (filename, &buf) == -1 && errno == ENOENT)
451  {
452  fd = open (filename, O_WRONLY|O_CREAT, 0600);
453  if (fd < 0)
454  {
455  builtin_error (_("%s: cannot create: %s"), filename, strerror (errno));
456  return (EXECUTION_FAILURE);
457  }
458  close (fd);
459  }
460  /* cap the number of lines we write at the length of the history list */
461  histlen = where_history ();
462  if (histlen > 0 && history_lines_this_session > histlen)
463  history_lines_this_session = histlen; /* reset below anyway */
464  result = append_history (history_lines_this_session, filename);
465  /* Pretend we already read these lines from the file because we just
466  added them */
467  history_lines_in_file += history_lines_this_session;
468  history_lines_this_session = 0;
469  }
470  else
471  history_lines_this_session = 0; /* reset if > where_history() */
472 
473  return (result);
474 }

547 /* Do pre-processing on LINE. If PRINT_CHANGES is non-zero, then
548  print the results of expanding the line if there were any changes.
549  If there is an error, return NULL, otherwise the expanded line is
550  returned. If ADDIT is non-zero the line is added to the history
551  list after history expansion. ADDIT is just a suggestion;
552  REMEMBER_ON_HISTORY can veto, and does.
553  Right now this does history expansion. */
554 char *
555 pre_process_line (line, print_changes, addit)
556  char *line;
557  int print_changes, addit;
558 {
559  char *history_value;
560  char *return_value;
561  int expanded;
562 
563  return_value = line;
564  expanded = 0;
565 
566 # if defined (BANG_HISTORY)
567  /* History expand the line. If this results in no errors, then
568  add that line to the history if ADDIT is non-zero. */
569  if (!history_expansion_inhibited && history_expansion && history_expansion_p (line))
570  {
571  int old_len;
572 
573  /* If we are expanding the second or later line of a multi-line
574  command, decrease history_length so references to history expansions
575  in these lines refer to the previous history entry and not the
576  current command. */
577  old_len = history_length;
578  if (history_length > 0 && command_oriented_history && current_command_first_line_saved && current_command_line_count > 1)
579  history_length--;
580  expanded = history_expand (line, &history_value);
581  if (history_length >= 0 && command_oriented_history && current_command_first_line_saved && current_command_line_count > 1)
582  history_length = old_len;
583 
584  if (expanded)
585  {
586  if (print_changes)
587  {
588  if (expanded < 0)
589  internal_error ("%s", history_value);
590 #if defined (READLINE)
591  else if (hist_verify == 0 || expanded == 2)
592 #else
593  else
594 #endif
595  fprintf (stderr, "%s\n", history_value);
596  }
597 
598  /* If there was an error, return NULL. */
599  if (expanded < 0 || expanded == 2) /* 2 == print only */
600  {
601 # if defined (READLINE)
602  if (expanded == 2 && rl_dispatching == 0 && *history_value)
603 # else
604  if (expanded == 2 && *history_value)
605 # endif /* !READLINE */
606  maybe_add_history (history_value);
607 
608  free (history_value);
609 
610 # if defined (READLINE)
611  /* New hack. We can allow the user to edit the
612  failed history expansion. */
613  if (history_reediting && expanded < 0 && rl_done)
614  re_edit (line);
615 # endif /* READLINE */
616  return ((char *)NULL);
617  }
618 
619 # if defined (READLINE)
620  if (hist_verify && expanded == 1)
621  {
622  re_edit (history_value);
623  free (history_value);
624  return ((char *)NULL);
625  }
626 # endif
627  }
628 
629  /* Let other expansions know that return_value can be free'ed,
630  and that a line has been added to the history list. Note
631  that we only add lines that have something in them. */
632  expanded = 1;
633  return_value = history_value;
634  }
635 # endif /* BANG_HISTORY */
636 
637  if (addit && remember_on_history && *return_value)
638  maybe_add_history (return_value);
639 
640 #if 0
641  if (expanded == 0)
642  return_value = savestring (line);
643 #endif
644 
645  return (return_value);
646 }

648 /* Return 1 if the first non-whitespace character in LINE is a `#', indicating
649  that the line is a shell comment. Return 2 if there is a comment after the
650  first non-whitespace character. Return 0 if the line does not contain a
651  comment. */
652 static int
653 shell_comment (line)
654  char *line;
655 {
656  char *p;
657  int n;
658 
659  if (dstack.delimiter_depth != 0 || (parser_state & PST_HEREDOC))
660  return 0;
661  if (line == 0)
662  return 0;
663  for (p = line; p && *p && whitespace (*p); p++)
664  ;
665  if (p && *p == '#')
666  return 1;
667  n = skip_to_delim (line, p - line, "#", SD_NOJMP|SD_GLOB|SD_EXTGLOB|SD_COMPLETE);
668  return (line[n] == '#') ? 2 : 0;
669 }

688 /* Check LINE against what HISTCONTROL says to do. Returns 1 if the line
689  should be saved; 0 if it should be discarded. */
690 static int
691 check_history_control (line)
692  char *line;
693 {
694  HIST_ENTRY *temp;
695  int r;
696 
697  if (history_control == 0)
698  return 1;
699 
700  /* ignorespace or ignoreboth */
701  if ((history_control & HC_IGNSPACE) && *line == ' ')
702  return 0;
703 
704  /* ignoredups or ignoreboth */
705  if (history_control & HC_IGNDUPS)
706  {
707  using_history ();
708  temp = previous_history ();
709 
710  r = (temp == 0 || STREQ (temp->line, line) == 0);
711 
712  using_history ();
713 
714  if (r == 0)
715  return r;
716  }
717 
718  return 1;
719 }

721 /* Remove all entries matching LINE from the history list. Triggered when
722  HISTCONTROL includes `erasedups'. */
723 static void
724 hc_erasedups (line)
725  char *line;
726 {
727  HIST_ENTRY *temp;
728  int r;
729 
730  using_history ();
731  while (temp = previous_history ())
732  {
733  if (STREQ (temp->line, line))
734  {

866 /* Add a line to the history list.
867  The variable COMMAND_ORIENTED_HISTORY controls the style of history
868  remembering; when non-zero, and LINE is not the first line of a
869  complete parser construct, append LINE to the last history line instead
870  of adding it as a new line. */
871 void
872 bash_add_history (line)
873  char *line;
874 {
875  int add_it, offset, curlen, is_comment;
876  HIST_ENTRY *current, *old;
877  char *chars_to_add, *new_line;
878 
879  add_it = 1;
880  if (command_oriented_history && current_command_line_count > 1)
881  {
882  is_comment = shell_comment (line);
883 
884  /* The second and subsequent lines of a here document have the trailing
885  newline preserved. We don't want to add extra newlines here, but we
886  do want to add one after the first line (which is the command that
887  contains the here-doc specifier). parse.y:history_delimiting_chars()
888  does the right thing to take care of this for us. We don't want to
889  add extra newlines if the user chooses to enable literal_history,
890  so we have to duplicate some of what that function does here. */
891  /* If we're in a here document and past the first line,
892  (current_command_line_count > 2)
893  don't add a newline here. This will also take care of the literal_history
894  case if the other conditions are met. */
895  if ((parser_state & PST_HEREDOC) && here_doc_first_line == 0 && line[strlen (line) - 1] == '\n')
896  chars_to_add = "";
897  else if (current_command_line_count == current_command_line_comment+1)
898  chars_to_add = "\n";
899  else if (literal_history)
900  chars_to_add = "\n";
901  else
902  chars_to_add = history_delimiting_chars (line);
903 
904  using_history ();
905  current = previous_history ();
906 
907  current_command_line_comment = is_comment ? current_command_line_count : -2;
908 
909  if (current)
910  {
911  /* If the previous line ended with an escaped newline (escaped
912  with backslash, but otherwise unquoted), then remove the quoted
913  newline, since that is what happens when the line is parsed. */
914  curlen = strlen (current->line);
915 
916  if (dstack.delimiter_depth == 0 && current->line[curlen - 1] == '\\' &&
917  current->line[curlen - 2] != '\\')
918  {
919  current->line[curlen - 1] = '\0';
920  curlen--;
921  chars_to_add = "";
922  }
923 
924  /* If we're not in some kind of quoted construct, the current history
925  entry ends with a newline, and we're going to add a semicolon,
926  don't. In some cases, it results in a syntax error (e.g., before
927  a close brace), and it should not be needed. */
928  if (dstack.delimiter_depth == 0 && current->line[curlen - 1] == '\n' && *chars_to_add == ';')
929  chars_to_add++;
930 
931  new_line = (char *)xmalloc (1
932  + curlen
933  + strlen (line)
934  + strlen (chars_to_add));
935  sprintf (new_line, "%s%s%s", current->line, chars_to_add, line);
936  offset = where_history ();
937  old = replace_history_entry (offset, new_line, current->data);
938  free (new_line);
939 
940  if (old)
941  free_history_entry (old);
942 
943  add_it = 0;
944  }
945  }
946 
947  if (add_it && history_is_stifled() && history_length == 0 && history_length == history_max_entries)
948  add_it = 0;
949 
950  if (add_it)
951  really_add_history (line);
952 
953 #if defined (SYSLOG_HISTORY)
954  if (syslog_history)
955  bash_syslog_history (line);
956 #endif
957 
958  using_history ();
959 }
