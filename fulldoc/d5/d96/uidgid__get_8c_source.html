32 {

37 
38  user = (char*)ug;
39  group = strchr(ug, ':');
40  if (group) {
41  int sz = (++group) - ug;
42  user = alloca(sz);
43  /* copies sz-1 bytes, stores terminating '\0' */
44  safe_strncpy(user, ug, sz);
45  }
46  n = bb_strtou(user, NULL, 10);
47  if (!errno) {
48  u->uid = n;
49  pwd = getpwuid(n);
50  /* If we have e.g. "500" string without user */

52  u->gid = pwd ? pwd->pw_gid : n;
53  goto skip;
54  }
55  /* it is not numeric */
56  pwd = getpwnam(user);
57  if (!pwd)
58  return 0;

60  u->gid = pwd->pw_gid;
61 
62  skip:

64  n = bb_strtou(group, NULL, 10);
65  if (!errno) {
66  u->gid = n;
67  return 1;
68  }

70  if (!gr)
71  return 0;
