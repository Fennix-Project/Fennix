111 void ParseConfig(char *ConfigString, KernelConfig *ModConfig)
112 {
113  if (ConfigString == NULL)
114  {
115  KPrint("Empty kernel parameters!");
116  return;
117  }
118  else if (strlen(ConfigString) == 0)
119  {
120  KPrint("Empty kernel parameters!");
121  return;
122  }
123 
124  if (ModConfig == NULL)
125  {
126  KPrint("ModConfig is NULL!");
127  return;
128  }
129 
130  KPrint("Kernel parameters: %s", ConfigString);
131  debug("Kernel parameters: %s", ConfigString);
132 
133  char *argv[32];
134  int argc = 0;
135  /* Not sure if the quotes are being parsed correctly. */
136  targp_parse(ConfigString, argv, &argc);
137 
138 #ifdef DEBUG
139  for (int i = 0; i < argc; i++)
140  debug("argv[%d] = %s", i, argv[i]);
141  debug("argc = %d", argc);
142 #endif
143 
144  char identifier;
145  const char *value;
146  cag_option_context context;
147 
148  cag_option_prepare(&context, ConfigOptions, CAG_ARRAY_SIZE(ConfigOptions), argc, argv);
149 
150  while (cag_option_fetch(&context))
151  {
152  identifier = cag_option_get(&context);
153  switch (identifier)
154  {
155  case 'a':
156  {
157  value = cag_option_get_value(&context);
158  if (strcmp(value, "xallocv1") == 0)
159  {
160  KPrint("\eAAFFAAUsing XallocV1 as memory allocator");
161  ModConfig->AllocatorType = Memory::MemoryAllocatorType::XallocV1;
162  }
163  else if (strcmp(value, "liballoc11") == 0)
164  {
165  KPrint("\eAAFFAAUsing Liballoc11 as memory allocator");
166  ModConfig->AllocatorType = Memory::MemoryAllocatorType::liballoc11;
167  }
168  else if (strcmp(value, "pages") == 0)
169  {
170  KPrint("\eAAFFAAUsing Pages as memory allocator");
171  ModConfig->AllocatorType = Memory::MemoryAllocatorType::Pages;
172  }
173  else
174  {
175  KPrint("\eAAFFAAUnknown memory allocator: %s", value);
176  ModConfig->AllocatorType = Memory::MemoryAllocatorType::None;
177  }
178  break;
179  }
180  case 'c':
181  {
182  value = cag_option_get_value(&context);
183  KPrint("\eAAFFAAUsing %s cores", atoi(value) ? value : "all");
184  ModConfig->Cores = atoi(value);
185  break;
186  }
187  case 'p':
188  {
189  value = cag_option_get_value(&context);
190  KPrint("\eAAFFAARedirecting I/O APIC interrupts to %s%s", atoi(value) ? "core " : "", atoi(value) ? value : "BSP");
191  ModConfig->IOAPICInterruptCore = atoi(value);
192  break;
193  }
194  case 't':
195  {
196  value = cag_option_get_value(&context);
197  if (strcmp(value, "multi") == 0)
198  {
199  KPrint("\eAAFFAAUsing Multi-Tasking Scheduler");
200  ModConfig->SchedulerType = 1;
201  }
202  else if (strcmp(value, "single") == 0)
203  {
204  KPrint("\eAAFFAAUsing Single-Tasking Scheduler");
205  ModConfig->SchedulerType = 0;
206  }
207  else
208  {
209  KPrint("\eAAFFAAUnknown scheduler: %s", value);
210  ModConfig->SchedulerType = 1;
211  }
212  break;
213  }
214  case 'd':
215  {
216  value = cag_option_get_value(&context);
217  strncpy(ModConfig->DriverDirectory, value, strlen(value));
218  KPrint("\eAAFFAAUsing %s as driver directory", value);
219  break;
220  }
221  case 'i':
222  {
223  value = cag_option_get_value(&context);
224  strncpy(ModConfig->InitPath, value, strlen(value));
225  KPrint("\eAAFFAAUsing %s as init program", value);
226  break;
227  }
228  case 'o':
229  {
230  value = cag_option_get_value(&context);
231  strcmp(value, "true") == 0 ? ModConfig->InterruptsOnCrash = true : ModConfig->InterruptsOnCrash = false;
232  KPrint("\eAAFFAAInterrupts on crash: %s", value);
233  break;
234  }
235  case 'l':
236  {
237  value = cag_option_get_value(&context);
238  strcmp(value, "true") == 0 ? ModConfig->UnlockDeadLock = true : ModConfig->UnlockDeadLock = false;
239  KPrint("\eAAFFAAUnlocking the deadlock after 10 retries");
240  break;
241  }
242  case 's':
243  {
244  value = cag_option_get_value(&context);
245  strcmp(value, "true") == 0 ? ModConfig->SIMD = true : ModConfig->SIMD = false;
246  KPrint("\eAAFFAASingle Instruction, Multiple Data (SIMD): %s", value);
247  break;
248  }
249  case 'b':
250  {
251  value = cag_option_get_value(&context);
252  strcmp(value, "true") == 0 ? ModConfig->BootAnimation = true : ModConfig->BootAnimation = false;
253  KPrint("\eAAFFAABoot animation: %s", value);
254  break;
255  }
256  case 'h':
257  {
258  KPrint("\n---------------------------------------------------------------------------\nUsage: kernel.fsys [OPTION]...\nKernel configuration.");
259  cag_option_print(ConfigOptions, CAG_ARRAY_SIZE(ConfigOptions), nullptr);
260  KPrint("\eFF2200System Halted.");
261  CPU::Stop();
262  }
263  default:
264  {
265  KPrint("\eFF2200Unknown option: %c", identifier);
266  break;
267  }
268  }
269  }
270  debug("Config loaded");
271 }
