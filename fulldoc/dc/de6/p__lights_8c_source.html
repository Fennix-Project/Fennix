39 void T_FireFlicker (fireflicker_t* flick)
40 {
41  int amount;
42 
43  if (--flick->count)
44  return;
45 
46  amount = (P_Random()&3)*16;
47 
48  if (flick->sector->lightlevel - amount < flick->minlight)
49  flick->sector->lightlevel = flick->minlight;

51  flick->sector->lightlevel = flick->maxlight - amount;
52 
53  flick->count = 4;

307 void T_Glow(glow_t* g)

313  g->sector->lightlevel -= GLOWSPEED;
314  if (g->sector->lightlevel <= g->minlight)
315  {
316  g->sector->lightlevel += GLOWSPEED;
317  g->direction = 1;
318  }
319  break;
320 
321  case 1:
322  // UP
323  g->sector->lightlevel += GLOWSPEED;
324  if (g->sector->lightlevel >= g->maxlight)
325  {
326  g->sector->lightlevel -= GLOWSPEED;
327  g->direction = -1;
328  }
329  break;
330  }
331 }
