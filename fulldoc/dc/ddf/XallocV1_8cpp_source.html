147  void *V1::malloc(Xsize_t Size)
148  {
149  if (Size == 0)
150  {
151  Xalloc_warn("Attempted to allocate 0 bytes!");
152  return nullptr;
153  }
154 
155  SmartSMAP;
156  Xalloc_lock;
157 
158  if (this->FirstBlock == nullptr)
159  {
160  this->FirstBlock = new Block(Size);
161  ((Block *)this->FirstBlock)->IsFree = false;
162  Xalloc_unlock;
163  return ((Block *)this->FirstBlock)->Address;
164  }
165 
166  Block *CurrentBlock = ((Block *)this->FirstBlock);
167  while (CurrentBlock != nullptr)
168  {
169  if (!CurrentBlock->Check())
170  {
171  Xalloc_err("Block %#lx has an invalid checksum! (%#x != %#x)",
172  (Xuint64_t)CurrentBlock, CurrentBlock->Checksum, Xalloc_BlockChecksum);
173  while (Xalloc_StopOnFail)
174  ;
175  }
176  else if (CurrentBlock->IsFree && CurrentBlock->Size >= Size)
177  {
178  CurrentBlock->IsFree = false;
179  Xmemset(CurrentBlock->Address, 0, Size);
180  Xalloc_unlock;
181  return CurrentBlock->Address;
182  }
183  CurrentBlock = CurrentBlock->Next;
184  }
185 
186  CurrentBlock = ((Block *)this->FirstBlock);
187  while (CurrentBlock->Next != nullptr)
188  CurrentBlock = CurrentBlock->Next;
189 
190  CurrentBlock->Next = new Block(Size);
191  ((Block *)CurrentBlock->Next)->Last = CurrentBlock;
192  ((Block *)CurrentBlock->Next)->IsFree = false;
193  Xalloc_unlock;
194  return ((Block *)CurrentBlock->Next)->Address;
195  }

197  void V1::free(void *Address)
198  {
199  if (Address == nullptr)
200  {
201  Xalloc_warn("Attempted to free a null pointer!");
202  return;
203  }
204 
205  SmartSMAP;
206  Xalloc_lock;
207 
208  Block *CurrentBlock = ((Block *)this->FirstBlock);
209  while (CurrentBlock != nullptr)
210  {
211  if (!CurrentBlock->Check())
212  {
213  Xalloc_err("Block %#lx has an invalid checksum! (%#x != %#x)",
214  (Xuint64_t)CurrentBlock, CurrentBlock->Checksum, Xalloc_BlockChecksum);
215  while (Xalloc_StopOnFail)
216  ;
217  }
218  else if (CurrentBlock->Address == Address)
219  {
220  if (CurrentBlock->IsFree)
221  {
222  Xalloc_warn("Attempted to free an already freed pointer!");
223  Xalloc_unlock;
224  return;
225  }
226 
227  CurrentBlock->IsFree = true;
228  Xalloc_unlock;
229  return;
230  }
231  CurrentBlock = CurrentBlock->Next;
232  }
233 
234  Xalloc_err("Invalid address %#lx.", Address);
235  Xalloc_unlock;
236  }
