151 static int
152 find_matching_open (char *string, int from, int closer)
153 {
154  register int i;
155  int opener, level, delimiter;
156 
157  switch (closer)
158  {
159  case ']': opener = '['; break;
160  case '}': opener = '{'; break;
161  case ')': opener = '('; break;
162  default:
163  return (-1);
164  }
165 
166  level = 1; /* The closer passed in counts as 1. */
167  delimiter = 0; /* Delimited state unknown. */
168 
169  for (i = from; i > -1; i--)
170  {
171  if (delimiter && (string[i] == delimiter))
172  delimiter = 0;
173  else if (rl_basic_quote_characters && strchr (rl_basic_quote_characters, string[i]))
174  delimiter = string[i];
175  else if (!delimiter && (string[i] == closer))
176  level++;
177  else if (!delimiter && (string[i] == opener))
178  level--;
179 
180  if (!level)
181  break;
182  }
183  return (i);
184 }
