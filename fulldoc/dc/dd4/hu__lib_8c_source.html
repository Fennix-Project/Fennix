93 void
94 HUlib_drawTextLine
95 ( hu_textline_t* l,
96  boolean drawcursor )
97 {
98 
99  int i;
100  int w;
101  int x;
102  unsigned char c;
103 
104  // draw the new stuff
105  x = l->x;
106  for (i=0;i<l->len;i++)
107  {
108  c = toupper((int)l->l[i]);
109  if (c != ' '
110  && c >= l->sc
111  && c <= '_')
112  {
113  w = SHORT(l->f[c - l->sc]->width);
114  if (x+w > SCREENWIDTH)
115  break;
116  V_DrawPatchDirect(x, l->y, l->f[c - l->sc]);
117  x += w;
118  }
119  else
120  {
121  x += 4;
122  if (x >= SCREENWIDTH)
123  break;
124  }
125  }
126 
127  // draw the cursor if requested
128  if (drawcursor
129  && x + SHORT(l->f['_' - l->sc]->width) <= SCREENWIDTH)
130  {
131  V_DrawPatchDirect(x, l->y, l->f['_' - l->sc]);
132  }
133 }

136 // sorta called by HU_Erase and just better darn get things straight
137 void HUlib_eraseTextLine(hu_textline_t* l)
138 {
139  int lh;
140  int y;
141  int yoffset;
142 
143  // Only erases when NOT in automap and the screen is reduced,
144  // and the text must either need updating or refreshing
145  // (because of a recent change back from the automap)
146 
147  if (!automapactive &&
148  viewwindowx && l->needsupdate)
149  {
150  lh = SHORT(l->f[0]->height) + 1;
151  for (y=l->y,yoffset=y*SCREENWIDTH ; y<l->y+lh ; y++,yoffset+=SCREENWIDTH)
152  {
153  if (y < viewwindowy || y >= viewwindowy + viewheight)
154  R_VideoErase(yoffset, SCREENWIDTH); // erase entire line
155  else
156  {
157  R_VideoErase(yoffset, viewwindowx); // erase left border
158  R_VideoErase(yoffset + viewwindowx + viewwidth, viewwindowx);
159  // erase right border
160  }
161  }
162  }
163 
164  if (l->needsupdate) l->needsupdate--;
165 
166 }

310 HUlib_keyInIText
311 ( hu_itext_t* it,
312  unsigned char ch )
313 {
314  ch = toupper(ch);
315 
316  if (ch >= ' ' && ch <= '_')
317  HUlib_addCharToTextLine(&it->l, (char) ch);
318  else
319  if (ch == KEY_BACKSPACE)
320  HUlib_delCharFromIText(it);
321  else
322  if (ch != KEY_ENTER)
323  return false; // did not eat key
324 
325  return true; // ate the key
