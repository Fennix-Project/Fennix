50  int FileDescriptorTable::AddFileDescriptor(const char *AbsolutePath,
51  mode_t Mode, int Flags)
52  {
53  Tasking::PCB *pcb = thisProcess;
54 
55  auto ProbeMode = [](mode_t Mode, int Flags) -> int
56  {
57  if (!(Flags & O_CREAT))
58  return 0;
59 
60  if (Flags & O_RDONLY)
61  {
62  if (!(Mode & S_IRUSR))
63  {
64  debug("No read permission (%d)", Mode);
65  return -EACCES;
66  }
67  }
68 
69  if (Flags & O_WRONLY)
70  {
71  if (!(Mode & S_IWUSR))
72  {
73  debug("No write permission (%d)", Mode);
74  return -EACCES;
75  }
76  }
77 
78  if (Flags & O_RDWR)
79  {
80  if (!(Mode & S_IRUSR) ||
81  !(Mode & S_IWUSR))
82  {
83  debug("No read/write permission (%d)", Mode);
84  return -EACCES;
85  }
86  }
87 
88  return 0;
89  };
90 
91  if (ProbeMode(Mode, Flags) < 0)
92  return -EACCES;
93 
94  fixme("Do not follow symlinks when O_CREAT and O_EXCL are set");
95 
96  if (Flags & O_CREAT)
97  {
98  FileNode *ret = fs->Create(pcb->CWD, AbsolutePath, Mode);
99  if (Flags & O_EXCL && ret == nullptr)
100  {
101  debug("%s: File already exists?, returning EEXIST",
102  AbsolutePath);
103  return -EEXIST;
104  }
105  }
106 
107  if (Flags & O_CLOEXEC)
108  {
109  fixme("O_CLOEXEC");
110  }
111 
112  FileNode *File = fs->GetByPath(AbsolutePath, pcb->CWD);
113 
114  if (!File)
115  {
116  error("Failed to open file %s", AbsolutePath);
117  return -ENOENT;
118  }
119 
120  if (Flags & O_TRUNC)
121  {
122  debug("Truncating file %s", AbsolutePath);
123  File->Truncate(0);
124  }
125 
126  Fildes fd{};
127 
128  if (Flags & O_APPEND)
129  {
130  debug("Appending to file %s", AbsolutePath);
131  struct kstat stat;
132  File->Stat(&stat);
133  fd.Offset = File->Seek(stat.Size);
134  }
135 
136  fd.Mode = Mode;
137  fd.Flags = Flags;
138  fd.Node = File;
139 
140  int fdn = this->GetFreeFileDescriptor();
141  if (fdn < 0)
142  return fdn;
143 
144  this->FileMap.insert({fdn, fd});
145 
146  char linkName[64];
147  snprintf(linkName, 64, "%d", fdn);
148  assert(fs->CreateLink(linkName, this->fdDir, AbsolutePath) != nullptr);
149 
150  File->Open(Flags, Mode);
151  return fdn;
152  }

233  off_t FileDescriptorTable::usr_lseek(int fd, off_t offset, int whence)
234  {
235  auto it = this->FileMap.find(fd);
236  if (it == this->FileMap.end())
237  ReturnLogError(-EBADF, "Invalid fd %d", fd);
238 
239  off_t &newOffset = it->second.Offset;
240 
241  switch (whence)
242  {
243  case SEEK_SET:
244  {
245  newOffset = it->second.Node->Seek(offset);
246  break;
247  }
248  case SEEK_CUR:
249  {
250  newOffset = it->second.Node->Seek(newOffset + offset);
251  break;
252  }
253  case SEEK_END:
254  {
255  struct kstat stat
256  {
257  };
258  it->second.Node->Stat(&stat);
259  newOffset = it->second.Node->Seek(stat.Size + offset);
260  break;
261  }
262  default:
263  return -EINVAL;
264  }
265 
266  return newOffset;
267  }
