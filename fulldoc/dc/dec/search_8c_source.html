81 /* Make the data from the history entry ENTRY be the contents of the
82  current line. This doesn't do anything with rl_point; the caller
83  must set it. */
84 static void
85 make_history_line_current (HIST_ENTRY *entry)
86 {
87  UNDO_LIST *xlist;
88 
89  xlist = _rl_saved_line_for_history ? (UNDO_LIST *)_rl_saved_line_for_history->data : 0;
90  /* At this point, rl_undo_list points to a private search string list. */
91  if (rl_undo_list && rl_undo_list != (UNDO_LIST *)entry->data && rl_undo_list != xlist)
92  rl_free_undo_list ();
93 
94  /* Now we create a new undo list with a single insert for this text.
95  WE DON'T CHANGE THE ORIGINAL HISTORY ENTRY UNDO LIST */
96  _rl_replace_text (entry->line, 0, rl_end);
97  _rl_fix_point (1);
98 #if defined (VI_MODE)
99  if (rl_editing_mode == vi_mode)
100  /* POSIX.2 says that the `U' command doesn't affect the copy of any
101  command lines to the edit line. We're going to implement that by
102  making the undo list start after the matching line is copied to the
103  current editing buffer. */
104  rl_free_undo_list ();
105 #endif
106 
107  /* This will need to free the saved undo list associated with the original
108  (pre-search) line buffer.
109  XXX - look at _rl_free_saved_history_line and consider calling it if
110  rl_undo_list != xlist (or calling rl_free_undo list directly on
111  _rl_saved_line_for_history->data) */
112  if (_rl_saved_line_for_history)
113  _rl_free_history_entry (_rl_saved_line_for_history);
114  _rl_saved_line_for_history = (HIST_ENTRY *)NULL;
115 }

117 /* Search the history list for STRING starting at absolute history position
118  POS. If STRING begins with `^', the search must match STRING at the
119  beginning of a history line, otherwise a full substring match is performed
120  for STRING. DIR < 0 means to search backwards through the history list,
121  DIR >= 0 means to search forward. */
122 static int
123 noninc_search_from_pos (char *string, int pos, int dir, int flags, int *ncp)
124 {
125  int ret, old, sflags;
126  char *s;
127 
128  if (pos < 0)
129  return -1;
130 
131  old = where_history ();
132  if (history_set_pos (pos) == 0)
133  return -1;
134 
135  RL_SETSTATE(RL_STATE_SEARCH);
136  /* These functions return the match offset in the line; history_offset gives
137  the matching line in the history list */
138  if (flags & SF_PATTERN)
139  {
140  s = string;
141  sflags = 0; /* Non-anchored search */
142  if (*s == '^')
143  {
144  sflags |= ANCHORED_SEARCH;
145  s++;
146  }
147  ret = _hs_history_patsearch (s, dir, sflags);
148  }
149  else if (*string == '^')
150  ret = history_search_prefix (string + 1, dir);
151  else
152  ret = history_search (string, dir);
153  RL_UNSETSTATE(RL_STATE_SEARCH);
154 
155  if (ncp)
156  *ncp = ret; /* caller will catch -1 to indicate no-op */
157 
158  if (ret != -1)
159  ret = where_history ();
160 
161  history_set_pos (old);
162  return (ret);
163 }

165 /* Search for a line in the history containing STRING. If DIR is < 0, the
166  search is backwards through previous entries, else through subsequent
167  entries. Returns 1 if the search was successful, 0 otherwise. */
168 static int
169 noninc_dosearch (char *string, int dir, int flags)
170 {
171  int oldpos, pos, ind;
172  HIST_ENTRY *entry;
173 
174  if (string == 0 || *string == '\0' || noninc_history_pos < 0)
175  {
176  rl_ding ();
177  return 0;
178  }
179 
180  pos = noninc_search_from_pos (string, noninc_history_pos + dir, dir, flags, &ind);
181  if (pos == -1)
182  {
183  /* Search failed, current history position unchanged. */
184  rl_maybe_unsave_line ();
185  rl_clear_message ();
186  rl_point = 0;
187  rl_ding ();
188  return 0;
189  }
190 
191  noninc_history_pos = pos;
192 
193  oldpos = where_history ();
194  history_set_pos (noninc_history_pos);
195  entry = current_history (); /* will never be NULL after successful search */
196 
197 #if defined (VI_MODE)
198  if (rl_editing_mode != vi_mode)
199 #endif
200  history_set_pos (oldpos);
201 
202  make_history_line_current (entry);
203 
204  if (_rl_enable_active_region && ((flags & SF_PATTERN) == 0) && ind > 0 && ind < rl_end)
205  {
206  rl_point = ind;
207  rl_mark = ind + strlen (string);
208  if (rl_mark > rl_end)
209  rl_mark = rl_end; /* can't happen? */
210  rl_activate_mark ();
211  }
212  else
213  {
214  rl_point = 0;
215  rl_mark = rl_end;
216  }
217 
218  rl_clear_message ();
219  return 1;
220 }

285 /* Process just-read character C according to search context CXT. Return -1
286  if the caller should abort the search, 0 if we should break out of the
287  loop, and 1 if we should continue to read characters. */
288 static int
289 _rl_nsearch_dispatch (_rl_search_cxt *cxt, int c)
290 {
291  int n;
292 
293  if (c < 0)
294  c = CTRL ('C');
295 
296  switch (c)
297  {
298  case CTRL('W'):
299  rl_unix_word_rubout (1, c);
300  break;
301 
302  case CTRL('U'):
303  rl_unix_line_discard (1, c);
304  break;
305 
306  case RETURN:
307  case NEWLINE:
308  return 0;
309 
310  case CTRL('H'):
311  case RUBOUT:
312  if (rl_point == 0)
313  {
314  _rl_nsearch_abort (cxt);
315  return -1;
316  }
317  _rl_rubout_char (1, c);
318  break;
319 
320  case CTRL('C'):
321  case CTRL('G'):
322  rl_ding ();
323  _rl_nsearch_abort (cxt);
324  return -1;
325 
326  case ESC:
327  /* XXX - experimental code to allow users to bracketed-paste into the
328  search string. Similar code is in isearch.c:_rl_isearch_dispatch().
329  The difference here is that the bracketed paste sometimes doesn't
330  paste everything, so checking for the prefix and the suffix in the
331  input queue doesn't work well. We just have to check to see if the
332  number of chars in the input queue is enough for the bracketed paste
333  prefix and hope for the best. */
334  if (_rl_enable_bracketed_paste && ((n = _rl_nchars_available ()) >= (BRACK_PASTE_SLEN-1)))
335  {
336  if (_rl_read_bracketed_paste_prefix (c) == 1)
337  rl_bracketed_paste_begin (1, c);
338  else
339  {
340  c = rl_read_key (); /* get the ESC that got pushed back */
341  _rl_insert_char (1, c);
342  }
343  }
344  else
345  _rl_insert_char (1, c);
346  break;
347 
348  default:
349 #if defined (HANDLE_MULTIBYTE)
350  if (MB_CUR_MAX > 1 && rl_byte_oriented == 0)
351  rl_insert_text (cxt->mb);
352  else
353 #endif
354  _rl_insert_char (1, c);
355  break;
356  }
357 
358  (*rl_redisplay_function) ();
359  rl_deactivate_mark ();
360  return 1;
361 }

363 /* Perform one search according to CXT, using NONINC_SEARCH_STRING. Return
364  -1 if the search should be aborted, any other value means to clean up
365  using _rl_nsearch_cleanup (). Returns 1 if the search was successful,
366  0 otherwise. */
367 static int
368 _rl_nsearch_dosearch (_rl_search_cxt *cxt)
369 {
370  rl_mark = cxt->save_mark;
371 
372  /* If rl_point == 0, we want to re-use the previous search string and
373  start from the saved history position. If there's no previous search
374  string, punt. */
375  if (rl_point == 0)
376  {
377  if (noninc_search_string == 0)
378  {
379  rl_ding ();
380  rl_restore_prompt ();
381  RL_UNSETSTATE (RL_STATE_NSEARCH);
382  return -1;
383  }
384  }
385  else
386  {
387  /* We want to start the search from the current history position. */
388  noninc_history_pos = cxt->save_line;
389  FREE (noninc_search_string);
390  noninc_search_string = savestring (rl_line_buffer);
391 
392  /* If we don't want the subsequent undo list generated by the search
393  matching a history line to include the contents of the search string,
394  we need to clear rl_line_buffer here. For now, we just clear the
395  undo list generated by reading the search string. (If the search
396  fails, the old undo list will be restored by rl_maybe_unsave_line.) */
397  rl_free_undo_list ();
398  }
399 
400  rl_restore_prompt ();
401  return (noninc_dosearch (noninc_search_string, cxt->direction, cxt->sflags&SF_PATTERN));
402 }

404 /* Search non-interactively through the history list. DIR < 0 means to
405  search backwards through the history of previous commands; otherwise
406  the search is for commands subsequent to the current position in the
407  history list. PCHAR is the character to use for prompting when reading
408  the search string; if not specified (0), it defaults to `:'. */
409 static int
410 noninc_search (int dir, int pchar)
411 {
412  _rl_search_cxt *cxt;
413  int c, r;
414 
415  cxt = _rl_nsearch_init (dir, pchar);
416 
417  if (RL_ISSTATE (RL_STATE_CALLBACK))
418  return (0);
419 
420  /* Read the search string. */
421  r = 0;
422  while (1)
423  {
424  c = _rl_search_getchar (cxt);
425 
426  if (c < 0)
427  {
428  _rl_nsearch_abort (cxt);
429  return 1;
430  }
431 
432  if (c == 0)
433  break;
434 
435  r = _rl_nsearch_dispatch (cxt, c);
436  if (r < 0)
437  return 1;
438  else if (r == 0)
439  break;
440  }
441 
442  r = _rl_nsearch_dosearch (cxt);
443  return ((r >= 0) ? _rl_nsearch_cleanup (cxt, r) : (r != 1));
444 }

528 static int
529 rl_history_search_internal (int count, int dir)
530 {
531  HIST_ENTRY *temp;
532  int ret, oldpos, newcol;
533  int had_saved_line;
534  char *t;
535 
536  had_saved_line = _rl_saved_line_for_history != 0;
537  rl_maybe_save_line ();
538  temp = (HIST_ENTRY *)NULL;
539 
540  /* Search COUNT times through the history for a line matching
541  history_search_string. If history_search_string[0] == '^', the
542  line must match from the start; otherwise any substring can match.
543  When this loop finishes, TEMP, if non-null, is the history line to
544  copy into the line buffer. */
545  while (count)
546  {
547  RL_CHECK_SIGNALS ();
548  ret = noninc_search_from_pos (history_search_string, _rl_history_search_pos + dir, dir, 0, &newcol);
549  if (ret == -1)
550  break;
551 
552  /* Get the history entry we found. */
553  _rl_history_search_pos = ret;
554  oldpos = where_history ();
555  history_set_pos (_rl_history_search_pos);
556  temp = current_history (); /* will never be NULL after successful search */
557  history_set_pos (oldpos);
558 
559  /* Don't find multiple instances of the same line. */
560  if (prev_line_found && STREQ (prev_line_found, temp->line))
561  continue;
562  prev_line_found = temp->line;
563  count--;
564  }
565 
566  /* If we didn't find anything at all, return. */
567  if (temp == 0)
568  {
569  /* XXX - check had_saved_line here? */
570  rl_maybe_unsave_line ();
571  rl_ding ();
572  /* If you don't want the saved history line (last match) to show up
573  in the line buffer after the search fails, change the #if 0 to
574  #if 1 */
575 #if 0
576  if (rl_point > _rl_history_search_len)
577  {
578  rl_point = rl_end = _rl_history_search_len;
579  rl_line_buffer[rl_end] = '\0';
580  rl_mark = 0;
581  }
582 #else
583  rl_point = _rl_history_search_len; /* rl_maybe_unsave_line changes it */
584  rl_mark = rl_end;
585 #endif
586  return 1;
587  }
588 
589  /* Copy the line we found into the current line buffer. */
590  make_history_line_current (temp);
591 
592  /* decide where to put rl_point -- need to change this for pattern search */
593  if (_rl_history_search_flags & ANCHORED_SEARCH)
594  rl_point = _rl_history_search_len; /* easy case */
595  else
596  {
597 #if 0
598  t = strstr (rl_line_buffer, history_search_string); /* XXX */
599  rl_point = t ? (int)(t - rl_line_buffer) + _rl_history_search_len : rl_end;
600 #else
601  rl_point = (newcol >= 0) ? newcol : rl_end;
602 #endif
603  }
604  rl_mark = rl_end;
605 
606  return 0;
607 }

609 static void
610 rl_history_search_reinit (int flags)
611 {
612  int sind;
613 
614  _rl_history_search_pos = where_history ();
615  _rl_history_search_len = rl_point;
616  _rl_history_search_flags = flags;
617 
618  prev_line_found = (char *)NULL;
619  if (rl_point)
620  {
621  /* Allocate enough space for anchored and non-anchored searches */
622  if (_rl_history_search_len >= history_string_size - 2)
623  {
624  history_string_size = _rl_history_search_len + 2;
625  history_search_string = (char *)xrealloc (history_search_string, history_string_size);
626  }
627  sind = 0;
628  if (flags & ANCHORED_SEARCH)
629  history_search_string[sind++] = '^';
630  strncpy (history_search_string + sind, rl_line_buffer, rl_point);
631  history_search_string[rl_point + sind] = '\0';
632  }
633  _rl_free_saved_history_line (); /* XXX rl_undo_list? */
634 }

637  from the start of the line to rl_point. This is a non-incremental
638  search. The search is anchored to the beginning of the history line. */
639 int
640 rl_history_search_forward (int count, int ignore)

651  return (rl_history_search_internal (abs (count), (count > 0) ? 1 : -1));
