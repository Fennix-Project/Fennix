60  int __fs_Lookup(struct Inode *_Parent, const char *Name, struct Inode **Result)
61  {
62  func("%#lx %s %#lx", _Parent, Name, Result);
63 
64  assert(_Parent != nullptr);
65 
66  const char *basename;
67  size_t length;
68  cwk_path_get_basename(Name, &basename, &length);
69  if (basename == NULL)
70  {
71  error("Invalid name %s", Name);
72  return -EINVAL;
73  }
74 
75  auto Parent = (Manager::DeviceInode *)_Parent;
76  for (const auto &child : Parent->Children)
77  {
78  debug("Comparing %s with %s", basename, child->Name.c_str());
79  if (strcmp(child->Name.c_str(), basename) != 0)
80  continue;
81 
82  *Result = &child->Node;
83  return 0;
84  }
85 
86  debug("Not found %s", Name);
87  return -ENOENT;
88  }

90  int __fs_Create(struct Inode *_Parent, const char *Name, mode_t Mode, struct Inode **Result)
91  {
92  func("%#lx %s %d", _Parent, Name, Mode);
93 
94  assert(_Parent != nullptr);
95 
96  /* We expect to be /dev or children of it */
97  auto Parent = (Manager::DeviceInode *)_Parent;
98  auto _dev = new Manager::DeviceInode;
99  _dev->Parent = nullptr;
100  _dev->ParentInode = _Parent;
101  _dev->Name = Name;
102  _dev->Node.Mode = Mode;
103  _dev->Node.Index = Parent->Node.Index + Parent->Children.size();
104  Parent->Children.push_back(_dev);
105 
106  *Result = &_dev->Node;
107  return 0;
108  }

110  ssize_t __fs_Read(struct Inode *Node, void *Buffer, size_t Size, off_t Offset)
111  {
112  func("%#lx %d %d", Node, Size, Offset);
113  switch (Node->GetMajor())
114  {
115  case 0:
116  {
117  switch (Node->GetMinor())
118  {
119  case 2: /* /dev/null */
120  {
121  return 0;
122  }
123  case 3: /* /dev/zero */
124  {
125  if (Size <= 0)
126  return 0;
127 
128  memset(Buffer, 0, Size);
129  return Size;
130  }
131  case 4: /* /dev/random */
132  {
133  if (Size <= 0)
134  return 0;
135 
136  if (Size < sizeof(uint64_t))
137  {
138  uint8_t *buf = (uint8_t *)Buffer;
139  for (size_t i = 0; i < Size; i++)
140  buf[i] = (uint8_t)(Random::rand16() & 0xFF);
141  return Size;
142  }
143 
144  uint64_t *buf = (uint64_t *)Buffer;
145  for (size_t i = 0; i < Size / sizeof(uint64_t); i++)
146  buf[i] = Random::rand64();
147  return Size;
148  }
149  case 5: /* /dev/mem */
150  {
151  stub;
152  return 0;
153  }
154  case 6: /* /dev/kcon */
155  return KernelConsole::CurrentTerminal.load()->Read(Buffer, Size, Offset);
156  case 7: /* /dev/tty */
157  {
158  TTY::TeletypeDriver *tty = (TTY::TeletypeDriver *)thisProcess->tty;
159  if (tty == nullptr)
160  return -ENOTTY;
161  return tty->Read(Buffer, Size, Offset);
162  }
163  case 8: /* /dev/ptmx */
164  return -ENOSYS;
165  default:
166  return -ENOENT;
167  };
168  break;
169  }
170  case 1:
171  {
172  switch (Node->GetMinor())
173  {
174  case 0: /* /dev/input/keyboard */
175  {
176  if (Size < sizeof(KeyboardReport))
177  return -EINVAL;
178 
179  size_t nReads = Size / sizeof(KeyboardReport);
180 
181  KeyboardReport *report = (KeyboardReport *)Buffer;
182 
183  while (DriverManager->GlobalKeyboardInputReports.Count() == 0)
184  TaskManager->Yield();
185 
186  DriverManager->GlobalKeyboardInputReports.Read(report, nReads);
187  return sizeof(KeyboardReport) * nReads;
188  }
189  case 1: /* /dev/input/mouse */
190  {
191  if (Size < sizeof(MouseReport))
192  return -EINVAL;
193 
194  size_t nReads = Size / sizeof(MouseReport);
195 
196  MouseReport *report = (MouseReport *)Buffer;
197 
198  while (DriverManager->GlobalMouseInputReports.Count() == 0)
199  TaskManager->Yield();
200 
201  DriverManager->GlobalMouseInputReports.Read(report, nReads);
202  return sizeof(MouseReport) * nReads;
203  }
204  default:
205  return -ENOENT;
206  };
207  }
208  default:
209  {
210  std::unordered_map<dev_t, Driver::DriverObject> &drivers =
211  DriverManager->GetDrivers();
212  const auto it = drivers.find(Node->GetMajor());
213  if (it == drivers.end())
214  ReturnLogError(-EINVAL, "Driver %d not found", Node->GetMajor());
215  const Driver::DriverObject *drv = &it->second;
216 
217  auto dop = drv->DeviceOperations;
218  auto dOps = dop->find(Node->GetMinor());
219  if (dOps == dop->end())
220  ReturnLogError(-EINVAL, "Device %d not found", Node->GetMinor());
221  AssertReturnError(dOps->second.Ops, -ENOTSUP);
222  AssertReturnError(dOps->second.Ops->Read, -ENOTSUP);
223  return dOps->second.Ops->Read(Node, Buffer, Size, Offset);
224  }
225  }
226  }

490  __no_sanitize("alignment")
491  ssize_t __fs_Readdir(struct Inode *_Node, struct kdirent *Buffer, size_t Size, off_t Offset, off_t Entries)
492  {
493  func("%#lx %#lx %d %d %d", _Node, Buffer, Size, Offset, Entries);
494 
495  auto Node = (Manager::DeviceInode *)_Node;
496 
497  off_t realOffset = Offset;
498 
499  size_t totalSize = 0;
500  uint16_t reclen = 0;
501  struct kdirent *ent = nullptr;
502 
503  if (Offset == 0)
504  {
505  reclen = (uint16_t)(offsetof(struct kdirent, d_name) + strlen(".") + 1);
506  if (totalSize + reclen >= Size)
507  return -EINVAL;
508 
509  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
510  ent->d_ino = Node->Node.Index;
511  ent->d_off = Offset++;
512  ent->d_reclen = reclen;
513  ent->d_type = DT_DIR;
514  strcpy(ent->d_name, ".");
515  totalSize += reclen;
516  }
517 
518  if (Offset <= 1)
519  {
520  reclen = (uint16_t)(offsetof(struct kdirent, d_name) + strlen("..") + 1);
521  if (totalSize + reclen >= Size)
522  {
523  if (realOffset == 1)
524  return -EINVAL;
525  return totalSize;
526  }
527 
528  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
529 
530  if (Node->Parent)
531  ent->d_ino = Node->Parent->Node->Index;
532  else if (Node->ParentInode)
533  ent->d_ino = Node->ParentInode->Index;
534  else
535  {
536  warn("Parent is null for %s", Node->Name.c_str());
537  ent->d_ino = Node->Node.Index;
538  }
539  ent->d_off = Offset++;
540  ent->d_reclen = reclen;
541  ent->d_type = DT_DIR;
542  strcpy(ent->d_name, "..");
543  totalSize += reclen;
544  }
545 
546  if (!S_ISDIR(Node->Node.Mode))
547  return -ENOTDIR;
548 
549  if ((Offset >= 2 ? (Offset - 2) : Offset) > (off_t)Node->Children.size())
550  return -EINVAL;
551 
552  off_t entries = 0;
553  for (const auto &var : Node->Children)
554  {
555  debug("iterating \"%s\" inside \"%s\"", var->Name.c_str(), Node->Name.c_str());
556  if (var->Node.Offset < realOffset)
557  {
558  debug("skipping \"%s\" (%d < %d)", var->Name.c_str(), var->Node.Offset, Offset);
559  continue;
560  }
561 
562  if (entries >= Entries)
563  break;
564 
565  reclen = (uint16_t)(offsetof(struct kdirent, d_name) + strlen(var->Name.c_str()) + 1);
566 
567  if (totalSize + reclen >= Size)
568  break;
569 
570  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
571  ent->d_ino = var->Node.Index;
572  ent->d_off = var->Node.Offset;
573  ent->d_reclen = reclen;
574  ent->d_type = IFTODT(var->Node.Mode);
575  strncpy(ent->d_name, var->Name.c_str(), strlen(var->Name.c_str()));
576 
577  totalSize += reclen;
578  entries++;
579  }
580 
581  if (totalSize + sizeof(struct kdirent) >= Size)
582  return totalSize;
583 
584  ent = (struct kdirent *)((uintptr_t)Buffer + totalSize);
585  ent->d_ino = 0;
586  ent->d_off = 0;
587  ent->d_reclen = 0;
588  ent->d_type = DT_UNKNOWN;
589  ent->d_name[0] = '\0';
590  return totalSize;
591  }

603  dev_t Manager::RegisterInputDevice(std::unordered_map<dev_t, DriverHandlers> *dop,
604  dev_t DriverID, size_t i, const InodeOperations *Operations)
605  {
606  std::string prefix = "event";
607  for (size_t j = 0; j < 128; j++)
608  {
609  std::string deviceName = prefix + std::to_string(j);
610  FileNode *node = fs->GetByPath(deviceName.c_str(), devInputNode);
611  if (node)
612  continue;
613 
614  /* c rwx r-- r-- */
615  mode_t mode = S_IRWXU |
616  S_IRGRP |
617  S_IROTH |
618  S_IFCHR;
619 
620  node = fs->ForceCreate(devInputNode, deviceName.c_str(), mode);
621  node->Node->SetDevice(DriverID, i);
622 
623  DriverHandlers dh{};
624  dh.Ops = Operations;
625  dh.Node = node->Node;
626  dh.InputReports = new RingBuffer<InputReport>(16);
627  dop->insert({i, std::move(dh)});
628  return i;
629  }
630 
631  ReturnLogError(-1, "No available slots for device %d", DriverID);
632  return -1; /* -Werror=return-type */
633  }

667  dev_t Manager::RegisterDevice(dev_t DriverID, DeviceType Type, const InodeOperations *Operations)
668  {
669  std::unordered_map<dev_t, Driver::DriverObject> &drivers =
670  DriverManager->GetDrivers();
671  const auto it = drivers.find(DriverID);
672  if (it == drivers.end())
673  ReturnLogError(-EINVAL, "Driver %d not found", DriverID);
674  const Driver::DriverObject *drv = &it->second;
675 
676  auto dop = drv->DeviceOperations;
677  for (size_t i = 0; i < 128; i++)
678  {
679  const auto dOps = dop->find(i);
680  const auto dOpsEnd = dop->end();
681  if (dOps != dOpsEnd)
682  continue;
683 
684  DeviceType devType = (DeviceType)(Type & DEVICE_TYPE_MASK);
685  switch (devType)
686  {
687  case DEVICE_TYPE_INPUT:
688  return RegisterInputDevice(dop, DriverID, i, Operations);
689  case DEVICE_TYPE_BLOCK:
690  return RegisterBlockDevice(dop, DriverID, i, Operations);
691  default:
692  ReturnLogError(-1, "Invalid device type %d", Type);
693  }
694  }
695 
696  ReturnLogError(-1, "No available slots for device %d", DriverID);
697  }
