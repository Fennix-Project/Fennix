164 bool IsE0 = false;

166 void PS2KbdInterruptHandler(TrapFrame *)
167 {
168  uint8_t sc = inb(PS2_DATA);
169  if (sc == PS2_KBD_RESP_ACK ||
170  sc == PS2_KBD_RESP_ECHO ||
171  sc == PS2_KBD_RESP_RESEND)
172  return;
173 
174  if (sc == 0xE0)
175  {
176  IsE0 = true;
177  return;
178  }
179 
180  if (sc == 0xE1)
181  {
182  IsE1 = true;
183  return;
184  }
185 
186  switch (KeyboardScanCodeSet)
187  {
188  case PS2_KBD_SC_SET_1:
189  case PS2_KBD_SC_SET_2:
190  {
191  if (IsE0)
192  {
193  IsE0 = false;
194  ReportKeyboardEvent(KeyboardDevID, ScanCodeSet1mm[sc], sc < 0x90);
195  return;
196  }
197  else
198  {
199  bool released = sc & 0x80;
200  uint8_t scFinal = released ? sc & 0x7F : sc;
201  ReportKeyboardEvent(KeyboardDevID, ScanCodeSet1[scFinal], !released);
202  return;
203  }
204  }
205  /* FIXME: https://wiki.osdev.org/PS/2_Keyboard */
206  // case PS2_KBD_SC_SET_2:
207  // {
208  // break;
209  // }
210  case PS2_KBD_SC_SET_3:
211  {
212  ReportKeyboardEvent(KeyboardDevID, ScanCodeSet3[sc], true);
213  ReportKeyboardEvent(KeyboardDevID, ScanCodeSet3[sc], false);
214  break;
215  }
216  default:
217  {
218  if (IsE0)
219  IsE0 = false;
220  Log("Unknown PS/2 Keyboard Scan Code Set: %#x", KeyboardScanCodeSet);
221  break;
222  }
223  }
224 }

226 int InitializeKeyboard()
227 {
228  // PS2WriteData(PS2_KBD_CMD_RESET);
229  // uint8_t test = PS2ReadData();

234  // return -EFAULT;

236 
237  PS2WriteData(PS2_KBD_CMD_DEFAULTS);
238  if (PS2ACKTimeout() != 0)
239  Log("PS/2 keyboard failed to set defaults");
240 
241  PS2WriteData(PS2_KBD_CMD_SCAN_CODE_SET);
242  if (PS2ACKTimeout() != 0)
243  Log("PS/2 keyboard failed to set scan code set");
244 
245  /* We want Scan Code Set 1 */
246  PS2WriteData(PS2_KBD_SCAN_CODE_SET_2); /* It will set to 1 but with translation? */
247  if (PS2ACKTimeout() != 0)
248  Log("PS/2 keyboard failed to set scan code set 2");
249 
250  PS2WriteData(PS2_KBD_CMD_SCAN_CODE_SET);
251  if (PS2ACKTimeout() != 0)
252  Log("PS/2 keyboard failed to set scan code set");
253 
254  PS2WriteData(PS2_KBD_SCAN_CODE_GET_CURRENT);
255  if (PS2ACKTimeout() != 0)
256  Log("PS/2 keyboard failed to get current scan code set");
257 
258  KeyboardScanCodeSet = PS2ReadAfterACK();
259  Log("PS/2 Keyboard Scan Code Set: 0x%X", KeyboardScanCodeSet);
260  PS2ClearOutputBuffer();
261 
262  PS2WriteData(PS2_KBD_CMD_ENABLE_SCANNING);
263 
264  RegisterInterruptHandler(1, PS2KbdInterruptHandler);
265  KeyboardDevID = RegisterInputDevice(ddt_Keyboard);
266  return 0;
267 }

275 int DetectPS2Keyboard()
276 {
277  PS2WriteData(PS2_KBD_CMD_DISABLE_SCANNING);
278  if (PS2ACKTimeout() != 0)
279  Log("PS/2 keyboard failed to disable scanning");
280 
281  PS2WriteData(PS2_KBD_CMD_IDENTIFY);
282  if (PS2ACKTimeout() != 0)
283  Log("PS/2 keyboard failed to identify");
284 
285  uint8_t recByte;
286  int timeout = 1000000;
287  while (timeout--)
288  {
289  recByte = PS2ReadData();
290  if (recByte != PS2_ACK)
291  break;
292  }
293  Device1ID[0] = recByte;
294 
295  timeout = 1000000;
296  while (timeout--)
297  {
298  recByte = PS2ReadData();
299  if (recByte != PS2_ACK)
300  break;
301  }
302  if (timeout == 0)
303  Log("PS/2 keyboard second byte timed out");
304  else
305  Device1ID[1] = recByte;
306 
307  Log("PS2 Keyboard Device: 0x%X 0x%X", Device1ID[0], Device1ID[1]);
308  return 0;
