172  thisProcess->Security.ProcessGroupID = *((pid_t *)Argp);

205  {

207  if (Lines == 0)

252  void VirtualTerminal::Append(char c)
253  {
254  if (c == '\n')
255  this->NewLine();
256  else if (c == '\r')
257  {
258  this->Cursor.X = 0;
259  if (this->CursorCB != nullptr)
260  this->CursorCB(&this->Cursor);
261  }
262  else if (c == '\t')
263  {
264  int n = 8 - (this->Cursor.X % 8);
265 
266  for (int i = 0; i < n; i++)
267  this->Append(' ');
268  }
269  else if (c == '\b')
270  {
271  if (this->Cursor.X > 0)
272  {
273  this->Cursor.X--;
274  }
275  else
276  {
277  this->Cursor.Y--;
278  this->Cursor.X = this->TerminalSize.ws_row - 1;
279  }
280 
281  if (this->CursorCB != nullptr)
282  this->CursorCB(&this->Cursor);
283  }
284  else
285  {
286  if (this->Cursor.X >= this->TerminalSize.ws_row)
287  this->NewLine();
288 
289  TerminalCell *cell = &this->Cells[this->Cursor.X + this->Cursor.Y * this->TerminalSize.ws_row];
290  cell->c = c;
291  cell->attr = this->Attribute;
292 
293  assert(this->PaintCB != nullptr);
294  this->PaintCB(cell, this->Cursor.X, this->Cursor.Y);
295 
296  this->Cursor.X++;
297 
298  if (this->CursorCB != nullptr)
299  this->CursorCB(&this->Cursor);
300  }

457  void VirtualTerminal::Process(char c)
458  {
459 #ifdef DEBUG
460  static int once = 0;
461  static uint8_t com4 = 0xFF;
462  if (!once++)
463  {
464  com4 = inb(0x2E8);
465  debug("COM4 is available");
466  outb(s_cst(uint16_t, 0x2E8 + 1), 0x00);
467  outb(s_cst(uint16_t, 0x2E8 + 3), 0x80);
468  outb(s_cst(uint16_t, 0x2E8 + 0), 0x01);
469  outb(s_cst(uint16_t, 0x2E8 + 1), 0x00);
470  outb(s_cst(uint16_t, 0x2E8 + 3), 0x03);
471  outb(s_cst(uint16_t, 0x2E8 + 2), 0xC7);
472  outb(s_cst(uint16_t, 0x2E8 + 4), 0x0B);
473  outb(s_cst(uint16_t, 0x2E8 + 4), 0x0F);
474  }
475  if (com4 != 0xFF)
476  {
477  while ((inb(s_cst(uint16_t, 0x2E8 + 5)) & 0x20) == 0)
478  ;
479  outb(0x2E8, c);
480  }
481 
482  // while (true)
483  // {
484  // while ((inb(0x2E8 + 5) & 1) == 0)
485  // ;
486  // outb(0x2E8, inb(0x2E8));
487  // }
488 #endif
489 
490  ANSIParser *parser = &this->Parser;
491 
492  switch (parser->State)
493  {
494  case ANSIParser::ParserState::Escape:
495  {
496  if (c == '\x1b')
497  {
498  parser->State = ANSIParser::ParserState::Bracket;
499  parser->Index = 0;
500  parser->Stack[parser->Index].Value = 0;
501  parser->Stack[parser->Index].Empty = true;
502  }
503  else
504  {
505  parser->State = ANSIParser::ParserState::Escape;
506  this->Append(c);
507  }
508  break;
509  }
510  case ANSIParser::ParserState::Bracket:
511  {
512  if (c == '[')
513  parser->State = ANSIParser::ParserState::Attribute;
514  else
515  {
516  parser->State = ANSIParser::ParserState::Escape;
517  this->Append(c);
518  }
519  break;
520  }
521  case ANSIParser::ParserState::Attribute:
522  {
523  if (isdigit(c))
524  {
525  parser->Stack[parser->Index].Value *= 10;
526  parser->Stack[parser->Index].Value += c - '0';
527  parser->Stack[parser->Index].Empty = false;
528  }
529  else
530  {
531  if (parser->Index < 8)
532  parser->Index++;
533 
534  parser->Stack[parser->Index].Value = 0;
535  parser->Stack[parser->Index].Empty = true;
536  parser->State = ANSIParser::ParserState::EndValue;
537  }
538  break;
539  }
540  case ANSIParser::ParserState::EndValue:
541  break;
542  default:
543  {
544  error("Invalid parser state: %d", parser->State);
545  assert(!"Invalid parser state");
546  }
547  }
548 
549  if (parser->State == ANSIParser::ParserState::EndValue)
550  {
551  if (c == ';')
552  parser->State = ANSIParser::ParserState::Attribute;
553  else
554  {
555  switch (c)
556  {
557  case 'A':
558  this->csi_cuu(parser->Stack, parser->Index);
559  break;
560  case 'B':
561  this->csi_cud(parser->Stack, parser->Index);
562  break;
563 
564  case 'C':
565  this->csi_cuf(parser->Stack, parser->Index);
566  break;
567  case 'D':
568  this->csi_cub(parser->Stack, parser->Index);
569  break;
570 
571  case 'E':
572  this->csi_cnl(parser->Stack, parser->Index);
573  break;
574  case 'F':
575  this->csi_cpl(parser->Stack, parser->Index);
576  break;
577 
578  case 'G':
579  this->csi_cha(parser->Stack, parser->Index);
580  break;
581  case 'd':
582  fixme("move cursor left P1 columns");
583  break;
584 
585  case 'H':
586  this->csi_cup(parser->Stack, parser->Index);
587  break;
588  case 'J':
589  this->csi_ed(parser->Stack, parser->Index);
590  break;
591  case 'K':
592  this->csi_el(parser->Stack, parser->Index);
593  break;
594  case 'm':
595  this->csi_sgr(parser->Stack, parser->Index);
596  break;
597  default:
598  break;
599  }
600  parser->State = ANSIParser::ParserState::Escape;
601  }
602  }
603  }

605  VirtualTerminal::VirtualTerminal(unsigned short Rows, unsigned short Columns,
606  unsigned short XPixels, unsigned short YPixels,
607  PaintCallback _Paint, CursorCallback _Print)
608  : PaintCB(_Paint), CursorCB(_Print)
609  {
610  this->TerminalSize = {
611  .ws_row = Rows,
612  .ws_col = Columns,
613  .ws_xpixel = XPixels,
614  .ws_ypixel = YPixels,
615  };
616 
617  /*
618  - ICRNL - Map Carriage Return to New Line
619  - IXON - Enable XON/XOFF flow control
620 
621  - OPOST - Enable output processing
622  - ONLCR - Map New Line to Carriage Return - New Line
623 
624  - CS8 - 8-bit characters
625  - CREAD - Enable receiver
626  - HUPCL - Hang up on last close
627 
628  - ECHO - Echo input characters
629  - ICANON - Enable canonical input (enable line editing)
630  */
631  this->TerminalConfig.c_iflag = /*ICRNL |*/ IXON;
632  this->TerminalConfig.c_oflag = OPOST | ONLCR;
633  this->TerminalConfig.c_cflag = CS8 | CREAD | HUPCL;
634  this->TerminalConfig.c_lflag = ECHO | ICANON;
635  this->TerminalConfig.c_lflag &= ~ICANON; /* FIXME: not ready for this yet */
636 
637  this->TerminalConfig.c_cc[VINTR] = 0x03; /* ^C */
638  this->TerminalConfig.c_cc[VQUIT] = 0x1C; /* ^\ */
639  this->TerminalConfig.c_cc[VERASE] = 0x7F; /* DEL */
640  this->TerminalConfig.c_cc[VKILL] = 0x15; /* ^U */
641  this->TerminalConfig.c_cc[VEOF] = 0x04; /* ^D */
642  this->TerminalConfig.c_cc[VTIME] = 0; /* Timeout for non-canonical read */
643  this->TerminalConfig.c_cc[VMIN] = 1; /* Minimum number of characters for non-canonical read */
644  this->TerminalConfig.c_cc[VSWTC] = 0; /* ^O */
645  this->TerminalConfig.c_cc[VSTART] = 0x11; /* ^Q */
646  this->TerminalConfig.c_cc[VSTOP] = 0x13; /* ^S */
647  this->TerminalConfig.c_cc[VSUSP] = 0x1A; /* ^Z */
648  this->TerminalConfig.c_cc[VEOL] = 0x00; /* NUL */
649  this->TerminalConfig.c_cc[VREPRINT] = 0x12; /* ^R */
650  this->TerminalConfig.c_cc[VDISCARD] = 0x14; /* ^T */
651  this->TerminalConfig.c_cc[VWERASE] = 0x17; /* ^W */
652  this->TerminalConfig.c_cc[VLNEXT] = 0x19; /* ^Y */
653  this->TerminalConfig.c_cc[VEOL2] = 0x7F; /* DEL (or sometimes EOF) */
654 
655  this->Cells = new TerminalCell[Rows * Columns];
656 
657  debug("Allocated %d bytes for terminal cells",
658  (Rows * Columns) * sizeof(TerminalCell));
659  }
