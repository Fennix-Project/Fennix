502 void civil_from_days(time_t days_since_epoch, int *year, unsigned int *month, unsigned int *day) {
503  time_t time = days_since_epoch + 719468;
504  int era = (time >= 0 ? time : time - 146096) / 146097;
505  unsigned int doe = static_cast<unsigned int>(time - era * 146097);
506  unsigned int yoe = (doe - doe/1460 + doe/36524 - doe/146096) / 365;
507  int y = static_cast<int>(yoe) + era * 400;
508  unsigned int doy = doe - (365*yoe + yoe/4 - yoe/100);
509  unsigned int mp = (5*doy + 2)/153;
510  unsigned int d = doy - (153*mp+2)/5 + 1;
511  unsigned int m = mp + (mp < 10 ? 3 : -9);
512 
513  *year = y + (m <= 2);
514  *month = m;
515  *day = d;
516 }

701  y -= m <= 2;
