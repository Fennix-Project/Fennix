27  MADT::MADT(ACPI::MADTHeader *madt)
28  {
29  trace("Initializing MADT");

31  {
32  error("MADT is NULL");
33  return;
34  }
35 
36  CPUCores = 0;
37  LAPICAddress = (LAPIC *)(uintptr_t)madt->LocalControllerAddress;
38  for (uint8_t *ptr = (uint8_t *)(madt->Entries);
39  (uintptr_t)(ptr) < (uintptr_t)(madt) + madt->Header.Length;
40  ptr += *(ptr + 1))
41  {
42  switch (*(ptr))
43  {
44  case 0:
45  {
46  if (ptr[4] & 1)
47  {
48  lapic.push_back((LocalAPIC *)ptr);
49  KPrint("Local APIC \e8888FF%d\eCCCCCC (APIC \e8888FF%d\eCCCCCC) found.", lapic.back()->ACPIProcessorId, lapic.back()->APICId);
50  CPUCores++;
51  }
52  break;
53  }
54  case 1:
55  {
56  ioapic.push_back((MADTIOApic *)ptr);
57  KPrint("I/O APIC \e8888FF%d\eCCCCCC (Address \e8888FF%#lx\eCCCCCC) found.", ioapic.back()->APICID, ioapic.back()->Address);
58  Memory::Virtual(KernelPageTable).Map((void *)(uintptr_t)ioapic.back()->Address, (void *)(uintptr_t)ioapic.back()->Address, Memory::PTFlag::RW | Memory::PTFlag::PCD); // Make sure that the address is mapped.
59  break;
60  }
61  case 2:
62  {
63  iso.push_back((MADTIso *)ptr);
64  KPrint("ISO (IRQ:\e8888FF%#lx\eCCCCCC, BUS:\e8888FF%#lx\eCCCCCC, GSI:\e8888FF%#lx\eCCCCCC, %s\eCCCCCC/%s\eCCCCCC) found.",
65  iso.back()->IRQSource, iso.back()->BuSSource, iso.back()->GSI,
66  iso.back()->Flags & 0x00000004 ? "\e1770FFActive High" : "\e475EFFActive Low",
67  iso.back()->Flags & 0x00000100 ? "\e00962DEdge Triggered" : "\e008F58Level Triggered");
68  break;
69  }
70  case 4:
71  {
72  nmi.push_back((MADTNmi *)ptr);
73  KPrint("NMI \e8888FF%#lx\eCCCCCC (lint:\e8888FF%#lx\eCCCCCC) found.", nmi.back()->processor, nmi.back()->lint);
74  break;
75  }
76  case 5:
77  {
78  LAPICAddress = (LAPIC *)ptr;
79  KPrint("APIC found at \e8888FF%#lx\eCCCCCC", LAPICAddress);
80  break;
81  }
82  default:
83  {
84  KPrint("Unknown MADT entry \e8888FF%#lx\eCCCCCC", *(ptr));
85  break;
86  }
87  }
88  Memory::Virtual(KernelPageTable).Map((void *)LAPICAddress, (void *)LAPICAddress, Memory::PTFlag::RW | Memory::PTFlag::PCD); // I should map more than one page?
89  }
90  CPUCores--; // We start at 0 (BSP) and end at 11 (APs), so we have 12 cores.
91  KPrint("Total CPU cores: %d", CPUCores + 1);
92  }
