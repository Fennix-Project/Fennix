35  }

123  mb2binfo.Memory.Size += (size_t)entry.len;
124  switch (entry.type)
125  {
126  case MULTIBOOT_MEMORY_AVAILABLE:
127  mb2binfo.Memory.Entry[i].BaseAddress = (void *)entry.addr;
128  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;
129  mb2binfo.Memory.Entry[i].Type = Usable;
130  break;
131  case MULTIBOOT_MEMORY_RESERVED:
132  mb2binfo.Memory.Entry[i].BaseAddress = (void *)entry.addr;
133  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;
134  mb2binfo.Memory.Entry[i].Type = Reserved;
135  break;
136  case MULTIBOOT_MEMORY_ACPI_RECLAIMABLE:
137  mb2binfo.Memory.Entry[i].BaseAddress = (void *)entry.addr;
138  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;
139  mb2binfo.Memory.Entry[i].Type = ACPIReclaimable;
140  break;
141  case MULTIBOOT_MEMORY_NVS:
142  mb2binfo.Memory.Entry[i].BaseAddress = (void *)entry.addr;
143  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;
144  mb2binfo.Memory.Entry[i].Type = ACPINVS;
145  break;
146  case MULTIBOOT_MEMORY_BADRAM:

148  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;

150  break;
151  default:

153  mb2binfo.Memory.Entry[i].Length = (size_t)entry.len;
154  mb2binfo.Memory.Entry[i].Type = Unknown;
155  break;

176  mb2binfo.Framebuffer[fb_count].Width = fb->common.framebuffer_width;

183  {
184  mb2binfo.Framebuffer[fb_count].Type = Indexed;
185  break;
186  }
187  case MULTIBOOT_FRAMEBUFFER_TYPE_RGB:
188  {
189  mb2binfo.Framebuffer[fb_count].Type = RGB;
190  mb2binfo.Framebuffer[fb_count].RedMaskSize = fb->framebuffer_red_mask_size;
191  mb2binfo.Framebuffer[fb_count].RedMaskShift = fb->framebuffer_red_field_position;
192  mb2binfo.Framebuffer[fb_count].GreenMaskSize = fb->framebuffer_green_mask_size;
193  mb2binfo.Framebuffer[fb_count].GreenMaskShift = fb->framebuffer_green_field_position;
194  mb2binfo.Framebuffer[fb_count].BlueMaskSize = fb->framebuffer_blue_mask_size;
195  mb2binfo.Framebuffer[fb_count].BlueMaskShift = fb->framebuffer_blue_field_position;
196  break;

202  }
203  default:

206  break;
207  }
208  }
209  debug("Framebuffer %d: %dx%d %d bpp", fb_count, fb->common.framebuffer_width, fb->common.framebuffer_height, fb->common.framebuffer_bpp);
210  debug("More info:\nAddress: %p\nPitch: %lld\nMemoryModel: %d\nRedMaskSize: %d\nRedMaskShift: %d\nGreenMaskSize: %d\nGreenMaskShift: %d\nBlueMaskSize: %d\nBlueMaskShift: %d",
211  fb->common.framebuffer_addr, fb->common.framebuffer_pitch, fb->common.framebuffer_type,
