79  uint32_t ReadEEPROM(uint8_t Address)
80  {
81  uint16_t Data = 0;
82  uint32_t temp = 0;
83  if (EEPROMAvailable)
84  {
85  WriteCMD(REG::EEPROM, (1) | ((uint32_t)(Address) << 8));
86  while (!((temp = ReadCMD(REG::EEPROM)) & (1 << 4)))
87  ;
88  }
89  else
90  {
91  WriteCMD(REG::EEPROM, (1) | ((uint32_t)(Address) << 2));
92  while (!((temp = ReadCMD(REG::EEPROM)) & (1 << 1)))
93  ;
94  }
95  Data = (uint16_t)((temp >> 16) & 0xFFFF);
96  return Data;

183  MediaAccessControl GetMAC()
184  {
185  MediaAccessControl mac;
186  if (EEPROMAvailable)
187  {
188  uint32_t temp;
189  temp = ReadEEPROM(0);
190  mac.Address[0] = temp & 0xff;
191  mac.Address[1] = (uint8_t)(temp >> 8);
192  temp = ReadEEPROM(1);
193  mac.Address[2] = temp & 0xff;
194  mac.Address[3] = (uint8_t)(temp >> 8);
195  temp = ReadEEPROM(2);
196  mac.Address[4] = temp & 0xff;
197  mac.Address[5] = (uint8_t)(temp >> 8);
198  }
199  else
200  {
201  uint8_t *BaseMac8 = (uint8_t *)(BAR.MemoryBase + 0x5400);
202  uint32_t *BaseMac32 = (uint32_t *)(BAR.MemoryBase + 0x5400);
203  if (BaseMac32[0] != 0)
204  for (int i = 0; i < 6; i++)
205  mac.Address[i] = BaseMac8[i];
206  else

232  void OnInterruptReceived(TrapFrame *)
233  {
234  WriteCMD(REG::IMASK, 0x1);
235  uint32_t status = ReadCMD(0xC0);
236  UNUSED(status);
237 
238  while ((RX[RXCurrent]->Status & 0x1))
239  {
240  uint8_t *data = (uint8_t *)RX[RXCurrent]->Address;
241  uint16_t dataSz = RX[RXCurrent]->Length;
242 
243  ReportNetworkPacket(ID, data, dataSz);
244 
245  RX[RXCurrent]->Status = 0;

247  RXCurrent = (uint16_t)((RXCurrent + 1) % E1000_NUM_RX_DESC);

259  E1000Device(PCIHeader0 *_Header, uint16_t _DeviceID)
260  : Header(_Header),
261  DeviceID(_DeviceID)
262  {
263  uint32_t PCIBAR0 = Header->BAR0;
264  uint32_t PCIBAR1 = Header->BAR1;
265  BAR.Type = PCIBAR0 & 1;
266  BAR.IOBase = (uint16_t)(PCIBAR0 & (~3));
267  BAR.MemoryBase = PCIBAR1 & (~15);
268 
269  switch (DeviceID)
270  {
271  case 0x100E:
272  {
273  Log("Found Intel 82540EM Gigabit Ethernet Controller.");
274 
275  /* Detect EEPROM */
276  WriteCMD(REG::EEPROM, 0x1);
277  for (int i = 0; i < 1000 && !EEPROMAvailable; i++)
278  if (ReadCMD(REG::EEPROM) & 0x10)
279  EEPROMAvailable = true;
280  else
281  EEPROMAvailable = false;
282 
283  if (!GetMAC().Valid())
284  {
285  Log("Failed to get MAC");
286  return;
287  }
288 
289  /* Start link */
290  uint32_t cmdret = ReadCMD(REG::CTRL);
291  WriteCMD(REG::CTRL, cmdret | ECTRL::SLU);
292 
293  for (int i = 0; i < 0x80; i++)
294  WriteCMD((uint16_t)(0x5200 + i * 4), 0);
295 
296  WriteCMD(REG::IMASK, 0x1F6DC);
297  WriteCMD(REG::IMASK, 0xFF & ~4);
298  ReadCMD(0xC0);
299 
300  InitializeRX();
301  InitializeTX();
302  break;
303  }
304  default:
305  {
306  Log("Unimplemented E1000 device.");
307  return;
308  }
309  }
310 
311  Initialized = true;
312  }
313 
314  ~E1000Device()
315  {
316  if (!Initialized)
317  return;
318 
319  switch (DeviceID)
320  {
321  case 0x100E:
322  {
323  // Clearing Enable bit in Receive Control Register
324  uint32_t cmdret = ReadCMD(REG::RCTRL);
325  WriteCMD(REG::RCTRL, cmdret & ~RCTL::EN);
326 
327  // Masking Interrupt Mask, Interrupt Throttling Rate & Interrupt Auto-Mask
328  WriteCMD(REG::IMASK, 0x00000000);
329  WriteCMD(REG::ITR, 0x00000000);
330  WriteCMD(REG::IAM, 0x00000000);
331 
332  // Clearing SLU bit in Device Control Register
333  cmdret = ReadCMD(REG::CTRL);
334  WriteCMD(REG::CTRL, cmdret & ~ECTRL::SLU);
335 
336  // Clear the Interrupt Cause Read register by reading it
337  ReadCMD(REG::ICR);
338 
339  // Powering down the device (?)
340  WriteCMD(REG::CTRL, PCTRL::POWER_DOWN);
341  /* TODO: Stop link; further testing required */
342  break;
343  }
344  default:
345  {
346  Log("Unimplemented E1000 device.");
347  return;
348  }
349  }
350  }
