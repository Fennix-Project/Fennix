143 static void
144 iso_8601_2000_year(char *buf, int year, size_t fw)
145 {
146  int extra;
147  char sign = '\0';
148 
149  if (year >= -9999 && year <= 9999) {
150  sprintf(buf, "%0*d", (int) fw, year);
151  return;
152  }
153 
154  /* now things get weird */
155  if (year > 9999) {
156  sign = '+';
157  } else {
158  sign = '-';
159  year = -year;
160  }
161 
162  extra = year / 10000;
163  year %= 10000;
164  sprintf(buf, "%c_%04d_%d", sign, extra, year);

226

228  if (strchr(format, '%') == NULL && strlen(format) + 1 >= maxsize)
229  return 0;

233  tzset();
234  first = 0;

251  }

267  }
268 #endif /* POSIX_SEMANTICS */

271  tbuf[0] = '\0';
272  if (*format != '%') {
273  *s++ = *format;
274  continue;
275  }
276 #ifdef POSIX_2008
277  pad = '\0';
278  fw = 0;
279  flag = '\0';
280  switch (*++format) {
281  case '+':
282  flag = '+';
283  /* fall through */
284  case '0':
285  pad = '0';
286  format++;
287  break;
288 
289  case '1':
290  case '2':
291  case '3':
292  case '4':
293  case '5':
294  case '6':
295  case '7':
296  case '8':
297  case '9':
298  break;
299

301  format--;
302  goto again;
303  }
304  for (; isdigit(*format); format++) {
305  fw = fw * 10 + (*format - '0');
306  }
307  format--;
308 #endif /* POSIX_2008 */
309 
310  again:
311  switch (*++format) {
312  case '\0':
313  *s++ = '%';
314  goto out;

316  case '%':
317  *s++ = '%';
318  continue;
319 
320  case 'a': /* abbreviated weekday name */
321  if (timeptr->tm_wday < 0 || timeptr->tm_wday > 6)
322  strcpy(tbuf, "?");
323  else
324  strcpy(tbuf, days_a[timeptr->tm_wday]);
325  break;
326 
327  case 'A': /* full weekday name */
328  if (timeptr->tm_wday < 0 || timeptr->tm_wday > 6)
329  strcpy(tbuf, "?");
330  else
331  strcpy(tbuf, days_l[timeptr->tm_wday]);
332  break;
333 
334  case 'b': /* abbreviated month name */
335  short_month:
336  if (timeptr->tm_mon < 0 || timeptr->tm_mon > 11)
337  strcpy(tbuf, "?");
338  else
339  strcpy(tbuf, months_a[timeptr->tm_mon]);
340  break;
341 
342  case 'B': /* full month name */
343  if (timeptr->tm_mon < 0 || timeptr->tm_mon > 11)
344  strcpy(tbuf, "?");
345  else
346  strcpy(tbuf, months_l[timeptr->tm_mon]);
347  break;
348 
349  case 'c': /* appropriate date and time representation */
350  /*
351  * This used to be:
352  *
353  * strftime(tbuf, sizeof tbuf, "%a %b %e %H:%M:%S %Y", timeptr);
354  *
355  * Now, per the ISO 1999 C standard, it this:
356  */
357  strftime(tbuf, sizeof tbuf, "%A %B %d %T %Y", timeptr);
358  break;
359 
360  case 'C':
361 #ifdef POSIX_2008
362  if (pad != '\0' && fw > 0) {
363  size_t min_fw = (flag ? 3 : 2);
364 
365  fw = max(fw, min_fw);
366  sprintf(tbuf, flag
367  ? "%+0*ld"
368  : "%0*ld", (int) fw,
369  (timeptr->tm_year + 1900L) / 100);
370  } else
371 #endif /* POSIX_2008 */
372  century:
373  sprintf(tbuf, "%02ld", (timeptr->tm_year + 1900L) / 100);
374  break;
375 
376  case 'd': /* day of the month, 01 - 31 */
377  i = range(1, timeptr->tm_mday, 31);
378  sprintf(tbuf, "%02d", i);
379  break;
380 
381  case 'D': /* date as %m/%d/%y */
382  strftime(tbuf, sizeof tbuf, "%m/%d/%y", timeptr);
383  break;
384 
385  case 'e': /* day of month, blank padded */
386  sprintf(tbuf, "%2d", range(1, timeptr->tm_mday, 31));
387  break;
388 
389  case 'E':
390  /* POSIX (now C99) locale extensions, ignored for now */
391  goto again;
392 
393  case 'F': /* ISO 8601 date representation */
394  {
395 #ifdef POSIX_2008
396  /*
397  * Field width for %F is for the whole thing.
398  * It must be at least 10.
399  */
400  char m_d[10];
401  strftime(m_d, sizeof m_d, "-%m-%d", timeptr);
402  size_t min_fw = 10;
403 
404  if (pad != '\0' && fw > 0) {
405  fw = max(fw, min_fw);
406  } else {
407  fw = min_fw;
408  }
409 
410  fw -= 6; /* -XX-XX at end are invariant */
411 
412  iso_8601_2000_year(tbuf, timeptr->tm_year + 1900, fw);
413  strcat(tbuf, m_d);
414 #else
415  strftime(tbuf, sizeof tbuf, "%Y-%m-%d", timeptr);
416 #endif /* POSIX_2008 */
417  }
418  break;
419 
420  case 'g':
421  case 'G':
422  /*
423  * Year of ISO week.
424  *
425  * If it's December but the ISO week number is one,
426  * that week is in next year.
427  * If it's January but the ISO week number is 52 or
428  * 53, that week is in last year.
429  * Otherwise, it's this year.
430  */
431  w = iso8601wknum(timeptr);
432  if (timeptr->tm_mon == 11 && w == 1)
433  y = 1900L + timeptr->tm_year + 1;
434  else if (timeptr->tm_mon == 0 && w >= 52)
435  y = 1900L + timeptr->tm_year - 1;
436  else
437  y = 1900L + timeptr->tm_year;
438 
439  if (*format == 'G') {
440 #ifdef POSIX_2008
441  if (pad != '\0' && fw > 0) {
442  size_t min_fw = 4;
443 
444  fw = max(fw, min_fw);
445  sprintf(tbuf, flag
446  ? "%+0*ld"
447  : "%0*ld", (int) fw,
448  y);
449  } else
450 #endif /* POSIX_2008 */
451  sprintf(tbuf, "%ld", y);
452  }
453  else
454  sprintf(tbuf, "%02ld", y % 100);
455  break;
456 
457  case 'h': /* abbreviated month name */
458  goto short_month;
459 
460  case 'H': /* hour, 24-hour clock, 00 - 23 */
461  i = range(0, timeptr->tm_hour, 23);
462  sprintf(tbuf, "%02d", i);
463  break;
464 
465  case 'I': /* hour, 12-hour clock, 01 - 12 */
466  i = range(0, timeptr->tm_hour, 23);
467  if (i == 0)
468  i = 12;
469  else if (i > 12)
470  i -= 12;
471  sprintf(tbuf, "%02d", i);
472  break;
473 
474  case 'j': /* day of the year, 001 - 366 */
475  sprintf(tbuf, "%03d", timeptr->tm_yday + 1);
476  break;
477 
478  case 'm': /* month, 01 - 12 */
479  i = range(0, timeptr->tm_mon, 11);
480  sprintf(tbuf, "%02d", i + 1);
481  break;
482 
483  case 'M': /* minute, 00 - 59 */
484  i = range(0, timeptr->tm_min, 59);
485  sprintf(tbuf, "%02d", i);
486  break;
487 
488  case 'n': /* same as \n */
489  tbuf[0] = '\n';
490  tbuf[1] = '\0';
491  break;
492 
493  case 'O':
494  /* POSIX (now C99) locale extensions, ignored for now */
495  goto again;
496 
497  case 'p': /* am or pm based on 12-hour clock */
498  i = range(0, timeptr->tm_hour, 23);
499  if (i < 12)
500  strcpy(tbuf, ampm[0]);
501  else
502  strcpy(tbuf, ampm[1]);
503  break;
504 
505  case 'r': /* time as %I:%M:%S %p */
506  strftime(tbuf, sizeof tbuf, "%I:%M:%S %p", timeptr);
507  break;
508 
509  case 'R': /* time as %H:%M */
510  strftime(tbuf, sizeof tbuf, "%H:%M", timeptr);
511  break;
512 
513 #if defined(HAVE_MKTIME)
514  case 's': /* time as seconds since the Epoch */
515  {
516  struct tm non_const_timeptr;
517 
518  non_const_timeptr = *timeptr;
519  sprintf(tbuf, "%ld", mktime(& non_const_timeptr));
520  break;
521  }
522 #endif /* defined(HAVE_MKTIME) */
523 
524  case 'S': /* second, 00 - 60 */
525  i = range(0, timeptr->tm_sec, 60);
526  sprintf(tbuf, "%02d", i);
527  break;
528 
529  case 't': /* same as \t */
530  tbuf[0] = '\t';
531  tbuf[1] = '\0';
532  break;
533 
534  case 'T': /* time as %H:%M:%S */
535  the_time:
536  strftime(tbuf, sizeof tbuf, "%H:%M:%S", timeptr);
537  break;
538 
539  case 'u':
540  /* ISO 8601: Weekday as a decimal number [1 (Monday) - 7] */
541  sprintf(tbuf, "%d", timeptr->tm_wday == 0 ? 7 :
542  timeptr->tm_wday);
543  break;
544 
545  case 'U': /* week of year, Sunday is first day of week */
546  sprintf(tbuf, "%02d", weeknumber(timeptr, 0));
547  break;
548 
549  case 'V': /* week of year according ISO 8601 */
550  sprintf(tbuf, "%02d", iso8601wknum(timeptr));
551  break;
552 
553  case 'w': /* weekday, Sunday == 0, 0 - 6 */
554  i = range(0, timeptr->tm_wday, 6);
555  sprintf(tbuf, "%d", i);
556  break;
557 
558  case 'W': /* week of year, Monday is first day of week */
559  sprintf(tbuf, "%02d", weeknumber(timeptr, 1));
560  break;

563  strftime(tbuf, sizeof tbuf, "%A %B %d %Y", timeptr);
564  break;

567  goto the_time;
568  break;

571  year:
572  i = timeptr->tm_year % 100;
573  sprintf(tbuf, "%02d", i);
574  break;

579  size_t min_fw = 4;
580

585  1900L + timeptr->tm_year);
586  } else
587 #endif /* POSIX_2008 */
588  sprintf(tbuf, "%ld", 1900L + timeptr->tm_year);
589  break;

592  * From: Chip Rosenthal <chip@chinacat.unicom.com>
593  * Date: Sun, 19 Mar 1995 00:33:29 -0600 (CST)
594  *
595  * Warning: the %z [code] is implemented by inspecting the
596  * timezone name conditional compile settings, and
597  * inferring a method to get timezone offsets. I've tried
598  * this code on a couple of machines, but I don't doubt
599  * there is some system out there that won't like it.
600  * Maybe the easiest thing to do would be to bracket this
601  * with an #ifdef that can turn it off. The %z feature
602  * would be an admittedly obscure one that most folks can
603  * live without, but it would be a great help to those of
604  * us that muck around with various message processors.
605  */
606  case 'z': /* time zone offset east of GMT e.g. -0600 */
607  if (timeptr->tm_isdst < 0)
608  break;
609 #ifdef HAVE_TM_NAME
610  /*
611  * Systems with tm_name probably have tm_tzadj as
612  * secs west of GMT. Convert to mins east of GMT.
613  */
614  off = -timeptr->tm_tzadj / 60;
615 #else /* !HAVE_TM_NAME */
616 #ifdef HAVE_TM_ZONE
617  /*
618  * Systems with tm_zone probably have tm_gmtoff as
619  * secs east of GMT. Convert to mins east of GMT.
620  */
621  off = timeptr->tm_gmtoff / 60;
622 #else /* !HAVE_TM_ZONE */
623 #if HAVE_TZNAME
624  /*
625  * Systems with tzname[] probably have timezone as
626  * secs west of GMT. Convert to mins east of GMT.
627  */
628 # if defined(__hpux) || defined (HPUX) || defined(__CYGWIN__)
629  off = -timezone / 60;
630 # else
631  /* ADR: 4 August 2001, fixed this per gazelle@interaccess.com */
632  off = -(daylight ? altzone : timezone) / 60;
633 # endif
634 #else /* !HAVE_TZNAME */
635  gettimeofday(& tv, & zone);
636  off = -zone.tz_minuteswest;
637 #endif /* !HAVE_TZNAME */
638 #endif /* !HAVE_TM_ZONE */
639 #endif /* !HAVE_TM_NAME */
640  if (off < 0) {
641  tbuf[0] = '-';
642  off = -off;
643  } else {
644  tbuf[0] = '+';
645  }
646  sprintf(tbuf+1, "%02ld%02ld", off/60, off%60);
647  break;
648 
649  case 'Z': /* time zone name or abbreviation */
650 #ifdef HAVE_TZNAME
651  i = (daylight && timeptr->tm_isdst > 0); /* 0 or 1 */
652  strcpy(tbuf, tzname[i]);
653 #else
654 #ifdef HAVE_TM_ZONE
655  strcpy(tbuf, timeptr->tm_zone);
656 #else
657 #ifdef HAVE_TM_NAME
658  strcpy(tbuf, timeptr->tm_name);
659 #else
660  gettimeofday(& tv, & zone);
661  strcpy(tbuf, timezone(zone.tz_minuteswest,
662  timeptr->tm_isdst > 0));
663 #endif /* HAVE_TM_NAME */
664 #endif /* HAVE_TM_ZONE */
665 #endif /* HAVE_TZNAME */
666  break;
667 
668 #ifdef SUNOS_EXT
669  case 'k': /* hour, 24-hour clock, blank pad */
670  sprintf(tbuf, "%2d", range(0, timeptr->tm_hour, 23));
671  break;
672 
673  case 'l': /* hour, 12-hour clock, 1 - 12, blank pad */
674  i = range(0, timeptr->tm_hour, 23);
675  if (i == 0)
676  i = 12;
677  else if (i > 12)
678  i -= 12;
679  sprintf(tbuf, "%2d", i);
680  break;
681 #endif
682 
683 #ifdef HPUX_EXT
684  case 'N': /* Emperor/Era name */
685  /* this is essentially the same as the century */
686  goto century; /* %C */
687 
688  case 'o': /* Emperor/Era year */
689  goto year; /* %y */
690 #endif /* HPUX_EXT */
691 
692 
693 #ifdef VMS_EXT
694  case 'v': /* date as dd-bbb-YYYY */
695  sprintf(tbuf, "%2d-%3.3s-%4ld",
696  range(1, timeptr->tm_mday, 31),
697  months_a[range(0, timeptr->tm_mon, 11)],
698  timeptr->tm_year + 1900L);
699  for (i = 3; i < 6; i++)
700  if (islower(tbuf[i]))
701  tbuf[i] = toupper(tbuf[i]);
702  break;
703 #endif
704 
705  default:
706  tbuf[0] = '%';
707  tbuf[1] = *format;
708  tbuf[2] = '\0';
709  break;
710  }
711  i = strlen(tbuf);
712  if (i) {
713  if (s + i < endp - 1) {
714  strcpy(s, tbuf);
715  s += i;
716  } else
717  return 0;
718  }
719  }
720 out:
721  if (s < endp && *format == '\0') {
722  *s = '\0';
723  if (s == start)
724  errno = oerrno;
725  return (s - start);
726  } else
727  return 0;
728 }

739 /* iso8601wknum --- compute week number according to ISO 8601 */
740 
741 static int
742 iso8601wknum(const struct tm *timeptr)
743 {
744  /*
745  * From 1003.2:
746  * If the week (Monday to Sunday) containing January 1
747  * has four or more days in the new year, then it is week 1;
748  * otherwise it is the highest numbered week of the previous
749  * year (52 or 53), and the next week is week 1.
750  *
751  * ADR: This means if Jan 1 was Monday through Thursday,
752  * it was week 1, otherwise week 52 or 53.
753  *
754  * XPG4 erroneously included POSIX.2 rationale text in the
755  * main body of the standard. Thus it requires week 53.
756  */
757 
758  int weeknum, jan1day, diff;
759 
760  /* get week number, Monday as first day of the week */
761  weeknum = weeknumber(timeptr, 1);
762 
763  /*
764  * With thanks and tip of the hatlo to tml@tik.vtt.fi
765  *
766  * What day of the week does January 1 fall on?
767  * We know that
768  * (timeptr->tm_yday - jan1.tm_yday) MOD 7 ==
769  * (timeptr->tm_wday - jan1.tm_wday) MOD 7
770  * and that
771  * jan1.tm_yday == 0
772  * and that
773  * timeptr->tm_wday MOD 7 == timeptr->tm_wday
774  * from which it follows that. . .
775  */
776  jan1day = timeptr->tm_wday - (timeptr->tm_yday % 7);
777  if (jan1day < 0)
778  jan1day += 7;
779 
780  /*
781  * If Jan 1 was a Monday through Thursday, it was in
782  * week 1. Otherwise it was last year's highest week, which is
783  * this year's week 0.
784  *
785  * What does that mean?
786  * If Jan 1 was Monday, the week number is exactly right, it can
787  * never be 0.
788  * If it was Tuesday through Thursday, the weeknumber is one
789  * less than it should be, so we add one.
790  * Otherwise, Friday, Saturday or Sunday, the week number is
791  * OK, but if it is 0, it needs to be 52 or 53.
792  */
793  switch (jan1day) {
794  case 1: /* Monday */
795  break;
796  case 2: /* Tuesday */
797  case 3: /* Wednesday */
798  case 4: /* Thursday */
799  weeknum++;
800  break;
801  case 5: /* Friday */
802  case 6: /* Saturday */
803  case 0: /* Sunday */
804  if (weeknum == 0) {
805 #ifdef USE_BROKEN_XPG4
806  /* XPG4 (as of March 1994) says 53 unconditionally */
807  weeknum = 53;
808 #else
809  /* get week number of last week of last year */
810  struct tm dec31ly; /* 12/31 last year */
811  dec31ly = *timeptr;
812  dec31ly.tm_year--;
813  dec31ly.tm_mon = 11;
814  dec31ly.tm_mday = 31;
815  dec31ly.tm_wday = (jan1day == 0) ? 6 : jan1day - 1;
816  dec31ly.tm_yday = 364 + isleap(dec31ly.tm_year + 1900L);
817  weeknum = iso8601wknum(& dec31ly);
818 #endif
819  }
820  break;
821  }
822 
823  if (timeptr->tm_mon == 11) {
824  /*
825  * The last week of the year
826  * can be in week 1 of next year.
827  * Sigh.
828  *
829  * This can only happen if
830  * M T W
831  * 29 30 31
832  * 30 31
833  * 31
834  */
835  int wday, mday;
836 
837  wday = timeptr->tm_wday;
838  mday = timeptr->tm_mday;
839  if ( (wday == 1 && (mday >= 29 && mday <= 31))
840  || (wday == 2 && (mday == 30 || mday == 31))
841  || (wday == 3 && mday == 31))
842  weeknum = 1;
843  }
844 
845  return weeknum;
846 }
