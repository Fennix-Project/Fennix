144  switch (wc)

150  case '5': case '6': case '7': case '8': case '9':
151  case ':': case ';': case '<': case '=': case '>':
152  case '?':
153  case 'A': case 'B': case 'C': case 'D': case 'E':
154  case 'F': case 'G': case 'H': case 'I': case 'J':
155  case 'K': case 'L': case 'M': case 'N': case 'O':
156  case 'P': case 'Q': case 'R': case 'S': case 'T':
157  case 'U': case 'V': case 'W': case 'X': case 'Y':
158  case 'Z':
159  case '[': case '\\': case ']': case '^': case '_':
160  case 'a': case 'b': case 'c': case 'd': case 'e':
161  case 'f': case 'g': case 'h': case 'i': case 'j':
162  case 'k': case 'l': case 'm': case 'n': case 'o':
163  case 'p': case 'q': case 'r': case 's': case 't':
164  case 'u': case 'v': case 'w': case 'x': case 'y':
165  case 'z': case '{': case '|': case '}': case '~':
166  return 1;
167  default:
