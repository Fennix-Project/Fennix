101 // R_MapPlane
102 //
103 // Uses global vars:
104 // planeheight

109 // viewy

111 // BASIC PRIMITIVE
112 //
113 void
114 R_MapPlane
115 ( int y,
116  int x1,
117  int x2 )
118 {
119  angle_t angle;
120  fixed_t distance;
121  fixed_t length;
122  unsigned index;
123 
124 #ifdef RANGECHECK
125  if (x2 < x1
126  || x1 < 0
127  || x2 >= viewwidth
128  || y > viewheight)
129  {
130  I_Error ("R_MapPlane: %i, %i at %i",x1,x2,y);
131  }
132 #endif
133 
134  if (planeheight != cachedheight[y])
135  {
136  cachedheight[y] = planeheight;
137  distance = cacheddistance[y] = FixedMul (planeheight, yslope[y]);
138  ds_xstep = cachedxstep[y] = FixedMul (distance,basexscale);
139  ds_ystep = cachedystep[y] = FixedMul (distance,baseyscale);
140  }
141  else
142  {
143  distance = cacheddistance[y];
144  ds_xstep = cachedxstep[y];
145  ds_ystep = cachedystep[y];
146  }
147 
148  length = FixedMul (distance,distscale[x1]);
149  angle = (viewangle + xtoviewangle[x1])>>ANGLETOFINESHIFT;
150  ds_xfrac = viewx + FixedMul(finecosine[angle], length);
151  ds_yfrac = -viewy - FixedMul(finesine[angle], length);
152 
153  if (fixedcolormap)
154  ds_colormap = fixedcolormap;
155  else
156  {
157  index = distance >> LIGHTZSHIFT;
158 
159  if (index >= MAXLIGHTZ )
160  index = MAXLIGHTZ-1;
161 
162  ds_colormap = planezlight[index];
163  }
164 
165  ds_y = y;
166  ds_x1 = x1;
167  ds_x2 = x2;
168 
169  // high or low detail
170  spanfunc ();
171 }

208 // R_FindPlane
209 //
210 visplane_t*
211 R_FindPlane
212 ( fixed_t height,
213  int picnum,
214  int lightlevel )
215 {
216  visplane_t* check;
217 
218  if (picnum == skyflatnum)
219  {
220  height = 0; // all skys map together
221  lightlevel = 0;
222  }
223 
224  for (check=visplanes; check<lastvisplane; check++)
225  {
226  if (height == check->height
227  && picnum == check->picnum
228  && lightlevel == check->lightlevel)
229  {
230  break;
231  }
232  }
233 
234 
235  if (check < lastvisplane)
236  return check;
237 
238  if (lastvisplane - visplanes == MAXVISPLANES)
239  I_Error ("R_FindPlane: no more visplanes");
240 
241  lastvisplane++;
242 
243  check->height = height;
244  check->picnum = picnum;
245  check->lightlevel = lightlevel;
246  check->minx = SCREENWIDTH;
247  check->maxx = -1;
248 
249  memset (check->top,0xff,sizeof(check->top));
250 
251  return check;
252 }

291

293  if (pl->top[x] != 0xff)
294  break;
295 
296  if (x > intrh)
297  {
298  pl->minx = unionl;
299  pl->maxx = unionh;
300 
301  // use the same one
302  return pl;
303  }
304 
305  // make a new visplane
306  lastvisplane->height = pl->height;
307  lastvisplane->picnum = pl->picnum;
308  lastvisplane->lightlevel = pl->lightlevel;
309 
310  pl = lastvisplane++;
311  pl->minx = start;
312  pl->maxx = stop;
313 
314  memset (pl->top,0xff,sizeof(pl->top));
315 
316  return pl;
