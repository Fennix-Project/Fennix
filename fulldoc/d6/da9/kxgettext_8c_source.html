14 {

20  if ((textlen > 0) && (text[textlen-1] == '\n'))
21  eol = 1;
22 
23  *bfp++ = '"';
24  --len;
25 
26  if (multiline) {

30  len -= 3;
31  }
32

34  if (*text == '"')

36  else if (*text == '\n') {
37  *bfp++ = '\\';
38  *bfp++ = 'n';
39  *bfp++ = '"';

41  *bfp++ = '"';

44  goto next;
45  }

49  }
