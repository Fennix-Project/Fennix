75 nsa bool UserModeExceptionHandler(CPU::ExceptionFrame *Frame)
76 {
77  CPUData *core = GetCurrentCPU();
78  Tasking::PCB *proc = core->CurrentProcess;
79  Tasking::TCB *thread = core->CurrentThread;
80  debug("Current process %s(%d) and thread %s(%d)",
81  proc->Name, proc->ID, thread->Name, thread->ID);
82  thread->SetState(Tasking::Waiting);
83 
84 #ifdef DEBUG
85  dbgPrint(Frame);
86 #endif
87 
88  int sigRet = -1;
89  switch (Frame->InterruptNumber)
90  {
91  case CPU::x86::PageFault:
92  {
93  bool Handled = proc->vma->HandleCoW(Frame->cr2);
94  if (!Handled)
95  Handled = thread->Stack->Expand(Frame->cr2);
96 
97  if (likely(Handled))
98  {
99  debug("Page fault handled");
100  Frame->cr2 = 0;
101  thread->SetState(Tasking::Ready);
102  return true;
103  }
104 
105  sigRet = proc->Signals.SendSignal(SIGSEGV,
106  {Tasking::KILL_CRASH});
107  break;
108  }
109  case CPU::x86::Debug:
110  case CPU::x86::Breakpoint:
111  {
112  sigRet = proc->Signals.SendSignal(SIGTRAP,
113  {Tasking::KILL_CRASH});
114  break;
115  }
116  case CPU::x86::DivideByZero:
117  case CPU::x86::Overflow:
118  case CPU::x86::BoundRange:
119  case CPU::x86::x87FloatingPoint:
120  case CPU::x86::SIMDFloatingPoint:
121  {
122  sigRet = proc->Signals.SendSignal(SIGFPE,
123  {Tasking::KILL_CRASH});
124  break;
125  }
126  case CPU::x86::InvalidOpcode:
127  case CPU::x86::GeneralProtectionFault:
128  {
129  sigRet = proc->Signals.SendSignal(SIGILL,
130  {Tasking::KILL_CRASH});
131  break;
132  }
133  case CPU::x86::DeviceNotAvailable:
134  {
135  sigRet = proc->Signals.SendSignal(SIGBUS,
136  {Tasking::KILL_CRASH});
137  break;
138  }
139  case CPU::x86::NonMaskableInterrupt:
140  case CPU::x86::DoubleFault:
141  case CPU::x86::CoprocessorSegmentOverrun:
142  case CPU::x86::InvalidTSS:
143  case CPU::x86::SegmentNotPresent:
144  case CPU::x86::StackSegmentFault:
145  case CPU::x86::AlignmentCheck:
146  case CPU::x86::MachineCheck:
147  case CPU::x86::Virtualization:
148  case CPU::x86::Security:
149  default:
150  {
151  error("Unhandled exception %d on CPU %d",
152  Frame->InterruptNumber, core->ID);
153  break;
154  }
155  }
156 
157  if (sigRet == 0)
158  {
159  trace("User mode exception handler handled");
160  thread->SetState(Tasking::Ready);
161  return true;
162  }
163 
164  error("User mode exception handler failed");
165  return false;
166 }
