40  std::string *TargetPath,
41  void *MemoryImage,
42  std::vector<const char *> *_NeededLibraries)
43  {
44  std::vector<const char *> NeededLibraries = *_NeededLibraries;
45  delete _NeededLibraries;
46 
47  debug("Interpreter thread started for %s", TargetPath->c_str());
48  // Interpreter will create an IPC with token "LOAD".
49  char UniqueToken[16] = {'L', 'O', 'A', 'D', '\0'};
50  InterProcessCommunication::IPCHandle *Handle = nullptr;
51  while (Handle == nullptr)
52  {
53  debug("Searching for IPC with token %s", UniqueToken);
54  Handle = TargetProcess->IPC->SearchByToken(UniqueToken);
55  if (Handle == nullptr)
56  debug("Failed");
57 
58  TaskManager->Sleep(200);
59  if (Handle == nullptr)
60  debug("Retrying...");
61  }
62  debug("IPC found, sending data...");
63  InterpreterIPCData *TmpBuffer = new InterpreterIPCData;
64  strncpy(TmpBuffer->Path, TargetPath->c_str(), sizeof(TmpBuffer->Path) - 1);
65  TmpBuffer->MemoryImage = MemoryImage;
66  size_t NeededLibsSize = NeededLibraries.size();
67 
68  for (size_t i = 0; i < NeededLibsSize; i++)
69  strncpy(TmpBuffer->Libraries[i].Name, NeededLibraries[i],
70  sizeof(TmpBuffer->Libraries[i].Name) - 1);
71 
72 #ifdef DEBUG
73  debug("Input:");
74  debug("Path: %s", TargetPath->c_str());
75  debug("MemoryImage: %p", MemoryImage);
76  for (size_t i = 0; i < NeededLibsSize; i++)
77  debug("Library: %s", NeededLibraries[i]);
78 
79  debug("Buffer:");
80  debug("Path: %s", TmpBuffer->Path);
81  debug("MemoryImage: %p", TmpBuffer->MemoryImage);
82  for (size_t i = 0; i < 64; i++)
83  {
84  if (TmpBuffer->Libraries[i].Name[0] != '\0')
85  break;
86  debug("Library: %s", TmpBuffer->Libraries[i].Name);
87  }
88 #endif
89 
90  RetryIPCWrite:
91  InterProcessCommunication::IPCErrorCode ret =
92  TargetProcess->IPC->Write(Handle->ID, TmpBuffer, sizeof(InterpreterIPCData));
93  debug("Write returned %d", ret);
94  if (ret == InterProcessCommunication::IPCErrorCode::IPCNotListening)
95  {
96  debug("IPC not listening, retrying...");
97  TaskManager->Sleep(100);
98  goto RetryIPCWrite;
99  }
100  delete TmpBuffer;
101 
102  while (!TargetProcess->IPC->SearchByToken(UniqueToken))
103  TaskManager->Schedule();
104 
105  debug("Interpreter thread finished for %s", TargetPath->c_str());
106 
107  for (size_t i = 0; i < NeededLibsSize; i++)
108  delete[] NeededLibraries[i];
109 
110  delete TargetPath;
111  TEXIT(0);
112  }
