44 
45  /* Check ELF header. */
46  if (ELFHeader.e_ident[EI_MAG0] == ELFMAG0 &&
47  ELFHeader.e_ident[EI_MAG1] == ELFMAG1 &&
48  ELFHeader.e_ident[EI_MAG2] == ELFMAG2 &&
49  ELFHeader.e_ident[EI_MAG3] == ELFMAG3)
50  {
51  debug("Image - ELF");

53  goto Success;
54  }
55 
56  if (MachHeader.magic == MH_MAGIC || MachHeader.magic == MH_CIGAM)
