31 void testMemoryIntegrity()
32 {
33  debug("Testing memory integrity...");
34 
35  for (size_t i = 1; i <= MAX_SIZE; i *= 2)
36  {
37  void *ptr1 = malloc(i);
38  if (ptr1 == NULL)
39  {
40  error("malloc failed for size %zu", i);
41  inf_loop;
42  }
43 
44  memset(ptr1, 0xAA, i);
45 
46  for (size_t j = 0; j < i; ++j)
47  {
48  if (*((unsigned char *)ptr1 + j) != 0xAA)
49  {
50  error("Memory corruption detected before free (size %zu)", i);
51  inf_loop;
52  }
53  }
54 
55  void *ptr2 = malloc(i * 2);
56  if (ptr2 == NULL)
57  {
58  error("malloc failed for size %zu", i * 2);
59  inf_loop;
60  }
61 
62  memset(ptr2, 0xBB, i * 2);
63 
64  for (size_t j = 0; j < i; ++j)
65  {
66  if (*((unsigned char *)ptr1 + j) != 0xAA)
67  {
68  error("Memory corruption detected in previous allocation (size %zu)", i);
69  inf_loop;
70  }
71  }
72 
73  free(ptr1);
74 
75  for (size_t j = 0; j < i; ++j)
76  {
77  if (*((unsigned char *)ptr2 + j) != 0xBB)
78  {
79  error("Memory corruption detected in current allocation (size %zu)", i * 2);
80  inf_loop;
81  }
82  }
83 
84  void *ptr3 = realloc(ptr2, i * 3);
85  if (ptr3 == NULL)
86  {
87  error("realloc failed for size %zu", i * 3);
88  inf_loop;
89  }
90 
91  memset(ptr3, 0xCC, i * 3);
92 
93  for (size_t j = 0; j < i; ++j)
94  {
95  if (*((unsigned char *)ptr3 + j) != 0xCC)
96  {
97  error("Memory corruption detected after realloc (size %zu)", i * 3);
98  inf_loop;
99  }
100  }
101 
102  free(ptr3);
103  }
104 
105  debug("Memory integrity test passed.");
106 }
