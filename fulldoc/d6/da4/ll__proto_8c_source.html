154 {
155  unsigned i;
156  id = ntohs(id);
157  for (i = 0; i < ARRAY_SIZE(llproto_ids); i++) {
158  if (llproto_ids[i] == id)

163 }

165 int FAST_FUNC ll_proto_a2n(unsigned short *id, char *buf)
166 {
