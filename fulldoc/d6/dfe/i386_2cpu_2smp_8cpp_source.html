48 nsa CPUData *GetCurrentCPU()
49 {
50  if (unlikely(!Interrupts::apic[0]))
51  return &CPUs[0]; /* No APIC means we are on the BSP. */
52 
53  APIC::APIC *apic = (APIC::APIC *)Interrupts::apic[0];
54  int CoreID = 0;
55  if (CPUEnabled.load(std::memory_order_acquire) == true)
56  {
57  if (apic->x2APIC)
58  CoreID = int(CPU::x32::rdmsr(CPU::x32::MSR_X2APIC_APICID));
59  else
60  CoreID = apic->Read(APIC::APIC_ID) >> 24;
61  }
62 
63  if (unlikely((&CPUs[CoreID])->IsActive != true))
64  {
65  error("CPU %d is not active!", CoreID);
66  assert((&CPUs[0])->IsActive == true); /* We can't continue without the BSP. */
67  return &CPUs[0];
68  }
69 
70  assert((&CPUs[CoreID])->Checksum == CPU_DATA_CHECKSUM); /* This should never happen. */
71  return &CPUs[CoreID];
72 }
