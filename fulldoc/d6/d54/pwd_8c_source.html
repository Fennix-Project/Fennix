31 {
32  struct stat st1;
33  struct stat st2;
34  char *wd;
35  char *p;
36 
37  wd = getenv("PWD");
38  if (!wd || wd[0] != '/')
39  return 0;
40 
41  p = wd;
42  while (*p) {
43  /* doing strstr(p, "/.") by hand is smaller and faster... */
44  if (*p++ != '/')
45  continue;
46  if (*p != '.')
47  continue;
48  /* we found "/.", skip to next char */
49  p++;
50  if (*p == '.')
51  p++; /* we found "/.." */
52  if (*p == '\0' || *p == '/')
53  return 0; /* "/./" or "/../" component: bad */
54  }
55 
56  if (stat(wd, &st1) != 0)
57  return 0;
58  if (stat(".", &st2) != 0)
59  return 0;
60  if (st1.st_ino != st2.st_ino)
61  return 0;
62  if (st1.st_dev != st2.st_dev)
63  return 0;
64 
65  puts(wd);
66  return 1;
67 }
