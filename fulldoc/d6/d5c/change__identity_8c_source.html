33 void FAST_FUNC change_identity(const struct passwd *pw)
34 {
35  int res;
36 
37  res = initgroups(pw->pw_name, pw->pw_gid);
38  endgrent(); /* helps to close a fd used internally by libc */
39 
40  if (res != 0) {
41  /*
42  * If initgroups() fails because a system call is unimplemented
43  * then we are running on a Linux kernel compiled without multiuser
44  * support (CONFIG_MULTIUSER is not defined).
45  *
46  * If we are running without multiuser support *and* the target uid
47  * already matches the current uid then we can skip the change of
48  * identity.
49  */
50  if (errno == ENOSYS && pw->pw_uid == getuid()) {
51  return;
52  }
53 
54  bb_simple_perror_msg_and_die("can't set groups");
55  }
56 
57  xsetgid(pw->pw_gid);
58  xsetuid(pw->pw_uid);
59 }
