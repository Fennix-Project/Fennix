54 // Turn vm_unmap calls in free into vm_map(..., PROT_NONE, ...) calls to prevent
55 // those addresses from being reused. This is useful for detecting situations like this:
56 // 1. Allocate object X at address Y
57 // 2. Do some computation using object X
58 // 3. Free object X at address Y
59 // 4. Allocate object Z at address W, and it so happens that W == Y
60 // 5. Try to use object X, but the memory which was backing it now contains object Z

86 void *MemoryAllocator::allocate(size_t size) {
87  size_t pg_size = (size + size_t{pageSize - 1}) & ~size_t{pageSize - 1};
88  size_t offset = (pg_size - size) & ~size_t{pointerAlignment - 1};
89 
90  void *ptr;
91 
92  // Two extra pages for metadata in front and guard page at the end
93  // Reserve the whole region as PROT_NONE...
94  if (int e = mlibc::sys_vm_map(nullptr, pg_size + pageSize * 2, PROT_NONE, MAP_ANONYMOUS | MAP_PRIVATE, -1, 0, &ptr))
95  mlibc::panicLogger() << "sys_vm_map failed in MemoryAllocator::allocate (errno " << e << ")" << frg::endlog;
96 
97  // ...Then replace pages to make them accessible, excluding the guard page
98  if (int e = mlibc::sys_vm_map(ptr, pg_size + pageSize, PROT_READ | PROT_WRITE, MAP_ANONYMOUS | MAP_PRIVATE | MAP_FIXED, -1, 0, &ptr))
99  mlibc::panicLogger() << "sys_vm_map failed in MemoryAllocator::allocate (errno " << e << ")" << frg::endlog;
100 
101  void *meta = ptr;
102  void *out_page = reinterpret_cast<void *>(reinterpret_cast<uintptr_t>(ptr) + pageSize);
103  void *out = reinterpret_cast<void *>(reinterpret_cast<uintptr_t>(out_page) + offset);
104  void *out_align_area = reinterpret_cast<void *>(reinterpret_cast<uintptr_t>(out) + size);
105 
106  AllocatorMeta metaData{size, pg_size, allocatorMagic};
107 
108  memset(meta, metaAreaValue, pageSize);
109  memcpy(meta, &metaData, sizeof(AllocatorMeta));
110 
111  memset(out_page, offsetAreaValue, offset);
112  memset(out, allocatedAreaValue, size);
113  memset(out_align_area, alignmentAreaValue, pg_size - offset - size);
114 
115  if constexpr (logAllocations)
116  mlibc::infoLogger() << "MemoryAllocator::allocate(" << size << ") = " << out << frg::endlog;
117 
118  return out;
119 }
