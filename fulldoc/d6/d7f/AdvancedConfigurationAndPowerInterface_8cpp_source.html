27  void *ACPI::FindTable(ACPI::ACPIHeader *ACPIHeader, char *Signature)
28  {
29  for (uint64_t t = 0; t < ((ACPIHeader->Length - sizeof(ACPI::ACPIHeader)) / (XSDTSupported ? 8 : 4)); t++)
30  {
31  // Should I be concerned about unaligned memory access?
32  ACPI::ACPIHeader *SDTHdr = nullptr;
33  if (XSDTSupported)
34  SDTHdr = (ACPI::ACPIHeader *)(*(uint64_t *)((uint64_t)ACPIHeader + sizeof(ACPI::ACPIHeader) + (t * 8)));
35  else
36  SDTHdr = (ACPI::ACPIHeader *)(*(uint32_t *)((uint64_t)ACPIHeader + sizeof(ACPI::ACPIHeader) + (t * 4)));
37 
38  for (uint64_t i = 0; i < 4; i++)
39  {
40  if (SDTHdr->Signature[i] != Signature[i])
41  break;
42  if (i == 3)
43  {
44  trace("%s found at address %p!", Signature, (uintptr_t)SDTHdr);
45  return SDTHdr;
46  }
47  }
48  }
49  // warn("%s not found!", Signature);
50  return nullptr;
51  }
