507 /* Expand a system dependent string segment. Return NULL if unsupported. */
508 static const char *
509 get_sysdep_segment_value (name)
510  const char *name;
511 {
512  /* Test for an ISO C 99 section 7.8.1 format string directive.
513  Syntax:
514  P R I { d | i | o | u | x | X }
515  { { | LEAST | FAST } { 8 | 16 | 32 | 64 } | MAX | PTR } */
516  /* We don't use a table of 14 times 6 'const char *' strings here, because
517  data relocations cost startup time. */
518  if (name[0] == 'P' && name[1] == 'R' && name[2] == 'I')
519  {
520  if (name[3] == 'd' || name[3] == 'i' || name[3] == 'o' || name[3] == 'u'
521  || name[3] == 'x' || name[3] == 'X')
522  {
523  if (name[4] == '8' && name[5] == '\0')
524  {
525  if (name[3] == 'd')
526  return PRId8;
527  if (name[3] == 'i')
528  return PRIi8;
529  if (name[3] == 'o')
530  return PRIo8;
531  if (name[3] == 'u')
532  return PRIu8;
533  if (name[3] == 'x')
534  return PRIx8;
535  if (name[3] == 'X')
536  return PRIX8;
537  abort ();
538  }
539  if (name[4] == '1' && name[5] == '6' && name[6] == '\0')
540  {
541  if (name[3] == 'd')
542  return PRId16;
543  if (name[3] == 'i')
544  return PRIi16;
545  if (name[3] == 'o')
546  return PRIo16;
547  if (name[3] == 'u')
548  return PRIu16;
549  if (name[3] == 'x')
550  return PRIx16;
551  if (name[3] == 'X')
552  return PRIX16;
553  abort ();
554  }
555  if (name[4] == '3' && name[5] == '2' && name[6] == '\0')
556  {
557  if (name[3] == 'd')
558  return PRId32;
559  if (name[3] == 'i')
560  return PRIi32;
561  if (name[3] == 'o')
562  return PRIo32;
563  if (name[3] == 'u')
564  return PRIu32;
565  if (name[3] == 'x')
566  return PRIx32;
567  if (name[3] == 'X')
568  return PRIX32;
569  abort ();
570  }
571  if (name[4] == '6' && name[5] == '4' && name[6] == '\0')
572  {
573  if (name[3] == 'd')
574  return PRId64;
575  if (name[3] == 'i')
576  return PRIi64;
577  if (name[3] == 'o')
578  return PRIo64;
579  if (name[3] == 'u')
580  return PRIu64;
581  if (name[3] == 'x')
582  return PRIx64;
583  if (name[3] == 'X')
584  return PRIX64;
585  abort ();
586  }
587  if (name[4] == 'L' && name[5] == 'E' && name[6] == 'A'
588  && name[7] == 'S' && name[8] == 'T')
589  {
590  if (name[9] == '8' && name[10] == '\0')
591  {
592  if (name[3] == 'd')
593  return PRIdLEAST8;
594  if (name[3] == 'i')
595  return PRIiLEAST8;
596  if (name[3] == 'o')
597  return PRIoLEAST8;
598  if (name[3] == 'u')
599  return PRIuLEAST8;
600  if (name[3] == 'x')
601  return PRIxLEAST8;
602  if (name[3] == 'X')
603  return PRIXLEAST8;
604  abort ();
605  }
606  if (name[9] == '1' && name[10] == '6' && name[11] == '\0')
607  {
608  if (name[3] == 'd')
609  return PRIdLEAST16;
610  if (name[3] == 'i')
611  return PRIiLEAST16;
612  if (name[3] == 'o')
613  return PRIoLEAST16;
614  if (name[3] == 'u')
615  return PRIuLEAST16;
616  if (name[3] == 'x')
617  return PRIxLEAST16;
618  if (name[3] == 'X')
619  return PRIXLEAST16;

771 {

773  This can be found (in textual form) in the entry "". If this
774  entry does not exist or if this does not contain the `charset='
775  information, we will assume the charset matches the one the

777  char *nullentry;

781  domain->codeset_cntr =
782  (domainbinding != NULL ? domainbinding->codeset_cntr : 0);

794

796  {
797 #if defined _LIBC || HAVE_ICONV
798  const char *charsetstr;
799

801  if (charsetstr != NULL)
802  {
803  size_t len;
804  char *charset;
805  const char *outcharset;
806 
807  charsetstr += strlen ("charset=");

837 # endif

848  domain->conv = (__gconv_t) -1;

853 # if (__GLIBC__ == 2 && __GLIBC_MINOR__ >= 2) || __GLIBC__ > 2 \
854  || _LIBICONV_VERSION >= 0x0105

903 /* Load the message catalogs specified by FILENAME. If it is no valid
904  message catalog do nothing. */
905 void
906 internal_function
907 _nl_load_domain (domain_file, domainbinding)
908  struct loaded_l10nfile *domain_file;
909  struct binding *domainbinding;
910 {
911  int fd;
912  size_t size;
913 #ifdef _LIBC
914  struct stat64 st;
915 #else
916  struct stat st;
917 #endif
918  struct mo_file_header *data = (struct mo_file_header *) -1;
919  int use_mmap = 0;
920  struct loaded_domain *domain;
921  int revision;
922  const char *nullentry;
923 
924  domain_file->decided = 1;
925  domain_file->data = NULL;
926 
927  /* Note that it would be useless to store domainbinding in domain_file
928  because domainbinding might be == NULL now but != NULL later (after
929  a call to bind_textdomain_codeset). */
930 
931  /* If the record does not represent a valid locale the FILENAME
932  might be NULL. This can happen when according to the given
933  specification the locale file name is different for XPG and CEN
934  syntax. */
935  if (domain_file->filename == NULL)
936  return;
937 
938  /* Try to open the addressed file. */
939  fd = open (domain_file->filename, O_RDONLY | O_BINARY);
940  if (fd == -1)
941  return;
942 
943  /* We must know about the size of the file. */
944  if (
945 #ifdef _LIBC
946  __builtin_expect (fstat64 (fd, &st) != 0, 0)
947 #else
948  __builtin_expect (fstat (fd, &st) != 0, 0)
949 #endif
950  || __builtin_expect ((size = (size_t) st.st_size) != st.st_size, 0)
951  || __builtin_expect (size < sizeof (struct mo_file_header), 0))
952  {
953  /* Something went wrong. */
954  close (fd);
955  return;
956  }
957 
958 #ifdef HAVE_MMAP
959  /* Now we are ready to load the file. If mmap() is available we try
960  this first. If not available or it failed we try to load it. */
961  data = (struct mo_file_header *) mmap (NULL, size, PROT_READ,
962  MAP_PRIVATE, fd, 0);
963 
964  if (__builtin_expect (data != (struct mo_file_header *) -1, 1))
965  {
966  /* mmap() call was successful. */
967  close (fd);
968  use_mmap = 1;
969  }
970 #endif
971 
972  /* If the data is not yet available (i.e. mmap'ed) we try to load
973  it manually. */
974  if (data == (struct mo_file_header *) -1)
975  {
976  size_t to_read;
977  char *read_ptr;
978 
979  data = (struct mo_file_header *) malloc (size);
980  if (data == NULL)
981  {
982  if (use_mmap == 0)
983  close (fd);
984  return;
985  }
986 
987  to_read = size;
988  read_ptr = (char *) data;
989  do
990  {
991  long int nb = (long int) read (fd, read_ptr, to_read);
992  if (nb <= 0)
993  {
994 #ifdef EINTR
995  if (nb == -1 && errno == EINTR)
996  continue;
997 #endif
998  close (fd);
999  return;
1000  }
1001  read_ptr += nb;
1002  to_read -= nb;
1003  }
1004  while (to_read > 0);
1005 
1006  close (fd);
1007  }
1008 
1009  /* Using the magic number we can test whether it really is a message
1010  catalog file. */
1011  if (__builtin_expect (data->magic != _MAGIC && data->magic != _MAGIC_SWAPPED,
1012  0))
1013  {
1014  /* The magic number is wrong: not a message catalog file. */
1015 #ifdef HAVE_MMAP
1016  if (use_mmap)
1017  munmap ((caddr_t) data, size);
1018  else
1019 #endif
1020  free (data);
1021  return;
1022  }
1023 
1024  domain = (struct loaded_domain *) malloc (sizeof (struct loaded_domain));
1025  if (domain == NULL)
1026  {
1027 #ifdef HAVE_MMAP
1028  if (use_mmap)
1029  munmap ((caddr_t) data, size);
1030  else
1031 #endif
1032  free (data);
1033  return;
1034  }
1035  domain_file->data = domain;
1036 
1037  domain->data = (char *) data;
1038  domain->use_mmap = use_mmap;
1039  domain->mmap_size = size;
1040  domain->must_swap = data->magic != _MAGIC;
1041  domain->malloced = NULL;
1042 
1043  /* Fill in the information about the available tables. */
1044  revision = W (domain->must_swap, data->revision);
1045  /* We support only the major revision 0. */
1046  switch (revision >> 16)
1047  {
1048  case 0:
1049  domain->nstrings = W (domain->must_swap, data->nstrings);
1050  domain->orig_tab = (const struct string_desc *)
1051  ((char *) data + W (domain->must_swap, data->orig_tab_offset));
1052  domain->trans_tab = (const struct string_desc *)
1053  ((char *) data + W (domain->must_swap, data->trans_tab_offset));
1054  domain->hash_size = W (domain->must_swap, data->hash_tab_size);
1055  domain->hash_tab =
1056  (domain->hash_size > 2
1057  ? (const nls_uint32 *)
1058  ((char *) data + W (domain->must_swap, data->hash_tab_offset))
1059  : NULL);
1060  domain->must_swap_hash_tab = domain->must_swap;
1061 
1062  /* Now dispatch on the minor revision. */
1063  switch (revision & 0xffff)
1064  {
1065  case 0:
1066  domain->n_sysdep_strings = 0;
1067  domain->orig_sysdep_tab = NULL;
1068  domain->trans_sysdep_tab = NULL;
1069  break;
1070  case 1:
1071  default:
1072  {
1073  nls_uint32 n_sysdep_strings;
1074 
1075  if (domain->hash_tab == NULL)
1076  /* This is invalid. These minor revisions need a hash table. */
1077  goto invalid;
1078 
1079  n_sysdep_strings =
1080  W (domain->must_swap, data->n_sysdep_strings);
1081  if (n_sysdep_strings > 0)
1082  {
1083  nls_uint32 n_sysdep_segments;
1084  const struct sysdep_segment *sysdep_segments;
1085  const char **sysdep_segment_values;
1086  const nls_uint32 *orig_sysdep_tab;
1087  const nls_uint32 *trans_sysdep_tab;
1088  size_t memneed;
1089  char *mem;
1090  struct sysdep_string_desc *inmem_orig_sysdep_tab;
1091  struct sysdep_string_desc *inmem_trans_sysdep_tab;
1092  nls_uint32 *inmem_hash_tab;
1093  unsigned int i;
1094 
1095  /* Get the values of the system dependent segments. */
1096  n_sysdep_segments =
1097  W (domain->must_swap, data->n_sysdep_segments);
1098  sysdep_segments = (const struct sysdep_segment *)
1099  ((char *) data
1100  + W (domain->must_swap, data->sysdep_segments_offset));
1101  sysdep_segment_values =
1102  alloca (n_sysdep_segments * sizeof (const char *));
1103  for (i = 0; i < n_sysdep_segments; i++)
1104  {
1105  const char *name =
1106  (char *) data
1107  + W (domain->must_swap, sysdep_segments[i].offset);
1108  nls_uint32 namelen =
1109  W (domain->must_swap, sysdep_segments[i].length);
1110 
1111  if (!(namelen > 0 && name[namelen - 1] == '\0'))
1112  {
1113  freea (sysdep_segment_values);
1114  goto invalid;
1115  }
1116 
1117  sysdep_segment_values[i] = get_sysdep_segment_value (name);
1118  }
1119 
1120  orig_sysdep_tab = (const nls_uint32 *)
1121  ((char *) data
1122  + W (domain->must_swap, data->orig_sysdep_tab_offset));
1123  trans_sysdep_tab = (const nls_uint32 *)
1124  ((char *) data
1125  + W (domain->must_swap, data->trans_sysdep_tab_offset));
1126 
1127  /* Compute the amount of additional memory needed for the
1128  system dependent strings and the augmented hash table. */
1129  memneed = 2 * n_sysdep_strings
1130  * sizeof (struct sysdep_string_desc)
1131  + domain->hash_size * sizeof (nls_uint32);
1132  for (i = 0; i < 2 * n_sysdep_strings; i++)
1133  {
1134  const struct sysdep_string *sysdep_string =
1135  (const struct sysdep_string *)
1136  ((char *) data
1137  + W (domain->must_swap,
1138  i < n_sysdep_strings
1139  ? orig_sysdep_tab[i]
1140  : trans_sysdep_tab[i - n_sysdep_strings]));
1141  size_t need = 0;
1142  const struct segment_pair *p = sysdep_string->segments;
1143 
1144  if (W (domain->must_swap, p->sysdepref) != SEGMENTS_END)
1145  for (p = sysdep_string->segments;; p++)
1146  {
1147  nls_uint32 sysdepref;
1148 
1149  need += W (domain->must_swap, p->segsize);
1150 
1151  sysdepref = W (domain->must_swap, p->sysdepref);
1152  if (sysdepref == SEGMENTS_END)
1153  break;
1154 
1155  if (sysdepref >= n_sysdep_segments)
1156  {
1157  /* Invalid. */
1158  freea (sysdep_segment_values);
1159  goto invalid;
1160  }
1161 
1162  need += strlen (sysdep_segment_values[sysdepref]);
1163  }
1164 
1165  memneed += need;
1166  }
1167 
1168  /* Allocate additional memory. */
1169  mem = (char *) malloc (memneed);
1170  if (mem == NULL)
1171  goto invalid;
1172 
1173  domain->malloced = mem;
1174  inmem_orig_sysdep_tab = (struct sysdep_string_desc *) mem;
1175  mem += n_sysdep_strings * sizeof (struct sysdep_string_desc);
1176  inmem_trans_sysdep_tab = (struct sysdep_string_desc *) mem;
1177  mem += n_sysdep_strings * sizeof (struct sysdep_string_desc);
1178  inmem_hash_tab = (nls_uint32 *) mem;
1179  mem += domain->hash_size * sizeof (nls_uint32);
1180 
1181  /* Compute the system dependent strings. */
1182  for (i = 0; i < 2 * n_sysdep_strings; i++)
1183  {
1184  const struct sysdep_string *sysdep_string =
1185  (const struct sysdep_string *)
1186  ((char *) data
1187  + W (domain->must_swap,
1188  i < n_sysdep_strings
1189  ? orig_sysdep_tab[i]
1190  : trans_sysdep_tab[i - n_sysdep_strings]));
1191  const char *static_segments =
1192  (char *) data
1193  + W (domain->must_swap, sysdep_string->offset);
1194  const struct segment_pair *p = sysdep_string->segments;
1195

1197  inmem_orig_sysdep_tab[i] (for i < n_sysdep_strings) and
1198  inmem_trans_sysdep_tab[i-n_sysdep_strings] (for
1199  i >= n_sysdep_strings). */
1200

1202  {

1204  inmem_orig_sysdep_tab[i].length =
1205  W (domain->must_swap, p->segsize);
1206  inmem_orig_sysdep_tab[i].pointer = static_segments;
1207  }
1208  else
1209  {
1210  inmem_orig_sysdep_tab[i].pointer = mem;
1211 
1212  for (p = sysdep_string->segments;; p++)
1213  {
1214  nls_uint32 segsize =
1215  W (domain->must_swap, p->segsize);

1217  W (domain->must_swap, p->sysdepref);
1218  size_t n;
1219

1221  {
1222  memcpy (mem, static_segments, segsize);
1223  mem += segsize;
1224  static_segments += segsize;
1225  }
1226 
1227  if (sysdepref == SEGMENTS_END)
1228  break;
1229 
1230  n = strlen (sysdep_segment_values[sysdepref]);
1231  memcpy (mem, sysdep_segment_values[sysdepref], n);
1232  mem += n;
1233  }
1234 
1235  inmem_orig_sysdep_tab[i].length =
1236  mem - inmem_orig_sysdep_tab[i].pointer;
1237  }
1238  }
1239

1241  for (i = 0; i < domain->hash_size; i++)
1242  inmem_hash_tab[i] =
1243  W (domain->must_swap_hash_tab, domain->hash_tab[i]);

1257  break;
1258  }

1261  idx -= domain->hash_size - incr;
1262  else
1263  idx += incr;
1264  }
1265  }
1266 
1267  freea (sysdep_segment_values);
1268 
1269  domain->n_sysdep_strings = n_sysdep_strings;
1270  domain->orig_sysdep_tab = inmem_orig_sysdep_tab;
1271  domain->trans_sysdep_tab = inmem_trans_sysdep_tab;
1272 
1273  domain->hash_tab = inmem_hash_tab;
1274  domain->must_swap_hash_tab = 0;
1275  }
1276  else
1277  {
1278  domain->n_sysdep_strings = 0;
1279  domain->orig_sysdep_tab = NULL;
1280  domain->trans_sysdep_tab = NULL;
1281  }
1282  }
1283  break;
