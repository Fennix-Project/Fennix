134 //
135 // V_DrawPatch
136 // Masks a column based masked pic to the screen.
137 //
138 
139 void V_DrawPatch(int x, int y, patch_t *patch)
140 {
141  int count;
142  int col;
143  column_t *column;
144  byte *desttop;
145  byte *dest;
146  byte *source;
147  int w;
148 
149  y -= SHORT(patch->topoffset);
150  x -= SHORT(patch->leftoffset);
151 
152  // haleyjd 08/28/10: Strife needs silent error checking here.
153  if(patchclip_callback)
154  {
155  if(!patchclip_callback(patch, x, y))
156  return;
157  }
158 
159 #ifdef RANGECHECK
160  if (x < 0
161  || x + SHORT(patch->width) > SCREENWIDTH
162  || y < 0
163  || y + SHORT(patch->height) > SCREENHEIGHT)
164  {
165  I_Error("Bad V_DrawPatch x=%i y=%i patch.width=%i patch.height=%i topoffset=%i leftoffset=%i", x, y, patch->width, patch->height, patch->topoffset, patch->leftoffset);
166  }
167 #endif
168 
169  V_MarkRect(x, y, SHORT(patch->width), SHORT(patch->height));
170 
171  col = 0;
172  desttop = dest_screen + y * SCREENWIDTH + x;
173 
174  w = SHORT(patch->width);
175 
176  for ( ; col<w ; x++, col++, desttop++)
177  {
178  column = (column_t *)((byte *)patch + LONG(patch->columnofs[col]));
179 
180  // step through the posts in a column
181  while (column->topdelta != 0xff)
182  {
183  source = (byte *)column + 3;
184  dest = desttop + column->topdelta*SCREENWIDTH;
185  count = column->length;
186 
187  while (count--)
188  {
189  *dest = *source++;
190  dest += SCREENWIDTH;
191  }
192  column = (column_t *)((byte *)column + column->length + 4);
193  }
194  }
195 }

649 //
650 // WritePCXfile
651 //
652 
653 void WritePCXfile(char *filename, byte *data,
654  int width, int height,
655  byte *palette)
656 {
657  int i;
658  int length;
659  pcx_t* pcx;
660  byte* pack;
661 
662  pcx = Z_Malloc (width*height*2+1000, PU_STATIC, NULL);
663 
664  pcx->manufacturer = 0x0a; // PCX id
665  pcx->version = 5; // 256 color
666  pcx->encoding = 1; // uncompressed
667  pcx->bits_per_pixel = 8; // 256 color
668  pcx->xmin = 0;
669  pcx->ymin = 0;
670  pcx->xmax = SHORT(width-1);
671  pcx->ymax = SHORT(height-1);
672  pcx->hres = SHORT(width);
673  pcx->vres = SHORT(height);
674  memset (pcx->palette,0,sizeof(pcx->palette));
675  pcx->color_planes = 1; // chunky image
676  pcx->bytes_per_line = SHORT(width);
677  pcx->palette_type = SHORT(2); // not a grey scale
678  memset (pcx->filler,0,sizeof(pcx->filler));
679 
680  // pack the image
681  pack = &pcx->data;
682 
683  for (i=0 ; i<width*height ; i++)
684  {
685  if ( (*data & 0xc0) != 0xc0)
686  *pack++ = *data++;
687  else
688  {
689  *pack++ = 0xc1;
690  *pack++ = *data++;
691  }
692  }
693 
694  // write the palette
695  *pack++ = 0x0c; // palette ID byte
696  for (i=0 ; i<768 ; i++)
697  *pack++ = *palette++;
698 
699  // write output file
700  length = pack - (byte *)pcx;
701  M_WriteFile (filename, pcx, length);
702 
703  Z_Free (pcx);
704 }

721 void WritePNGfile(char *filename, byte *data,
722  int width, int height,
723  byte *palette)
724 {
725  png_structp ppng;
726  png_infop pinfo;
727  png_colorp pcolor;
728  FILE *handle;
729  int i;
730 
731  handle = fopen(filename, "wb");
732  if (!handle)
733  {
734  return;
735  }
736 
737  ppng = png_create_write_struct(PNG_LIBPNG_VER_STRING, NULL,
738  error_fn, warning_fn);
739  if (!ppng)
740  {
741  return;
742  }
743 
744  pinfo = png_create_info_struct(ppng);
745  if (!pinfo)
746  {
747  png_destroy_write_struct(&ppng, NULL);
748  return;
749  }
750 
751  png_init_io(ppng, handle);
752 
753  png_set_IHDR(ppng, pinfo, width, height,
754  8, PNG_COLOR_TYPE_PALETTE, PNG_INTERLACE_NONE,
755  PNG_COMPRESSION_TYPE_DEFAULT, PNG_FILTER_TYPE_DEFAULT);
756 
757  pcolor = malloc(sizeof(*pcolor) * 256);
758  if (!pcolor)
759  {
760  png_destroy_write_struct(&ppng, &pinfo);
761  return;
762  }
763 
764  for (i = 0; i < 256; i++)
765  {
766  pcolor[i].red = *(palette + 3 * i);
767  pcolor[i].green = *(palette + 3 * i + 1);
768  pcolor[i].blue = *(palette + 3 * i + 2);
769  }
770 
771  png_set_PLTE(ppng, pinfo, pcolor, 256);
772  free(pcolor);
773 
774  png_write_info(ppng, pinfo);
775 
776  for (i = 0; i < SCREENHEIGHT; i++)
777  {
778  png_write_row(ppng, data + i*SCREENWIDTH);
779  }
780 
781  png_write_end(ppng, pinfo);
782  png_destroy_write_struct(&ppng, &pinfo);
783  fclose(handle);
784 }

846 void V_DrawMouseSpeedBox(int speed)
847 {
848  extern int usemouse;
849  int bgcolor, bordercolor, red, black, white, yellow;
850  int box_x, box_y;
851  int original_speed;
852  int redline_x;
853  int linelen;
854 
855  // Get palette indices for colors for widget. These depend on the
856  // palette of the game being played.
857 
858  bgcolor = I_GetPaletteIndex(0x77, 0x77, 0x77);
859  bordercolor = I_GetPaletteIndex(0x55, 0x55, 0x55);
860  red = I_GetPaletteIndex(0xff, 0x00, 0x00);
861  black = I_GetPaletteIndex(0x00, 0x00, 0x00);
862  yellow = I_GetPaletteIndex(0xff, 0xff, 0x00);
863  white = I_GetPaletteIndex(0xff, 0xff, 0xff);
864 
865  // If the mouse is turned off or acceleration is turned off, don't
866  // draw the box at all.
867 
868  if (!usemouse || fabs(mouse_acceleration - 1) < 0.01)
869  {
870  return;
871  }
872 
873  // Calculate box position
874 
875  box_x = SCREENWIDTH - MOUSE_SPEED_BOX_WIDTH - 10;
876  box_y = 15;
877 
878  V_DrawFilledBox(box_x, box_y,
879  MOUSE_SPEED_BOX_WIDTH, MOUSE_SPEED_BOX_HEIGHT, bgcolor);
880  V_DrawBox(box_x, box_y,
881  MOUSE_SPEED_BOX_WIDTH, MOUSE_SPEED_BOX_HEIGHT, bordercolor);
882 
883  // Calculate the position of the red line. This is 1/3 of the way
884  // along the box.
885 
886  redline_x = MOUSE_SPEED_BOX_WIDTH / 3;
887 
888  // Undo acceleration and get back the original mouse speed
889 
890  if (speed < mouse_threshold)
891  {
892  original_speed = speed;
893  }
894  else
895  {
896  original_speed = speed - mouse_threshold;
897  original_speed = (int) (original_speed / mouse_acceleration);
898  original_speed += mouse_threshold;
899  }
900 
901  // Calculate line length
902 
903  linelen = (original_speed * redline_x) / mouse_threshold;
904 
905  // Draw horizontal "thermometer"
906 
907  if (linelen > MOUSE_SPEED_BOX_WIDTH - 1)
908  {
909  linelen = MOUSE_SPEED_BOX_WIDTH - 1;
910  }
911 
912  V_DrawHorizLine(box_x + 1, box_y + 4, MOUSE_SPEED_BOX_WIDTH - 2, black);
913 
914  if (linelen < redline_x)
915  {
916  V_DrawHorizLine(box_x + 1, box_y + MOUSE_SPEED_BOX_HEIGHT / 2,
917  linelen, white);
918  }
919  else
920  {
921  V_DrawHorizLine(box_x + 1, box_y + MOUSE_SPEED_BOX_HEIGHT / 2,
922  redline_x, white);
923  V_DrawHorizLine(box_x + redline_x, box_y + MOUSE_SPEED_BOX_HEIGHT / 2,
924  linelen - redline_x, yellow);
925  }
926 
927  // Draw red line
928 
929  V_DrawVertLine(box_x + redline_x, box_y + 1,
930  MOUSE_SPEED_BOX_HEIGHT - 2, red);
931 }
