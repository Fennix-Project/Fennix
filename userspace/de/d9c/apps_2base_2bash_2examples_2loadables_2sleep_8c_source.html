56 static int
57 parse_gnutimefmt (char *string, long *sp, long *up)
58 {
59  int c, r;
60  char *s, *ep;
61  long tsec, tusec, accumsec, accumusec, t;
62  int mult;
63 
64  tsec = tusec = 0;
65  accumsec = accumusec = 0;
66  mult = 1;
67 
68  for (s = string; s && *s; s++) {
69  r = uconvert(s, &accumsec, &accumusec, &ep);
70  if (r == 0 && *ep == 0)
71  return r;
72  c = *ep;
73  mult = 1;
74  switch (c) {
75  case '\0':
76  case 's':
77  mult = S_SEC;
78  break;
79  case 'm':
80  mult = S_MIN;
81  break;
82  case 'h':
83  mult = S_HOUR;
84  break;
85  case 'd':
86  mult = S_DAY;
87  break;
88  default:
89  return 0;
90  }
91 
92  /* multiply the accumulated value by the multiplier */
93  t = accumusec * mult;
94  accumsec = accumsec * mult + (t / 1000000);
95  accumusec = t % 1000000;
96 
97  /* add to running total */
98  tsec += accumsec;
99  tusec += accumusec;
100  if (tusec >= 1000000) {
101  tsec++;
102  tusec -= 1000000;
103  }
104 
105  /* reset and continue */
106  accumsec = accumusec = 0;
107  mult = 1;
108  if (c == 0)
109  break;
110  s = ep;
111  }
112 
113  if (sp)
114  *sp = tsec;
115  if (up)
116  *up = tusec;
117 
118  return 1;
119 }

121 int
122 sleep_builtin (WORD_LIST *list)
123 {
124  long sec, usec;
125  char *ep;
126  int r, mul;
127  time_t t;
128 
129  if (list == 0) {
130  builtin_usage();
131  return(EX_USAGE);
132  }
133 
134  /* Skip over `--' */
135  if (list->word && ISOPTION (list->word->word, '-'))
136  list = list->next;
137 
138  if (*list->word->word == '-' || list->next) {
139  builtin_usage ();
140  return (EX_USAGE);
141  }
142 
143  r = uconvert(list->word->word, &sec, &usec, &ep);
144  /*
145  * Maybe postprocess conversion failures here based on EP
146  *
147  * A heuristic: if the conversion failed, but the argument appears to
148  * contain a GNU-like interval specifier (e.g. "1m30s"), try to parse
149  * it. If we can't, return the right exit code to tell
150  * execute_builtin to try and execute a disk command instead.
151  */
152  if (r == 0 && (strchr ("dhms", *ep) || strpbrk (list->word->word, "dhms")))
153  r = parse_gnutimefmt (list->word->word, &sec, &usec);
154 
155  if (r) {
156  fsleep(sec, usec);
157  QUIT;
158  return(EXECUTION_SUCCESS);
159  }
160  builtin_error("%s: bad sleep interval", list->word->word);
161  return (EXECUTION_FAILURE);
162 }
