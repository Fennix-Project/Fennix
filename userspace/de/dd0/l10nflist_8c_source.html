172 /* Return number of bits set in X. */
173 static int pop PARAMS ((int x));
174 
175 static inline int
176 pop (x)
177  int x;
178 {
179  /* We assume that no more than 16 bits are used. */
180  x = ((x & ~0x5555) >> 1) + (x & 0x5555);
181  x = ((x & ~0x3333) >> 2) + (x & 0x3333);
182  x = ((x >> 4) + x) & 0x0f0f;
183  x = ((x >> 8) + x) & 0xff;
184 
185  return x;
186 }
