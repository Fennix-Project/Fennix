126 /* Handle signed arithmetic overflow and underflow. Have to do it this way
127  to avoid compilers optimizing out simpler overflow checks. */

129 /* Make sure that a+b does not exceed MAXV or is smaller than MINV (if b < 0).
130  Assumes that b > 0 if a > 0 and b < 0 if a < 0 */
131 #define ADDOVERFLOW(a,b,minv,maxv) \
132  ((((a) > 0) && ((b) > ((maxv) - (a)))) || \
