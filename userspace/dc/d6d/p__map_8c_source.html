95 // PIT_StompThing
96 //
97 boolean PIT_StompThing (mobj_t* thing)
98 {
99  fixed_t blockdist;
100 
101  if (!(thing->flags & MF_SHOOTABLE) )
102  return true;
103 
104  blockdist = thing->radius + tmthing->radius;
105 
106  if ( abs(thing->x - tmx) >= blockdist
107  || abs(thing->y - tmy) >= blockdist )
108  {
109  // didn't hit it
110  return true;
111  }
112 
113  // don't clip against self
114  if (thing == tmthing)
115  return true;
116 
117  // monsters don't stomp things except on boss level
118  if ( !tmthing->player && gamemap != 30)
119  return false;
120 
121  P_DamageMobj (thing, tmthing, tmthing, 10000);
122 
123  return true;
124 }

127 //
128 // P_TeleportMove
129 //
130 boolean
131 P_TeleportMove
132 ( mobj_t* thing,
133  fixed_t x,
134  fixed_t y )
135 {
136  int xl;
137  int xh;
138  int yl;
139  int yh;
140  int bx;
141  int by;
142 
143  subsector_t* newsubsec;
144 
145  // kill anything occupying the position
146  tmthing = thing;
147  tmflags = thing->flags;
148 
149  tmx = x;
150  tmy = y;
151 
152  tmbbox[BOXTOP] = y + tmthing->radius;
153  tmbbox[BOXBOTTOM] = y - tmthing->radius;
154  tmbbox[BOXRIGHT] = x + tmthing->radius;
155  tmbbox[BOXLEFT] = x - tmthing->radius;
156 
157  newsubsec = R_PointInSubsector (x,y);
158  ceilingline = NULL;
159 
160  // The base floor/ceiling is from the subsector
161  // that contains the point.
162  // Any contacted lines the step closer together
163  // will adjust them.
164  tmfloorz = tmdropoffz = newsubsec->sector->floorheight;
165  tmceilingz = newsubsec->sector->ceilingheight;
166 
167  validcount++;
168  numspechit = 0;
169 
170  // stomp on any things contacted
171  xl = (tmbbox[BOXLEFT] - bmaporgx - MAXRADIUS)>>MAPBLOCKSHIFT;
172  xh = (tmbbox[BOXRIGHT] - bmaporgx + MAXRADIUS)>>MAPBLOCKSHIFT;
173  yl = (tmbbox[BOXBOTTOM] - bmaporgy - MAXRADIUS)>>MAPBLOCKSHIFT;
174  yh = (tmbbox[BOXTOP] - bmaporgy + MAXRADIUS)>>MAPBLOCKSHIFT;
175 
176  for (bx=xl ; bx<=xh ; bx++)
177  for (by=yl ; by<=yh ; by++)
178  if (!P_BlockThingsIterator(bx,by,PIT_StompThing))
179  return false;
180 
181  // the move is ok,
182  // so link the thing into its new position
183  P_UnsetThingPosition (thing);
184 
185  thing->floorz = tmfloorz;
186  thing->ceilingz = tmceilingz;
187  thing->x = x;
188  thing->y = y;
189 
190  P_SetThingPosition (thing);
191 
192  return true;
193 }

202 //
203 // PIT_CheckLine
204 // Adjusts tmfloorz and tmceilingz as lines are contacted
205 //
206 boolean PIT_CheckLine (line_t* ld)
207 {
208  if (tmbbox[BOXRIGHT] <= ld->bbox[BOXLEFT]
209  || tmbbox[BOXLEFT] >= ld->bbox[BOXRIGHT]
210  || tmbbox[BOXTOP] <= ld->bbox[BOXBOTTOM]
211  || tmbbox[BOXBOTTOM] >= ld->bbox[BOXTOP] )
212  return true;
213 
214  if (P_BoxOnLineSide (tmbbox, ld) != -1)
215  return true;
216 
217  // A line has been hit
218 
219  // The moving thing's destination position will cross
220  // the given line.
221  // If this should not be allowed, return false.
222  // If the line is special, keep track of it
223  // to process later if the move is proven ok.
224  // NOTE: specials are NOT sorted by order,
225  // so two special lines that are only 8 pixels apart
226  // could be crossed in either order.
227 
228  if (!ld->backsector)
229  return false; // one sided line
230 
231  if (!(tmthing->flags & MF_MISSILE) )
232  {
233  if ( ld->flags & ML_BLOCKING )
234  return false; // explicitly blocking everything
235 
236  if ( !tmthing->player && ld->flags & ML_BLOCKMONSTERS )
237  return false; // block monsters only
238  }
239 
240  // set openrange, opentop, openbottom
241  P_LineOpening (ld);
242 
243  // adjust floor / ceiling heights
244  if (opentop < tmceilingz)
245  {
246  tmceilingz = opentop;
247  ceilingline = ld;
248  }
249 
250  if (openbottom > tmfloorz)
251  tmfloorz = openbottom;
252 
253  if (lowfloor < tmdropoffz)
254  tmdropoffz = lowfloor;
255 
256  // if contacted a special line, add it to the list
257  if (ld->special)
258  {
259  spechit[numspechit] = ld;
260  numspechit++;
261 
262  // fraggle: spechits overrun emulation code from prboom-plus
263  if (numspechit > MAXSPECIALCROSS_ORIGINAL)
264  {
265  SpechitOverrun(ld);
266  }
267  }
268 
269  return true;
270 }

272 //
273 // PIT_CheckThing
274 //
275 boolean PIT_CheckThing (mobj_t* thing)
276 {
277  fixed_t blockdist;
278  boolean solid;
279  int damage;
280 
281  if (!(thing->flags & (MF_SOLID|MF_SPECIAL|MF_SHOOTABLE) ))
282  return true;
283 
284  blockdist = thing->radius + tmthing->radius;
285 
286  if ( abs(thing->x - tmx) >= blockdist
287  || abs(thing->y - tmy) >= blockdist )
288  {
289  // didn't hit it
290  return true;
291  }
292 
293  // don't clip against self
294  if (thing == tmthing)
295  return true;
296 
297  // check for skulls slamming into things
298  if (tmthing->flags & MF_SKULLFLY)
299  {
300  damage = ((P_Random()%8)+1)*tmthing->info->damage;
301 
302  P_DamageMobj (thing, tmthing, tmthing, damage);
303 
304  tmthing->flags &= ~MF_SKULLFLY;
305  tmthing->momx = tmthing->momy = tmthing->momz = 0;
306 
307  P_SetMobjState (tmthing, tmthing->info->spawnstate);
308 
309  return false; // stop moving
310  }
311 
312 
313  // missiles can hit other things
314  if (tmthing->flags & MF_MISSILE)
315  {
316  // see if it went over / under
317  if (tmthing->z > thing->z + thing->height)
318  return true; // overhead
319  if (tmthing->z+tmthing->height < thing->z)
320  return true; // underneath
321 
322  if (tmthing->target
323  && (tmthing->target->type == thing->type ||
324  (tmthing->target->type == MT_KNIGHT && thing->type == MT_BRUISER)||
325  (tmthing->target->type == MT_BRUISER && thing->type == MT_KNIGHT) ) )
326  {
327  // Don't hit same species as originator.
328  if (thing == tmthing->target)
329  return true;
330 
331  // sdh: Add deh_species_infighting here. We can override the
332  // "monsters of the same species cant hurt each other" behavior
333  // through dehacked patches
334 
335  if (thing->type != MT_PLAYER && !deh_species_infighting)
336  {
337  // Explode, but do no damage.
338  // Let players missile other players.
339  return false;
340  }
341  }
342 
343  if (! (thing->flags & MF_SHOOTABLE) )
344  {
345  // didn't do any damage
346  return !(thing->flags & MF_SOLID);
347  }
348 
349  // damage / explode
350  damage = ((P_Random()%8)+1)*tmthing->info->damage;
351  P_DamageMobj (thing, tmthing, tmthing->target, damage);
352 
353  // don't traverse any more
354  return false;
355  }
356 
357  // check for special pickup
358  if (thing->flags & MF_SPECIAL)
359  {
360  solid = thing->flags&MF_SOLID;
361  if (tmflags&MF_PICKUP)
362  {
363  // can remove thing
364  P_TouchSpecialThing (thing, tmthing);
365  }
366  return !solid;
367  }
368 
369  return !(thing->flags & MF_SOLID);
370 }

377 //
378 // P_CheckPosition
379 // This is purely informative, nothing is modified
380 // (except things picked up).
381 //
382 // in:
383 // a mobj_t (can be valid or invalid)
384 // a position to be checked
385 // (doesn't need to be related to the mobj_t->x,y)
386 //
387 // during:
388 // special things are touched if MF_PICKUP
389 // early out on solid lines?
390 //
391 // out:
392 // newsubsec
393 // floorz
394 // ceilingz
395 // tmdropoffz
396 // the lowest point contacted
397 // (monsters won't move to a dropoff)
398 // speciallines[]
399 // numspeciallines
400 //
401 boolean
402 P_CheckPosition
403 ( mobj_t* thing,
404  fixed_t x,
405  fixed_t y )
406 {
407  int xl;
408  int xh;
409  int yl;
410  int yh;
411  int bx;
412  int by;
413  subsector_t* newsubsec;
414 
415  tmthing = thing;
416  tmflags = thing->flags;
417 
418  tmx = x;
419  tmy = y;
420 
421  tmbbox[BOXTOP] = y + tmthing->radius;
422  tmbbox[BOXBOTTOM] = y - tmthing->radius;
423  tmbbox[BOXRIGHT] = x + tmthing->radius;
424  tmbbox[BOXLEFT] = x - tmthing->radius;
425 
426  newsubsec = R_PointInSubsector (x,y);
427  ceilingline = NULL;
428 
429  // The base floor / ceiling is from the subsector
430  // that contains the point.
431  // Any contacted lines the step closer together
432  // will adjust them.
433  tmfloorz = tmdropoffz = newsubsec->sector->floorheight;
434  tmceilingz = newsubsec->sector->ceilingheight;
435 
436  validcount++;
437  numspechit = 0;
438 
439  if ( tmflags & MF_NOCLIP )
440  return true;
441 
442  // Check things first, possibly picking things up.
443  // The bounding box is extended by MAXRADIUS
444  // because mobj_ts are grouped into mapblocks
445  // based on their origin point, and can overlap
446  // into adjacent blocks by up to MAXRADIUS units.
447  xl = (tmbbox[BOXLEFT] - bmaporgx - MAXRADIUS)>>MAPBLOCKSHIFT;
448  xh = (tmbbox[BOXRIGHT] - bmaporgx + MAXRADIUS)>>MAPBLOCKSHIFT;
449  yl = (tmbbox[BOXBOTTOM] - bmaporgy - MAXRADIUS)>>MAPBLOCKSHIFT;
450  yh = (tmbbox[BOXTOP] - bmaporgy + MAXRADIUS)>>MAPBLOCKSHIFT;
451 
452  for (bx=xl ; bx<=xh ; bx++)
453  for (by=yl ; by<=yh ; by++)
454  if (!P_BlockThingsIterator(bx,by,PIT_CheckThing))
455  return false;
456 
457  // check lines
458  xl = (tmbbox[BOXLEFT] - bmaporgx)>>MAPBLOCKSHIFT;
459  xh = (tmbbox[BOXRIGHT] - bmaporgx)>>MAPBLOCKSHIFT;
460  yl = (tmbbox[BOXBOTTOM] - bmaporgy)>>MAPBLOCKSHIFT;
461  yh = (tmbbox[BOXTOP] - bmaporgy)>>MAPBLOCKSHIFT;
462 
463  for (bx=xl ; bx<=xh ; bx++)
464  for (by=yl ; by<=yh ; by++)
465  if (!P_BlockLinesIterator (bx,by,PIT_CheckLine))
466  return false;
467 
468  return true;
469 }

472 //
473 // P_TryMove
474 // Attempt to move to a new position,
475 // crossing special lines unless MF_TELEPORT is set.
476 //
477 boolean
478 P_TryMove
479 ( mobj_t* thing,
480  fixed_t x,
481  fixed_t y )
482 {
483  fixed_t oldx;
484  fixed_t oldy;
485  int side;
486  int oldside;
487  line_t* ld;
488 
489  floatok = false;
490  if (!P_CheckPosition (thing, x, y))
491  return false; // solid wall or thing
492 
493  if ( !(thing->flags & MF_NOCLIP) )
494  {
495  if (tmceilingz - tmfloorz < thing->height)
496  return false; // doesn't fit
497 
498  floatok = true;
499 
500  if ( !(thing->flags&MF_TELEPORT)
501  &&tmceilingz - thing->z < thing->height)
502  return false; // mobj must lower itself to fit
503 
504  if ( !(thing->flags&MF_TELEPORT)
505  && tmfloorz - thing->z > 24*FRACUNIT )
506  return false; // too big a step up
507 
508  if ( !(thing->flags&(MF_DROPOFF|MF_FLOAT))
509  && tmfloorz - tmdropoffz > 24*FRACUNIT )
510  return false; // don't stand over a dropoff
511  }
512 
513  // the move is ok,
514  // so link the thing into its new position
515  P_UnsetThingPosition (thing);
516 
517  oldx = thing->x;
518  oldy = thing->y;
519  thing->floorz = tmfloorz;
520  thing->ceilingz = tmceilingz;
521  thing->x = x;
522  thing->y = y;
523 
524  P_SetThingPosition (thing);
525 
526  // if any special lines were hit, do the effect
527  if (! (thing->flags&(MF_TELEPORT|MF_NOCLIP)) )
528  {
529  while (numspechit--)
530  {
531  // see if the line was crossed
532  ld = spechit[numspechit];
533  side = P_PointOnLineSide (thing->x, thing->y, ld);
534  oldside = P_PointOnLineSide (oldx, oldy, ld);
535  if (side != oldside)
536  {
537  if (ld->special)
538  P_CrossSpecialLine (ld-lines, oldside, thing);
539  }
540  }
541  }
542 
543  return true;
544 }

547 //
548 // P_ThingHeightClip
549 // Takes a valid thing and adjusts the thing->floorz,
550 // thing->ceilingz, and possibly thing->z.
551 // This is called for all nearby monsters
552 // whenever a sector changes height.
553 // If the thing doesn't fit,
554 // the z will be set to the lowest value
555 // and false will be returned.
556 //
557 boolean P_ThingHeightClip (mobj_t* thing)
558 {
559  boolean onfloor;
560 
561  onfloor = (thing->z == thing->floorz);
562 
563  P_CheckPosition (thing, thing->x, thing->y);
564  // what about stranding a monster partially off an edge?
565 
566  thing->floorz = tmfloorz;
567  thing->ceilingz = tmceilingz;
568 
569  if (onfloor)
570  {
571  // walking monsters rise and fall with the floor
572  thing->z = thing->floorz;
573  }
574  else
575  {
576  // don't adjust a floating monster unless forced to
577  if (thing->z+thing->height > thing->ceilingz)
578  thing->z = thing->ceilingz - thing->height;
579  }
580 
581  if (thing->ceilingz - thing->floorz < thing->height)
582  return false;
583 
584  return true;
585 }

606 //
607 // P_HitSlideLine
608 // Adjusts the xmove / ymove
609 // so that the next move will slide along the wall.
610 //
611 void P_HitSlideLine (line_t* ld)
612 {
613  int side;
614 
615  angle_t lineangle;
616  angle_t moveangle;
617  angle_t deltaangle;
618 
619  fixed_t movelen;
620  fixed_t newlen;
621 
622 
623  if (ld->slopetype == ST_HORIZONTAL)
624  {
625  tmymove = 0;
626  return;
627  }
628 
629  if (ld->slopetype == ST_VERTICAL)
630  {
631  tmxmove = 0;
632  return;
633  }
634 
635  side = P_PointOnLineSide (slidemo->x, slidemo->y, ld);
636 
637  lineangle = R_PointToAngle2 (0,0, ld->dx, ld->dy);
638 
639  if (side == 1)
640  lineangle += ANG180;
641 
642  moveangle = R_PointToAngle2 (0,0, tmxmove, tmymove);
643  deltaangle = moveangle-lineangle;
644 
645  if (deltaangle > ANG180)
646  deltaangle += ANG180;
647  // I_Error ("SlideLine: ang>ANG180");
648 
649  lineangle >>= ANGLETOFINESHIFT;
650  deltaangle >>= ANGLETOFINESHIFT;
651 
652  movelen = P_AproxDistance (tmxmove, tmymove);
653  newlen = FixedMul (movelen, finecosine[deltaangle]);
654 
655  tmxmove = FixedMul (newlen, finecosine[lineangle]);
656  tmymove = FixedMul (newlen, finesine[lineangle]);
657 }

668  I_Error ("PTR_SlideTraverse: not a line?");
669

671 
672  if ( ! (li->flags & ML_TWOSIDED) )
673  {
674  if (P_PointOnLineSide (slidemo->x, slidemo->y, li))
675  {
676  // don't hit the back side
677  return true;
678  }
679  goto isblocking;
680  }

683  P_LineOpening (li);
684

686  goto isblocking; // doesn't fit

692  goto isblocking; // too big a step up
693 
694  // this line doesn't block movement
695  return true;
696

698  // see if it is closer than best so far
699  isblocking:
700  if (in->frac < bestslidefrac)
701  {
702  secondslidefrac = bestslidefrac;
703  secondslideline = bestslideline;
704  bestslidefrac = in->frac;
705  bestslideline = li;
706  }
707

713 //
714 // P_SlideMove
715 // The momx / momy move is bad, so try to slide
716 // along a wall.
717 // Find the first line hit, move flush to it,
718 // and slide along it
719 //
720 // This is a kludgy mess.
721 //
722 void P_SlideMove (mobj_t* mo)
723 {
724  fixed_t leadx;
725  fixed_t leady;
726  fixed_t trailx;
727  fixed_t traily;
728  fixed_t newx;
729  fixed_t newy;
730  int hitcount;
731 
732  slidemo = mo;
733  hitcount = 0;
734 
735  retry:
736  if (++hitcount == 3)
737  goto stairstep; // don't loop forever
738 
739 
740  // trace along the three leading corners
741  if (mo->momx > 0)
742  {
743  leadx = mo->x + mo->radius;
744  trailx = mo->x - mo->radius;
745  }
746  else
747  {
748  leadx = mo->x - mo->radius;
749  trailx = mo->x + mo->radius;
750  }
751 
752  if (mo->momy > 0)
753  {
754  leady = mo->y + mo->radius;
755  traily = mo->y - mo->radius;
756  }
757  else
758  {
759  leady = mo->y - mo->radius;
760  traily = mo->y + mo->radius;
761  }
762 
763  bestslidefrac = FRACUNIT+1;
764 
765  P_PathTraverse ( leadx, leady, leadx+mo->momx, leady+mo->momy,
766  PT_ADDLINES, PTR_SlideTraverse );
767  P_PathTraverse ( trailx, leady, trailx+mo->momx, leady+mo->momy,
768  PT_ADDLINES, PTR_SlideTraverse );
769  P_PathTraverse ( leadx, traily, leadx+mo->momx, traily+mo->momy,
770  PT_ADDLINES, PTR_SlideTraverse );
771 
772  // move up to the wall
773  if (bestslidefrac == FRACUNIT+1)
774  {
