175 /* Interpretation of offset for color fields: All offsets are from the right,
176  * inside a "pixel" value, which is exactly 'bits_per_pixel' wide (means: you
177  * can use the offset as right argument to <<). A pixel afterwards is a bit
178  * stream and is written to video memory as that unmodified.
179  *
180  * For pseudocolor: offset and length should be the same for all color
181  * components. Offset specifies the position of the least significant bit
182  * of the palette index in a pixel value. Length indicates the number
183  * of available palette entries (i.e. # of entries = 1 << length).
184  */
185 struct fb_bitfield {
186  uint32_t offset; /* beginning of bitfield */
187  uint32_t length; /* length of bitfield */
188  uint32_t msb_right; /* != 0 : Most significant bit is */
189  /* right */
190 };
