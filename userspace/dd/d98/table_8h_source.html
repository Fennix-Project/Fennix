33  * Memory table entry.
34  *
35  * MEM is the address of the allocated pointer.
36  * SIZE is the requested allocation size.
37  * FLAGS includes either MT_ALLOC (MEM is allocated) or MT_FREE (MEM is
38  * not allocated). Other flags later.
39  * FUNC is set to the name of the function doing the allocation (from the
40  * `tag' argument to register_alloc().
41  * FILE and LINE are the filename and line number of the last allocation
42  * and free (depending on STATUS) of MEM.
43  * NALLOC and NFREE are incremented on each allocation that returns MEM or
44  * each free of MEM, respectively (way to keep track of memory reuse

47  */

80 /*
81  * void HASH_MIX
82  *
83  * DESCRIPTION:
84  *
85  * Mix 3 32-bit values reversibly. For every delta with one or two
86  * bits set, and the deltas of all three high bits or all three low
87  * bits, whether the original value of a,b,c is almost all zero or is
88  * uniformly distributed.
89  *
90  * If HASH_MIX() is run forward or backward, at least 32 bits in a,b,c
91  * have at least 1/4 probability of changing. If mix() is run
92  * forward, every bit of c will change between 1/3 and 2/3 of the
93  * time. (Well, 22/100 and 78/100 for some 2-bit deltas.)
94  *
95  * HASH_MIX() takes 36 machine instructions, but only 18 cycles on a
96  * superscalar machine (like a Pentium or a Sparc). No faster mixer
97  * seems to work, that's the result of my brute-force search. There
98  * were about 2^68 hashes to choose from. I only tested about a
99  * billion of those.
100  */
101 #define HASH_MIX(a, b, c) \
102  do { \
103  a -= b; a -= c; a ^= (c >> 13); \
104  b -= c; b -= a; b ^= (a << 8); \
105  c -= a; c -= b; c ^= (b >> 13); \
106  a -= b; a -= c; a ^= (c >> 12); \
107  b -= c; b -= a; b ^= (a << 16); \
108  c -= a; c -= b; c ^= (b >> 5); \
109  a -= b; a -= c; a ^= (c >> 3); \
110  b -= c; b -= a; b ^= (a << 10); \
111  c -= a; c -= b; c ^= (b >> 15); \
112  } while(0)
