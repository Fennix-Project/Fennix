239 wctype_t wctype(const char *cs) {
240  frg::string_view s{cs};
241  if(s == "alnum") return ct_alnum;
242  if(s == "alpha") return ct_alpha;
243  if(s == "blank") return ct_blank;
244  if(s == "cntrl") return ct_cntrl;
245  if(s == "digit") return ct_digit;
246  if(s == "graph") return ct_graph;
247  if(s == "lower") return ct_lower;
248  if(s == "print") return ct_print;
249  if(s == "punct") return ct_punct;
250  if(s == "space") return ct_space;
251  if(s == "upper") return ct_upper;
252  if(s == "xdigit") return ct_xdigit;
253  mlibc::infoLogger() << "mlibc: wctype(\"" << cs << "\") is not supported" << frg::endlog;
254  return ct_null;
255 }
