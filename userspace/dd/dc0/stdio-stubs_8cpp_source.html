1004 ssize_t getdelim(char **line, size_t *n, int delim, FILE *stream) {
1005  // Otherwise, we cannot store the buffer / size.
1006  if(!line || !n) {
1007  errno = EINVAL;
1008  return -1;
1009  }
1010 
1011  char *buffer = *line;
1012  /* set the starting capacity to 512 if buffer = NULL */
1013  size_t capacity = (!buffer) ? 512 : *n;
1014  size_t nwritten = 0;
1015 
1016  auto file = static_cast<mlibc::abstract_file *>(stream);
1017  frg::unique_lock lock(file->_lock);
1018 
1019  // Avoid allocating if we've already hit the end
1020  auto c = fgetc_unlocked(stream);
1021  if (c == EOF || ferror(stream)) {
1022  return -1;
1023  } else {
1024  file->unget(c);
1025  }
1026 
1027  while (true) {
1028  // Fill the buffer
1029  while (buffer && capacity > 0 && nwritten < capacity - 1) {
1030  auto c = fgetc_unlocked(stream);
1031  if (ferror(stream)) {
1032  return -1;
1033  } else if (c == EOF) {
1034  buffer[nwritten] = 0;
1035  return nwritten;
1036  }
1037 
1038  buffer[nwritten++] = c;
1039 
1040  if (c == delim) {
1041  buffer[nwritten] = 0;
1042  return nwritten;
1043  }
1044  }
1045 
1046  // Double the size of the buffer (but make sure it's at least 1024)
1047  capacity = (capacity >= 1024) ? capacity * 2 : 1024;
1048  buffer = reinterpret_cast<char *>(getAllocator().reallocate(*line, capacity));
1049  if (!buffer) {
1050  errno = ENOMEM;
1051  return -1;
1052  }
1053 
1054  *line = buffer;
1055  *n = capacity;
1056  }
