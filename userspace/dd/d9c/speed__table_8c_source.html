124 unsigned FAST_FUNC tty_baud_to_value(speed_t speed)
125 {
126  int i = 0;
127 
128  do {
129  if (speed == speeds[i].speed) {
130  if (speeds[i].value & 0x8000u) {
131  return ((unsigned)(speeds[i].value) & 0x7fffU) * 200;
132  }
133  return speeds[i].value;
134  }
135  } while (++i < NUM_SPEEDS);
136 
137  return 0;
138 }
