161 /* This is called to handle a signal when it is safe to do so (out of the
162  signal handler execution path). Called by _rl_signal_handler for all the
163  signals readline catches except SIGWINCH. */
164 static void
165 _rl_handle_signal (int sig)
166 {
167  int block_sig;
168 
169 #if defined (HAVE_POSIX_SIGNALS)
170  sigset_t set, oset;
171 #else /* !HAVE_POSIX_SIGNALS */
172 # if defined (HAVE_BSD_SIGNALS)
173  long omask;
174 # else /* !HAVE_BSD_SIGNALS */
175  sighandler_cxt dummy_cxt; /* needed for rl_set_sighandler call */
176 # endif /* !HAVE_BSD_SIGNALS */
177 #endif /* !HAVE_POSIX_SIGNALS */
178 
179  RL_SETSTATE(RL_STATE_SIGHANDLER);
180 
181 #if !defined (HAVE_BSD_SIGNALS) && !defined (HAVE_POSIX_SIGNALS)
182  /* Since the signal will not be blocked while we are in the signal
183  handler, ignore it until rl_clear_signals resets the catcher. */
184 # if defined (SIGALRM)
185  if (sig == SIGINT || sig == SIGALRM)
186 # else
187  if (sig == SIGINT)
188 # endif
189  rl_set_sighandler (sig, SIG_IGN, &dummy_cxt);
190 #endif /* !HAVE_BSD_SIGNALS && !HAVE_POSIX_SIGNALS */
191 
192  /* If there's a sig cleanup function registered, call it and `deregister'
193  the cleanup function to avoid multiple calls */
194  if (_rl_sigcleanup)
195  {
196  (*_rl_sigcleanup) (sig, _rl_sigcleanarg);
197  _rl_sigcleanup = 0;
198  _rl_sigcleanarg = 0;
199  }
200 
201 #if defined (HAVE_POSIX_SIGNALS)
202  /* Get the current set of blocked signals. If we want to block a signal for
203  the duration of the cleanup functions, make sure to add it to SET and
204  set block_sig = 1 (see the SIGHUP case below). */
205  block_sig = 0; /* sentinel to block signals with sigprocmask */
206  sigemptyset (&set);
207  sigprocmask (SIG_BLOCK, (sigset_t *)NULL, &set);
208 #endif
209 
210  switch (sig)
211  {
212  case SIGINT:
213  /* We will end up blocking SIGTTOU while we are resetting the tty, so
214  watch out for this if it causes problems. We could prevent this by
215  setting block_sig to 1 without modifying SET. */
216  _rl_reset_completion_state ();
217  rl_free_line_state ();
218 #if defined (READLINE_CALLBACKS)
219  rl_callback_sigcleanup ();
220 #endif
221 
222  /* FALLTHROUGH */
223 
224 #if defined (SIGTSTP)
225  case SIGTSTP:
226  case SIGTTIN:
227  case SIGTTOU:
228 # if defined (HAVE_POSIX_SIGNALS)
229  /* Block SIGTTOU so we can restore the terminal settings to something
230  sane without stopping on SIGTTOU if we have been placed into the
231  background. Even trying to get the current terminal pgrp with
232  tcgetpgrp() will generate SIGTTOU, so we don't bother. We still do
233  this even if we've been stopped on SIGTTOU, since we handle signals
234  when we have returned from the signal handler and the signal is no
235  longer blocked. */
236  if (block_sig == 0)
237  {
238  sigaddset (&set, SIGTTOU);
239  block_sig = 1;
240  }
241 # endif
242 #endif /* SIGTSTP */
243  /* Any signals that should be blocked during cleanup should go here. */
244 #if defined (SIGHUP)
245  case SIGHUP:
246 # if defined (_AIX)
247  if (block_sig == 0)
248  {
249  sigaddset (&set, sig);
250  block_sig = 1;
251  }
252 # endif // _AIX
253 #endif
254  /* Signals that don't require blocking during cleanup should go here. */
255  case SIGTERM:
256 #if defined (SIGALRM)
257  case SIGALRM:
258  if (sig == SIGALRM)
259  _rl_timeout_handle_sigalrm ();
260 #endif
261 #if defined (SIGQUIT)
262  case SIGQUIT:
263 #endif
264 
265 #if defined (HAVE_POSIX_SIGNALS)
266  if (block_sig)
267  sigprocmask (SIG_BLOCK, &set, &oset);
268 #endif
269 
270  rl_echo_signal_char (sig);
271  rl_cleanup_after_signal ();
272 
273  /* At this point, the application's signal handler, if any, is the
274  current handler. */
275 
276 #if defined (HAVE_POSIX_SIGNALS)
277  /* Unblock any signal(s) blocked above */
278  if (block_sig)
279  sigprocmask (SIG_UNBLOCK, &oset, (sigset_t *)NULL);
280 #endif
281 
282  /* We don't have to bother unblocking the signal because we are not
283  running in a signal handler context. */
284 
285 #if defined (__EMX__)
286  signal (sig, SIG_ACK);
287 #endif
288 
289 #if defined (HAVE_KILL)
290  kill (getpid (), sig);
291 #else
292  raise (sig); /* assume we have raise */
293 #endif
294 
295  /* We don't need to modify the signal mask now that this is not run in
296  a signal handler context. */
297 
298  rl_reset_after_signal ();
299  }
300 
301  RL_UNSETSTATE(RL_STATE_SIGHANDLER);
302  SIGHANDLER_RETURN;
303 }

345 /* Set up a readline-specific signal handler, saving the old signal
346  information in OHANDLER. Return the old signal handler, like
347  signal(). */
348 static SigHandler *
349 rl_set_sighandler (int sig, SigHandler *handler, sighandler_cxt *ohandler)
350 {
351  sighandler_cxt old_handler;
352 #if defined (HAVE_POSIX_SIGNALS)
353  struct sigaction act;
354 
355  act.sa_handler = handler;
356 # if defined (SIGWINCH)
357  act.sa_flags = (sig == SIGWINCH) ? SA_RESTART : 0;
358 # else
359  act.sa_flags = 0;
360 # endif /* SIGWINCH */
361  sigemptyset (&act.sa_mask);
362  sigemptyset (&ohandler->sa_mask);
363  sigaction (sig, &act, &old_handler);
364 #else
365  old_handler.sa_handler = (SigHandler *)signal (sig, handler);
366 #endif /* !HAVE_POSIX_SIGNALS */
367 
368  /* XXX -- assume we have memcpy */
369  /* If rl_set_signals is called twice in a row, don't set the old handler to
370  rl_signal_handler, because that would cause infinite recursion. */
371  if (handler != rl_signal_handler || old_handler.sa_handler != rl_signal_handler)
372  memcpy (ohandler, &old_handler, sizeof (sighandler_cxt));
373 
374  return (ohandler->sa_handler);
375 }

377 /* Set disposition of SIG to HANDLER, returning old state in OHANDLER. Don't
378  change disposition if OHANDLER indicates the signal was ignored. */
379 static void
380 rl_maybe_set_sighandler (int sig, SigHandler *handler, sighandler_cxt *ohandler)
381 {
382  sighandler_cxt dummy;
383  SigHandler *oh;
384 
385  sigemptyset (&dummy.sa_mask);
386  dummy.sa_flags = 0;
387  oh = rl_set_sighandler (sig, handler, ohandler);
388  if (oh == (SigHandler *)SIG_IGN)
389  rl_sigaction (sig, ohandler, &dummy);
390 }

392 /* Set the disposition of SIG to HANDLER, if HANDLER->sa_handler indicates the
393  signal was not being ignored. MUST only be called for signals whose
394  disposition was changed using rl_maybe_set_sighandler or for which the
395  SIG_IGN check was performed inline (e.g., SIGALRM below). */
396 static void
397 rl_maybe_restore_sighandler (int sig, sighandler_cxt *handler)
398 {
399  sighandler_cxt dummy;
400 
401  sigemptyset (&dummy.sa_mask);
402  dummy.sa_flags = 0;
403  if (handler->sa_handler != SIG_IGN)
404  rl_sigaction (sig, handler, &dummy);
405 }

407 int
408 rl_set_signals (void)
409 {
410  sighandler_cxt dummy;
411  SigHandler *oh;
412 #if defined (HAVE_POSIX_SIGNALS)
413  static int sigmask_set = 0;
414  static sigset_t bset, oset;
415 #endif
416 
417 #if defined (HAVE_POSIX_SIGNALS)
418  if (rl_catch_signals && sigmask_set == 0)
419  {
420  sigemptyset (&bset);
421 
422  sigaddset (&bset, SIGINT);
423  sigaddset (&bset, SIGTERM);
424 #if defined (SIGHUP)
425  sigaddset (&bset, SIGHUP);
426 #endif
427 #if defined (SIGQUIT)
428  sigaddset (&bset, SIGQUIT);
429 #endif
430 #if defined (SIGALRM)
431  sigaddset (&bset, SIGALRM);
432 #endif
433 #if defined (SIGTSTP)
434  sigaddset (&bset, SIGTSTP);
435 #endif
436 #if defined (SIGTTIN)
437  sigaddset (&bset, SIGTTIN);
438 #endif
439 #if defined (SIGTTOU)
440  sigaddset (&bset, SIGTTOU);
441 #endif
442  sigmask_set = 1;
443  }
444 #endif /* HAVE_POSIX_SIGNALS */
445 
446  if (rl_catch_signals && signals_set_flag == 0)
447  {
448 #if defined (HAVE_POSIX_SIGNALS)
449  sigemptyset (&_rl_orig_sigset);
450  sigprocmask (SIG_BLOCK, &bset, &_rl_orig_sigset);
451 #endif
452 
453  rl_maybe_set_sighandler (SIGINT, rl_signal_handler, &old_int);
454  rl_maybe_set_sighandler (SIGTERM, rl_signal_handler, &old_term);
455 #if defined (SIGHUP)
456  rl_maybe_set_sighandler (SIGHUP, rl_signal_handler, &old_hup);
457 #endif
458 #if defined (SIGQUIT)
459  rl_maybe_set_sighandler (SIGQUIT, rl_signal_handler, &old_quit);
460 #endif
461 
462 #if defined (SIGALRM)
463  oh = rl_set_sighandler (SIGALRM, rl_signal_handler, &old_alrm);
464  if (oh == (SigHandler *)SIG_IGN)
465  rl_sigaction (SIGALRM, &old_alrm, &dummy);
466 #if defined (HAVE_POSIX_SIGNALS) && defined (SA_RESTART)
467  /* If the application using readline has already installed a signal
468  handler with SA_RESTART, SIGALRM will cause reads to be restarted
469  automatically, so readline should just get out of the way. Since
470  we tested for SIG_IGN above, we can just test for SIG_DFL here. */
471  if (oh != (SigHandler *)SIG_DFL && (old_alrm.sa_flags & SA_RESTART))
472  rl_sigaction (SIGALRM, &old_alrm, &dummy);
473 #endif /* HAVE_POSIX_SIGNALS */
474 #endif /* SIGALRM */
475 
476 #if defined (SIGTSTP)
477  rl_maybe_set_sighandler (SIGTSTP, rl_signal_handler, &old_tstp);
478 #endif /* SIGTSTP */
479 
480 #if defined (SIGTTOU)
481  rl_maybe_set_sighandler (SIGTTOU, rl_signal_handler, &old_ttou);
482 #endif /* SIGTTOU */
483 
484 #if defined (SIGTTIN)
485  rl_maybe_set_sighandler (SIGTTIN, rl_signal_handler, &old_ttin);
486 #endif /* SIGTTIN */
487 
488  signals_set_flag = 1;
489 
490 #if defined (HAVE_POSIX_SIGNALS)
491  sigprocmask (SIG_SETMASK, &_rl_orig_sigset, (sigset_t *)NULL);
492 #endif
493  }
494  else if (rl_catch_signals == 0)
495  {
496 #if defined (HAVE_POSIX_SIGNALS)
497  sigemptyset (&_rl_orig_sigset);
498  sigprocmask (SIG_BLOCK, (sigset_t *)NULL, &_rl_orig_sigset);
499 #endif
500  }
501 
502 #if defined (SIGWINCH)
503  if (rl_catch_sigwinch && sigwinch_set_flag == 0)
504  {
505  rl_maybe_set_sighandler (SIGWINCH, rl_sigwinch_handler, &old_winch);
506  sigwinch_set_flag = 1;
507  }
508 #endif /* SIGWINCH */
509 
510  return 0;
511 }

513 int
514 rl_clear_signals (void)
515 {
516  sighandler_cxt dummy;
517 
518  if (rl_catch_signals && signals_set_flag == 1)
519  {
520  /* Since rl_maybe_set_sighandler doesn't override a SIG_IGN handler,
521  we should in theory not have to restore a handler where
522  old_xxx.sa_handler == SIG_IGN. That's what rl_maybe_restore_sighandler
523  does. Fewer system calls should reduce readline's per-line
524  overhead */
525  rl_maybe_restore_sighandler (SIGINT, &old_int);
526  rl_maybe_restore_sighandler (SIGTERM, &old_term);
527 #if defined (SIGHUP)
528  rl_maybe_restore_sighandler (SIGHUP, &old_hup);
529 #endif
530 #if defined (SIGQUIT)
531  rl_maybe_restore_sighandler (SIGQUIT, &old_quit);
532 #endif
533 #if defined (SIGALRM)
534  rl_maybe_restore_sighandler (SIGALRM, &old_alrm);
535 #endif
536 
537 #if defined (SIGTSTP)
538  rl_maybe_restore_sighandler (SIGTSTP, &old_tstp);
539 #endif /* SIGTSTP */
540 
541 #if defined (SIGTTOU)
542  rl_maybe_restore_sighandler (SIGTTOU, &old_ttou);
543 #endif /* SIGTTOU */
544 
545 #if defined (SIGTTIN)
546  rl_maybe_restore_sighandler (SIGTTIN, &old_ttin);
547 #endif /* SIGTTIN */
548 
549  signals_set_flag = 0;
550  }
551 
552 #if defined (SIGWINCH)
553  if (rl_catch_sigwinch && sigwinch_set_flag == 1)
554  {
555  sigemptyset (&dummy.sa_mask);
556  rl_sigaction (SIGWINCH, &old_winch, &dummy);
557  sigwinch_set_flag = 0;
558  }
559 #endif
560 
561  return 0;
