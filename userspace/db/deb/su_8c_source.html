113  if (ENABLE_FEATURE_SU_SYSLOG) {
114  /* The utmp entry (via getlogin) is probably the best way to
115  * identify the user, especially if someone su's from a su-shell.
116  * But getlogin can fail -- usually due to lack of utmp entry.
117  * in this case resort to getpwuid. */
118 #if ENABLE_FEATURE_UTMP
119  old_user = user_buf;
120  if (getlogin_r(user_buf, sizeof(user_buf)) != 0)
121 #endif

140  }

144  } else {

147  syslog(LOG_NOTICE, "%c %s %s:%s",
148  '-', tty, old_user, opt_username);

159  opt_shell = getenv("SHELL");
160  }
161 
162 #if ENABLE_FEATURE_SU_CHECKS_SHELLS
163  if (opt_shell && cur_uid != 0 && pw->pw_shell && restricted_shell(pw->pw_shell)) {
164  /* The user being su'd to has a nonstandard shell, and so is
165  * probably a uucp account or has restricted access. Don't
166  * compromise the account by allowing access with a standard
167  * shell. */
168  bb_simple_error_msg("using restricted shell");
169  opt_shell = NULL; /* ignore -s PROG */

173 #endif

176

179  ((flags & SU_OPT_l) ? (SETUP_ENV_CLEARENV + SETUP_ENV_CHDIR) : 0)
180  + (!(flags & SU_OPT_mp) * SETUP_ENV_CHANGEENV),
181  pw);
182  IF_SELINUX(set_current_security_context(NULL);)
183 
184  if (opt_command) {
185  *--argv = opt_command;
186  *--argv = (char*)"-c";
187  }
188 
189  /* A nasty ioctl exists which can stuff data into input queue:
190  * #include <sys/ioctl.h>
191  * int main() {
192  * const char *msg = "echo $UID\n";
193  * while (*msg) ioctl(0, TIOCSTI, *msg++);
194  * return 0;
195  * }
196  * With "su USER -c EXPLOIT" run by root, exploit can make root shell
197  * read as input and execute arbitrary command.
198  * It's debatable whether we need to protect against this
199  * (root may hesitate to run unknown scripts interactively).
200  *
201  * Some versions of su run -c CMD in a different session:
202  * ioctl(TIOCSTI) works only on the controlling tty.
203  */
204 
205  /* Never returns */
206  exec_shell(opt_shell, flags & SU_OPT_l, (const char**)argv);
207 
208  /* return EXIT_FAILURE; - not reached */
