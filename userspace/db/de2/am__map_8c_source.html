270 // Calculates the slope and slope according to the x-axis of a line
271 // segment in map coordinates (with the upright y-axis n' all) so
272 // that it can be used with the brain-dead drawing stuff.
273 
274 void
275 AM_getIslope
276 ( mline_t* ml,
277  islope_t* is )
278 {
279  int dx, dy;
280 
281  dy = ml->a.y - ml->b.y;
282  dx = ml->b.x - ml->a.x;
283  if (!dy) is->islp = (dx<0?-__INT_MAX__:__INT_MAX__);
284  else is->islp = FixedDiv(dx, dy);
285  if (!dx) is->slp = (dy<0?-__INT_MAX__:__INT_MAX__);
286  else is->slp = FixedDiv(dy, dx);
287 
288 }

351 //
352 // Determines bounding box of all vertices,
353 // sets global variables controlling zoom range.
354 //
355 void AM_findMinMaxBoundaries(void)
356 {
357  int i;
358  fixed_t a;
359  fixed_t b;
360 
361  min_x = min_y = __INT_MAX__;
362  max_x = max_y = -__INT_MAX__;
363 
364  for (i=0;i<numvertexes;i++)
365  {
366  if (vertexes[i].x < min_x)
367  min_x = vertexes[i].x;
368  else if (vertexes[i].x > max_x)
369  max_x = vertexes[i].x;
370 
371  if (vertexes[i].y < min_y)
372  min_y = vertexes[i].y;
373  else if (vertexes[i].y > max_y)
374  max_y = vertexes[i].y;
375  }
376 
377  max_w = max_x - min_x;
378  max_h = max_y - min_y;
379 
380  min_w = 2*PLAYERRADIUS; // const? never changed?
381  min_h = 2*PLAYERRADIUS;
382 
383  a = FixedDiv(f_w<<FRACBITS, max_w);
384  b = FixedDiv(f_h<<FRACBITS, max_h);
385 
386  min_scale_mtof = a < b ? a : b;
387  max_scale_mtof = FixedDiv(f_h<<FRACBITS, 2*PLAYERRADIUS);
388 
389 }

591 //
592 // Handle events (user inputs) in automap mode
593 //
594 boolean
595 AM_Responder
596 ( event_t* ev )
597 {
598 
599  int rc;
600  static int bigstate=0;
601  static char buffer[20];
602  int key;
603 
604  rc = false;
605 
606  if (!automapactive)
607  {
608  if (ev->type == ev_keydown && ev->data1 == key_map_toggle)
609  {
610  AM_Start ();
611  viewactive = false;
612  rc = true;
613  }
614  }
615  else if (ev->type == ev_keydown)
616  {
617  rc = true;
618  key = ev->data1;
619 
620  if (key == key_map_east) // pan right
621  {
622  if (!followplayer) m_paninc.x = FTOM(F_PANINC);
623  else rc = false;
624  }
625  else if (key == key_map_west) // pan left
626  {
627  if (!followplayer) m_paninc.x = -FTOM(F_PANINC);
628  else rc = false;
629  }
630  else if (key == key_map_north) // pan up
631  {
632  if (!followplayer) m_paninc.y = FTOM(F_PANINC);
633  else rc = false;
634  }
635  else if (key == key_map_south) // pan down
636  {
637  if (!followplayer) m_paninc.y = -FTOM(F_PANINC);
638  else rc = false;
639  }
640  else if (key == key_map_zoomout) // zoom out
641  {
642  mtof_zoommul = M_ZOOMOUT;
643  ftom_zoommul = M_ZOOMIN;
644  }
645  else if (key == key_map_zoomin) // zoom in
646  {
647  mtof_zoommul = M_ZOOMIN;
648  ftom_zoommul = M_ZOOMOUT;
649  }
650  else if (key == key_map_toggle)
651  {
652  bigstate = 0;
653  viewactive = true;
654  AM_Stop ();
655  }
656  else if (key == key_map_maxzoom)
657  {
658  bigstate = !bigstate;
659  if (bigstate)
660  {
661  AM_saveScaleAndLoc();
662  AM_minOutWindowScale();
663  }
664  else AM_restoreScaleAndLoc();
665  }
666  else if (key == key_map_follow)
667  {
668  followplayer = !followplayer;
669  f_oldloc.x = __INT_MAX__;
670  if (followplayer)
671  plr->message = DEH_String(AMSTR_FOLLOWON);
672  else
673  plr->message = DEH_String(AMSTR_FOLLOWOFF);
674  }
675  else if (key == key_map_grid)
676  {
677  grid = !grid;
678  if (grid)
679  plr->message = DEH_String(AMSTR_GRIDON);
680  else
681  plr->message = DEH_String(AMSTR_GRIDOFF);
682  }
683  else if (key == key_map_mark)
684  {
685  M_snprintf(buffer, sizeof(buffer), "%s %d",
686  DEH_String(AMSTR_MARKEDSPOT), markpointnum);
687  plr->message = buffer;
688  AM_addMark();
689  }
690  else if (key == key_map_clearmark)
691  {
692  AM_clearMarks();
693  plr->message = DEH_String(AMSTR_MARKSCLEARED);
694  }
695  else
696  {
697  rc = false;
698  }
699 
700  if (!deathmatch && cht_CheckCheat(&cheat_amap, ev->data2))
701  {
702  rc = false;
703  cheating = (cheating+1) % 3;
704  }

840 //
841 // Automap clipping of lines.
842 //
843 // Based on Cohen-Sutherland clipping algorithm but with a slightly
844 // faster reject and precalculated slopes. If the speed is needed,
845 // use a hash algorithm to handle the common cases.
846 //
847 boolean
848 AM_clipMline
849 ( mline_t* ml,
850  fline_t* fl )
851 {
852  enum
853  {
854  LEFT =1,
855  RIGHT =2,
856  BOTTOM =4,
857  TOP =8
858  };
859 
860  register int outcode1 = 0;
861  register int outcode2 = 0;
862  register int outside;
863 
864  fpoint_t tmp;
865  int dx;
866  int dy;
867 
868 
869 #define DOOUTCODE(oc, mx, my) \
870  (oc) = 0; \
871  if ((my) < 0) (oc) |= TOP; \
872  else if ((my) >= f_h) (oc) |= BOTTOM; \
873  if ((mx) < 0) (oc) |= LEFT; \
874  else if ((mx) >= f_w) (oc) |= RIGHT;
875 
876 
877  // do trivial rejects and outcodes
878  if (ml->a.y > m_y2)
879  outcode1 = TOP;
880  else if (ml->a.y < m_y)
881  outcode1 = BOTTOM;
882 
883  if (ml->b.y > m_y2)
884  outcode2 = TOP;
885  else if (ml->b.y < m_y)
886  outcode2 = BOTTOM;
887 
888  if (outcode1 & outcode2)
889  return false; // trivially outside
890 
891  if (ml->a.x < m_x)
892  outcode1 |= LEFT;
893  else if (ml->a.x > m_x2)
894  outcode1 |= RIGHT;
895 
896  if (ml->b.x < m_x)
897  outcode2 |= LEFT;
898  else if (ml->b.x > m_x2)
899  outcode2 |= RIGHT;
900 
901  if (outcode1 & outcode2)
902  return false; // trivially outside
903 
904  // transform to frame-buffer coordinates.
905  fl->a.x = CXMTOF(ml->a.x);
906  fl->a.y = CYMTOF(ml->a.y);
907  fl->b.x = CXMTOF(ml->b.x);
908  fl->b.y = CYMTOF(ml->b.y);
909 
910  DOOUTCODE(outcode1, fl->a.x, fl->a.y);
911  DOOUTCODE(outcode2, fl->b.x, fl->b.y);
912 
913  if (outcode1 & outcode2)
914  return false;
915 
916  while (outcode1 | outcode2)
917  {
918  // may be partially inside box
919  // find an outside point
920  if (outcode1)
921  outside = outcode1;
922  else
923  outside = outcode2;
924 
925  // clip to each side
926  if (outside & TOP)
927  {
928  dy = fl->a.y - fl->b.y;
929  dx = fl->b.x - fl->a.x;
930  tmp.x = fl->a.x + (dx*(fl->a.y))/dy;
931  tmp.y = 0;
932  }
933  else if (outside & BOTTOM)
934  {
935  dy = fl->a.y - fl->b.y;
936  dx = fl->b.x - fl->a.x;
937  tmp.x = fl->a.x + (dx*(fl->a.y-f_h))/dy;
938  tmp.y = f_h-1;
939  }
940  else if (outside & RIGHT)
941  {
942  dy = fl->b.y - fl->a.y;
943  dx = fl->b.x - fl->a.x;
944  tmp.y = fl->a.y + (dy*(f_w-1 - fl->a.x))/dx;
945  tmp.x = f_w-1;
946  }
947  else if (outside & LEFT)
948  {
949  dy = fl->b.y - fl->a.y;
950  dx = fl->b.x - fl->a.x;
951  tmp.y = fl->a.y + (dy*(-fl->a.x))/dx;
952  tmp.x = 0;
953  }
954  else
955  {
956  tmp.x = 0;
957  tmp.y = 0;
958  }
959 
960  if (outside == outcode1)
961  {
962  fl->a = tmp;
963  DOOUTCODE(outcode1, fl->a.x, fl->a.y);
964  }
965  else
966  {
967  fl->b = tmp;
968  DOOUTCODE(outcode2, fl->b.x, fl->b.y);
969  }
970 
971  if (outcode1 & outcode2)
972  return false; // trivially outside
973  }
974 
975  return true;
976 }
