39 FCT (pattern, string, flags)
40  CHAR *pattern;
41  CHAR *string;
42  int flags;

45

55 /* Match STRING against the filename pattern PATTERN, returning zero if
56  it matches, FNM_NOMATCH if not. */
57 static int
58 GMATCH (string, se, pattern, pe, ends, flags)
59  CHAR *string, *se;
60  CHAR *pattern, *pe;
61  struct STRUCT *ends;
62  int flags;
63 {
64  CHAR *p, *n; /* pattern, string */
65  INT c; /* current pattern character - XXX U_CHAR? */
66  INT sc; /* current string character - XXX U_CHAR? */
67 
68  p = pattern;
69  n = string;
70 
71  if (string == 0 || pattern == 0)
72  return FNM_NOMATCH;
73 
74 #if DEBUG_MATCHING
75 fprintf(stderr, "gmatch: string = %s; se = %s\n", string, se);
76 fprintf(stderr, "gmatch: pattern = %s; pe = %s\n", pattern, pe);
77 #endif
78 
79  while (p < pe)
80  {
81  c = *p++;
82  c = FOLD (c);
83 
84  sc = n < se ? *n : '\0';
85 
86  if (interrupt_state || terminating_signal)
87  return FNM_NOMATCH;
88 
89 #ifdef EXTENDED_GLOB
90  /* EXTMATCH () will handle recursively calling GMATCH, so we can
91  just return what EXTMATCH() returns. */
92  if ((flags & FNM_EXTMATCH) && *p == L('(') &&
93  (c == L('+') || c == L('*') || c == L('?') || c == L('@') || c == L('!'))) /* ) */
94  {
95  int lflags;
96  /* If we're not matching the start of the string, we're not
97  concerned about the special cases for matching `.' */
98  lflags = (n == string) ? flags : (flags & ~(FNM_PERIOD|FNM_DOTDOT));
99  return (EXTMATCH (c, n, se, p, pe, lflags));
100  }
101 #endif /* EXTENDED_GLOB */
102 
103  switch (c)
104  {
105  case L('?'): /* Match single character */
106  if (sc == '\0')
107  return FNM_NOMATCH;
108  else if ((flags & FNM_PATHNAME) && sc == L('/'))
109  /* If we are matching a pathname, `?' can never match a `/'. */
110  return FNM_NOMATCH;
111  else if ((flags & FNM_PERIOD) && sc == L('.') &&
112  (n == string || ((flags & FNM_PATHNAME) && n[-1] == L('/'))))
113  /* `?' cannot match a `.' if it is the first character of the
114  string or if it is the first character following a slash and
115  we are matching a pathname. */
116  return FNM_NOMATCH;
117 
118  /* `?' cannot match `.' or `..' if it is the first character of the
119  string or if it is the first character following a slash and
120  we are matching a pathname. */
121  if ((flags & FNM_DOTDOT) &&
122  ((n == string && SDOT_OR_DOTDOT(n)) ||
123  ((flags & FNM_PATHNAME) && n[-1] == L('/') && PDOT_OR_DOTDOT(n))))
124  return FNM_NOMATCH;
125 
126  break;
127 
128  case L('\\'): /* backslash escape removes special meaning */
129  if (p == pe && sc == '\\' && (n+1 == se))
130  break;
131 
132  if (p == pe)
133  return FNM_NOMATCH;
134 
135  if ((flags & FNM_NOESCAPE) == 0)
136  {
137  c = *p++;
138  /* A trailing `\' cannot match. */
139  if (p > pe)
140  return FNM_NOMATCH;
141  c = FOLD (c);
142  }
143  if (FOLD (sc) != (U_CHAR)c)
144  return FNM_NOMATCH;
145  break;
146 
147  case L('*'): /* Match zero or more characters */
148  /* See below for the reason for using this. It avoids backtracking
149  back to a previous `*'. Picked up from glibc. */
150  if (ends != NULL)
151  {
152  ends->pattern = p - 1;
153  ends->string = n;
154  return (0);
155  }
156 
157  if ((flags & FNM_PERIOD) && sc == L('.') &&
158  (n == string || ((flags & FNM_PATHNAME) && n[-1] == L('/'))))
159  /* `*' cannot match a `.' if it is the first character of the
160  string or if it is the first character following a slash and
161  we are matching a pathname. */
162  return FNM_NOMATCH;
163 
164  /* `*' cannot match `.' or `..' if it is the first character of the
165  string or if it is the first character following a slash and
166  we are matching a pathname. */
167  if ((flags & FNM_DOTDOT) &&
168  ((n == string && SDOT_OR_DOTDOT(n)) ||
169  ((flags & FNM_PATHNAME) && n[-1] == L('/') && PDOT_OR_DOTDOT(n))))
170  return FNM_NOMATCH;
171 
172  if (p == pe)
173  return 0;
174 
175  /* Collapse multiple consecutive `*' and `?', but make sure that
176  one character of the string is consumed for each `?'. */
177  for (c = *p++; (c == L('?') || c == L('*')); c = *p++)
178  {
179  if ((flags & FNM_PATHNAME) && sc == L('/'))
180  /* A slash does not match a wildcard under FNM_PATHNAME. */
181  return FNM_NOMATCH;
182 #ifdef EXTENDED_GLOB
183  else if ((flags & FNM_EXTMATCH) && c == L('?') && *p == L('(')) /* ) */
184  {
185  CHAR *newn;
186 
187  /* We can match 0 or 1 times. If we match, return success */
188  if (EXTMATCH (c, n, se, p, pe, flags) == 0)
189  return (0);
190 
191  /* We didn't match the extended glob pattern, but
192  that's OK, since we can match 0 or 1 occurrences.
193  We need to skip the glob pattern and see if we
194  match the rest of the string. */
195  newn = PATSCAN (p + 1, pe, 0);
196  /* If NEWN is 0, we have an ill-formed pattern. */
197  p = newn ? newn : pe;
198  }
199 #endif
200  else if (c == L('?'))
201  {
202  if (sc == L('\0'))
203  return FNM_NOMATCH;
204  /* One character of the string is consumed in matching
205  this ? wildcard, so *??? won't match if there are
206  fewer than three characters. */
207  n++;
208  sc = n < se ? *n : '\0';
209  }
210 
211 #ifdef EXTENDED_GLOB
212  /* Handle ******(patlist) */
213  if ((flags & FNM_EXTMATCH) && c == L('*') && *p == L('(')) /*)*/
214  {
215  CHAR *newn;
216  /* We need to check whether or not the extended glob
217  pattern matches the remainder of the string.
218  If it does, we match the entire pattern. */
219  for (newn = n; newn < se; ++newn)
220  {
221  if (EXTMATCH (c, newn, se, p, pe, flags) == 0)
222  return (0);
223  }
224  /* We didn't match the extended glob pattern, but
225  that's OK, since we can match 0 or more occurrences.
226  We need to skip the glob pattern and see if we
227  match the rest of the string. */
228  newn = PATSCAN (p + 1, pe, 0);
229  /* If NEWN is 0, we have an ill-formed pattern. */
230  p = newn ? newn : pe;
231  }
232 #endif
233  if (p == pe)
234  break;
235  }
236 
237  /* The wildcards are the last element of the pattern. The name
238  cannot match completely if we are looking for a pathname and
239  it contains another slash, unless FNM_LEADING_DIR is set. */
240  if (c == L('\0'))
241  {
242  int r = (flags & FNM_PATHNAME) == 0 ? 0 : FNM_NOMATCH;
243  if (flags & FNM_PATHNAME)
244  {
245  if (flags & FNM_LEADING_DIR)
246  r = 0;
247  else if (MEMCHR (n, L('/'), se - n) == NULL)
248  r = 0;
249  }
250  return r;
251  }
252 
253  /* If we've hit the end of the pattern and the last character of
254  the pattern was handled by the loop above, we've succeeded.
255  Otherwise, we need to match that last character. */
256  if (p == pe && (c == L('?') || c == L('*')))
257  return (0);
258 
259  /* If we've hit the end of the string and the rest of the pattern
260  is something that matches the empty string, we can succeed. */
261 #if defined (EXTENDED_GLOB)
262  if (n == se && ((flags & FNM_EXTMATCH) && (c == L('!') || c == L('?')) && *p == L('(')))
263  {
264  --p;
265  if (EXTMATCH (c, n, se, p, pe, flags) == 0)
266  return (c == L('!') ? FNM_NOMATCH : 0);
267  return (c == L('!') ? 0 : FNM_NOMATCH);
268  }
269 #endif
270 
271  /* If we stop at a slash in the pattern and we are looking for a
272  pathname ([star]/foo), then consume enough of the string to stop
273  at any slash and then try to match the rest of the pattern. If
274  the string doesn't contain a slash, fail */
275  if (c == L('/') && (flags & FNM_PATHNAME))
276  {
277  while (n < se && *n != L('/'))
278  ++n;
279  if (n < se && *n == L('/') && (GMATCH (n+1, se, p, pe, NULL, flags) == 0))
280  return 0;
281  return FNM_NOMATCH; /* XXX */
282  }
283 
284  /* General case, use recursion. */
285  {
286  U_CHAR c1;
287  const CHAR *endp;
288  struct STRUCT end;
289 
290  end.pattern = NULL;
291  endp = MEMCHR (n, (flags & FNM_PATHNAME) ? L('/') : L('\0'), se - n);
292  if (endp == 0)
293  endp = se;
294 
295  c1 = ((flags & FNM_NOESCAPE) == 0 && c == L('\\')) ? *p : c;
296  c1 = FOLD (c1);
297  for (--p; n < endp; ++n)
298  {
299  /* Only call strmatch if the first character indicates a
300  possible match. We can check the first character if
301  we're not doing an extended glob match. */
302  if ((flags & FNM_EXTMATCH) == 0 && c != L('[') && FOLD (*n) != c1) /*]*/
303  continue;
304 
305  /* If we're doing an extended glob match and the pattern is not
306  one of the extended glob patterns, we can check the first
307  character. */
308  if ((flags & FNM_EXTMATCH) && p[1] != L('(') && /*)*/
309  STRCHR (L("?*+@!"), *p) == 0 && c != L('[') && FOLD (*n) != c1) /*]*/
310  continue;
311 
312  /* Otherwise, we just recurse. */
313  if (GMATCH (n, se, p, pe, &end, flags & ~(FNM_PERIOD|FNM_DOTDOT)) == 0)
314  {
315  if (end.pattern == NULL)
316  return (0);
317  break;
318  }
319  }
320  /* This is a clever idea from glibc, used to avoid backtracking
321  to a `*' that appears earlier in the pattern. We get away
322  without saving se and pe because they are always the same,
323  even in the recursive calls to gmatch */
324  if (end.pattern != NULL)
325  {
326  p = end.pattern;
327  n = end.string;
328  continue;
329  }
330 
331  return FNM_NOMATCH;
332  }
333 
334  case L('['):
335  {
336  if (sc == L('\0') || n == se)
337  return FNM_NOMATCH;
338 
339  /* A character class cannot match a `.' if it is the first
340  character of the string or if it is the first character
341  following a slash and we are matching a pathname. */
342  if ((flags & FNM_PERIOD) && sc == L('.') &&
343  (n == string || ((flags & FNM_PATHNAME) && n[-1] == L('/'))))
344  return (FNM_NOMATCH);
345 
346  /* `?' cannot match `.' or `..' if it is the first character of the
347  string or if it is the first character following a slash and
348  we are matching a pathname. */
349  if ((flags & FNM_DOTDOT) &&
350  ((n == string && SDOT_OR_DOTDOT(n)) ||
351  ((flags & FNM_PATHNAME) && n[-1] == L('/') && PDOT_OR_DOTDOT(n))))
352  return FNM_NOMATCH;
353 
354  p = BRACKMATCH (p, sc, flags);
355  if (p == 0)
356  return FNM_NOMATCH;
357  }
358  break;
359 
360  default:
361  if ((U_CHAR)c != FOLD (sc))
362  return (FNM_NOMATCH);
363  }
364 
365  ++n;
366  }
367 
368  if (n == se)
369  return (0);
370 
371  if ((flags & FNM_LEADING_DIR) && *n == L('/'))
372  /* The FNM_LEADING_DIR flag says that "foo*" matches "foobar/frobozz". */
373  return 0;
374 
375  return (FNM_NOMATCH);
376 }

378 /* Parse a bracket expression collating symbol ([.sym.]) starting at P, find
379  the value of the symbol, and move P past the collating symbol expression.
380  The value is returned in *VP, if VP is not null. */
381 static CHAR *
382 PARSE_COLLSYM (p, vp)
383  CHAR *p;
384  INT *vp;
385 {
386  register int pc;
387  INT val;
388 
389  p++; /* move past the `.' */
390 
391  for (pc = 0; p[pc]; pc++)
392  if (p[pc] == L('.') && p[pc+1] == L(']'))
393  break;
394  if (p[pc] == 0)
395  {
396  if (vp)
397  *vp = INVALID;
398  return (p + pc);
399  }
400  val = COLLSYM (p, pc);
401  if (vp)
402  *vp = val;
403  return (p + pc + 2);
404 }

406 /* Use prototype definition here because of type promotion. */
407 static CHAR *
408 #if defined (PROTOTYPES)
409 BRACKMATCH (CHAR *p, U_CHAR test, int flags)
410 #else
411 BRACKMATCH (p, test, flags)
412  CHAR *p;
413  U_CHAR test;
414  int flags;
415 #endif
416 {
417  register CHAR cstart, cend, c;
418  register int not; /* Nonzero if the sense of the character class is inverted. */
419  int brcnt, forcecoll, isrange;
420  INT pc;
421  CHAR *savep;
422  CHAR *brchrp;
423  U_CHAR orig_test;
424 
425  orig_test = test;
426  test = FOLD (orig_test);
427 
428  savep = p;
429 
430  /* POSIX.2 3.13.1 says that an exclamation mark (`!') shall replace the
431  circumflex (`^') in its role in a `nonmatching list'. A bracket
432  expression starting with an unquoted circumflex character produces
433  unspecified results. This implementation treats the two identically. */
434  if (not = (*p == L('!') || *p == L('^')))
435  ++p;
436 
437  c = *p++;
438  for (;;)
439  {
440  /* Initialize cstart and cend in case `-' is the last
441  character of the pattern. */
442  cstart = cend = c;
443  forcecoll = 0;
444 
445  /* POSIX.2 equivalence class: [=c=]. See POSIX.2 2.8.3.2. Find
446  the end of the equivalence class, move the pattern pointer past
447  it, and check for equivalence. XXX - this handles only
448  single-character equivalence classes, which is wrong, or at
449  least incomplete. */
450  if (c == L('[') && *p == L('=') && p[2] == L('=') && p[3] == L(']'))
451  {
452  pc = FOLD (p[1]);
453  p += 4;
454  if (COLLEQUIV (test, pc))
455  {
456 /*[*/ /* Move past the closing `]', since the first thing we do at
457  the `matched:' label is back p up one. */
458  p++;
459  goto matched;
460  }
461  else
462  {
463  c = *p++;
464  if (c == L('\0'))
465  return ((test == L('[')) ? savep : (CHAR *)0); /*]*/
466  c = FOLD (c);
467  continue;
468  }
469  }
470 
471  /* POSIX.2 character class expression. See POSIX.2 2.8.3.2. */
472  if (c == L('[') && *p == L(':'))
473  {
474  CHAR *close, *ccname;
475 
476  pc = 0; /* make sure invalid char classes don't match. */
477  /* Find end of character class name */
478  for (close = p + 1; *close != '\0'; close++)
479  if (*close == L(':') && *(close+1) == L(']'))
480  break;
481 
482  if (*close != L('\0'))
483  {
484  ccname = (CHAR *)malloc ((close - p) * sizeof (CHAR));
485  if (ccname == 0)
486  pc = 0;
487  else
488  {
489  bcopy (p + 1, ccname, (close - p - 1) * sizeof (CHAR));
490  *(ccname + (close - p - 1)) = L('\0');
491  /* As a result of a POSIX discussion, char class names are
492  allowed to be quoted (?) */
493  DEQUOTE_PATHNAME (ccname);
494  pc = IS_CCLASS (orig_test, (XCHAR *)ccname);
495  }
496  if (pc == -1)
497  {
498  /* CCNAME is not a valid character class in the current
499  locale. In addition to noting no match (pc = 0), we have
500  a choice about what to do with the invalid charclass.
501  Posix leaves the behavior unspecified, but we're going
502  to skip over the charclass and keep going instead of
503  testing ORIG_TEST against each character in the class
504  string. If we don't want to do that, take out the update
505  of P. */
506  pc = 0;
507  p = close + 2;
508  }
509  else
510  p = close + 2; /* move past the closing `]' */
511 
512  free (ccname);
513  }
514 
515  if (pc)
516  {
517 /*[*/ /* Move past the closing `]', since the first thing we do at
518  the `matched:' label is back p up one. */
519  p++;
520  goto matched;
521  }
522  else
523  {
524  /* continue the loop here, since this expression can't be
525  the first part of a range expression. */
526  c = *p++;
527  if (c == L('\0'))
528  return ((test == L('[')) ? savep : (CHAR *)0);
529  else if (c == L(']'))
530  break;
531  c = FOLD (c);
532  continue;
533  }
534  }
535 
536  /* POSIX.2 collating symbols. See POSIX.2 2.8.3.2. Find the end of
537  the symbol name, make sure it is terminated by `.]', translate
538  the name to a character using the external table, and do the
539  comparison. */
540  if (c == L('[') && *p == L('.'))
541  {
542  p = PARSE_COLLSYM (p, &pc);
543  /* An invalid collating symbol cannot be the first point of a
544  range. If it is, we set cstart to one greater than `test',
545  so any comparisons later will fail. */
546  cstart = (pc == INVALID) ? test + 1 : pc;
547  forcecoll = 1;
548  }
549 
550  if (!(flags & FNM_NOESCAPE) && c == L('\\'))
551  {
552  if (*p == '\0')
553  return (CHAR *)0;
554  cstart = cend = *p++;
555  }
556 
557  cstart = cend = FOLD (cstart);
558  isrange = 0;
559 
560  /* POSIX.2 2.8.3.1.2 says: `An expression containing a `[' that
561  is not preceded by a backslash and is not part of a bracket
562  expression produces undefined results.' This implementation
563  treats the `[' as just a character to be matched if there is
564  not a closing `]'. */
565  if (c == L('\0'))
566  return ((test == L('[')) ? savep : (CHAR *)0);
567 
568  c = *p++;
569  c = FOLD (c);
570 
571  if (c == L('\0'))
572  return ((test == L('[')) ? savep : (CHAR *)0);
573 
574  if ((flags & FNM_PATHNAME) && c == L('/'))
575  /* [/] can never match when matching a pathname. */
576  return (CHAR *)0;
577 
578  /* This introduces a range, unless the `-' is the last
579  character of the class. Find the end of the range
580  and move past it. */
581  if (c == L('-') && *p != L(']'))
582  {
583  cend = *p++;
584  if (!(flags & FNM_NOESCAPE) && cend == L('\\'))
585  cend = *p++;
586  if (cend == L('\0'))
587  return (CHAR *)0;
588  if (cend == L('[') && *p == L('.'))
589  {
590  p = PARSE_COLLSYM (p, &pc);
591  /* An invalid collating symbol cannot be the second part of a
592  range expression. If we get one, we set cend to one fewer
593  than the test character to make sure the range test fails. */
594  cend = (pc == INVALID) ? test - 1 : pc;
595  forcecoll = 1;
596  }
597  cend = FOLD (cend);
598 
599  c = *p++;
600 
601  /* POSIX.2 2.8.3.2: ``The ending range point shall collate
602  equal to or higher than the starting range point; otherwise
603  the expression shall be treated as invalid.'' Note that this
604  applies to only the range expression; the rest of the bracket
605  expression is still checked for matches. */
606  if (RANGECMP (cstart, cend, forcecoll) > 0)
607  {
608  if (c == L(']'))
609  break;
610  c = FOLD (c);
611  continue;
612  }
613  isrange = 1;
614  }
615 
616  if (isrange == 0 && test == cstart)
617  goto matched;
618  if (isrange && RANGECMP (test, cstart, forcecoll) >= 0 && RANGECMP (test, cend, forcecoll) <= 0)
619  goto matched;
620 
621  if (c == L(']'))
622  break;
623  }
624  /* No match. */
625  return (!not ? (CHAR *)0 : p);
626 
627 matched:
628  /* Skip the rest of the [...] that already matched. */
629  c = *--p;
630  brcnt = 1;
631  brchrp = 0;
632  while (brcnt > 0)
633  {
634  int oc;
635 
636  /* A `[' without a matching `]' is just another character to match. */
637  if (c == L('\0'))
638  return ((test == L('[')) ? savep : (CHAR *)0);
639 
640  oc = c;
641  c = *p++;
642  if (c == L('[') && (*p == L('=') || *p == L(':') || *p == L('.')))
643  {
644  brcnt++;
645  brchrp = p++; /* skip over the char after the left bracket */
646  if ((c = *p) == L('\0'))
647  return ((test == L('[')) ? savep : (CHAR *)0);
648  /* If *brchrp == ':' we should check that the rest of the characters
649  form a valid character class name. We don't do that yet, but we
650  keep BRCHRP in case we want to. */
651  }
652  /* We only want to check brchrp if we set it above. */
653  else if (c == L(']') && brcnt > 1 && brchrp != 0 && oc == *brchrp)
654  {
655  brcnt--;
656  brchrp = 0; /* just in case */
657  }
658  /* Left bracket loses its special meaning inside a bracket expression.
659  It is only valid when followed by a `.', `=', or `:', which we check
660  for above. Technically the right bracket can appear in a collating
661  symbol, so we check for that here. Otherwise, it terminates the

691 /* Scan a pattern starting at STRING and ending at END, keeping track of
692  embedded () and []. If DELIM is 0, we scan until a matching `)'
693  because we're scanning a `patlist'. Otherwise, we scan until we see
694  DELIM. In all cases, we never scan past END. The return value is the
695  first character after the matching DELIM or NULL if the pattern is
696  empty or invalid. */
697 /*static*/ CHAR *
698 PATSCAN (string, end, delim)
699  CHAR *string, *end;
700  INT delim;
701 {
702  int pnest, bnest, skip;
703  INT cchar;
704  CHAR *s, c, *bfirst;
705 
706  pnest = bnest = skip = 0;
707  cchar = 0;
708  bfirst = NULL;
709 
710  if (string == end)
711  return (NULL);
712 
713  for (s = string; c = *s; s++)
714  {
715  if (s >= end)
716  return (s);
717  if (skip)
718  {
719  skip = 0;
720  continue;
721  }
722  switch (c)
723  {
724  case L('\\'):
725  skip = 1;
726  break;
727 
728  case L('\0'):
729  return ((CHAR *)NULL);
730 
731  /* `[' is not special inside a bracket expression, but it may
732  introduce one of the special POSIX bracket expressions
733  ([.SYM.], [=c=], [: ... :]) that needs special handling. */
734  case L('['):
735  if (bnest == 0)
736  {
737  bfirst = s + 1;
738  if (*bfirst == L('!') || *bfirst == L('^'))
739  bfirst++;
740  bnest++;
741  }
742  else if (s[1] == L(':') || s[1] == L('.') || s[1] == L('='))
743  cchar = s[1];
744  break;
745 
746  /* `]' is not special if it's the first char (after a leading `!'
747  or `^') in a bracket expression or if it's part of one of the
748  special POSIX bracket expressions ([.SYM.], [=c=], [: ... :]) */
749  case L(']'):
750  if (bnest)
751  {
752  if (cchar && s[-1] == cchar)
753  cchar = 0;
754  else if (s != bfirst)
755  {
756  bnest--;
757  bfirst = 0;
758  }
759  }
760  break;
761 
762  case L('('):
763  if (bnest == 0)
764  pnest++;
765  break;
766 
767  case L(')'):
768  if (bnest == 0 && pnest-- <= 0)
769  return ++s;
770  break;
771 
772  case L('|'):
773  if (bnest == 0 && pnest == 0 && delim == L('|'))
774  return ++s;
775  break;
776  }
777  }
778 
779  return (NULL);
780 }

819 /* Match a ksh extended pattern specifier. Return FNM_NOMATCH on failure or
820  0 on success. This is handed the entire rest of the pattern and string
821  the first time an extended pattern specifier is encountered, so it calls
822  gmatch recursively. */
823 static int
824 EXTMATCH (xc, s, se, p, pe, flags)
825  INT xc; /* select which operation */
826  CHAR *s, *se;
827  CHAR *p, *pe;
828  int flags;
829 {
830  CHAR *prest; /* pointer to rest of pattern */
831  CHAR *psub; /* pointer to sub-pattern */
832  CHAR *pnext; /* pointer to next sub-pattern */
833  CHAR *srest; /* pointer to rest of string */
834  int m1, m2, xflags; /* xflags = flags passed to recursive matches */
835 
836 #if DEBUG_MATCHING
837 fprintf(stderr, "extmatch: xc = %c\n", xc);
838 fprintf(stderr, "extmatch: s = %s; se = %s\n", s, se);
839 fprintf(stderr, "extmatch: p = %s; pe = %s\n", p, pe);
840 fprintf(stderr, "extmatch: flags = %d\n", flags);
841 #endif
842 
843  prest = PATSCAN (p + (*p == L('(')), pe, 0); /* ) */
844  if (prest == 0)
845  /* If PREST is 0, we failed to scan a valid pattern. In this
846  case, we just want to compare the two as strings. */
847  return (STRCOMPARE (p - 1, pe, s, se));
848 
849  switch (xc)
850  {
851  case L('+'): /* match one or more occurrences */
852  case L('*'): /* match zero or more occurrences */
853  /* If we can get away with no matches, don't even bother. Just
854  call GMATCH on the rest of the pattern and return success if
855  it succeeds. */
856  if (xc == L('*') && (GMATCH (s, se, prest, pe, NULL, flags) == 0))
857  return 0;
858 
859  /* OK, we have to do this the hard way. First, we make sure one of
860  the subpatterns matches, then we try to match the rest of the
861  string. */
862  for (psub = p + 1; ; psub = pnext)
863  {
864  pnext = PATSCAN (psub, pe, L('|'));
865  for (srest = s; srest <= se; srest++)
866  {
867  /* Match this substring (S -> SREST) against this
868  subpattern (psub -> pnext - 1) */
869  m1 = GMATCH (s, srest, psub, pnext - 1, NULL, flags) == 0;
870  /* OK, we matched a subpattern, so make sure the rest of the
871  string matches the rest of the pattern. Also handle
872  multiple matches of the pattern. */
873  if (m1)
874  {
875  /* if srest > s, we are not at start of string */
876  xflags = (srest > s) ? (flags & ~(FNM_PERIOD|FNM_DOTDOT)) : flags;
877  m2 = (GMATCH (srest, se, prest, pe, NULL, xflags) == 0) ||
878  (s != srest && GMATCH (srest, se, p - 1, pe, NULL, xflags) == 0);
879  }
880  if (m1 && m2)
881  return (0);
882  }
883  if (pnext == prest)
884  break;
885  }
886  return (FNM_NOMATCH);
887 
888  case L('?'): /* match zero or one of the patterns */
889  case L('@'): /* match one (or more) of the patterns */
890  /* If we can get away with no matches, don't even bother. Just
891  call gmatch on the rest of the pattern and return success if
892  it succeeds. */
893  if (xc == L('?') && (GMATCH (s, se, prest, pe, NULL, flags) == 0))
894  return 0;
895 
896  /* OK, we have to do this the hard way. First, we see if one of
897  the subpatterns matches, then, if it does, we try to match the
898  rest of the string. */
899  for (psub = p + 1; ; psub = pnext)
900  {
901  pnext = PATSCAN (psub, pe, L('|'));
902  srest = (prest == pe) ? se : s;
903  for ( ; srest <= se; srest++)
904  {
905  /* if srest > s, we are not at start of string */
906  xflags = (srest > s) ? (flags & ~(FNM_PERIOD|FNM_DOTDOT)) : flags;
907  if (GMATCH (s, srest, psub, pnext - 1, NULL, flags) == 0 &&
908  GMATCH (srest, se, prest, pe, NULL, xflags) == 0)
909  return (0);
910  }
911  if (pnext == prest)
912  break;
913  }
914  return (FNM_NOMATCH);
915 
916  case '!': /* match anything *except* one of the patterns */
917  for (srest = s; srest <= se; srest++)
918  {
919  m1 = 0;
920  for (psub = p + 1; ; psub = pnext)
921  {
922  pnext = PATSCAN (psub, pe, L('|'));
923  /* If one of the patterns matches, just bail immediately. */
924  if (m1 = (GMATCH (s, srest, psub, pnext - 1, NULL, flags) == 0))
925  break;
926  if (pnext == prest)
927  break;
928  }
929 
930  /* If nothing matched, but the string starts with a period and we
931  need to match periods explicitly, don't return this as a match,
932  even for negation. */
933  if (m1 == 0 && (flags & FNM_PERIOD) && *s == '.')
934  return (FNM_NOMATCH);
935 
936  if (m1 == 0 && (flags & FNM_DOTDOT) &&

938  ((flags & FNM_PATHNAME) && s[-1] == L('/') && PDOT_OR_DOTDOT(s))))
939  return (FNM_NOMATCH);
940

942  xflags = (srest > s) ? (flags & ~(FNM_PERIOD|FNM_DOTDOT)) : flags;
943  if (m1 == 0 && GMATCH (srest, se, prest, pe, NULL, xflags) == 0)
944  return (0);
945  }
946  return (FNM_NOMATCH);
947  }
