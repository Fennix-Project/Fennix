262 /* Read a range of lines from FILENAME, adding them to the history list.
263  Start reading at the FROM'th line and end at the TO'th. If FROM
264  is zero, start at the beginning. If TO is less than FROM, read
265  until the end of the file. If FILENAME is NULL, then read from
266  ~/.history. Returns 0 if successful, or errno if not. */
267 int
268 read_history_range (const char *filename, int from, int to)
269 {
270  register char *line_start, *line_end, *p;
271  char *input, *buffer, *bufend, *last_ts;
272  int file, current_line, chars_read, has_timestamps, reset_comment_char;
273  struct stat finfo;
274  size_t file_size;
275 #if defined (EFBIG)
276  int overflow_errno = EFBIG;
277 #elif defined (EOVERFLOW)
278  int overflow_errno = EOVERFLOW;
279 #else
280  int overflow_errno = EIO;
281 #endif
282 
283  history_lines_read_from_file = 0;
284 
285  buffer = last_ts = (char *)NULL;
286  input = history_filename (filename);
287  file = input ? open (input, O_RDONLY|O_BINARY, 0666) : -1;
288 
289  if ((file < 0) || (fstat (file, &finfo) == -1))
290  goto error_and_exit;
291 
292  if (S_ISREG (finfo.st_mode) == 0)
293  {
294 #ifdef EFTYPE
295  errno = EFTYPE;
296 #else
297  errno = EINVAL;
298 #endif
299  goto error_and_exit;
300  }
301 
302  file_size = (size_t)finfo.st_size;
303 
304  /* check for overflow on very large files */
305  if (file_size != finfo.st_size || file_size + 1 < file_size)
306  {
307  errno = overflow_errno;
308  goto error_and_exit;
309  }
310 
311  if (file_size == 0)
312  {
313  xfree (input);
314  close (file);
315  return 0; /* don't waste time if we don't have to */
316  }
317 
318 #ifdef HISTORY_USE_MMAP
319  /* We map read/write and private so we can change newlines to NULs without
320  affecting the underlying object. */
321  buffer = (char *)mmap (0, file_size, PROT_READ|PROT_WRITE, MAP_RFLAGS, file, 0);
322  if ((void *)buffer == MAP_FAILED)
323  {
324  errno = overflow_errno;
325  goto error_and_exit;
326  }
327  chars_read = file_size;
328 #else
329  buffer = (char *)malloc (file_size + 1);
330  if (buffer == 0)
331  {
332  errno = overflow_errno;
333  goto error_and_exit;
334  }
335 
336  chars_read = read (file, buffer, file_size);
337 #endif
338  if (chars_read < 0)
339  {
340  error_and_exit:
341  if (errno != 0)
342  chars_read = errno;
343  else
344  chars_read = EIO;
345  if (file >= 0)
346  close (file);
347 
348  FREE (input);
349 #ifndef HISTORY_USE_MMAP
350  FREE (buffer);
351 #endif
352 
353  return (chars_read);
354  }
355 
356  close (file);
357 
358  /* Set TO to larger than end of file if negative. */
359  if (to < 0)
360  to = chars_read;
361 
362  /* Start at beginning of file, work to end. */
363  bufend = buffer + chars_read;
364  *bufend = '\0'; /* null-terminate buffer for timestamp checks */
365  current_line = 0;
366 
367  /* Heuristic: the history comment character rarely changes, so assume we
368  have timestamps if the buffer starts with `#[:digit:]' and temporarily
369  set history_comment_char so timestamp parsing works right */
370  reset_comment_char = 0;
371  if (history_comment_char == '\0' && buffer[0] == '#' && isdigit ((unsigned char)buffer[1]))
372  {
373  history_comment_char = '#';
374  reset_comment_char = 1;
375  }
376 
377  has_timestamps = HIST_TIMESTAMP_START (buffer);
378  history_multiline_entries += has_timestamps && history_write_timestamps;
379 
380  /* Skip lines until we are at FROM. */
381  if (has_timestamps)
382  last_ts = buffer;
383  for (line_start = line_end = buffer; line_end < bufend && current_line < from; line_end++)
384  if (*line_end == '\n')
385  {
386  p = line_end + 1;
387  /* If we see something we think is a timestamp, continue with this
388  line. We should check more extensively here... */
389  if (HIST_TIMESTAMP_START(p) == 0)
390  current_line++;
391  else
392  last_ts = p;
393  line_start = p;
394  /* If we are at the last line (current_line == from) but we have
395  timestamps (has_timestamps), then line_start points to the
396  text of the last command, and we need to skip to its end. */
397  if (current_line >= from && has_timestamps)
398  {
399  for (line_end = p; line_end < bufend && *line_end != '\n'; line_end++)
400  ;
401  line_start = (*line_end == '\n') ? line_end + 1 : line_end;
402  }
403  }
404 
405  /* If there are lines left to gobble, then gobble them now. */
406  for (line_end = line_start; line_end < bufend; line_end++)
407  if (*line_end == '\n')
408  {
409  /* Change to allow Windows-like \r\n end of line delimiter. */
410  if (line_end > line_start && line_end[-1] == '\r')
411  line_end[-1] = '\0';
412  else
413  *line_end = '\0';
414 
415  if (*line_start)
416  {
417  if (HIST_TIMESTAMP_START(line_start) == 0)
418  {
419  if (last_ts == NULL && history_length > 0 && history_multiline_entries)
420  _hs_append_history_line (history_length - 1, line_start);
421  else
422  add_history (line_start);
423  if (last_ts)
424  {
425  add_history_time (last_ts);
426  last_ts = NULL;
427  }
428  }
429  else
430  {
431  last_ts = line_start;
432  current_line--;
433  }
434  }
435 
436  current_line++;
437 
438  if (current_line >= to)
439  break;
440 
441  line_start = line_end + 1;
442  }
443 
444  history_lines_read_from_file = current_line;
445  if (reset_comment_char)
446  history_comment_char = '\0';
447 
448  FREE (input);
449 #ifndef HISTORY_USE_MMAP
450  FREE (buffer);
451 #else
452  munmap (buffer, file_size);
453 #endif
454 
455  return (0);
456 }

514 /* Truncate the history file FNAME, leaving only LINES trailing lines.
515  If FNAME is NULL, then use ~/.history. Writes a new file and renames
516  it to the original name. Returns 0 on success, errno on failure. */
517 int
518 history_truncate_file (const char *fname, int lines)
519 {
520  char *buffer, *filename, *tempname, *bp, *bp1; /* bp1 == bp+1 */
521  int file, chars_read, rv, orig_lines, exists, r;
522  struct stat finfo, nfinfo;
523  size_t file_size;
524 
525  history_lines_written_to_file = 0;
526 
527  buffer = (char *)NULL;
528  filename = history_filename (fname);
529  tempname = 0;
530  file = filename ? open (filename, O_RDONLY|O_BINARY, 0666) : -1;
531  rv = exists = 0;
532 
533  /* Don't try to truncate non-regular files. */
534  if (file == -1 || fstat (file, &finfo) == -1)
535  {
536  rv = errno;
537  if (file != -1)
538  close (file);
539  goto truncate_exit;
540  }
541  exists = 1;
542 
543  nfinfo.st_uid = finfo.st_uid;
544  nfinfo.st_gid = finfo.st_gid;
545 
546  if (S_ISREG (finfo.st_mode) == 0)
547  {
548  close (file);
549 #ifdef EFTYPE
550  rv = EFTYPE;
551 #else
552  rv = EINVAL;
553 #endif
554  goto truncate_exit;
555  }
556 
557  file_size = (size_t)finfo.st_size;
558 
559  /* check for overflow on very large files */
560  if (file_size != finfo.st_size || file_size + 1 < file_size)
561  {
562  close (file);
563 #if defined (EFBIG)
564  rv = errno = EFBIG;
565 #elif defined (EOVERFLOW)
566  rv = errno = EOVERFLOW;
567 #else
568  rv = errno = EINVAL;
569 #endif
570  goto truncate_exit;
571  }
572 
573  buffer = (char *)malloc (file_size + 1);
574  if (buffer == 0)
575  {
576  rv = errno;
577  close (file);
578  goto truncate_exit;
579  }
580 
581  chars_read = read (file, buffer, file_size);
582  close (file);
583 
584  if (chars_read <= 0)
585  {
586  rv = (chars_read < 0) ? errno : 0;
587  goto truncate_exit;
588  }
589 
590  orig_lines = lines;
591  /* Count backwards from the end of buffer until we have passed
592  LINES lines. bp1 is set funny initially. But since bp[1] can't
593  be a comment character (since it's off the end) and *bp can't be
594  both a newline and the history comment character, it should be OK. */
595  for (bp1 = bp = buffer + chars_read - 1; lines && bp > buffer; bp--)
596  {
597  if (*bp == '\n' && HIST_TIMESTAMP_START(bp1) == 0)
598  lines--;
599  bp1 = bp;
600  }
601 
602  /* If this is the first line, then the file contains exactly the
603  number of lines we want to truncate to, so we don't need to do
604  anything. It's the first line if we don't find a newline between
605  the current value of i and 0. Otherwise, write from the start of
606  this line until the end of the buffer. */
607  for ( ; bp > buffer; bp--)
608  {
609  if (*bp == '\n' && HIST_TIMESTAMP_START(bp1) == 0)
610  {
611  bp++;
612  break;
613  }
614  bp1 = bp;
615  }
616 
617  /* Write only if there are more lines in the file than we want to
618  truncate to. */
619  if (bp <= buffer)
620  {
621  rv = 0;
622  /* No-op if LINES == 0 at this point */
623  history_lines_written_to_file = orig_lines - lines;
624  goto truncate_exit;
625  }
626 
627  tempname = history_tempfile (filename);
628 
629  if ((file = open (tempname, O_WRONLY|O_CREAT|O_TRUNC|O_BINARY, 0600)) != -1)
630  {
631  if (write (file, bp, chars_read - (bp - buffer)) < 0)
632  rv = errno;
633 
634  if (fstat (file, &nfinfo) < 0 && rv == 0)
635  rv = errno;
636 
637  if (close (file) < 0 && rv == 0)
638  rv = errno;
639  }
640  else
641  rv = errno;
642 
643  truncate_exit:
644  FREE (buffer);
645 
646  history_lines_written_to_file = orig_lines - lines;
647 
648  if (rv == 0 && filename && tempname)
649  rv = histfile_restore (tempname, filename);
650 
651  if (rv != 0)
652  {
653  rv = errno;
654  if (tempname)
655  unlink (tempname);
656  history_lines_written_to_file = 0;
657  }
658 
659 #if defined (HAVE_CHOWN)
660  /* Make sure the new filename is owned by the same user as the old. If one
661  user is running this, it's a no-op. If the shell is running after sudo
662  with a shared history file, we don't want to leave the history file
663  owned by root. */
664  if (rv == 0 && exists && SHOULD_CHOWN (finfo, nfinfo))
665  r = chown (filename, finfo.st_uid, finfo.st_gid);
666 #endif
667 
668  xfree (filename);
669  FREE (tempname);
670 
671  return rv;
672 }

674 /* Workhorse function for writing history. Writes the last NELEMENT entries
675  from the history list to FILENAME. OVERWRITE is non-zero if you
676  wish to replace FILENAME with the entries. */
677 static int
678 history_do_write (const char *filename, int nelements, int overwrite)
679 {
680  register int i;
681  char *output, *tempname, *histname;
682  int file, mode, rv, exists;
683  struct stat finfo, nfinfo;
684 #ifdef HISTORY_USE_MMAP
685  size_t cursize;
686 
687  history_lines_written_to_file = 0;
688 
689  mode = overwrite ? O_RDWR|O_CREAT|O_TRUNC|O_BINARY : O_RDWR|O_APPEND|O_BINARY;
690 #else
691  mode = overwrite ? O_WRONLY|O_CREAT|O_TRUNC|O_BINARY : O_WRONLY|O_APPEND|O_BINARY;
692 #endif
693  histname = history_filename (filename);
694  exists = histname ? (stat (histname, &finfo) == 0) : 0;
695 
696  tempname = (overwrite && exists && S_ISREG (finfo.st_mode)) ? history_tempfile (histname) : 0;
697  output = tempname ? tempname : histname;
698 
699  file = output ? open (output, mode, 0600) : -1;
700  rv = 0;
701 
702  if (file == -1)
703  {
704  rv = errno;
705  FREE (histname);
706  FREE (tempname);
707  return (rv);
708  }
709 
710 #ifdef HISTORY_USE_MMAP
711  cursize = overwrite ? 0 : lseek (file, 0, SEEK_END);
712 #endif
713 
714  if (nelements > history_length)
