85 /* sin(pi*x) assuming x > 2^-100, if sin(pi*x)==0 the sign is arbitrary */
86 static float sin_pi(float x)
87 {
88  double_t y;
89  int n;
90 
91  /* spurious inexact if odd int */
92  x = 2*(x*0.5f - floorf(x*0.5f)); /* x mod 2.0 */
93 
94  n = (int)(x*4);
95  n = (n+1)/2;
96  y = x - n*0.5f;
97  y *= 3.14159265358979323846;
98  switch (n) {
99  default: /* case 4: */
100  case 0: return __sindf(y);
101  case 1: return __cosdf(y);
102  case 2: return __sindf(-y);
103  case 3: return -__cosdf(y);
104  }
105 }

116  sign = u.i>>31;

118  if (ix >= 0x7f800000)

121  if (sign) {

124  }

137  }
