45 #if defined (HAVE_SETITIMER)
46 unsigned int
47 falarm(secs, usecs)
48  unsigned int secs, usecs;
49 {
50  struct itimerval it, oit;
51 
52  it.it_interval.tv_sec = 0;
53  it.it_interval.tv_usec = 0;
54 
55  it.it_value.tv_sec = secs;
56  it.it_value.tv_usec = usecs;
57 
58  if (setitimer(ITIMER_REAL, &it, &oit) < 0)
59  return (-1); /* XXX will be converted to unsigned */
60 
61  /* Backwards compatibility with alarm(3) */
62  if (oit.it_value.tv_usec)
63  oit.it_value.tv_sec++;
64  return (oit.it_value.tv_sec);
65 }
66 #else

69  unsigned int secs, usecs;

84  `usleep', but it's already taken */

86 #if defined (HAVE_TIMEVAL) && (defined (HAVE_SELECT) || defined (HAVE_PSELECT))
87 int
88 fsleep(sec, usec)
89  unsigned int sec, usec;
90 {
91  int e, r;
92  sigset_t blocked_sigs, prevmask;
93 #if defined (HAVE_PSELECT)
94  struct timespec ts;
95 #else
96  struct timeval tv;
97 #endif
98 
99  sigemptyset (&blocked_sigs);
100 # if defined (SIGCHLD)
101  sigaddset (&blocked_sigs, SIGCHLD);
102 # endif
103 
104 #if defined (HAVE_PSELECT)
105  ts.tv_sec = sec;
106  ts.tv_nsec = usec * 1000;
107 #else
108  sigemptyset (&prevmask);
109  tv.tv_sec = sec;
110  tv.tv_usec = usec;
111 #endif /* !HAVE_PSELECT */
112 
113  do
114  {
115 #if defined (HAVE_PSELECT)
116  r = pselect(0, (fd_set *)0, (fd_set *)0, (fd_set *)0, &ts, &blocked_sigs);
117 #else
118  sigprocmask (SIG_SETMASK, &blocked_sigs, &prevmask);
119  r = select(0, (fd_set *)0, (fd_set *)0, (fd_set *)0, &tv);
120  sigprocmask (SIG_SETMASK, &prevmask, NULL);
121 #endif
122  e = errno;
123  if (r < 0 && errno == EINTR)
124  return -1; /* caller will handle */
125  errno = e;
126  }
127  while (r < 0 && errno == EINTR);
128 
129  return r;
130 }
131 #else /* !HAVE_TIMEVAL || !HAVE_SELECT */
132 int
133 fsleep(sec, usec)
134  long sec, usec;
135 {
136  if (usec >= 500000) /* round */
137  sec++;
138  return (sleep(sec));
139 }
140 #endif /* !HAVE_TIMEVAL || !HAVE_SELECT */
