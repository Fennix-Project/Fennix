50 static NOINLINE int timeout_wait(int timeout, pid_t pid)
51 {
52  /* Just sleep(HUGE_NUM); kill(parent) may kill wrong process! */
53  while (1) {
54  sleep1();
55  if (--timeout <= 0)
56  break;
57  if (kill(pid, 0)) {
58  /* process is gone */
59  return EXIT_SUCCESS;
60  }
61  }
62  return EXIT_FAILURE;
63 }
