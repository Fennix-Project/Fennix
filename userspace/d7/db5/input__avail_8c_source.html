63 /* Return >= 1 if select/FIONREAD indicates data available for reading on
64  file descriptor FD; 0 if no data available. Return -1 on error. */
65 int
66 input_avail (fd)
67  int fd;
68 {
69  int result, chars_avail;
70 #if defined(HAVE_SELECT)
71  fd_set readfds, exceptfds;
72  struct timeval timeout;
73 #endif
74 
75  if (fd < 0)
76  return -1;
77 
78  chars_avail = 0;
79 
80 #if defined (HAVE_SELECT)
81  FD_ZERO (&readfds);
82  FD_ZERO (&exceptfds);
83  FD_SET (fd, &readfds);
84  FD_SET (fd, &exceptfds);
85  timeout.tv_sec = 0;
86  timeout.tv_usec = 0;
87  result = select (fd + 1, &readfds, (fd_set *)NULL, &exceptfds, &timeout);
88  return ((result <= 0) ? 0 : 1);
89 #endif
90 
91 #if defined (FIONREAD)
92  errno = 0;
93  result = ioctl (fd, FIONREAD, &chars_avail);
94  if (result == -1 && errno == EIO)
95  return -1;
96  return (chars_avail);
97 #endif
98 
99  return 0;
100 }
