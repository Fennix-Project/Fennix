34 int
35 mktemp_builtin (list)
36  WORD_LIST *list;
37 {
38  WORD_LIST *l;
39  int rval, opt, fd, mflags, base_mflags;
40  int dflag, qflag, tflag, uflag, onetime;
41  char *prefix, *varname, *filename, *template;
42  SHELL_VAR *v;
43 
44  dflag = qflag = uflag = tflag = onetime = 0;
45  prefix = varname = 0;
46  rval = EXECUTION_SUCCESS;
47 
48  reset_internal_getopt ();
49  while ((opt = internal_getopt (list, "dqut:v:")) != -1)
50  {
51  switch (opt)
52  {
53  case 'd':
54  dflag = 1;
55  break;
56  case 'q':
57  qflag = 1;
58  break;
59  case 't':
60  tflag = 1;
61  prefix = list_optarg;
62  break;
63  case 'u':
64  uflag = 1;
65  break;
66  case 'v':
67  varname = list_optarg;
68  break;
69  CASE_HELPOPT;
70  default:
71  builtin_usage ();
72  return (EX_USAGE);
73  }
74  }
75  list = loptend;
76 
77  if (varname) /* check for validity, not readonly */
78  {
79  if (legal_identifier (varname) == 0)
80  {
81  if (qflag == 0)
82  sh_invalidid (varname);
83  return (EXECUTION_FAILURE);
84  }
85  v = find_variable (varname);
86  if (v && readonly_p (v))
87  {
88  if (qflag == 0)
89  sh_readonly (varname);
90  return (EXECUTION_FAILURE);
91  }
92  }
93 
94  onetime = (list == 0); /* once through the loop, $TMPDIR/prefix.XXXXXX */
95 
96  if (prefix == 0)
97  prefix = DEFAULT_PREFIX;
98  base_mflags = MT_USETMPDIR|MT_USERANDOM; /* USERANDOM not strictly needed */
99 
100  while (list || onetime)
101  {
102  mflags = base_mflags;
103  onetime = 0;
104 #if defined (USE_MKTEMP) && defined (USE_MKSTEMP)
105  if (list)
106  {
107  template = list->word->word;
108  mflags |= MT_TEMPLATE;
109  }
110 #else
111  /* This is sub-optimal. */
112  if (list)
113  {
114  /* Treat the basename as a prefix */
115  template = strrchr (list->word->word, '/');
116  if (template)
117  template++;
118  else
119  template = list->word->word;
120  }
121 #endif
122  else
123  template = prefix;
124 
125  if (dflag)
126  {
127  filename = sh_mktmpdir (template, mflags);
128  if (filename == 0)
129  {
130  if (qflag == 0)
131  builtin_error ("%s: cannot create directory", template);
132  rval = EXECUTION_FAILURE;
133  }
134  else
135  {
136  if (uflag)
137  rmdir (filename);
138  printf ("%s\n", filename);
139  }
140  }
141  else /* filename */
142  {
143  fd = sh_mktmpfd (template, mflags, &filename);
144  if (fd < 0)
145  {
146  if (qflag == 0)
147  builtin_error ("%s: cannot create file", template);
148  rval = EXECUTION_FAILURE;
149  }
150  else
151  {
152  close (fd);
153  if (uflag)
154  unlink (filename);
155  printf ("%s\n", filename);
156  }
157  }
158 
159  /* Assign variable if requested */
160  if (filename && varname)
161  {
162  v = builtin_bind_variable (varname, filename, 0);
163  if (v == 0 || readonly_p (v) || noassign_p (v))
164  {
165  builtin_error ("%s: cannot set variable", varname);
166  rval = EXECUTION_FAILURE;
167  }
168  }
169 
170  FREE (filename);
171 
172  if (list)
173  list = list->next;
174  }
175 
176  return (rval);
177 }
