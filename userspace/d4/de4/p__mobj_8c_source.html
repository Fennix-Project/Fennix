47 boolean
48 P_SetMobjState
49 ( mobj_t* mobj,
50  statenum_t state )
51 {
52  state_t* st;
53 
54  do
55  {
56  if (state == S_NULL)
57  {
58  mobj->state = (state_t *) S_NULL;
59  P_RemoveMobj (mobj);
60  return false;
61  }
62 
63  st = &states[state];
64  mobj->state = st;
65  mobj->tics = st->tics;
66  mobj->sprite = st->sprite;
67  mobj->frame = st->frame;
68 
69  // Modified handling.
70  // Call action functions when the state is set
71  if (st->action.acp1)
72  st->action.acp1(mobj);
73 
74  state = st->nextstate;
75  } while (!mobj->tics);
76 
77  return true;
78 }

103 // P_XYMovement
104 //
105 #define STOPSPEED 0x1000
106 #define FRICTION 0xe800
107 
108 void P_XYMovement (mobj_t* mo)
109 {
110  fixed_t ptryx;
111  fixed_t ptryy;
112  player_t* player;
113  fixed_t xmove;
114  fixed_t ymove;
115 
116  if (!mo->momx && !mo->momy)
117  {
118  if (mo->flags & MF_SKULLFLY)
119  {
120  // the skull slammed into something
121  mo->flags &= ~MF_SKULLFLY;
122  mo->momx = mo->momy = mo->momz = 0;
123 
124  P_SetMobjState (mo, mo->info->spawnstate);
125  }
126  return;
127  }
128 
129  player = mo->player;
130 
131  if (mo->momx > MAXMOVE)
132  mo->momx = MAXMOVE;
133  else if (mo->momx < -MAXMOVE)
134  mo->momx = -MAXMOVE;
135 
136  if (mo->momy > MAXMOVE)
137  mo->momy = MAXMOVE;
138  else if (mo->momy < -MAXMOVE)
139  mo->momy = -MAXMOVE;
140 
141  xmove = mo->momx;
142  ymove = mo->momy;
143 
144  do
145  {
146  if (xmove > MAXMOVE/2 || ymove > MAXMOVE/2)
147  {
148  ptryx = mo->x + xmove/2;
149  ptryy = mo->y + ymove/2;
150  xmove >>= 1;
151  ymove >>= 1;
152  }
153  else
154  {
155  ptryx = mo->x + xmove;
156  ptryy = mo->y + ymove;
157  xmove = ymove = 0;
158  }
159 
160  if (!P_TryMove (mo, ptryx, ptryy))
161  {
162  // blocked move
163  if (mo->player)
164  { // try to slide along it
165  P_SlideMove (mo);
166  }
167  else if (mo->flags & MF_MISSILE)
168  {
169  // explode a missile
170  if (ceilingline &&
171  ceilingline->backsector &&
172  ceilingline->backsector->ceilingpic == skyflatnum)
173  {
174  // Hack to prevent missiles exploding
175  // against the sky.
176  // Does not handle sky floors.
177  P_RemoveMobj (mo);
178  return;
179  }
180  P_ExplodeMissile (mo);
181  }
182  else
183  mo->momx = mo->momy = 0;
184  }
185  } while (xmove || ymove);
186 
187  // slow down
188  if (player && player->cheats & CF_NOMOMENTUM)
189  {
190  // debug option for no sliding at all
191  mo->momx = mo->momy = 0;
192  return;
193  }
194 
195  if (mo->flags & (MF_MISSILE | MF_SKULLFLY) )
196  return; // no friction for missiles ever
197 
198  if (mo->z > mo->floorz)
199  return; // no friction when airborne
200 
201  if (mo->flags & MF_CORPSE)
202  {
203  // do not stop sliding
204  // if halfway off a step with some momentum
205  if (mo->momx > FRACUNIT/4
206  || mo->momx < -FRACUNIT/4
207  || mo->momy > FRACUNIT/4
208  || mo->momy < -FRACUNIT/4)
209  {
210  if (mo->floorz != mo->subsector->sector->floorheight)
211  return;
212  }
213  }
214 
215  if (mo->momx > -STOPSPEED
216  && mo->momx < STOPSPEED
217  && mo->momy > -STOPSPEED
218  && mo->momy < STOPSPEED
219  && (!player
220  || (player->cmd.forwardmove== 0
221  && player->cmd.sidemove == 0 ) ) )
222  {
223  // if in a walking frame, stop moving
224  if ( player&&(unsigned)((player->mo->state - states)- S_PLAY_RUN1) < 4)
225  P_SetMobjState (player->mo, S_PLAY);
226 
227  mo->momx = 0;
228  mo->momy = 0;
229  }
230  else
231  {
232  mo->momx = FixedMul (mo->momx, FRICTION);
233  mo->momy = FixedMul (mo->momy, FRICTION);
234  }
235 }

237 //
238 // P_ZMovement
239 //
240 void P_ZMovement (mobj_t* mo)
241 {
242  fixed_t dist;
243  fixed_t delta;
244 
245  // check for smooth step up
246  if (mo->player && mo->z < mo->floorz)
247  {
248  mo->player->viewheight -= mo->floorz-mo->z;
249 
250  mo->player->deltaviewheight
251  = (VIEWHEIGHT - mo->player->viewheight)>>3;
252  }
253 
254  // adjust height
255  mo->z += mo->momz;
256 
257  if ( mo->flags & MF_FLOAT
258  && mo->target)
259  {
260  // float down towards target if too close
261  if ( !(mo->flags & MF_SKULLFLY)
262  && !(mo->flags & MF_INFLOAT) )
263  {
264  dist = P_AproxDistance (mo->x - mo->target->x,
265  mo->y - mo->target->y);
266 
267  delta =(mo->target->z + (mo->height>>1)) - mo->z;
268 
269  if (delta<0 && dist < -(delta*3) )
270  mo->z -= FLOATSPEED;
271  else if (delta>0 && dist < (delta*3) )
272  mo->z += FLOATSPEED;
273  }
274 
275  }
276 
277  // clip movement
278  if (mo->z <= mo->floorz)
279  {
280  // hit the floor
281 
282  // Note (id):
283  // somebody left this after the setting momz to 0,
284  // kinda useless there.
285  //
286  // cph - This was the a bug in the linuxdoom-1.10 source which
287  // caused it not to sync Doom 2 v1.9 demos. Someone
288  // added the above comment and moved up the following code. So
289  // demos would desync in close lost soul fights.
290  // Note that this only applies to original Doom 1 or Doom2 demos - not
291  // Final Doom and Ultimate Doom. So we test demo_compatibility *and*
292  // gamemission. (Note we assume that Doom1 is always Ult Doom, which
293  // seems to hold for most published demos.)
294  //
295  // fraggle - cph got the logic here slightly wrong. There are three
296  // versions of Doom 1.9:
297  //
298  // * The version used in registered doom 1.9 + doom2 - no bounce
299  // * The version used in ultimate doom - has bounce
300  // * The version used in final doom - has bounce
301  //
302  // So we need to check that this is either retail or commercial
303  // (but not doom2)
304 
305  int correct_lost_soul_bounce = gameversion >= exe_ultimate;
306 
307  if (correct_lost_soul_bounce && mo->flags & MF_SKULLFLY)
308  {
309  // the skull slammed into something
310  mo->momz = -mo->momz;
311  }
312 
313  if (mo->momz < 0)
314  {
315  if (mo->player
316  && mo->momz < -GRAVITY*8)
317  {
318  // Squat down.
319  // Decrease viewheight for a moment
320  // after hitting the ground (hard),
321  // and utter appropriate sound.
322  mo->player->deltaviewheight = mo->momz>>3;
323  S_StartSound (mo, sfx_oof);
324  }
325  mo->momz = 0;
326  }
327  mo->z = mo->floorz;
328 
329 
330  // cph 2001/05/26 -
331  // See lost soul bouncing comment above. We need this here for bug
332  // compatibility with original Doom2 v1.9 - if a soul is charging and
333  // hit by a raising floor this incorrectly reverses its Y momentum.
334  //
335 
336  if (!correct_lost_soul_bounce && mo->flags & MF_SKULLFLY)
337  mo->momz = -mo->momz;
338 
339  if ( (mo->flags & MF_MISSILE)
340  && !(mo->flags & MF_NOCLIP) )
341  {
342  P_ExplodeMissile (mo);
343  return;
344  }
345  }
346  else if (! (mo->flags & MF_NOGRAVITY) )
347  {
348  if (mo->momz == 0)
349  mo->momz = -GRAVITY*2;
350  else
351  mo->momz -= GRAVITY;
352  }
353 
354  if (mo->z + mo->height > mo->ceilingz)
355  {
356  // hit the ceiling
357  if (mo->momz > 0)
358  mo->momz = 0;
359  {
360  mo->z = mo->ceilingz - mo->height;
361  }
362 
363  if (mo->flags & MF_SKULLFLY)
364  { // the skull slammed into something
365  mo->momz = -mo->momz;
366  }
367 
368  if ( (mo->flags & MF_MISSILE)
369  && !(mo->flags & MF_NOCLIP) )
370  {
371  P_ExplodeMissile (mo);
372  return;
373  }
374  }
375 }

379 //
380 // P_NightmareRespawn
381 //
382 void
383 P_NightmareRespawn (mobj_t* mobj)
384 {
385  fixed_t x;
386  fixed_t y;
387  fixed_t z;
388  subsector_t* ss;
389  mobj_t* mo;
390  mapthing_t* mthing;
391 
392  x = mobj->spawnpoint.x << FRACBITS;
393  y = mobj->spawnpoint.y << FRACBITS;
394 
395  // somthing is occupying it's position?
396  if (!P_CheckPosition (mobj, x, y) )
397  return; // no respwan
398 
399  // spawn a teleport fog at old spot
400  // because of removal of the body?
401  mo = P_SpawnMobj (mobj->x,
402  mobj->y,
403  mobj->subsector->sector->floorheight , MT_TFOG);
404  // initiate teleport sound
405  S_StartSound (mo, sfx_telept);
406 
407  // spawn a teleport fog at the new spot
408  ss = R_PointInSubsector (x,y);
409 
410  mo = P_SpawnMobj (x, y, ss->sector->floorheight , MT_TFOG);
411 
412  S_StartSound (mo, sfx_telept);
413 
414  // spawn the new monster
415  mthing = &mobj->spawnpoint;
416 
417  // spawn it
418  if (mobj->info->flags & MF_SPAWNCEILING)
419  z = ONCEILINGZ;
420  else
421  z = ONFLOORZ;
422 
423  // inherit attributes from deceased one
424  mo = P_SpawnMobj (x,y,z, mobj->type);
425  mo->spawnpoint = mobj->spawnpoint;
426  mo->angle = ANG45 * (mthing->angle/45);
427 
428  if (mthing->options & MTF_AMBUSH)
429  mo->flags |= MF_AMBUSH;
430 
431  mo->reactiontime = 18;
432 
433  // remove the old monster,
434  P_RemoveMobj (mobj);
435 }

438 //
439 // P_MobjThinker
440 //
441 void P_MobjThinker (mobj_t* mobj)
442 {
443  // momentum movement
444  if (mobj->momx
445  || mobj->momy
446  || (mobj->flags&MF_SKULLFLY) )
447  {
448  P_XYMovement (mobj);
449 
450  // FIXME: decent NOP/NULL/Nil function pointer please.
451  if (mobj->thinker.function.acv == (actionf_v) (-1))
452  return; // mobj was removed
453  }
454  if ( (mobj->z != mobj->floorz)
455  || mobj->momz )
456  {
457  P_ZMovement (mobj);
458 
459  // FIXME: decent NOP/NULL/Nil function pointer please.
460  if (mobj->thinker.function.acv == (actionf_v) (-1))
461  return; // mobj was removed
462  }
463 
464 
465  // cycle through states,
466  // calling action functions at transitions
467  if (mobj->tics != -1)
468  {
469  mobj->tics--;
470 
471  // you can cycle through multiple states in a tic
472  if (!mobj->tics)
473  if (!P_SetMobjState (mobj, mobj->state->nextstate) )
474  return; // freed itself
475  }
476  else
477  {
478  // check for nightmare respawn
479  if (! (mobj->flags & MF_COUNTKILL) )
480  return;
481 
482  if (!respawnmonsters)
483  return;
484 
485  mobj->movecount++;
486 
487  if (mobj->movecount < 12*TICRATE)
488  return;
489 
490  if ( leveltime&31 )
491  return;
492 
493  if (P_Random () > 4)
494  return;
495 
496  P_NightmareRespawn (mobj);
497  }
498

530  mobj->reactiontime = info->reactiontime;
531 
532  mobj->lastlook = P_Random () % MAXPLAYERS;
533  // do not set the state with P_SetMobjState,
534  // because action routines can not be called yet
535  st = &states[info->spawnstate];
536 
537  mobj->state = st;
538  mobj->tics = st->tics;
539  mobj->sprite = st->sprite;
540  mobj->frame = st->frame;
541 
542  // set subsector and/or block links
543  P_SetThingPosition (mobj);
544 
545  mobj->floorz = mobj->subsector->sector->floorheight;
546  mobj->ceilingz = mobj->subsector->sector->ceilingheight;
547 
548  if (z == ONFLOORZ)
549  mobj->z = mobj->floorz;
550  else if (z == ONCEILINGZ)
551  mobj->z = mobj->ceilingz - mobj->info->height;
552  else
553  mobj->z = z;
554 
555  mobj->thinker.function.acp1 = (actionf_p1)P_MobjThinker;
556 
557  P_AddThinker (&mobj->thinker);
558 
559  return mobj;
560 }

734 //
735 // P_SpawnMapThing
736 // The fields of the mapthing should
737 // already be in host byte order.
738 //
739 void P_SpawnMapThing (mapthing_t* mthing)
740 {
741  int i;
742  int bit;
743  mobj_t* mobj;
744  fixed_t x;
745  fixed_t y;
746  fixed_t z;
747 
748  // count deathmatch start positions
749  if (mthing->type == 11)
750  {
751  if (deathmatch_p < &deathmatchstarts[10])
752  {
753  memcpy (deathmatch_p, mthing, sizeof(*mthing));
754  deathmatch_p++;
755  }
756  return;
757  }
758 
759  if (mthing->type <= 0)
760  {
761  // Thing type 0 is actually "player -1 start".
762  // For some reason, Vanilla Doom accepts/ignores this.
763 
764  return;
765  }
766 
767  // check for players specially
768  if (mthing->type <= 4)
769  {
770  // save spots for respawning in network games
771  playerstarts[mthing->type-1] = *mthing;
772  if (!deathmatch)
773  P_SpawnPlayer (mthing);
774 
775  return;
776  }
777 
778  // check for apropriate skill level
779  if (!netgame && (mthing->options & 16) )
780  return;
781 
782  if (gameskill == sk_baby)
783  bit = 1;
784  else if (gameskill == sk_nightmare)
785  bit = 4;
786  else
787  bit = 1<<(gameskill-1);
788 
789  if (!(mthing->options & bit) )
790  return;
791 
792  // find which type to spawn
793  for (i=0 ; i< NUMMOBJTYPES ; i++)
794  if (mthing->type == mobjinfo[i].doomednum)
795  break;
