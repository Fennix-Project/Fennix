239 static int tftp_blksize_check(const char *blksize_str, int maxsize)
240 {
241  /* Check if the blksize is valid:
242  * RFC2348 says between 8 and 65464,
243  * but our implementation makes it impossible
244  * to use blksizes smaller than 22 octets. */
245  unsigned blksize = bb_strtou(blksize_str, NULL, 10);
246  if (errno
247  || (blksize < 24) || (blksize > maxsize)
248  ) {
249  bb_error_msg("bad blocksize '%s'", blksize_str);
250  return -1;
251  }
252 # if ENABLE_TFTP_DEBUG
253  bb_info_msg("using blksize %u", blksize);
254 # endif
255  return blksize;
256 }

258 static char *tftp_get_option(const char *option, char *buf, int len)
259 {
260  int opt_val = 0;
261  int opt_found = 0;
262  int k;
263 
264  /* buf points to:
265  * "opt_name<NUL>opt_val<NUL>opt_name2<NUL>opt_val2<NUL>..." */
266 
267  while (len > 0) {
268  /* Make sure options are terminated correctly */
269  for (k = 0; k < len; k++) {
270  if (buf[k] == '\0') {
271  goto nul_found;
272  }
273  }
274  return NULL;
275  nul_found:
276  if (opt_val == 0) { /* it's "name" part */
277  if (strcasecmp(buf, option) == 0) {
278  opt_found = 1;
279  }
280  } else if (opt_found) {
281  return buf;
282  }
283 
284  k++;
285  buf += k;
286  len -= k;
287  opt_val ^= 1;
288  }
289 
290  return NULL;
291 }

295 static int tftp_protocol(
296  /* NULL if tftp, !NULL if tftpd: */
297  len_and_sockaddr *our_lsa,
298  len_and_sockaddr *peer_lsa,
299  const char *local_file
300  IF_TFTP(, const char *remote_file)
301 #if !ENABLE_TFTP
302 # define remote_file NULL
303 #endif
304  /* 1 for tftp; 1/0 for tftpd depending whether client asked about it: */
305  IF_FEATURE_TFTP_BLOCKSIZE(, int want_transfer_size)
306  IF_FEATURE_TFTP_BLOCKSIZE(, int blksize))
307 {
308 #if !ENABLE_FEATURE_TFTP_BLOCKSIZE
309  enum { blksize = TFTP_BLKSIZE_DEFAULT };
310 #endif
311 
312  struct pollfd pfd[1];
313 #define socket_fd (pfd[0].fd)
314  int len;
315  int send_len;
316  IF_FEATURE_TFTP_BLOCKSIZE(smallint expect_OACK = 0;)
317  smallint finished = 0;
318  uint16_t opcode;
319  uint16_t block_nr;
320  uint16_t recv_blk;
321  int local_fd = -1;
322  int retries, waittime_ms;
323  int io_bufsize = blksize + 4;
324  char *cp;
325  /* Can't use RESERVE_CONFIG_BUFFER here since the allocation
326  * size varies meaning BUFFERS_GO_ON_STACK would fail.
327  *
328  * We must keep the transmit and receive buffers separate
329  * in case we rcv a garbage pkt - we need to rexmit the last pkt.
330  */
331  char *xbuf = xmalloc(io_bufsize);
332  char *rbuf = xmalloc(io_bufsize);
333 
334  socket_fd = xsocket(peer_lsa->u.sa.sa_family, SOCK_DGRAM, 0);
335  setsockopt_reuseaddr(socket_fd);
336 
337  if (!ENABLE_TFTP || our_lsa) { /* tftpd */
338  /* Create a socket which is:
339  * 1. bound to IP:port peer sent 1st datagram to,
340  * 2. connected to peer's IP:port
341  * This way we will answer from the IP:port peer
342  * expects, will not get any other packets on
343  * the socket, and also plain read/write will work. */
344  xbind(socket_fd, &our_lsa->u.sa, our_lsa->len);
345  xconnect(socket_fd, &peer_lsa->u.sa, peer_lsa->len);
346 
347  /* Is there an error already? Send pkt and bail out */
348  if (G_error_pkt_reason || G_error_pkt_str[0])
349  goto send_err_pkt;
350 
351  if (G.pw) {
352  change_identity(G.pw); /* initgroups, setgid, setuid */
353  }
354  }
355 
356  /* Examples of network traffic.
357  * Note two cases when ACKs with block# of 0 are sent.
358  *
359  * Download without options:
360  * tftp -> "\0\1FILENAME\0octet\0"
361  * "\0\3\0\1FILEDATA..." <- tftpd
362  * tftp -> "\0\4\0\1"
363  * ...
364  * Download with option of blksize 16384:
365  * tftp -> "\0\1FILENAME\0octet\0blksize\00016384\0"
366  * "\0\6blksize\00016384\0" <- tftpd
367  * tftp -> "\0\4\0\0"
368  * "\0\3\0\1FILEDATA..." <- tftpd
369  * tftp -> "\0\4\0\1"
370  * ...
371  * Upload without options:
372  * tftp -> "\0\2FILENAME\0octet\0"
373  * "\0\4\0\0" <- tftpd
374  * tftp -> "\0\3\0\1FILEDATA..."
375  * "\0\4\0\1" <- tftpd
376  * ...
377  * Upload with option of blksize 16384:
378  * tftp -> "\0\2FILENAME\0octet\0blksize\00016384\0"
379  * "\0\6blksize\00016384\0" <- tftpd
380  * tftp -> "\0\3\0\1FILEDATA..."
381  * "\0\4\0\1" <- tftpd
382  * ...
383  */
384  block_nr = 1;
385  cp = xbuf + 2;
386 
387  if (!ENABLE_TFTP || our_lsa) { /* tftpd */
388  /* Open file (must be after changing user) */
389  int open_mode = O_RDONLY;
390  if (CMD_GET(option_mask32)) {
391  open_mode = O_WRONLY | O_TRUNC | O_CREAT;
392  if ((option_mask32 & (TFTPD_OPT+TFTPD_OPT_c)) == TFTPD_OPT) {
393  /* tftpd without -c */
394  open_mode = O_WRONLY | O_TRUNC;
395  }
396  }
397  local_fd = open(local_file, open_mode, 0666);
398  if (local_fd < 0) {
399  /* sanitize name, it came from untrusted remote side */
400  unsigned char *p = (void *) local_file;
401  while (*p) {
402  if (*p < ' ')
403  *p = '?';
404  p++;
405  }
406  bb_perror_msg("can't open '%s'", local_file);
407  G_error_pkt_reason = ERR_NOFILE;
408  strcpy(G_error_pkt_str, "can't open file");
409  goto send_err_pkt_nomsg;
410  }
411 
412 /* gcc 4.3.1 would NOT optimize it out as it should! */
413 #if ENABLE_FEATURE_TFTP_BLOCKSIZE
414  if (blksize != TFTP_BLKSIZE_DEFAULT || want_transfer_size) {
415  /* Create and send OACK packet. */
416  /* For the download case, block_nr is still 1 -
417  * we expect 1st ACK from peer to be for (block_nr-1),
418  * that is, for "block 0" which is our OACK pkt */
419  opcode = TFTP_OACK;
420  goto add_blksize_opt;
421  }
422 #endif
423  if (CMD_GET(option_mask32)) {
424  /* It's upload and we don't send OACK.
425  * We must ACK 1st packet (with filename)
426  * as if it is "block 0" */
427  block_nr = 0;
428  }
429  } else { /* tftp */
430  if (CMD_PUT(option_mask32)) {
431  local_fd = STDIN_FILENO;
432  if (local_file)
433  local_fd = xopen(local_file, O_RDONLY);
434  }
435 /* Removing #if, or using if() statement instead of #if may lead to
436  * "warning: null argument where non-null required": */
437 #if ENABLE_TFTP
438  /* tftp */
439 
440  /* We can't (and don't really need to) bind the socket:
441  * we don't know from which local IP datagrams will be sent,
442  * but kernel will pick the same IP every time (unless routing
443  * table is changed), thus peer will see dgrams consistently
444  * coming from the same IP.
445  * We would like to connect the socket, but since peer's
446  * UDP code can be less perfect than ours, _peer's_ IP:port
447  * in replies may differ from IP:port we used to send
448  * our first packet. We can connect() only when we get
449  * first reply. */
450 
451  /* build opcode */
452  opcode = TFTP_WRQ;
453  if (CMD_GET(option_mask32)) {
454  opcode = TFTP_RRQ;
455  }
456  /* add filename and mode */
457  /* fill in packet if the filename fits into xbuf */
458  len = strlen(remote_file);
459  if (len + 3 + sizeof("octet") >= io_bufsize) {
460  bb_simple_error_msg("remote filename is too long");
461  goto ret;
462  }
463  cp = stpcpy(cp, remote_file) + 1;
464  /* add "mode" part of the packet */
465  cp = stpcpy(cp, "octet") + 1;
466 
467 # if ENABLE_FEATURE_TFTP_BLOCKSIZE
468  if (blksize == TFTP_BLKSIZE_DEFAULT && !want_transfer_size)
469  goto send_pkt;
470 
471  /* Need to add option to pkt */
472  if ((&xbuf[io_bufsize - 1] - cp) < sizeof("blksize NNNNN tsize ") + sizeof(off_t)*3) {
473  bb_simple_error_msg("remote filename is too long");
474  goto ret;
475  }
476  expect_OACK = 1;
477 # endif
478 #endif /* ENABLE_TFTP */
479 
480 #if ENABLE_FEATURE_TFTP_BLOCKSIZE
481  add_blksize_opt:
482  if (blksize != TFTP_BLKSIZE_DEFAULT) {
483  /* add "blksize", <nul>, blksize, <nul> */
484  strcpy(cp, "blksize");
485  cp += sizeof("blksize");
486  cp += snprintf(cp, 6, "%d", blksize) + 1;
487  }
488  if (want_transfer_size) {
489  /* add "tsize", <nul>, size, <nul> (see RFC2349) */
490  /* if tftp and downloading, we send "0" (local_fd is not open yet)
491  * and this makes server to send "tsize" option with the size */
492  /* if tftp and uploading, we send file size (maybe dont, to not confuse old servers???) */
493  /* if tftpd and downloading, we are answering to client's request */
494  /* if tftpd and uploading: !want_transfer_size, this code is not executed */
495  struct stat st;
496  strcpy(cp, "tsize");
497  cp += sizeof("tsize");
498  st.st_size = 0;
499  if (local_fd >= 0)
500  fstat(local_fd, &st);
501  cp += sprintf(cp, "%"OFF_FMT"u", (off_t)st.st_size) + 1;
502 # if ENABLE_FEATURE_TFTP_PROGRESS_BAR
503  /* Save for progress bar. If 0 (tftp downloading),
504  * we look at server's reply later */
505  G.size = st.st_size;
506  if (remote_file && st.st_size)
507  tftp_progress_init();
508 # endif
509  }
510 #endif
511  /* First packet is built, so skip packet generation */
512  goto send_pkt;
513  }
514 
515  /* Using mostly goto's - continue/break will be less clear
516  * in where we actually jump to */
517  while (1) {
518  /* Build ACK or DATA */
519  cp = xbuf + 2;
520  *((uint16_t*)cp) = htons(block_nr);
521  cp += 2;
522  block_nr++;
523  opcode = TFTP_ACK;
524  if (CMD_PUT(option_mask32)) {
525  opcode = TFTP_DATA;
526  len = full_read(local_fd, cp, blksize);
527  if (len < 0) {
528  goto send_read_err_pkt;
529  }
530  if (len != blksize) {
531  finished = 1;
532  }
533  cp += len;
534  IF_FEATURE_TFTP_PROGRESS_BAR(G.pos += len;)
535  }
536  send_pkt:
537  /* Send packet */
538  *((uint16_t*)xbuf) = htons(opcode); /* fill in opcode part */
539  send_len = cp - xbuf;
540  /* NB: send_len value is preserved in code below
541  * for potential resend */
542 
543  retries = TFTP_NUM_RETRIES; /* re-initialize */
544  waittime_ms = TFTP_TIMEOUT_MS;
545 
546  send_again:
547 #if ENABLE_TFTP_DEBUG
548  fprintf(stderr, "sending %u bytes\n", send_len);
549  for (cp = xbuf; cp < &xbuf[send_len]; cp++)
550  fprintf(stderr, "%02x ", (unsigned char) *cp);
551  fprintf(stderr, "\n");
552 #endif
553  xsendto(socket_fd, xbuf, send_len, &peer_lsa->u.sa, peer_lsa->len);
554 
555 #if ENABLE_FEATURE_TFTP_PROGRESS_BAR
556  if (is_bb_progress_inited(&G.pmt))
557  tftp_progress_update();
558 #endif
559  /* Was it final ACK? then exit */
560  if (finished && (opcode == TFTP_ACK))
561  goto ret;
562 
563  recv_again:
564  /* Receive packet */
565  /*pfd[0].fd = socket_fd;*/
566  pfd[0].events = POLLIN;
567  switch (safe_poll(pfd, 1, waittime_ms)) {
568  default:
569  /*bb_perror_msg("poll"); - done in safe_poll */
570  goto ret;
571  case 0:
572  retries--;
573  if (retries == 0) {
574  tftp_progress_done();
575  bb_simple_error_msg("timeout");
576  goto ret; /* no err packet sent */
577  }
578 
579  /* exponential backoff with limit */
580  waittime_ms += waittime_ms/2;
581  if (waittime_ms > TFTP_MAXTIMEOUT_MS) {
582  waittime_ms = TFTP_MAXTIMEOUT_MS;
583  }
584 
585  goto send_again; /* resend last sent pkt */
586  case 1:
587  if (!our_lsa) {
588  /* tftp (not tftpd!) receiving 1st packet */
589  our_lsa = ((void*)(ptrdiff_t)-1); /* not NULL */

599  * socket is connect()ed, can just read from it. */

601  * This is not TCP, one read == one pkt! */

635  } else if (recv_blk <= 8) {
636  msg = nth_string(errcode_str, recv_blk);

650  if (res) {

674  }

676  * "An option not acknowledged by the server
677  * must be ignored by the client and server
678  * as if it were never requested." */

682  io_bufsize = TFTP_BLKSIZE_DEFAULT + 4;
683  }
684 #endif

686  * to get / block# we are about to send next time */
687

689  if (recv_blk == block_nr) {
690  int sz;
691  if (local_fd == -1) {
692  local_fd = STDOUT_FILENO;
693  if (local_file)
694  local_fd = xopen(local_file, O_WRONLY | O_TRUNC | O_CREAT);
695  }
696  sz = full_write(local_fd, &rbuf[4], len - 4);
697  if (sz != len - 4) {
698  strcpy(G_error_pkt_str, bb_msg_write_error);
699  G_error_pkt_reason = ERR_WRITE;
700  goto send_err_pkt;
701  }
702  if (sz != blksize) {
703  finished = 1;
704  }
705  IF_FEATURE_TFTP_PROGRESS_BAR(G.pos += sz;)
706  continue; /* send ACK */
707  }
708 /* Disabled to cope with servers with Sorcerer's Apprentice Syndrome */
709 #if 0
710  if (recv_blk == (block_nr - 1)) {
711  /* Server lost our TFTP_ACK. Resend it */
712  block_nr = recv_blk;
713  continue;
714  }
715 #endif
716  }
717 
718  if (CMD_PUT(option_mask32) && (opcode == TFTP_ACK)) {
719  /* did peer ACK our last DATA pkt? */
720  if (recv_blk == (uint16_t) (block_nr - 1)) {
721  if (finished)
722  goto ret;
723  continue; /* send next block */
724  }
725  }
726  /* Awww... recv'd packet is not recognized! */
727  goto recv_again;
728  /* why recv_again? - rfc1123 says:
729  * "The sender (i.e., the side originating the DATA packets)
730  * must never resend the current DATA packet on receipt
731  * of a duplicate ACK".
732  * DATA pkts are resent ONLY on timeout.
733  * Thus "goto send_again" will be a bad mistake above.
734  * See:
735  * http://en.wikipedia.org/wiki/Sorcerer's_Apprentice_Syndrome
736  */
737  } /* end of "while (1)" */
738  ret:
739  if (ENABLE_FEATURE_CLEAN_UP) {
740  close(local_fd);
741  close(socket_fd);
742  free(xbuf);
743  free(rbuf);
744  }
745  if (!finished)
746  goto err;
747  return EXIT_SUCCESS;
748 
749  send_read_err_pkt:
750  strcpy(G_error_pkt_str, bb_msg_read_error);
751  send_err_pkt:
752  if (G_error_pkt_str[0])
753  bb_simple_error_msg(G_error_pkt_str);
754  send_err_pkt_nomsg:
755  G.error_pkt[1] = TFTP_ERROR;
756  xsendto(socket_fd, G.error_pkt, 4 + 1 + strlen(G_error_pkt_str),
757  &peer_lsa->u.sa, peer_lsa->len);
758  err:
759  if (local_fd >= 0 && CMD_GET(option_mask32) && local_file)
760  unlink(local_file);
761  return EXIT_FAILURE;
762 #undef remote_file
763 }
