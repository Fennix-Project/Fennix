244 static boolean WeaponSelectable(weapontype_t weapon)
245 {
246  // Can't select the super shotgun in Doom 1.
247 
248  if (weapon == wp_supershotgun && logical_gamemission == doom)
249  {
250  return false;
251  }
252 
253  // These weapons aren't available in shareware.
254 
255  if ((weapon == wp_plasma || weapon == wp_bfg)
256  && gamemission == doom && gamemode == shareware)
257  {
258  return false;
259  }
260 
261  // Can't select a weapon if we don't own it.
262 
263  if (!players[consoleplayer].weaponowned[weapon])
264  {
265  return false;
266  }
267 
268  // Can't select the fist if we have the chainsaw, unless
269  // we also have the berserk pack.

272  && players[consoleplayer].weaponowned[wp_chainsaw]
273  && !players[consoleplayer].powers[pw_strength])
274  {
275  return false;
276  }
277 
278  return true;
279 }

281 static int G_NextWeapon(int direction)
282 {
283  weapontype_t weapon;
284  int start_i, i;
285 
286  // Find index in the table.
287 
288  if (players[consoleplayer].pendingweapon == wp_nochange)
289  {
290  weapon = players[consoleplayer].readyweapon;
291  }
292  else
293  {
294  weapon = players[consoleplayer].pendingweapon;
295  }
296 
297  for (i=0; i<arrlen(weapon_order_table); ++i)
298  {
299  if (weapon_order_table[i].weapon == weapon)
300  {
301  break;
302  }
303  }
304 
305  // Switch weapon. Don't loop forever.
306  start_i = i;
307  do
308  {
309  i += direction;
310  i = (i + arrlen(weapon_order_table)) % arrlen(weapon_order_table);
311  } while (i != start_i && !WeaponSelectable(weapon_order_table[i].weapon));
312 
313  return weapon_order_table[i].weapon_num;
314 }

317 // G_BuildTiccmd
318 // Builds a ticcmd from all of the available inputs
319 // or reads it from the demo buffer.
320 // If recording a demo, write it out
321 //
322 void G_BuildTiccmd (ticcmd_t* cmd, int maketic)
323 {
324  int i;
325  boolean strafe;
326  boolean bstrafe;
327  int speed;
328  int tspeed;
329  int forward;
330  int side;
331 
332  memset(cmd, 0, sizeof(ticcmd_t));
333 
334  cmd->consistancy =
335  consistancy[consoleplayer][maketic%BACKUPTICS];
336 
337  strafe = gamekeydown[key_strafe] || mousebuttons[mousebstrafe]
338  || joybuttons[joybstrafe];
339 
340  // fraggle: support the old "joyb_speed = 31" hack which
341  // allowed an autorun effect
342 
343  speed = key_speed >= NUMKEYS
344  || joybspeed >= MAX_JOY_BUTTONS
345  || gamekeydown[key_speed]
346  || joybuttons[joybspeed];
347 
348  forward = side = 0;
349 
350  // use two stage accelerative turning
351  // on the keyboard and joystick
352  if (joyxmove < 0
353  || joyxmove > 0
354  || gamekeydown[key_right]
355  || gamekeydown[key_left])
356  turnheld += ticdup;
357  else
358  turnheld = 0;
359 
360  if (turnheld < SLOWTURNTICS)
361  tspeed = 2; // slow turn
362  else
363  tspeed = speed;
364 
365  // let movement keys cancel each other out
366  if (strafe)
367  {
368  if (gamekeydown[key_right])
369  {
370  // fprintf(stderr, "strafe right\n");
371  side += sidemove[speed];
372  }
373  if (gamekeydown[key_left])
374  {
375  // fprintf(stderr, "strafe left\n");
376  side -= sidemove[speed];
377  }
378  if (joyxmove > 0)
379  side += sidemove[speed];
380  if (joyxmove < 0)
381  side -= sidemove[speed];
382 
383  }
384  else
385  {
386  if (gamekeydown[key_right])
387  cmd->angleturn -= angleturn[tspeed];
388  if (gamekeydown[key_left])
389  cmd->angleturn += angleturn[tspeed];
390  if (joyxmove > 0)
391  cmd->angleturn -= angleturn[tspeed];
392  if (joyxmove < 0)
393  cmd->angleturn += angleturn[tspeed];
394  }
395 
396  if (gamekeydown[key_up])
397  {
398  // fprintf(stderr, "up\n");
399  forward += forwardmove[speed];
400  }
401  if (gamekeydown[key_down])
402  {
403  // fprintf(stderr, "down\n");
404  forward -= forwardmove[speed];
405  }
406 
407  if (joyymove < 0)
408  forward += forwardmove[speed];
409  if (joyymove > 0)
410  forward -= forwardmove[speed];
411 
412  if (gamekeydown[key_strafeleft]
413  || joybuttons[joybstrafeleft]
414  || mousebuttons[mousebstrafeleft]
415  || joystrafemove < 0)
416  {
417  side -= sidemove[speed];
418  }
419 
420  if (gamekeydown[key_straferight]
421  || joybuttons[joybstraferight]
422  || mousebuttons[mousebstraferight]
423  || joystrafemove > 0)
424  {
425  side += sidemove[speed];
426  }
427 
428  // buttons
429  cmd->chatchar = HU_dequeueChatChar();
430 
431  if (gamekeydown[key_fire] || mousebuttons[mousebfire]
432  || joybuttons[joybfire])
433  cmd->buttons |= BT_ATTACK;
434 
435  if (gamekeydown[key_use]
436  || joybuttons[joybuse]
437  || mousebuttons[mousebuse])
438  {
439  cmd->buttons |= BT_USE;
440  // clear double clicks if hit use button
441  dclicks = 0;
442  }
443 
444  // If the previous or next weapon button is pressed, the
445  // next_weapon variable is set to change weapons when
446  // we generate a ticcmd. Choose a new weapon.
447 
448  if (gamestate == GS_LEVEL && next_weapon != 0)
449  {
450  i = G_NextWeapon(next_weapon);
451  cmd->buttons |= BT_CHANGE;
452  cmd->buttons |= i << BT_WEAPONSHIFT;
453  }
454  else
455  {
456  // Check weapon keys.
457 
458  for (i=0; i<arrlen(weapon_keys); ++i)
459  {
460  int key = *weapon_keys[i];
461 
462  if (gamekeydown[key])
463  {
464  cmd->buttons |= BT_CHANGE;
465  cmd->buttons |= i<<BT_WEAPONSHIFT;
466  break;
467  }
468  }
469  }
470 
471  next_weapon = 0;
472 
473  // mouse
474  if (mousebuttons[mousebforward])
475  {
476  forward += forwardmove[speed];
477  }
478  if (mousebuttons[mousebbackward])
479  {
480  forward -= forwardmove[speed];
481  }
482 
483  if (dclick_use)
484  {
485  // forward double click
486  if (mousebuttons[mousebforward] != dclickstate && dclicktime > 1 )
487  {
488  dclickstate = mousebuttons[mousebforward];
489  if (dclickstate)
490  dclicks++;
491  if (dclicks == 2)
492  {
493  cmd->buttons |= BT_USE;
494  dclicks = 0;
495  }
496  else
497  dclicktime = 0;
498  }
499  else
500  {
501  dclicktime += ticdup;
502  if (dclicktime > 20)
503  {
504  dclicks = 0;
505  dclickstate = 0;
506  }
507  }
508 
509  // strafe double click
510  bstrafe =
511  mousebuttons[mousebstrafe]
512  || joybuttons[joybstrafe];
513  if (bstrafe != dclickstate2 && dclicktime2 > 1 )
514  {
515  dclickstate2 = bstrafe;
516  if (dclickstate2)
517  dclicks2++;
518  if (dclicks2 == 2)
519  {
520  cmd->buttons |= BT_USE;
521  dclicks2 = 0;
522  }
523  else
524  dclicktime2 = 0;
525  }
526  else
527  {
528  dclicktime2 += ticdup;
529  if (dclicktime2 > 20)
530  {
531  dclicks2 = 0;
532  dclickstate2 = 0;
533  }
534  }
535  }
536 
537  forward += mousey;
538 
539  if (strafe)
540  side += mousex*2;
541  else
542  cmd->angleturn -= mousex*0x8;
543 
544  if (mousex == 0)
545  {
546  // No movement in the previous frame
547 
548  testcontrols_mousespeed = 0;
549  }
550 
551  mousex = mousey = 0;
552 
553  if (forward > MAXPLMOVE)
554  forward = MAXPLMOVE;
555  else if (forward < -MAXPLMOVE)
556  forward = -MAXPLMOVE;
557  if (side > MAXPLMOVE)
558  side = MAXPLMOVE;
559  else if (side < -MAXPLMOVE)
560  side = -MAXPLMOVE;
561 
562  cmd->forwardmove += forward;
563  cmd->sidemove += side;
564 
565  // special buttons
566  if (sendpause)
567  {
568  sendpause = false;
569  cmd->buttons = BT_SPECIAL | BTS_PAUSE;
570  }
571 
572  if (sendsave)
573  {
574  sendsave = false;
575  cmd->buttons = BT_SPECIAL | BTS_SAVEGAME | (savegameslot<<BTS_SAVESHIFT);
576  }
577 
578  // low-res turning
579 
580  if (lowres_turn)
581  {
582  static signed short carry = 0;
583  signed short desired_angleturn;
584 
585  desired_angleturn = cmd->angleturn + carry;
586 
587  // round angleturn to the nearest 256 unit boundary
588  // for recording demos with single byte values for turn
589 
590  cmd->angleturn = (desired_angleturn + 128) & 0xff00;
591 
592  // Carry forward the error from the reduced resolution to the
593  // next tic, so that successive small movements can accumulate.
594 
595  carry = desired_angleturn - cmd->angleturn;
596  }
597 }

750  if (gameaction == ga_nothing && !singledemo &&
751  (demoplayback || gamestate == GS_DEMOSCREEN)
752  )
753  {
754  if (ev->type == ev_keydown ||
755  (ev->type == ev_mouse && ev->data1) ||
756  (ev->type == ev_joystick && ev->data1) )
757  {

766 #if 0
