508 /* Return values of ACTFs ('action functions') are a bit mask:
509  * bit 1=1: prune (use SKIP constant for setting it)
510  * bit 0=1: matched successfully (TRUE)
511  */
512 
513 static int exec_actions(action ***appp, const char *fileName, const struct stat *statbuf)
514 {
515  int cur_group;
516  int cur_action;
517  int rc = 0;
518  action **app, *ap;
519 
520  /* "action group" is a set of actions ANDed together.
521  * groups are ORed together.
522  * We simply evaluate each group until we find one in which all actions
523  * succeed. */
524 
525  /* -prune is special: if it is encountered, then we won't
526  * descend into current directory. It doesn't matter whether
527  * action group (in which -prune sits) will succeed or not:
528  * find * -prune -name 'f*' -o -name 'm*' -- prunes every dir
529  * find * -name 'f*' -o -prune -name 'm*' -- prunes all dirs
530  * not starting with 'f' */
531 
532  /* We invert TRUE bit (bit 0). Now 1 there means 'failure'.
533  * and bitwise OR in "rc |= TRUE ^ ap->f()" will:
534  * (1) make SKIP (-prune) bit stick; and (2) detect 'failure'.
535  * On return, bit is restored. */
536 
537  cur_group = -1;
538  while ((app = appp[++cur_group]) != NULL) {
539  rc &= ~TRUE; /* 'success' so far, clear TRUE bit */
540  cur_action = -1;
541  while (1) {
542  ap = app[++cur_action];
543  if (!ap) /* all actions in group were successful */
544  return rc ^ TRUE; /* restore TRUE bit */
545  rc |= TRUE ^ ap->f(fileName, statbuf, ap);
546 #if ENABLE_FEATURE_FIND_NOT
547  if (ap->invert) rc ^= TRUE;
548 #endif
549  dbg("grp %d action %d rc:0x%x", cur_group, cur_action, rc);
550  if (rc & TRUE) /* current group failed, try next */
551  break;
552  }
553  }
554  dbg("returning:0x%x", rc ^ TRUE);
555  return rc ^ TRUE; /* restore TRUE bit */
556 }

574 ACTF(name)
575 {
576  int r;
577  const char *tmp = bb_basename(fileName);
578  /* GNU findutils: find DIR/ -name DIR
579  * prints "DIR/" (DIR// prints "DIR//" etc).
580  * Need to strip trailing "/".
581  * Such names can come only from top-level names, but
582  * we can't do this before recursive_action() call,
583  * since then "find FILE/ -name FILE"
584  * would also work (on non-directories), which is wrong.
585  */
586  char *trunc_slash = NULL;
587 
588  if (*tmp == '\0') {
589  /* "foo/bar/[//...]" */
590  while (tmp != fileName && tmp[-1] == '/')
591  tmp--;
592  if (tmp == fileName) { /* entire fileName is "//.."? */
593  /* yes, convert "//..." to "/"
594  * Testcases:
595  * find / -maxdepth 1 -name /: prints /
596  * find // -maxdepth 1 -name /: prints //
597  * find / -maxdepth 1 -name //: prints nothing
598  * find // -maxdepth 1 -name //: prints nothing
599  */
600  if (tmp[1])
601  trunc_slash = (char*)tmp + 1;
602  } else {
603  /* no, it's "foo/bar/[//...]", go back to 'b' */
604  trunc_slash = (char*)tmp;
605  while (tmp != fileName && tmp[-1] != '/')
606  tmp--;
607  }
608  }

611  * but somewhere between 4.1.20 and 4.4.0 GNU find stopped using it.
612  * find -name '*foo' should match .foo too:
613  */
614  if (trunc_slash) *trunc_slash = '\0';
615 #if FNM_CASEFOLD
616  r = fnmatch(ap->pattern, tmp, (ap->iname ? FNM_CASEFOLD : 0));
617 #else
618  if (ap->iname)
619  tmp = strcpy_upcase(alloca(strlen(tmp) + 1), tmp);
620  r = fnmatch(ap->pattern, tmp, 0);
621 #endif
