124 /* Query the right environment variables and call setlocale() to initialize
125  the C library locale settings. */
126 char *
127 _rl_init_locale (void)
128 {
129  char *ret, *lspec;
130 
131  /* Set the LC_CTYPE locale category from environment variables. */
132  lspec = _rl_get_locale_var ("LC_CTYPE");
133  /* Since _rl_get_locale_var queries the right environment variables,
134  we query the current locale settings with setlocale(), and, if
135  that doesn't return anything, we set lspec to the empty string to
136  force the subsequent call to setlocale() to define the `native'
137  environment. */
138 #if defined (HAVE_SETLOCALE)
139  if (lspec == 0 || *lspec == 0)
140  lspec = setlocale (LC_CTYPE, (char *)NULL);
141  if (lspec == 0)
142  lspec = "";
143  ret = setlocale (LC_CTYPE, lspec); /* ok, since it does not change locale */
144  if (ret == 0 || *ret == 0)
145  ret = setlocale (LC_CTYPE, (char *)NULL);
146  if (ret == 0 || *ret == 0)
147  ret = RL_DEFAULT_LOCALE;
148 #else
149  ret = (lspec == 0 || *lspec == 0) ? RL_DEFAULT_LOCALE : lspec;
150 #endif
151 
152  _rl_utf8locale = (ret && *ret) ? utf8locale (ret) : 0;
153 
154  _rl_current_locale = savestring (ret);
155  return ret;
156 }

158 /* If we have setlocale(3), just check the current LC_CTYPE category
159  value (passed as LOCALESTR), and go into eight-bit mode if it's not "C"
160  or "POSIX". If FORCE is non-zero, we reset the locale variables to values
161  appropriate for the C locale if the locale is "C" or "POSIX". FORCE is 0
162  when this is called from _rl_init_eightbit, since we're modifying the
163  default initial values and don't need to change anything else. If we
164  don't have setlocale(3), we check the codeset portion of LOCALESTR against
165  a set of known values and go into eight-bit mode if it matches one of those.
166  Returns 1 if we set eight-bit (multibyte) mode. */
167 static int
168 _rl_set_localevars (char *localestr, int force)
169 {
170 #if defined (HAVE_SETLOCALE)
171  if (localestr && *localestr && (localestr[0] != 'C' || localestr[1]) && (STREQ (localestr, "POSIX") == 0))
172  {
173  _rl_meta_flag = 1;
174  _rl_convert_meta_chars_to_ascii = 0;
175  _rl_output_meta_chars = 1;
176  return (1);
177  }
178  else if (force)
179  {
180  /* Default "C" locale settings. */
181  _rl_meta_flag = 0;
182  _rl_convert_meta_chars_to_ascii = 1;
183  _rl_output_meta_chars = 0;
184  return (0);
185  }
186  else
187  return (0);
188 
189 #else /* !HAVE_SETLOCALE */
190  char *t;
191  int i;
192 
193  /* We don't have setlocale. Finesse it. Check the environment for the
194  appropriate variables and set eight-bit mode if they have the right
195  values. */
196  if (localestr == 0 || (t = normalize_codeset (localestr)) == 0)
197  return (0);
198  for (i = 0; t && legal_lang_values[i]; i++)
199  if (STREQ (t, legal_lang_values[i]))
200  {
201  _rl_meta_flag = 1;
202  _rl_convert_meta_chars_to_ascii = 0;
203  _rl_output_meta_chars = 1;
204  break;
205  }
206 
207  if (force && legal_lang_values[i] == 0) /* didn't find it */
208  {
209  /* Default "C" locale settings. */
210  _rl_meta_flag = 0;
211  _rl_convert_meta_chars_to_ascii = 1;
212  _rl_output_meta_chars = 0;
213  }
214 
215  _rl_utf8locale = *t ? STREQ (t, "utf8") : 0;
216 
217  xfree (t);
218  return (legal_lang_values[i] ? 1 : 0);
219 #endif /* !HAVE_SETLOCALE */
220 }

238 static char *
239 normalize_codeset (char *codeset)
240 {
241  size_t namelen, i;
242  int len, all_digits;
243  char *wp, *retval;
244 
245  codeset = find_codeset (codeset, &namelen);
246 
247  if (codeset == 0)
248  return (codeset);
249 
250  all_digits = 1;
251  for (len = 0, i = 0; i < namelen; i++)
252  {
253  if (ISALNUM ((unsigned char)codeset[i]))
254  {
255  len++;
256  all_digits &= _rl_digit_p (codeset[i]);
257  }
258  }
259 
260  retval = (char *)malloc ((all_digits ? 3 : 0) + len + 1);
261  if (retval == 0)
262  return ((char *)0);
263 
264  wp = retval;
265  /* Add `iso' to beginning of an all-digit codeset */
266  if (all_digits)
267  {
268  *wp++ = 'i';
269  *wp++ = 's';
270  *wp++ = 'o';
271  }
272 
273  for (i = 0; i < namelen; i++)
274  if (ISALPHA ((unsigned char)codeset[i]))
275  *wp++ = _rl_to_lower (codeset[i]);
276  else if (_rl_digit_p (codeset[i]))
277  *wp++ = codeset[i];
278  *wp = '\0';
279 
280  return retval;
281 }

284 /* Isolate codeset portion of locale specification. */
285 static char *
286 find_codeset (char *name, size_t *lenp)
287 {
288  char *cp, *language, *result;
289 
290  cp = language = name;
291  result = (char *)0;
292 
293  while (*cp && *cp != '_' && *cp != '@' && *cp != '+' && *cp != ',')
294  cp++;
295 
296  /* This does not make sense: language has to be specified. As
297  an exception we allow the variable to contain only the codeset
298  name. Perhaps there are funny codeset names. */
299  if (language == cp)
300  {
301  *lenp = strlen (language);
302  result = language;
303  }
304  else
305  {
306  /* Next is the territory. */
307  if (*cp == '_')
308  do
309  ++cp;
310  while (*cp && *cp != '.' && *cp != '@' && *cp != '+' && *cp != ',' && *cp != '_');
311 
312  /* Now, finally, is the codeset. */
313  result = cp;
314  if (*cp == '.')
315  do
316  ++cp;
317  while (*cp && *cp != '@');
318 
319  if (cp - result > 2)
320  {
321  result++;
322  *lenp = cp - result;
323  }
324  else
325  {
326  *lenp = strlen (language);
327  result = language;
328  }
329  }
330 
331  return result;
332 }
