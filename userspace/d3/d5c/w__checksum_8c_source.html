31 static int GetFileNumber(wad_file_t *handle)
32 {
33  int i;
34  int result;
35 
36  for (i=0; i<num_open_wadfiles; ++i)
37  {
38  if (open_wadfiles[i] == handle)
39  {
40  return i;
41  }
42  }
43 
44  // Not found in list. This is a new file we haven't seen yet.
45  // Allocate another slot for this file.
46 
47  open_wadfiles = realloc(open_wadfiles,
48  sizeof(wad_file_t *) * (num_open_wadfiles + 1));
49  open_wadfiles[num_open_wadfiles] = handle;
50 
51  result = num_open_wadfiles;
52  ++num_open_wadfiles;
53 
54  return result;
55 }
