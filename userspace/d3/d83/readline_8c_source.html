477 STATIC_CALLBACK char *
478 readline_internal_teardown (int eof)
479 {
480  char *temp;
481  HIST_ENTRY *entry;
482 
483  RL_CHECK_SIGNALS ();
484 
485  if (eof)
486  RL_SETSTATE (RL_STATE_EOF); /* XXX */
487 
488  /* Restore the original of this history line, iff the line that we
489  are editing was originally in the history, AND the line has changed. */
490  entry = current_history ();
491 
492  /* We don't want to do this if we executed functions that call
493  history_set_pos to set the history offset to the line containing the
494  non-incremental search string. */
495  if (entry && rl_undo_list)
496  {
497  temp = savestring (the_line);
498  rl_revert_line (1, 0);
499  entry = replace_history_entry (where_history (), the_line, (histdata_t)NULL);
500  _rl_free_history_entry (entry);
501 
502  strcpy (the_line, temp);
503  xfree (temp);
504  }
505 
506  if (_rl_revert_all_at_newline)
507  _rl_revert_all_lines ();
508 
509  /* At any rate, it is highly likely that this line has an undo list. Get
510  rid of it now. */
511  if (rl_undo_list)
512  rl_free_undo_list ();
513 
514  /* Disable the meta key, if this terminal has one and we were told to use it.
515  The check whether or not we sent the enable string is in
516  _rl_disable_meta_key(); the flag is set in _rl_enable_meta_key */
517  _rl_disable_meta_key ();
518 
519  /* Restore normal cursor, if available. */
520  _rl_set_insert_mode (RL_IM_INSERT, 0);
521 
522  return (eof ? (char *)NULL : savestring (the_line));

560 STATIC_CALLBACK int
561 #if defined (READLINE_CALLBACKS)
562 readline_internal_char (void)
563 #else
564 readline_internal_charloop (void)
565 #endif
566 {
567  static int lastc, eof_found;
568  int c, code, lk, r;
569 
570  lastc = EOF;
571 
572 #if !defined (READLINE_CALLBACKS)
573  eof_found = 0;
574  while (rl_done == 0)
575  {
576 #endif
577  lk = _rl_last_command_was_kill;
578 
579 #if defined (HAVE_POSIX_SIGSETJMP)
580  code = sigsetjmp (_rl_top_level, 0);
581 #else
582  code = setjmp (_rl_top_level);
583 #endif
584 
585  if (code)
586  {
587  (*rl_redisplay_function) ();
588  _rl_want_redisplay = 0;
589 
590  /* If we longjmped because of a timeout, handle it here. */
591  if (RL_ISSTATE (RL_STATE_TIMEOUT))
592  {
593  RL_SETSTATE (RL_STATE_DONE);
594  rl_done = 1;
595  return 1;
596  }
597 
598  /* If we get here, we're not being called from something dispatched
599  from _rl_callback_read_char(), which sets up its own value of
600  _rl_top_level (saving and restoring the old, of course), so
601  we can just return here. */
602  if (RL_ISSTATE (RL_STATE_CALLBACK))
603  return (0);
604  }
605 
606  if (rl_pending_input == 0)
607  {
608  /* Then initialize the argument and number of keys read. */
609  _rl_reset_argument ();
610  rl_executing_keyseq[rl_key_sequence_length = 0] = '\0';
611  }
612 
613  RL_SETSTATE(RL_STATE_READCMD);
614  c = rl_read_key ();
615  RL_UNSETSTATE(RL_STATE_READCMD);
616 
617  /* look at input.c:rl_getc() for the circumstances under which this will
618  be returned; punt immediately on read error without converting it to
619  a newline; assume that rl_read_key has already called the signal
620  handler. */
621  if (c == READERR)
622  {
623 #if defined (READLINE_CALLBACKS)
624  RL_SETSTATE(RL_STATE_DONE);
625  return (rl_done = 1);
626 #else
627  RL_SETSTATE(RL_STATE_EOF);
628  eof_found = 1;
629  break;
630 #endif
631  }
632 
633  /* EOF typed to a non-blank line is ^D the first time, EOF the second
634  time in a row. This won't return any partial line read from the tty.
635  If we want to change this, to force any existing line to be returned
636  when read(2) reads EOF, for example, this is the place to change. */
637  if (c == EOF && rl_end)
638  {
639  if (RL_SIG_RECEIVED ())
640  {
641  RL_CHECK_SIGNALS ();
642  if (rl_signal_event_hook)
643  (*rl_signal_event_hook) (); /* XXX */
644  }
645 
646  /* XXX - reading two consecutive EOFs returns EOF */
647  if (RL_ISSTATE (RL_STATE_TERMPREPPED))
648  {
649  if (lastc == _rl_eof_char || lastc == EOF)
650  rl_end = 0;
651  else
652  c = _rl_eof_char;
653  }
654  else
655  c = NEWLINE;
656  }
657 
658  /* The character _rl_eof_char typed to blank line, and not as the
659  previous character is interpreted as EOF. This doesn't work when
660  READLINE_CALLBACKS is defined, so hitting a series of ^Ds will
661  erase all the chars on the line and then return EOF. */
662  if (((c == _rl_eof_char && lastc != c) || c == EOF) && rl_end == 0)
663  {
664 #if defined (READLINE_CALLBACKS)
665  RL_SETSTATE(RL_STATE_DONE);
666  return (rl_done = 1);
667 #else
668  RL_SETSTATE(RL_STATE_EOF);
669  eof_found = 1;
670  break;
671 #endif
672  }
673 
674  lastc = c;
675  r = _rl_dispatch ((unsigned char)c, _rl_keymap);
676  RL_CHECK_SIGNALS ();
677 
678  if (_rl_command_to_execute)
679  {
680  (*rl_redisplay_function) ();
681 
682  rl_executing_keymap = _rl_command_to_execute->map;
683  rl_executing_key = _rl_command_to_execute->key;
684 
685  _rl_executing_func = _rl_command_to_execute->func;
686

807 {
808  int nkey, r;
809 
810  /* For now */
811  /* The first time this context is used, we want to read input and dispatch
812  on it. When traversing the chain of contexts back `up', we want to use
813  the value from the next context down. We're simulating recursion using
814  a chain of contexts. */

824  cxt->flags |= KSEQ_DISPATCHED;
825  }
826  else

828 
829  /* For now */
830  if (r != -3) /* don't do this if we indicate there will be other matches */
831  r = _rl_subseq_result (r, cxt->oldmap, cxt->okey, (cxt->flags & KSEQ_SUBSEQ));
832 
833  RL_CHECK_SIGNALS ();

863 int
864 _rl_dispatch_subseq (register int key, Keymap map, int got_subseq)
865 {
866  int r, newkey;
867  char *macro;
868  rl_command_func_t *func;
869 #if defined (READLINE_CALLBACKS)
870  _rl_keyseq_cxt *cxt;
871 #endif
872 
873  if (META_CHAR (key) && _rl_convert_meta_chars_to_ascii)
874  {
875  if (map[ESC].type == ISKMAP)
876  {
877  if (RL_ISSTATE (RL_STATE_MACRODEF))
878  _rl_add_macro_char (ESC);
879  RESIZE_KEYSEQ_BUFFER ();
880  rl_executing_keyseq[rl_key_sequence_length++] = ESC;
881  map = FUNCTION_TO_KEYMAP (map, ESC);
882  key = UNMETA (key);
883  return (_rl_dispatch (key, map));
884  }
885  else
886  rl_ding ();
887  return 0;
888  }
889 
890  if (RL_ISSTATE (RL_STATE_MACRODEF))
891  _rl_add_macro_char (key);
892 
893  r = 0;
894  switch (map[key].type)
895  {
896  case ISFUNC:
897  func = map[key].function;
898  if (func)
899  {
900  /* Special case rl_do_lowercase_version (). */
901  if (func == rl_do_lowercase_version)
902  /* Should we do anything special if key == ANYOTHERKEY? */
903  return (_rl_dispatch (_rl_to_lower ((unsigned char)key), map));
904 
905  rl_executing_keymap = map;
906  rl_executing_key = key;
907 
908  _rl_executing_func = func;
909 
910  RESIZE_KEYSEQ_BUFFER();
911  rl_executing_keyseq[rl_key_sequence_length++] = key;
912  rl_executing_keyseq[rl_key_sequence_length] = '\0';
913 
914  rl_dispatching = 1;
915  RL_SETSTATE(RL_STATE_DISPATCHING);
916  r = (*func) (rl_numeric_arg * rl_arg_sign, key);
917  RL_UNSETSTATE(RL_STATE_DISPATCHING);
918  rl_dispatching = 0;
919 
920  /* If we have input pending, then the last command was a prefix
921  command. Don't change the state of rl_last_func. Otherwise,
922  remember the last command executed in this variable. */
923 #if defined (VI_MODE)
924  if (rl_pending_input == 0 && map[key].function != rl_digit_argument && map[key].function != rl_vi_arg_digit)
925 #else
926  if (rl_pending_input == 0 && map[key].function != rl_digit_argument)
927 #endif
928  rl_last_func = map[key].function;
929 
930  RL_CHECK_SIGNALS ();
931  }
932  else if (map[ANYOTHERKEY].function)
933  {
934  /* OK, there's no function bound in this map, but there is a
935  shadow function that was overridden when the current keymap
936  was created. Return -2 to note that. */
937  if (RL_ISSTATE (RL_STATE_MACROINPUT))
938  _rl_prev_macro_key ();
939  else
940  _rl_unget_char (key);
941  if (rl_key_sequence_length > 0)
942  rl_executing_keyseq[--rl_key_sequence_length] = '\0';
943  return -2;
944  }
945  else if (got_subseq)
946  {
947  /* Return -1 to note that we're in a subsequence, but we don't
948  have a matching key, nor was one overridden. This means
949  we need to back up the recursion chain and find the last
950  subsequence that is bound to a function. */
951  if (RL_ISSTATE (RL_STATE_MACROINPUT))
952  _rl_prev_macro_key ();
953  else
954  _rl_unget_char (key);
955  if (rl_key_sequence_length > 0)
956  rl_executing_keyseq[--rl_key_sequence_length] = '\0';
957  return -1;
958  }
959  else
960  {
961 #if defined (READLINE_CALLBACKS)
962  RL_UNSETSTATE (RL_STATE_MULTIKEY);
963  _rl_keyseq_chain_dispose ();
964 #endif
965  _rl_abort_internal ();
966  return -1;
967  }
968  break;
969 
970  case ISKMAP:
971  if (map[key].function != 0)
972  {
973 #if defined (VI_MODE)
974  /* The only way this test will be true is if a subsequence has been
975  bound starting with ESC, generally the arrow keys. What we do is
976  check whether there's input in the queue, which there generally
977  will be if an arrow key has been pressed, and, if there's not,
978  just dispatch to (what we assume is) rl_vi_movement_mode right
979  away. This is essentially an input test with a zero timeout (by
980  default) or a timeout determined by the value of `keyseq-timeout' */
981  /* _rl_keyseq_timeout specified in milliseconds; _rl_input_queued
982  takes microseconds, so multiply by 1000 */
983  if (rl_editing_mode == vi_mode && key == ESC && map == vi_insertion_keymap &&
984  (RL_ISSTATE (RL_STATE_INPUTPENDING|RL_STATE_MACROINPUT) == 0) &&
985  _rl_pushed_input_available () == 0 &&
986  _rl_input_queued ((_rl_keyseq_timeout > 0) ? _rl_keyseq_timeout*1000 : 0) == 0)
987  return (_rl_dispatch (ANYOTHERKEY, FUNCTION_TO_KEYMAP (map, key)));
988  /* This is a very specific test. It can possibly be generalized in
989  the future, but for now it handles a specific case of ESC being
990  the last character in a keyboard macro. */
991  if (rl_editing_mode == vi_mode && key == ESC && map == vi_insertion_keymap &&
992  (RL_ISSTATE (RL_STATE_INPUTPENDING) == 0) &&
993  (RL_ISSTATE (RL_STATE_MACROINPUT) && _rl_peek_macro_key () == 0) &&
994  _rl_pushed_input_available () == 0 &&
995  _rl_input_queued ((_rl_keyseq_timeout > 0) ? _rl_keyseq_timeout*1000 : 0) == 0)
996  return (_rl_dispatch (ANYOTHERKEY, FUNCTION_TO_KEYMAP (map, key)));
997 #endif
998 
999  RESIZE_KEYSEQ_BUFFER ();
1000  rl_executing_keyseq[rl_key_sequence_length++] = key;
1001  _rl_dispatching_keymap = FUNCTION_TO_KEYMAP (map, key);
1002 
1003  /* Allocate new context here. Use linked contexts (linked through
1004  cxt->ocxt) to simulate recursion */
1005 #if defined (READLINE_CALLBACKS)
1006 # if defined (VI_MODE)
1007  /* If we're redoing a vi mode command and we know there is a shadowed
1008  function corresponding to this key, just call it -- all the redoable
1009  vi mode commands already have all the input they need, and rl_vi_redo
1010  assumes that one call to rl_dispatch is sufficient to complete the
1011  command. */
1012  if (_rl_vi_redoing && RL_ISSTATE (RL_STATE_CALLBACK) &&
1013  map[ANYOTHERKEY].function != 0)
1014  return (_rl_subseq_result (-2, map, key, got_subseq));
1015 # endif
1016  if (RL_ISSTATE (RL_STATE_CALLBACK))
1017  {
1018  /* Return 0 only the first time, to indicate success to
1019  _rl_callback_read_char. The rest of the time, we're called
1020  from _rl_dispatch_callback, so we return -3 to indicate
1021  special handling is necessary. */
1022  r = RL_ISSTATE (RL_STATE_MULTIKEY) ? -3 : 0;
1023  cxt = _rl_keyseq_cxt_alloc ();
1024 
1025  if (got_subseq)
1026  cxt->flags |= KSEQ_SUBSEQ;
1027  cxt->okey = key;
1028  cxt->oldmap = map;
1029  cxt->dmap = _rl_dispatching_keymap;
1030  cxt->subseq_arg = got_subseq || cxt->dmap[ANYOTHERKEY].function;
1031 
1032  RL_SETSTATE (RL_STATE_MULTIKEY);
1033  _rl_kscxt = cxt;
1034 
1035  return r; /* don't indicate immediate success */
1036  }
1037 #endif
1038 
1039  /* Tentative inter-character timeout for potential multi-key
1040  sequences? If no input within timeout, abort sequence and
1041  act as if we got non-matching input. */
1042  /* _rl_keyseq_timeout specified in milliseconds; _rl_input_queued
1043  takes microseconds, so multiply by 1000 */
1044  if (_rl_keyseq_timeout > 0 &&
1045  (RL_ISSTATE (RL_STATE_INPUTPENDING|RL_STATE_MACROINPUT) == 0) &&
1046  _rl_pushed_input_available () == 0 &&
1047  _rl_dispatching_keymap[ANYOTHERKEY].function &&
1048  _rl_input_queued (_rl_keyseq_timeout*1000) == 0)
1049  {
1050  if (rl_key_sequence_length > 0)
1051  rl_executing_keyseq[--rl_key_sequence_length] = '\0';
1052  return (_rl_subseq_result (-2, map, key, got_subseq));
1053  }
1054 
1055  newkey = _rl_subseq_getchar (key);
1056  if (newkey < 0)
1057  {
1058  _rl_abort_internal ();
1059  return -1;
1060  }
1061 
1062  r = _rl_dispatch_subseq (newkey, _rl_dispatching_keymap, got_subseq || map[ANYOTHERKEY].function);
1063  return _rl_subseq_result (r, map, key, got_subseq);
1064  }
1065  else
1066  {
1067  _rl_abort_internal (); /* XXX */
1068  return -1;
1069  }
1070  break;
1071 
1072  case ISMACR:
1073  if (map[key].function != 0)
1074  {
1075  rl_executing_keyseq[rl_key_sequence_length] = '\0';
1076  macro = savestring ((char *)map[key].function);
1077  _rl_with_macro_input (macro);
1078  return 0;
1079  }
1080  break;
1081  }
1082 
1083 #if defined (VI_MODE)
1084  if (rl_editing_mode == vi_mode && _rl_keymap == vi_movement_keymap &&
1085  key != ANYOTHERKEY &&
1086  _rl_dispatching_keymap == vi_movement_keymap &&
1087  _rl_vi_textmod_command (key))
1088  _rl_vi_set_last (key, rl_numeric_arg, rl_arg_sign);
1089 #endif
1090 
1091  return (r);
1092 }

1094 static int
1095 _rl_subseq_result (int r, Keymap map, int key, int got_subseq)
1096 {
1097  Keymap m;
1098  int type, nt;
1099  rl_command_func_t *func, *nf;
1100 
1101  if (r == -2)
1102  /* We didn't match anything, and the keymap we're indexed into
1103  shadowed a function previously bound to that prefix. Call
1104  the function. The recursive call to _rl_dispatch_subseq has
1105  already taken care of pushing any necessary input back onto
1106  the input queue with _rl_unget_char. */
1107  {
1108  m = _rl_dispatching_keymap;
1109  type = m[ANYOTHERKEY].type;
1110  func = m[ANYOTHERKEY].function;
1111  if (type == ISFUNC && func == rl_do_lowercase_version)
1112  r = _rl_dispatch (_rl_to_lower ((unsigned char)key), map);
1113  else if (type == ISFUNC)
1114  {
1115  /* If we shadowed a function, whatever it is, we somehow need a
1116  keymap with map[key].func == shadowed-function.
1117  Let's use this one. Then we can dispatch using the original
1118  key, since there are commands (e.g., in vi mode) for which it
1119  matters. */
1120  nt = m[key].type;
1121  nf = m[key].function;
1122 
1123  m[key].type = type;
1124  m[key].function = func;
1125  /* Don't change _rl_dispatching_keymap, set it here */
1126  _rl_dispatching_keymap = map; /* previous map */
1127  r = _rl_dispatch_subseq (key, m, 0);
1128  m[key].type = nt;
1129  m[key].function = nf;
1130  }
1131  else
1132  /* We probably shadowed a keymap, so keep going. */
1133  r = _rl_dispatch (ANYOTHERKEY, m);
1134  }
1135  else if (r < 0 && map[ANYOTHERKEY].function)
1136  {
1137  /* We didn't match (r is probably -1), so return something to
1138  tell the caller that it should try ANYOTHERKEY for an
1139  overridden function. */
1140  if (RL_ISSTATE (RL_STATE_MACROINPUT))
1141  _rl_prev_macro_key ();
1142  else
1143  _rl_unget_char (key);
1144  if (rl_key_sequence_length > 0)
1145  rl_executing_keyseq[--rl_key_sequence_length] = '\0';
1146  _rl_dispatching_keymap = map;
1147  return -2;
1148  }
1149  else if (r < 0 && got_subseq) /* XXX */
1150  {
1151  /* OK, back up the chain. */
1152  if (RL_ISSTATE (RL_STATE_MACROINPUT))
1153  _rl_prev_macro_key ();
1154  else
1155  _rl_unget_char (key);
1156  if (rl_key_sequence_length > 0)
1157  rl_executing_keyseq[--rl_key_sequence_length] = '\0';
1158  _rl_dispatching_keymap = map;
1159  return -1;
1160  }
1161 
1162  return r;
1163 }
