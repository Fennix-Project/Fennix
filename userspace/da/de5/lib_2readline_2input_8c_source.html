241 /* If a character is available to be read, then read it and stuff it into
242  IBUFFER. Otherwise, just return. Returns number of characters read
243  (0 if none available) and -1 on error (EIO). */
244 static int
245 rl_gather_tyi (void)
246 {
247  int tty;
248  register int tem, result;
249  int chars_avail, k;
250  char input;
251 #if defined(HAVE_SELECT)
252  fd_set readfds, exceptfds;
253  struct timeval timeout;
254 #endif
255 
256  chars_avail = 0;
257  input = 0;
258  tty = fileno (rl_instream);
259 
260 #if defined (HAVE_PSELECT) || defined (HAVE_SELECT)
261  FD_ZERO (&readfds);
262  FD_ZERO (&exceptfds);
263  FD_SET (tty, &readfds);
264  FD_SET (tty, &exceptfds);
265  USEC_TO_TIMEVAL (_keyboard_input_timeout, timeout);
266 #if defined (RL_TIMEOUT_USE_SELECT)
267  result = _rl_timeout_select (tty + 1, &readfds, (fd_set *)NULL, &exceptfds, &timeout, NULL);
268 #else
269  result = select (tty + 1, &readfds, (fd_set *)NULL, &exceptfds, &timeout);
270 #endif
271  if (result <= 0)
272  return 0; /* Nothing to read. */
273 #endif
274 
275  result = -1;
276  errno = 0;
277 #if defined (FIONREAD)
278  result = ioctl (tty, FIONREAD, &chars_avail);
279  if (result == -1 && errno == EIO)
280  return -1;
281  if (result == -1)
282  chars_avail = 0;
283 #endif
284 
285 #if defined (O_NDELAY)
286  if (result == -1)
287  {
288  tem = fcntl (tty, F_GETFL, 0);
289 
290  fcntl (tty, F_SETFL, (tem | O_NDELAY));
291  chars_avail = read (tty, &input, 1);
292 
293  fcntl (tty, F_SETFL, tem);
294  if (chars_avail == -1 && errno == EAGAIN)
295  return 0;
296  if (chars_avail == -1 && errno == EIO)
297  return -1;
298  if (chars_avail == 0) /* EOF */
299  {
300  rl_stuff_char (EOF);
301  return (0);
302  }
303  }
304 #endif /* O_NDELAY */
305 
306 #if defined (__MINGW32__)
307  /* Use getch/_kbhit to check for available console input, in the same way
308  that we read it normally. */
309  chars_avail = isatty (tty) ? _kbhit () : 0;
310  result = 0;
311 #endif
312 
313  /* If there's nothing available, don't waste time trying to read
314  something. */
315  if (chars_avail <= 0)
316  return 0;
317 
318  tem = ibuffer_space ();
319 
320  if (chars_avail > tem)
321  chars_avail = tem;
322 
323  /* One cannot read all of the available input. I can only read a single
324  character at a time, or else programs which require input can be
325  thwarted. If the buffer is larger than one character, I lose.
326  Damn! */
327  if (tem < ibuffer_len)
328  chars_avail = 0;
329 
330  if (result != -1)
331  {
332  while (chars_avail--)
333  {
334  RL_CHECK_SIGNALS ();
335  k = (*rl_getc_function) (rl_instream);
336  if (rl_stuff_char (k) == 0)
337  break; /* some problem; no more room */
338  if (k == NEWLINE || k == RETURN)
339  break;
340  }
341  }
342  else
343  {
344  if (chars_avail)
345  rl_stuff_char (input);
346  }
347 
348  return 1;
349 }

565 /* Start measuring the time. Returns 0 on success. Returns -1 on
566  error. */
567 int
568 _rl_timeout_init (void)
569 {
570  unsigned int secs, usecs;
571 
572  /* Clear the timeout state of the previous edit */
573  RL_UNSETSTATE(RL_STATE_TIMEOUT);
574  timerclear (&timeout_point);
575 
576  /* Return 0 when timeout is unset. */
577  if (timerisunset (&timeout_duration))
578  return 0;
579 
580  /* Return -1 on gettimeofday error. */
581  if (gettimeofday(&timeout_point, 0) != 0)
582  {
583  timerclear (&timeout_point);
584  return -1;
585  }
586 
587  secs = timeout_duration.tv_sec;
588  usecs = timeout_duration.tv_usec;
589 
590 #if defined (RL_TIMEOUT_USE_SIGALRM)
591  /* If select(2)/pselect(2) is unavailable, use SIGALRM. */
592  if (set_alarm (&secs, &usecs) < 0)
593  return -1;
594 #endif
595 
596  timeout_point.tv_sec += secs;
597  timeout_point.tv_usec += usecs;
598  if (timeout_point.tv_usec >= USEC_PER_SEC)
599  {
600  timeout_point.tv_sec++;
601  timeout_point.tv_usec -= USEC_PER_SEC;
602  }
603 
604  return 0;
605 }

607 /* Get the remaining time until the scheduled timeout. Returns -1 on
608  error or no timeout set with secs and usecs unchanged. Returns 0
609  on an expired timeout with secs and usecs unchanged. Returns 1
610  when the timeout has not yet expired. The remaining time is stored
611  in secs and usecs. When NULL is specified to either of the
612  arguments, just the expiration is tested. */
613 int
614 rl_timeout_remaining (unsigned int *secs, unsigned int *usecs)
615 {
616  struct timeval current_time;
617 
618  /* Return -1 when timeout is unset. */
619  if (timerisunset (&timeout_point))
620  {
621  errno = 0;
622  return -1;
623  }
624 
625  /* Return -1 on error. errno is set by gettimeofday. */
626  if (gettimeofday(&current_time, 0) != 0)
627  return -1;
628 
629  /* Return 0 when timeout has already expired. */
630  /* could use timercmp (&timeout_point, &current_time, <) here */
631  if (current_time.tv_sec > timeout_point.tv_sec ||
632  (current_time.tv_sec == timeout_point.tv_sec &&
633  current_time.tv_usec >= timeout_point.tv_usec))
634  return 0;
635 
636  if (secs && usecs)
637  {
638  *secs = timeout_point.tv_sec - current_time.tv_sec;
639  *usecs = timeout_point.tv_usec - current_time.tv_usec;
640  if (timeout_point.tv_usec < current_time.tv_usec)
641  {
642  (*secs)--;
643  *usecs += USEC_PER_SEC;
644  }
645  }
646 
647  return 1;
648 }

650 /* This should only be called if RL_TIMEOUT_USE_SELECT is defined. */
651 
652 #if defined (HAVE_PSELECT) || defined (HAVE_SELECT)
653 int
654 _rl_timeout_select (int nfds, fd_set *readfds, fd_set *writefds, fd_set *exceptfds, const struct timeval *timeout, const sigset_t *sigmask)
655 {
656  int result;
657 #if defined (HAVE_PSELECT)
658  struct timespec ts;
659 #else
660  sigset_t origmask;
661  struct timeval tv;
662 #endif
663  int tmout_status;
664  struct timeval tmout;
665  unsigned int sec, usec;
666 
667  /* When the remaining time for rl_timeout is shorter than the
668  keyboard input timeout, replace `timeout' with the remaining time
669  for `rl_timeout' and set `tmout_status = 1'. */
670  tmout_status = rl_timeout_remaining (&sec, &usec);
671  tmout.tv_sec = sec;
672  tmout.tv_usec = usec;
673 
674  if (tmout_status == 0)
675  _rl_timeout_handle ();
676  else if (tmout_status == 1)
677  {
678  if (timeout == NULL || timercmp (&tmout, timeout, <))
679  timeout = &tmout;
680  else
681  tmout_status = -1;
682  }
683 
684 #if defined (HAVE_PSELECT)
685  if (timeout)
686  {
687  TIMEVAL_TO_TIMESPEC (timeout, &ts);
688  result = pselect (nfds, readfds, writefds, exceptfds, &ts, sigmask);
689  }
690  else
691  result = pselect (nfds, readfds, writefds, exceptfds, NULL, sigmask);
692 #else
693  if (sigmask)
694  sigprocmask (SIG_SETMASK, sigmask, &origmask);
695 
696  if (timeout)
697  {
698  tv.tv_sec = timeout->tv_sec;
699  tv.tv_usec = timeout->tv_usec;
700  result = select (nfds, readfds, writefds, exceptfds, &tv);
701  }
702  else
703  result = select (nfds, readfds, writefds, exceptfds, NULL);
704 
705  if (sigmask)
706  sigprocmask (SIG_SETMASK, &origmask, NULL);
707 #endif
708 
709  if (tmout_status == 1 && result == 0)
710  _rl_timeout_handle ();
711 
712  return result;
713 }
714 #endif

802 int
803 rl_getc (FILE *stream)
804 {
805  int result;
806  unsigned char c;
807  int fd;
808 #if defined (HAVE_PSELECT)
809  sigset_t empty_set;
810  fd_set readfds;
811 #endif
812 
813  fd = fileno (stream);
814  while (1)
815  {
816  RL_CHECK_SIGNALS ();
817 
818  /* We know at this point that _rl_caught_signal == 0 */
819 
820 #if defined (__MINGW32__)
821  if (isatty (fd)
822  return (_getch ()); /* "There is no error return." */
823 #endif
824  result = 0;
825 #if defined (HAVE_PSELECT) || defined (HAVE_SELECT)
826  /* At this point, if we have pselect, we're using select/pselect for the
827  timeouts. We handled MinGW above. */
828  FD_ZERO (&readfds);
829  FD_SET (fd, &readfds);
830 # if defined (HANDLE_SIGNALS)
831  result = _rl_timeout_select (fd + 1, &readfds, NULL, NULL, NULL, &_rl_orig_sigset);
832 # else
833  sigemptyset (&empty_set);
834  sigprocmask (SIG_BLOCK, (sigset_t *)NULL, &empty_set);
835  result = _rl_timeout_select (fd + 1, &readfds, NULL, NULL, NULL, &empty_set);
836 # endif /* HANDLE_SIGNALS */
837  if (result == 0)
838  _rl_timeout_handle (); /* check the timeout */
839 #endif
840  if (result >= 0)
841  result = read (fd, &c, sizeof (unsigned char));
842 
843  if (result == sizeof (unsigned char))
844  return (c);
845 
846  /* If zero characters are returned, then the file that we are
847  reading from is empty! Return EOF in that case. */
848  if (result == 0)
849  return (EOF);
850 
851 #if defined (__BEOS__)
852  if (errno == EINTR)
853  continue;
854 #endif
855 
856 #if defined (EWOULDBLOCK)
857 # define X_EWOULDBLOCK EWOULDBLOCK
858 #else
859 # define X_EWOULDBLOCK -99
860 #endif
861 
862 #if defined (EAGAIN)
863 # define X_EAGAIN EAGAIN
864 #else
865 # define X_EAGAIN -99
866 #endif
867 
868  if (errno == X_EWOULDBLOCK || errno == X_EAGAIN)
869  {
870  if (sh_unset_nodelay_mode (fd) < 0)
871  return (EOF);
872  continue;
873  }
874 
875 #undef X_EWOULDBLOCK
876 #undef X_EAGAIN
877 
878 /* fprintf(stderr, "rl_getc: result = %d errno = %d\n", result, errno); */
879 
880 handle_error:
881  /* If the error that we received was EINTR, then try again,
882  this is simply an interrupted system call to read (). We allow
883  the read to be interrupted if we caught SIGHUP, SIGTERM, or any
884  of the other signals readline treats specially. If the
885  application sets an event hook, call it for other signals.
886  Otherwise (not EINTR), some error occurred, also signifying EOF. */
887  if (errno != EINTR)
888  return (RL_ISSTATE (RL_STATE_READCMD) ? READERR : EOF);
889  /* fatal signals of interest */
890 #if defined (SIGHUP)
891  else if (_rl_caught_signal == SIGHUP || _rl_caught_signal == SIGTERM)
892 #else
893  else if (_rl_caught_signal == SIGTERM)
894 #endif
895  return (RL_ISSTATE (RL_STATE_READCMD) ? READERR : EOF);
896  /* keyboard-generated signals of interest */
897 #if defined (SIGQUIT)
898  else if (_rl_caught_signal == SIGINT || _rl_caught_signal == SIGQUIT)
899 #else
900  else if (_rl_caught_signal == SIGINT)
901 #endif
902  RL_CHECK_SIGNALS ();
903 #if defined (SIGTSTP)
904  else if (_rl_caught_signal == SIGTSTP)
905  RL_CHECK_SIGNALS ();
906 #endif
907  /* non-keyboard-generated signals of interest */
908 #if defined (SIGWINCH)
909  else if (_rl_caught_signal == SIGWINCH)
910  RL_CHECK_SIGNALS ();
911 #endif /* SIGWINCH */
912 #if defined (SIGALRM)
913  else if (_rl_caught_signal == SIGALRM
914 # if defined (SIGVTALRM)
915  || _rl_caught_signal == SIGVTALRM
916 # endif
917  )
918  RL_CHECK_SIGNALS ();
919 #endif /* SIGALRM */
920 
921  if (rl_signal_event_hook)
922  (*rl_signal_event_hook) ();
923  }
924 }
