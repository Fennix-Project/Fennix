530 /* The user must press "y" or "n". Non-zero return means "y" pressed. */
531 static int
532 get_y_or_n (int for_pager)
533 {
534  int c;
535 
536  /* For now, disable pager in callback mode, until we later convert to state
537  driven functions. Have to wait until next major version to add new
538  state definition, since it will change value of RL_STATE_DONE. */
539 #if defined (READLINE_CALLBACKS)
540  if (RL_ISSTATE (RL_STATE_CALLBACK))
541  return 1;
542 #endif
543 
544  for (;;)
545  {
546  RL_SETSTATE(RL_STATE_MOREINPUT);
547  c = rl_read_key ();
548  RL_UNSETSTATE(RL_STATE_MOREINPUT);
549 
550  if (c == 'y' || c == 'Y' || c == ' ')
551  return (1);
552  if (c == 'n' || c == 'N' || c == RUBOUT)
553  return (0);
554  if (c == ABORT_CHAR || c < 0)
555  _rl_abort_internal ();
556  if (for_pager && (c == NEWLINE || c == RETURN))
557  return (2);
558  if (for_pager && (c == 'q' || c == 'Q'))
559  return (0);
560  rl_ding ();
561  }
562 }

589 #if defined (VISIBLE_STATS)
590 /* Return the character which best describes FILENAME.
591  `@' for symbolic links
592  `/' for directories
593  `*' for executables
594  `=' for sockets
595  `|' for FIFOs
596  `%' for character special devices
597  `#' for block special devices */
598 static int
599 stat_char (char *filename)
600 {
601  struct stat finfo;
602  int character, r;
603  char *f;
604  const char *fn;
605 
606  /* Short-circuit a //server on cygwin, since that will always behave as
607  a directory. */
608 #if __CYGWIN__
609  if (filename[0] == '/' && filename[1] == '/' && strchr (filename+2, '/') == 0)
610  return '/';
611 #endif
612 
613  f = 0;
614  if (rl_filename_stat_hook)
615  {
616  f = savestring (filename);
617  (*rl_filename_stat_hook) (&f);
618  fn = f;
619  }
620  else
621  fn = filename;
622 
623 #if defined (HAVE_LSTAT) && defined (S_ISLNK)
624  r = lstat (fn, &finfo);
625 #else
626  r = stat (fn, &finfo);
627 #endif
628 
629  if (r == -1)
630  {
631  xfree (f);
632  return (0);
633  }
634 
635  character = 0;
636  if (S_ISDIR (finfo.st_mode))
637  character = '/';
638 #if defined (S_ISCHR)
639  else if (S_ISCHR (finfo.st_mode))
640  character = '%';

1063 /* Find the bounds of the current word for completion purposes, and leave
1064  rl_point set to the end of the word. This function skips quoted
1065  substrings (characters between matched pairs of characters in
1066  rl_completer_quote_characters). First we try to find an unclosed
1067  quoted substring on which to do matching. If one is not found, we use
1068  the word break characters to find the boundaries of the current word.
1069  We call an application-specific function to decide whether or not a
1070  particular word break character is quoted; if that function returns a
1071  non-zero result, the character does not break a word. This function
1072  returns the opening quote character if we found an unclosed quoted
1073  substring, '\0' otherwise. FP, if non-null, is set to a value saying
1074  which (shell-like) quote characters we found (single quote, double
1075  quote, or backslash) anywhere in the string. DP, if non-null, is set to
1076  the value of the delimiter character that caused a word break. */
1077 
1078 char
1079 _rl_find_completion_word (int *fp, int *dp)
1080 {
1081  int scan, end, found_quote, delimiter, pass_next, isbrk;
1082  char quote_char;
1083  const char *brkchars;
1084 
1085  end = rl_point;
1086  found_quote = delimiter = 0;
1087  quote_char = '\0';
1088 
1089  brkchars = 0;
1090  if (rl_completion_word_break_hook)
1091  brkchars = (*rl_completion_word_break_hook) ();
1092  if (brkchars == 0)
1093  brkchars = rl_completer_word_break_characters;
1094 
1095  if (rl_completer_quote_characters)
1096  {
1097  /* We have a list of characters which can be used in pairs to
1098  quote substrings for the completer. Try to find the start
1099  of an unclosed quoted substring. */
1100  /* FOUND_QUOTE is set so we know what kind of quotes we found. */
1101  for (scan = pass_next = 0; scan < end; scan = MB_NEXTCHAR (rl_line_buffer, scan, 1, MB_FIND_ANY))
1102  {
1103  if (pass_next)
1104  {
1105  pass_next = 0;
1106  continue;
1107  }
1108 
1109  /* Shell-like semantics for single quotes -- don't allow backslash
1110  to quote anything in single quotes, especially not the closing
1111  quote. If you don't like this, take out the check on the value
1112  of quote_char. */
1113  if (quote_char != '\'' && rl_line_buffer[scan] == '\\')
1114  {
1115  pass_next = 1;
1116  found_quote |= RL_QF_BACKSLASH;
1117  continue;
1118  }
1119 
1120  if (quote_char != '\0')
1121  {
1122  /* Ignore everything until the matching close quote char. */
1123  if (rl_line_buffer[scan] == quote_char)
1124  {
1125  /* Found matching close. Abandon this substring. */
1126  quote_char = '\0';
1127  rl_point = end;
1128  }
1129  }
1130  else if (strchr (rl_completer_quote_characters, rl_line_buffer[scan]))
1131  {
1132  /* Found start of a quoted substring. */
1133  quote_char = rl_line_buffer[scan];
1134  rl_point = scan + 1;
1135  /* Shell-like quoting conventions. */
1136  if (quote_char == '\'')
1137  found_quote |= RL_QF_SINGLE_QUOTE;
1138  else if (quote_char == '"')
1139  found_quote |= RL_QF_DOUBLE_QUOTE;
1140  else
1141  found_quote |= RL_QF_OTHER_QUOTE;
1142  }
1143  }
1144  }
1145 
1146  if (rl_point == end && quote_char == '\0')
1147  {
1148  /* We didn't find an unclosed quoted substring upon which to do
1149  completion, so use the word break characters to find the
1150  substring on which to complete. */
1151  while (rl_point = MB_PREVCHAR (rl_line_buffer, rl_point, MB_FIND_ANY))
1152  {
1153  scan = rl_line_buffer[rl_point];
1154 
1155  if (strchr (brkchars, scan) == 0)
1156  continue;
1157 
1158  /* Call the application-specific function to tell us whether
1159  this word break character is quoted and should be skipped. */
1160  if (rl_char_is_quoted_p && found_quote &&
1161  (*rl_char_is_quoted_p) (rl_line_buffer, rl_point))
1162  continue;
1163 
1164  /* Convoluted code, but it avoids an n^2 algorithm with calls
1165  to char_is_quoted. */
1166  break;
1167  }
1168  }
1169 
1170  /* If we are at an unquoted word break, then advance past it. */
1171  scan = rl_line_buffer[rl_point];
1172 
1173  /* If there is an application-specific function to say whether or not
1174  a character is quoted and we found a quote character, let that
1175  function decide whether or not a character is a word break, even
1176  if it is found in rl_completer_word_break_characters. Don't bother
1177  if we're at the end of the line, though. */
1178  if (scan)
1179  {
1180  if (rl_char_is_quoted_p)
1181  isbrk = (found_quote == 0 ||
1182  (*rl_char_is_quoted_p) (rl_line_buffer, rl_point) == 0) &&
1183  strchr (brkchars, scan) != 0;
1184  else
1185  isbrk = strchr (brkchars, scan) != 0;
1186 
1187  if (isbrk)
1188  {
1189  /* If the character that caused the word break was a quoting
1190  character, then remember it as the delimiter. */
1191  if (rl_basic_quote_characters &&
1192  strchr (rl_basic_quote_characters, scan) &&
1193  (end - rl_point) > 1)
1194  delimiter = scan;
1195 
1196  /* If the character isn't needed to determine something special
1197  about what kind of completion to perform, then advance past it. */
1198  if (rl_special_prefixes == 0 || strchr (rl_special_prefixes, scan) == 0)
1199  rl_point++;
1200  }
1201  }
1202 
1203  if (fp)
1204  *fp = found_quote;
1205  if (dp)
1206  *dp = delimiter;
1207 
1208  return (quote_char);
1209 }
