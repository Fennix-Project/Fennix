539  file_error (filename);
540 
541  fd = open (filename, O_RDONLY, 0666);
542 
543  if (fd == -1)
544  file_error (filename);
545 
546  file_size = (size_t)finfo.st_size;
547  buffer = xmalloc (1 + file_size);
548 
549  if ((nr = read (fd, buffer, file_size)) < 0)

566  defs = (DEF_FILE *)xmalloc (sizeof (DEF_FILE));
567  defs->filename = filename;

576  while (i < file_size)

586  until we have a file to write output to. */

590  for (i = 0; line = defs->lines->array[i]; i++)
