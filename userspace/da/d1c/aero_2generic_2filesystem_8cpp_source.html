397  fd_set *except_set, const struct timespec *timeout,
398  const sigset_t *sigmask, int *num_events) {

400  if (fd == -1)
401  return -1;

404  struct epoll_event ev;
405  memset(&ev, 0, sizeof(struct epoll_event));
406

409  if (write_set && FD_ISSET(k, write_set))

412  ev.events |= EPOLLPRI;

416

418  if (epoll_ctl(fd, EPOLL_CTL_ADD, k, &ev))
419  return -1;
420  }

423  int n = epoll_pwait(
424  fd, evnts, 16,
425  timeout ? (timeout->tv_sec * 1000 + timeout->tv_nsec / 100) : -1,
426  sigmask);
427 
428  if (n == -1)
429  return -1;
430 
431  fd_set res_read_set;
432  fd_set res_write_set;
433  fd_set res_except_set;
434  FD_ZERO(&res_read_set);
435  FD_ZERO(&res_write_set);
436  FD_ZERO(&res_except_set);
437

439

441  int k = evnts[i].data.u32;
442
