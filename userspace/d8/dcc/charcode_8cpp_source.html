20  charcode_error operator() (code_seq<const char> &seq) {
21  auto uc = static_cast<unsigned char>(*seq.it);
22  if(!_progress) {
23  if(!(uc & 0b1000'0000)) {
24  // ASCII-compatible.
25  _cpoint = uc;
26  }else if((uc & 0b1110'0000) == 0b1100'0000) {
27  _cpoint = uc & 0b1'1111;
28  _progress = 1;
29  }else if((uc & 0b1111'0000) == 0b1110'0000) {
30  _cpoint = uc & 0b1111;
31  _progress = 2;
32  }else if((uc & 0b1111'1000) == 0b1111'0000) {
33  _cpoint = uc & 0b111;
34  _progress = 3;
35  }else{
36  // If the highest two bits are 0b10, this is the second (or later) unit.
37  // Units with highest five bits = 0b11111 do not occur in valid UTF-8.
38  __ensure((uc & 0b1100'0000) == 0b1000'0000
39  || (uc & 0b1111'1000) == 0b1111'1000);

41  }
42  }else{
43  // TODO: Return an error.
44  __ensure((uc & 0b1100'0000) == 0b1000'0000);
45  _cpoint = (_cpoint << 6) | (uc & 0x3F);
46  --_progress;
47  }
