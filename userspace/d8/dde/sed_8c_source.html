15 /* Code overview.
16  *
17  * Files are laid out to avoid unnecessary function declarations. So for
18  * example, every function add_cmd calls occurs before add_cmd in this file.
19  *
20  * add_cmd() is called on each line of sed command text (from a file or from
21  * the command line). It calls get_address() and parse_cmd_args(). The
22  * resulting sed_cmd_t structures are appended to a linked list
23  * (G.sed_cmd_head/G.sed_cmd_tail).
24  *
25  * process_files() does actual sedding, reading data lines from each input FILE*
26  * (which could be stdin) and applying the sed command list (sed_cmd_head) to
27  * each of the resulting lines.

32  *
33  * - comments ('#')
34  * - address matching: num|/matchstr/[,num|/matchstr/|$]command

257 /* strcpy, replacing "\from" with 'to'. If to is NUL, replacing "\any" with 'any' */
258 static unsigned parse_escapes(char *dest, const char *string, int len, char from, char to)
259 {
260  char *d = dest;
261  int i = 0;
262 
263  if (len == -1)
264  len = strlen(string);
265 
266  while (i < len) {
267  if (string[i] == '\\') {
268  if (!to || string[i+1] == from) {
269  if ((*d = to ? to : string[i+1]) == '\0')
270  return d - dest;
271  i += 2;
272  d++;
273  continue;
274  }
275  i++; /* skip backslash in string[] */
276  *d++ = '\\';
277  /* fall through: copy next char verbatim */
278  }
279  if ((*d = string[i++]) == '\0')
280  return d - dest;
281  d++;
282  }
283  *d = '\0';
284  return d - dest;
285 }

288 {

307 /*
308  * index_of_next_unescaped_regexp_delim - walks left to right through a string
309  * beginning at a specified index and returns the index of the next regular
310  * expression delimiter (typically a forward slash ('/')) not preceded by
311  * a backslash ('\'). A negative delimiter disables square bracket checking.
312  */
313 static int index_of_next_unescaped_regexp_delim(int delimiter, const char *str)
314 {
315  int bracket = -1;
316  int escaped = 0;
317  int idx = 0;
318  char ch;
319 
320  if (delimiter < 0) {
321  bracket--;
322  delimiter = -delimiter;
323  }
324 
325  for (; (ch = str[idx]) != '\0'; idx++) {
326  if (bracket >= 0) {
327  if (ch == ']'
328  && !(bracket == idx - 1 || (bracket == idx - 2 && str[idx - 1] == '^'))
329  ) {
330  bracket = -1;
331  }
332  } else if (escaped)
333  escaped = 0;
334  else if (ch == '\\')
335  escaped = 1;
336  else if (bracket == -1 && ch == '[')
337  bracket = idx;
338  else if (ch == delimiter)
339  return idx;
340  }
341 
342  /* if we make it to here, we've hit the end of the string */
343  bb_error_msg_and_die("unmatched '%c'", delimiter);
344 }

346 /*
347  * Returns the index of the third delimiter
348  */
349 static int parse_regex_delim(const char *cmdstr, char **match, char **replace)
350 {
351  const char *cmdstr_ptr = cmdstr;
352  unsigned char delimiter;
353  int idx = 0;
354 
355  /* verify that the 's' or 'y' is followed by something. That something
356  * (typically a 'slash') is now our regexp delimiter... */
357  if (*cmdstr == '\0')
358  bb_simple_error_msg_and_die("bad format in substitution expression");
359  delimiter = *cmdstr_ptr++;
360 
361  /* save the match string */
362  idx = index_of_next_unescaped_regexp_delim(delimiter, cmdstr_ptr);
363  *match = copy_parsing_escapes(cmdstr_ptr, idx, delimiter);
364  /* save the replacement string */
365  cmdstr_ptr += idx + 1;
366  idx = index_of_next_unescaped_regexp_delim(- (int)delimiter, cmdstr_ptr);
367 //GNU sed 4.8:
368 // echo 789 | sed 's&8&\&&' - 7&9 ("\&" remained "\&")
369 // echo 789 | sed 's1\‍(8\‍)1\1\11' - 7119 ("\1\1" become "11")
370  *replace = copy_parsing_escapes(cmdstr_ptr, idx, delimiter != '&' ? delimiter : 0);
371 
372  return ((cmdstr_ptr - cmdstr) + idx);
373 }

378 static int get_address(const char *my_str, int *linenum, regex_t ** regex)
379 {
380  const char *pos = my_str;
381 
382  if (isdigit(*my_str)) {
383  *linenum = strtol(my_str, (char**)&pos, 10);
384  /* endstr shouldn't ever equal NULL */
385  } else if (*my_str == '$') {
386  *linenum = -1;
387  pos++;
388  } else if (*my_str == '/' || *my_str == '\\') {
389  int next;
390  char delimiter;
391  char *temp;
392 
393  delimiter = '/';
394  if (*my_str == '\\')
395  delimiter = *++pos;
396  next = index_of_next_unescaped_regexp_delim(delimiter, ++pos);
397  if (next != 0) {
398  temp = copy_parsing_escapes(pos, next, 0);
399  G.previous_regex_ptr = *regex = xzalloc(sizeof(regex_t));
400  xregcomp(*regex, temp, G.regex_type);
401  free(temp);
402  } else {
403  *regex = G.previous_regex_ptr;
404  if (!G.previous_regex_ptr)
405  bb_simple_error_msg_and_die("no previous regexp");
406  }
407  /* Move position to next character after last delimiter */
408  pos += (next+1);
409  }
410  return pos - my_str;
411 }

437 static int parse_subst_cmd(sed_cmd_t *sed_cmd, const char *substr)
438 {
439  int cflags = G.regex_type;
440  char *match;
441  int idx;
442 
443  /*
444  * A substitution command should look something like this:
445  * s/match/replace/ #giIpw
446  * || | |||
447  * mandatory optional
448  */
449  idx = parse_regex_delim(substr, &match, &sed_cmd->string);
450 
451  /* determine the number of back references in the match string */
452  /* Note: we compute this here rather than in the do_subst_command()
453  * function to save processor time, at the expense of a little more memory
454  * (4 bits) per sed_cmd */
455 
456  /* process the flags */
457 
458  sed_cmd->which_match = 1;
459  dbg("s flags:'%s'", substr + idx + 1);
460  while (substr[++idx]) {
461  dbg("s flag:'%c'", substr[idx]);
462  /* Parse match number */
463  if (isdigit(substr[idx])) {
464  if (match[0] != '^') {
465  /* Match 0 treated as all, multiple matches we take the last one. */
466  const char *pos = substr + idx;
467 /* FIXME: error check? */
468  sed_cmd->which_match = (unsigned)strtol(substr+idx, (char**) &pos, 10);
469  idx = pos - substr - 1;
470  }
471  continue;
472  }
473  /* Skip spaces */
474  if (isspace(substr[idx]))
475  continue;
476 
477  switch (substr[idx]) {
478  /* Replace all occurrences */
479  case 'g':
480  sed_cmd->which_match = 0;
481  break;
482  /* Print pattern space */
483  case 'p':
484  sed_cmd->sub_p = 1;
485  break;
486  /* Write to file */
487  case 'w':
488  {
489  char *fname;
490  idx += parse_file_cmd(/*sed_cmd,*/ substr+idx+1, &fname);
491  sed_cmd->sw_file = sed_xfopen_w(fname);
492  sed_cmd->sw_last_char = '\n';
493  free(fname);
494  break;
495  }
496  /* Ignore case (gnu extension) */
497  case 'i':
498  case 'I':
499  cflags |= REG_ICASE;
500  break;
501  /* Comment */
502  case '#':
503  // while (substr[++idx]) continue;
504  idx += strlen(substr + idx); // same
505  /* Fall through */
506  /* End of command */
507  case ';':
508  case '}':
509  goto out;
510  default:
511  dbg("s bad flags:'%s'", substr + idx);
512  bb_simple_error_msg_and_die("bad option in substitution expression");
513  }
514  }
515  out:
516  /* compile the match string into a regex */
517  if (*match != '\0') {
518  /* If match is empty, we use last regex used at runtime */
519  sed_cmd->sub_match = xzalloc(sizeof(regex_t));
520  dbg("xregcomp('%s',%x)", match, cflags);
521  xregcomp(sed_cmd->sub_match, match, cflags);
522  dbg("regcomp ok");
523  }
524  free(match);
525 
526  return idx;
527 }

529 /*
530  * Process the commands arguments
531  */
532 static const char *parse_cmd_args(sed_cmd_t *sed_cmd, const char *cmdstr)
533 {
534  static const char cmd_letters[] ALIGN1 = "saicrw:btTydDgGhHlnNpPqx={}";
535  enum {
536  IDX_s = 0,
537  IDX_a,
538  IDX_i,
539  IDX_c,
540  IDX_r,
541  IDX_w,
542  IDX_colon,
543  IDX_b,
544  IDX_t,
545  IDX_T,
546  IDX_y,
547  IDX_d,
548  IDX_D,
549  IDX_g,
550  IDX_G,
551  IDX_h,
552  IDX_H,
553  IDX_l,
554  IDX_n,
555  IDX_N,
556  IDX_p,
557  IDX_P,
558  IDX_q,
559  IDX_x,
560  IDX_equal,
561  IDX_lbrace,
562  IDX_rbrace,
563  IDX_nul
564  };
565  unsigned idx;
566 
567  BUILD_BUG_ON(sizeof(cmd_letters)-1 != IDX_nul);
568 
569  idx = strchrnul(cmd_letters, sed_cmd->cmd) - cmd_letters;
570 
571  /* handle (s)ubstitution command */
572  if (idx == IDX_s) {
573  cmdstr += parse_subst_cmd(sed_cmd, cmdstr);
574  }
575  /* handle edit cmds: (a)ppend, (i)nsert, and (c)hange */
576  else if (idx <= IDX_c) { /* a,i,c */
577  unsigned len;
578 
579  if (idx < IDX_c) { /* a,i */
580  if (sed_cmd->end_line || sed_cmd->end_match)
581  bb_error_msg_and_die("command '%c' uses only one address", sed_cmd->cmd);
582  }
583  for (;;) {
584  if (*cmdstr == '\n' || *cmdstr == '\\') {
585  cmdstr++;
586  break;
587  }
588  if (!isspace(*cmdstr))
589  break;
590  cmdstr++;
591  }
592  len = strlen(cmdstr);
593  sed_cmd->string = copy_parsing_escapes(cmdstr, len, 0);
594  cmdstr += len;
595  /* "\anychar" -> "anychar" */
596  parse_escapes(sed_cmd->string, sed_cmd->string, -1, '\0', '\0');
597  }
598  /* handle file cmds: (r)ead */
599  else if (idx <= IDX_w) { /* r,w */
600  if (idx < IDX_w) { /* r */
601  if (sed_cmd->end_line || sed_cmd->end_match)
602  bb_error_msg_and_die("command '%c' uses only one address", sed_cmd->cmd);
603  }
604  cmdstr += parse_file_cmd(/*sed_cmd,*/ cmdstr, &sed_cmd->string);
605  if (sed_cmd->cmd == 'w') {
606  sed_cmd->sw_file = sed_xfopen_w(sed_cmd->string);
607  sed_cmd->sw_last_char = '\n';
608  }
609  }
610  /* handle branch commands */
611  else if (idx <= IDX_T) { /* :,b,t,T */
612  int length;
613 
614  cmdstr = skip_whitespace(cmdstr);
615  length = strcspn(cmdstr, semicolon_whitespace);
616  if (length) {
617  sed_cmd->string = xstrndup(cmdstr, length);
618  cmdstr += length;
619  }
620  }
621  /* translation command */
622  else if (idx == IDX_y) {
623  char *match, *replace;
624  int i = cmdstr[0];
625 
626  cmdstr += parse_regex_delim(cmdstr, &match, &replace)+1;
627  /* \n already parsed, but \delimiter needs unescaping. */
628  parse_escapes(match, match, -1, i, i);
629  parse_escapes(replace, replace, -1, i, i);
630 
631  sed_cmd->string = xzalloc((strlen(match) + 1) * 2);
632  for (i = 0; match[i] && replace[i]; i++) {
633  sed_cmd->string[i*2] = match[i];
634  sed_cmd->string[i*2+1] = replace[i];
635  }
636  free(match);
637  free(replace);
638  }
639  /* if it wasn't a single-letter command that takes no arguments
640  * then it must be an invalid command.
641  */
642  else if (idx >= IDX_nul) { /* not d,D,g,G,h,H,l,n,N,p,P,q,x,=,{,} */
643  bb_error_msg_and_die("unsupported command %c", sed_cmd->cmd);
644  }
645 
646  /* give back whatever's left over */
647  return cmdstr;
648 }

656  unsigned len, n;

658  /* Append this line to any unfinished line from last time. */
659  if (G.add_cmd_line) {
660  char *tp = xasprintf("%s\n%s", G.add_cmd_line, cmdstr);
661  free(G.add_cmd_line);
662  cmdstr = G.add_cmd_line = tp;
663  }
664 
665  /* If this line ends with unescaped backslash, request next line. */
666  n = len = strlen(cmdstr);
667  while (n && cmdstr[n-1] == '\\')
668  n--;
669  if ((len - n) & 1) { /* if odd number of trailing backslashes */
670  if (!G.add_cmd_line)
671  G.add_cmd_line = xstrdup(cmdstr);
672  G.add_cmd_line[len-1] = '\0';
673  return;
674  }
675 
676  /* Loop parsing all commands in this line. */
677  while (*cmdstr) {
678  /* Skip leading whitespace and semicolons */
679  cmdstr += strspn(cmdstr, semicolon_whitespace);
680 
681  /* If no more commands, exit. */
682  if (!*cmdstr) break;
683 
684  /* if this is a comment, jump past it and keep going */
685  if (*cmdstr == '#') {
686  /* "#n" is the same as using -n on the command line */
687  if (cmdstr[1] == 'n')
688  G.be_quiet++;
689  cmdstr = strpbrk(cmdstr, "\n\r");
690  if (!cmdstr) break;
691  continue;
692  }
693 
694  /* parse the command
695  * format is: [addr][,addr][!]cmd
696  * |----||-----||-|
697  * part1 part2 part3
698  */
699 
700  sed_cmd = xzalloc(sizeof(sed_cmd_t));
701 
702  /* first part (if present) is an address: either a '$', a number or a /regex/ */
703  cmdstr += get_address(cmdstr, &sed_cmd->beg_line, &sed_cmd->beg_match);
704  sed_cmd->beg_line_orig = sed_cmd->beg_line;
705 
706  /* second part (if present) will begin with a comma */
707  if (*cmdstr == ',') {
708  int idx;
709 
710  cmdstr++;
711  if (*cmdstr == '+' && isdigit(cmdstr[1])) {
712  /* http://sed.sourceforge.net/sedfaq3.html#s3.3
713  * Under GNU sed 3.02+, ssed, and sed15+, <address2>
714  * may also be a notation of the form +num,
715  * indicating the next num lines after <address1> is
716  * matched.
717  * GNU sed 4.2.1 accepts even "+" (meaning "+0").
718  * We don't (we check for isdigit, see above), think
719  * about the "+-3" case.
720  */
721  char *end;
722  /* code is smaller compared to using &cmdstr here: */
723  idx = strtol(cmdstr+1, &end, 10);
724  sed_cmd->end_line = -2 - idx;
725  cmdstr = end;
726  } else {
727  idx = get_address(cmdstr, &sed_cmd->end_line, &sed_cmd->end_match);
728  cmdstr += idx;
729  idx--; /* if 0, trigger error check below */
730  }
731  if (idx < 0)
732  bb_simple_error_msg_and_die("no address after comma");
733  sed_cmd->end_line_orig = sed_cmd->end_line;
734  }
735 
736  /* skip whitespace before the command */
737  cmdstr = skip_whitespace(cmdstr);
738 
739  /* Check for inversion flag */
740  if (*cmdstr == '!') {
741  sed_cmd->invert = 1;
742  cmdstr++;
743 
744  /* skip whitespace before the command */
745  cmdstr = skip_whitespace(cmdstr);
746  }
747 
748  /* last part (mandatory) will be a command */
749  if (!*cmdstr)
750  bb_simple_error_msg_and_die("missing command");
751  sed_cmd->cmd = *cmdstr++;
752  cmdstr = parse_cmd_args(sed_cmd, cmdstr);
753 
754  /* cmdstr now points past args.
755  * GNU sed requires a separator, if there are more commands,
756  * else it complains "char N: extra characters after command".
757  * Example: "sed 'p;d'". We also allow "sed 'pd'".
758  */
759 
760  /* Add the command to the command array */
761  *G.sed_cmd_tail = sed_cmd;
762  G.sed_cmd_tail = &sed_cmd->next;
763  }
764 
765  /* If we glued multiple lines together, free the memory. */
766  free(G.add_cmd_line);
767  G.add_cmd_line = NULL;
768 }

784 static void do_subst_w_backrefs(char *line, char *replace)
785 {
786  int i, j;
787 
788  /* go through the replacement string */
789  for (i = 0; replace[i]; i++) {
790  /* if we find a backreference (\1, \2, etc.) print the backref'ed text */
791  if (replace[i] == '\\') {
792  unsigned backref = replace[++i] - '0';
793  if (backref <= 9) {
794  /* print out the text held in G.regmatch[backref] */
795  if (G.regmatch[backref].rm_so != -1) {
796  j = G.regmatch[backref].rm_so;
797  while (j < G.regmatch[backref].rm_eo)
798  pipe_putc(line[j++]);
799  }
800  continue;
801  }
802  /* I _think_ it is impossible to get '\' to be
803  * the last char in replace string. Thus we don't check
804  * for replace[i] == NUL. (counterexample anyone?) */
805  /* if we find a backslash escaped character, print the character */
806  pipe_putc(replace[i]);
807  continue;
808  }
809  /* if we find an unescaped '&' print out the whole matched text. */
810  if (replace[i] == '&') {
811  j = G.regmatch[0].rm_so;
812  while (j < G.regmatch[0].rm_eo)
813  pipe_putc(line[j++]);
814  continue;
815  }
816  /* Otherwise just output the character. */
817  pipe_putc(replace[i]);
818  }

821 static int do_subst_command(sed_cmd_t *sed_cmd, char **line_p)
822 {
823  char *line = *line_p;
824  unsigned match_count = 0;
825  bool altered = 0;
826  bool prev_match_empty = 1;
827  bool tried_at_eol = 0;
828  regex_t *current_regex;
829 
830  current_regex = sed_cmd->sub_match;
831  /* Handle empty regex. */
832  if (!current_regex) {
833  current_regex = G.previous_regex_ptr;
834  if (!current_regex)
835  bb_simple_error_msg_and_die("no previous regexp");
836  }
837  G.previous_regex_ptr = current_regex;
838 
839  /* Find the first match */
840  dbg("matching '%s'", line);
841  if (REG_NOMATCH == regexec(current_regex, line, 10, G.regmatch, 0)) {
842  dbg("no match");
843  return 0;
844  }
845  dbg("match");
846 
847  /* Initialize temporary output buffer. */
848  G.pipeline.buf = xmalloc(PIPE_GROW);
849  G.pipeline.len = PIPE_GROW;
850  G.pipeline.idx = 0;
851 
852  /* Now loop through, substituting for matches */
853  do {
854  int start = G.regmatch[0].rm_so;
855  int end = G.regmatch[0].rm_eo;
856  int i;
857 
858  match_count++;
859 
860  /* If we aren't interested in this match, output old line to
861  * end of match and continue */
862  if (sed_cmd->which_match
863  && (sed_cmd->which_match != match_count)
864  ) {
865  for (i = 0; i < end; i++)
866  pipe_putc(*line++);
867  /* Null match? Print one more char */
868  if (start == end && *line)
869  pipe_putc(*line++);
870  goto next;
871  }
872 
873  /* Print everything before the match */
874  for (i = 0; i < start; i++)
875  pipe_putc(line[i]);
876 
877  /* Then print the substitution string,
878  * unless we just matched empty string after non-empty one.
879  * Example: string "cccd", pattern "c*", repl "R":
880  * result is "RdR", not "RRdR": first match "ccc",
881  * second is "" before "d", third is "" after "d".
882  * Second match is NOT replaced!
883  */
884  if (prev_match_empty || start != 0 || start != end) {
885  //dbg("%d %d %d", prev_match_empty, start, end);
886  dbg("inserting replacement at %d in '%s'", start, line);
887  do_subst_w_backrefs(line, sed_cmd->string);
888  /* Flag that something has changed */
889  altered = 1;
890  } else {
891  dbg("NOT inserting replacement at %d in '%s'", start, line);
892  }
893 
894  /* If matched string is empty (f.e. "c*" pattern),
895  * copy verbatim one char after it before attempting more matches
896  */
897  prev_match_empty = (start == end);
898  if (prev_match_empty) {
899  if (!line[end]) {
900  tried_at_eol = 1;
901  } else {
902  pipe_putc(line[end]);
903  end++;
904  }
905  }
906 
907  /* Advance past the match */
908  dbg("line += %d", end);
909  line += end;
910 
911  /* if we're not doing this globally, get out now */
912  if (sed_cmd->which_match != 0)
913  break;
914  next:
915  /* Exit if we are at EOL and already tried matching at it */
916  if (*line == '\0') {
917  if (tried_at_eol)
918  break;
919  tried_at_eol = 1;
920  }
921 
922 //maybe (end ? REG_NOTBOL : 0) instead of unconditional REG_NOTBOL?
923  } while (regexec(current_regex, line, 10, G.regmatch, REG_NOTBOL) != REG_NOMATCH);
924 
925  /* Copy rest of string into output pipeline */
926  while (1) {
927  char c = *line++;
928  pipe_putc(c);
929  if (c == '\0')
930  break;
931  }
932 
933  free(*line_p);
934  *line_p = G.pipeline.buf;
935  return altered;
936 }
