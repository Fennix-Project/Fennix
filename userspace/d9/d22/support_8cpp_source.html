85  void *dequeueSingle() {
86  while(true) {
87  bool done;

89  if(done) {

116  void _waitProgressFutex(bool *done) {
117  while(true) {
118  auto futex = __atomic_load_n(&_chunk->progressFutex, __ATOMIC_ACQUIRE);
119  __ensure(!(futex & ~(kHelProgressMask | kHelProgressWaiters | kHelProgressDone)));
120  do {
121  if(_lastProgress != (futex & kHelProgressMask)) {
122  *done = false;
123  return;
124  }else if(futex & kHelProgressDone) {
125  *done = true;
126  return;
127  }
128 
129  if(futex & kHelProgressWaiters)
130  break; // Waiters bit is already set (in a previous iteration).
131  } while(!__atomic_compare_exchange_n(&_chunk->progressFutex, &futex,
132  _lastProgress | kHelProgressWaiters,
133  false, __ATOMIC_ACQUIRE, __ATOMIC_ACQUIRE));
134 
135  HEL_CHECK(helFutexWait(&_chunk->progressFutex,
136  _lastProgress | kHelProgressWaiters, -1));
137  }
