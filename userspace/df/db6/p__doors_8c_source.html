54 //
55 // T_VerticalDoor
56 //
57 void T_VerticalDoor (vldoor_t* door)
58 {
59  result_e res;
60 
61  switch(door->direction)
62  {
63  case 0:
64  // WAITING
65  if (!--door->topcountdown)
66  {
67  switch(door->type)
68  {
69  case vld_blazeRaise:
70  door->direction = -1; // time to go back down
71  S_StartSound(&door->sector->soundorg, sfx_bdcls);
72  break;
73 
74  case vld_normal:
75  door->direction = -1; // time to go back down
76  S_StartSound(&door->sector->soundorg, sfx_dorcls);
77  break;
78 
79  case vld_close30ThenOpen:
80  door->direction = 1;
81  S_StartSound(&door->sector->soundorg, sfx_doropn);
82  break;
83 
84  default:
85  break;
86  }
87  }
88  break;
89 
90  case 2:
91  // INITIAL WAIT
92  if (!--door->topcountdown)
93  {
94  switch(door->type)
95  {
96  case vld_raiseIn5Mins:
97  door->direction = 1;
98  door->type = vld_normal;
99  S_StartSound(&door->sector->soundorg, sfx_doropn);
100  break;
101 
102  default:
103  break;
104  }
105  }
106  break;
107 
108  case -1:
109  // DOWN
110  res = T_MovePlane(door->sector,
111  door->speed,
112  door->sector->floorheight,
113  false,1,door->direction);
114  if (res == pastdest)
115  {
116  switch(door->type)
117  {
118  case vld_blazeRaise:
119  case vld_blazeClose:
120  door->sector->specialdata = NULL;
121  P_RemoveThinker (&door->thinker); // unlink and free
122  S_StartSound(&door->sector->soundorg, sfx_bdcls);
123  break;
124 
125  case vld_normal:
126  case vld_close:
127  door->sector->specialdata = NULL;
128  P_RemoveThinker (&door->thinker); // unlink and free
129  break;
130 
131  case vld_close30ThenOpen:
132  door->direction = 0;
133  door->topcountdown = TICRATE*30;
134  break;
135 
136  default:
137  break;
138  }
139  }
140  else if (res == crushed)
141  {
142  switch(door->type)
143  {
144  case vld_blazeClose:
145  case vld_close: // DO NOT GO BACK UP!
146  break;
147 
148  default:
149  door->direction = 1;
150  S_StartSound(&door->sector->soundorg, sfx_doropn);
151  break;
152  }
153  }
154  break;
155 
156  case 1:
157  // UP
158  res = T_MovePlane(door->sector,
159  door->speed,
160  door->topheight,
161  false,1,door->direction);
162 
163  if (res == pastdest)
164  {
165  switch(door->type)
166  {
167  case vld_blazeRaise:
168  case vld_normal:
169  door->direction = 0; // wait at top
170  door->topcountdown = door->topwait;
171  break;
172 
173  case vld_close30ThenOpen:
174  case vld_blazeOpen:
175  case vld_open:
176  door->sector->specialdata = NULL;
177  P_RemoveThinker (&door->thinker); // unlink and free
178  break;
179 
180  default:
181  break;
182  }
183  }
184  break;
185  }
186 }

252 EV_DoDoor
253 ( line_t* line,
254  vldoor_e type )
255 {
256  int secnum,rtn;
257  sector_t* sec;
258  vldoor_t* door;
259 
260  secnum = -1;
261  rtn = 0;
262 
263  while ((secnum = P_FindSectorFromLineTag(line,secnum)) >= 0)
264  {
265  sec = &sectors[secnum];
266  if (sec->specialdata)
267  continue;
268 
269 
270  // new door thinker
271  rtn = 1;
272  door = Z_Malloc (sizeof(*door), PU_LEVSPEC, 0);
273  P_AddThinker (&door->thinker);
274  sec->specialdata = door;
275 
276  door->thinker.function.acp1 = (actionf_p1) T_VerticalDoor;
277  door->sector = sec;
278  door->type = type;
279  door->topwait = VDOORWAIT;
280  door->speed = VDOORSPEED;
281 
282  switch(type)
283  {
284  case vld_blazeClose:
285  door->topheight = P_FindLowestCeilingSurrounding(sec);
286  door->topheight -= 4*FRACUNIT;
287  door->direction = -1;
288  door->speed = VDOORSPEED * 4;
289  S_StartSound(&door->sector->soundorg, sfx_bdcls);
290  break;
291 
292  case vld_close:
293  door->topheight = P_FindLowestCeilingSurrounding(sec);
294  door->topheight -= 4*FRACUNIT;
295  door->direction = -1;
296  S_StartSound(&door->sector->soundorg, sfx_dorcls);
297  break;
298 
299  case vld_close30ThenOpen:
300  door->topheight = sec->ceilingheight;
301  door->direction = -1;
302  S_StartSound(&door->sector->soundorg, sfx_dorcls);
303  break;
304 
305  case vld_blazeRaise:
306  case vld_blazeOpen:
307  door->direction = 1;
308  door->topheight = P_FindLowestCeilingSurrounding(sec);
309  door->topheight -= 4*FRACUNIT;
310  door->speed = VDOORSPEED * 4;
311  if (door->topheight != sec->ceilingheight)
312  S_StartSound(&door->sector->soundorg, sfx_bdopn);
313  break;
314 
315  case vld_normal:
316  case vld_open:
317  door->direction = 1;
318  door->topheight = P_FindLowestCeilingSurrounding(sec);
319  door->topheight -= 4*FRACUNIT;
320  if (door->topheight != sec->ceilingheight)
321  S_StartSound(&door->sector->soundorg, sfx_doropn);
322  break;
323 
324  default:
325  break;
326  }
327 
328  }
329  return rtn;
330 }

334 // EV_VerticalDoor : open a door manually, no tag value
335 //
336 void
337 EV_VerticalDoor
338 ( line_t* line,
339  mobj_t* thing )
340 {
341  player_t* player;
342  sector_t* sec;
343  vldoor_t* door;
344  int side;
345 
346  side = 0; // only front sides can be used
347 
348  // Check for locks
349  player = thing->player;
350 
351  switch(line->special)
352  {
353  case 26: // Blue Lock
354  case 32:
355  if ( !player )
356  return;
357 
358  if (!player->cards[it_bluecard] && !player->cards[it_blueskull])
359  {
360  player->message = DEH_String(PD_BLUEK);
361  S_StartSound(NULL,sfx_oof);
362  return;
363  }
364  break;
365 
366  case 27: // Yellow Lock
367  case 34:
368  if ( !player )
369  return;
370 
371  if (!player->cards[it_yellowcard] &&
372  !player->cards[it_yellowskull])
373  {
374  player->message = DEH_String(PD_YELLOWK);
375  S_StartSound(NULL,sfx_oof);
376  return;
377  }
378  break;
379 
380  case 28: // Red Lock
381  case 33:
382  if ( !player )
383  return;
384 
385  if (!player->cards[it_redcard] && !player->cards[it_redskull])
386  {
387  player->message = DEH_String(PD_REDK);
388  S_StartSound(NULL,sfx_oof);
389  return;
390  }
391  break;
392  }
393 
394  // if the sector has an active thinker, use it
395  sec = sides[ line->sidenum[side^1]] .sector;
396 
397  if (sec->specialdata)
398  {
399  door = sec->specialdata;
400  switch(line->special)
401  {
402  case 1: // ONLY FOR "RAISE" DOORS, NOT "OPEN"s
403  case 26:
404  case 27:
405  case 28:
406  case 117:
407  if (door->direction == -1)
408  door->direction = 1; // go back up
409  else
410  {
411  if (!thing->player)
412  return; // JDC: bad guys never close doors
413 
414  // When is a door not a door?
415  // In Vanilla, door->direction is set, even though
416  // "specialdata" might not actually point at a door.
417 
418  if (door->thinker.function.acp1 == (actionf_p1) T_VerticalDoor)
419  {
420  door->direction = -1; // start going down immediately
421  }
422  else if (door->thinker.function.acp1 == (actionf_p1) T_PlatRaise)
423  {
424  // Erm, this is a plat, not a door.
425  // This notably causes a problem in ep1-0500.lmp where
426  // a plat and a door are cross-referenced; the door
427  // doesn't open on 64-bit.
428  // The direction field in vldoor_t corresponds to the wait
429  // field in plat_t. Let's set that to -1 instead.
430 
431  plat_t *plat;
432 
433  plat = (plat_t *) door;
434  plat->wait = -1;
435  }
436  else
437  {
438  // This isn't a door OR a plat. Now we're in trouble.
439 
440  fprintf(stderr, "EV_VerticalDoor: Tried to close "
441  "something that wasn't a door.\n");
442 
443  // Try closing it anyway. At least it will work on 32-bit
444  // machines.
445 
446  door->direction = -1;
447  }
448  }
449  return;
450  }
451  }
452 
453  // for proper sound
454  switch(line->special)
455  {
456  case 117: // BLAZING DOOR RAISE
457  case 118: // BLAZING DOOR OPEN
458  S_StartSound(&sec->soundorg,sfx_bdopn);
459  break;
460 
461  case 1: // NORMAL DOOR SOUND
462  case 31:
463  S_StartSound(&sec->soundorg,sfx_doropn);
464  break;
465 
466  default: // LOCKED DOOR SOUND
467  S_StartSound(&sec->soundorg,sfx_doropn);
468  break;
469  }
470 
471 
472  // new door thinker
473  door = Z_Malloc (sizeof(*door), PU_LEVSPEC, 0);
474  P_AddThinker (&door->thinker);
475  sec->specialdata = door;
476  door->thinker.function.acp1 = (actionf_p1) T_VerticalDoor;
477  door->sector = sec;
478  door->direction = 1;
479  door->speed = VDOORSPEED;
480  door->topwait = VDOORWAIT;
481 
482  switch(line->special)
483  {
484  case 1:
485  case 26:
486  case 27:
487  case 28:
488  door->type = vld_normal;
489  break;
490 
491  case 31:
492  case 32:
493  case 33:
494  case 34:
495  door->type = vld_open;
496  line->special = 0;
497  break;
498 
499  case 117: // blazing door raise
500  door->type = vld_blazeRaise;
501  door->speed = VDOORSPEED*4;
502  break;
503  case 118: // blazing door open
504  door->type = vld_blazeOpen;
505  line->special = 0;
506  door->speed = VDOORSPEED*4;
507  break;
508  }
509 
510  // find the top and bottom of the movement range
511  door->topheight = P_FindLowestCeilingSurrounding(sec);
512  door->topheight -= 4*FRACUNIT;
513 }

639 void T_SlidingDoor (slidedoor_t* door)
640 {
641  switch(door->status)
642  {
643  case sd_opening:
644  if (!door->timer--)
645  {
646  if (++door->frame == SNUMFRAMES)
647  {
648  // IF DOOR IS DONE OPENING...
649  sides[door->line->sidenum[0]].midtexture = 0;
650  sides[door->line->sidenum[1]].midtexture = 0;
651  door->line->flags &= ML_BLOCKING^0xff;
652 
653  if (door->type == sdt_openOnly)
654  {
655  door->frontsector->specialdata = NULL;
656  P_RemoveThinker (&door->thinker);
657  break;
658  }
659 
660  door->timer = SDOORWAIT;
661  door->status = sd_waiting;
662  }
663  else
664  {
665  // IF DOOR NEEDS TO ANIMATE TO NEXT FRAME...
666  door->timer = SWAITTICS;
667 
668  sides[door->line->sidenum[0]].midtexture =
669  slideFrames[door->whichDoorIndex].
670  frontFrames[door->frame];
671  sides[door->line->sidenum[1]].midtexture =
672  slideFrames[door->whichDoorIndex].
673  backFrames[door->frame];
674  }
675  }
676  break;
677 
678  case sd_waiting:
679  // IF DOOR IS DONE WAITING...
680  if (!door->timer--)
681  {
682  // CAN DOOR CLOSE?
683  if (door->frontsector->thinglist != NULL ||
684  door->backsector->thinglist != NULL)
685  {
686  door->timer = SDOORWAIT;
687  break;
688  }
689 
690  //door->frame = SNUMFRAMES-1;
691  door->status = sd_closing;
692  door->timer = SWAITTICS;
693  }
694  break;
695 
696  case sd_closing:
697  if (!door->timer--)
698  {
699  if (--door->frame < 0)
700  {
701  // IF DOOR IS DONE CLOSING...
702  door->line->flags |= ML_BLOCKING;
703  door->frontsector->specialdata = NULL;
704  P_RemoveThinker (&door->thinker);
705  break;
706  }
707  else
708  {
709  // IF DOOR NEEDS TO ANIMATE TO NEXT FRAME...
710  door->timer = SWAITTICS;
711 
712  sides[door->line->sidenum[0]].midtexture =
713  slideFrames[door->whichDoorIndex].
714  frontFrames[door->frame];
715  sides[door->line->sidenum[1]].midtexture =
716  slideFrames[door->whichDoorIndex].
717  backFrames[door->frame];
718  }
719  }
720  break;
721  }
