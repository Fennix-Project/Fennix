75 /*
76 From initial port to busybox, removed most of the redundancy by
77 converting implementation of a polymorphic interface to the strict
78 functional style. The main role is run a script when link state
79 changed, other activities like audio signal or detailed reports
80 are on the script itself.
81 
82 One questionable point of the design is netlink usage:
83 
84 We have 1 second timeout by default to poll the link status,
85 it is short enough so that there are no real benefits in
86 using netlink to get "instantaneous" interface creation/deletion
87 notifications. We can check for interface existence by just
88 doing some fast ioctl using its name.
89 
90 Netlink code then can be just dropped (1k or more?)

247 static smallint detect_link_iff(void)
248 {
249  struct ifreq ifreq;
250 
251  set_ifreq_to_ifname(&ifreq);
252 
253  if (network_ioctl(SIOCGIFFLAGS, &ifreq, "SIOCGIFFLAGS") < 0) {
254  return IFSTATUS_ERR;
255  }
256 
257  /* If IFF_UP is not set (interface is down), IFF_RUNNING is never set
258  * regardless of link status. Simply continue to report last status -
259  * no point in reporting spurious link downs if interface is disabled
260  * by admin. When/if it will be brought up,
261  * we'll report real link status.

264  return G.iface_last_status;

266  return (ifreq.ifr_flags & IFF_RUNNING) ? IFSTATUS_UP : IFSTATUS_DOWN;

269 static smallint detect_link_wlan(void)
270 {
271  int i;
272  struct iwreq iwrequest;
273  uint8_t mac[ETH_ALEN];
274 
275  memset(&iwrequest, 0, sizeof(iwrequest));
276  strncpy_IFNAMSIZ(iwrequest.ifr_ifrn.ifrn_name, G.iface);
277 
278  if (network_ioctl(SIOCGIWAP, &iwrequest, "SIOCGIWAP") < 0) {
279  return IFSTATUS_ERR;
280  }
281 
282  memcpy(mac, &iwrequest.u.ap_addr.sa_data, ETH_ALEN);
283 
284  if (mac[0] == 0xFF || mac[0] == 0x44 || mac[0] == 0x00) {
285  for (i = 1; i < ETH_ALEN; ++i) {
286  if (mac[i] != mac[0])
287  return IFSTATUS_UP;
288  }
289  return IFSTATUS_DOWN;
290  }
291 
292  return IFSTATUS_UP;

336  argv[4] = NULL;
337 
338  env_PREVIOUS = xasprintf("%s=%s", IFPLUGD_ENV_PREVIOUS, strstatus(G.iface_prev_status));
339  putenv(env_PREVIOUS);
340  env_CURRENT = xasprintf("%s=%s", IFPLUGD_ENV_CURRENT, strstatus(G.iface_last_status));
341  putenv(env_CURRENT);
342 
343  /* r < 0 - can't exec, 0 <= r < 0x180 - exited, >=0x180 - killed by sig (r-0x180) */
344  r = spawn_and_wait(argv);
345 
346  bb_unsetenv_and_free(env_PREVIOUS);
347  bb_unsetenv_and_free(env_CURRENT);
348 
349  bb_info_msg("exit code: %d", r & 0xff);
350  return (option_mask32 & FLAG_IGNORE_RETVAL) ? 0 : r;

353 static void up_iface(void)
354 {
355  struct ifreq ifrequest;
356 
357  if (!G.iface_exists)
358  return;
359 
360  set_ifreq_to_ifname(&ifrequest);
361  if (network_ioctl(SIOCGIFFLAGS, &ifrequest, "getting interface flags") < 0) {
362  G.iface_exists = 0;
363  return;
364  }
365 
366  if (!(ifrequest.ifr_flags & IFF_UP)) {
367  ifrequest.ifr_flags |= IFF_UP;
368  /* Let user know we mess up with interface */
369  bb_simple_info_msg("upping interface");
370  if (network_ioctl(SIOCSIFFLAGS, &ifrequest, "setting interface flags") < 0) {
371  if (errno != ENODEV && errno != EADDRNOTAVAIL)
372  xfunc_die();
373  G.iface_exists = 0;
374  return;
375  }
376  }
377 
378 #if 0 /* why do we mess with IP addr? It's not our business */
379  if (network_ioctl(SIOCGIFADDR, &ifrequest, "can't get interface address") < 0) {
380  } else if (ifrequest.ifr_addr.sa_family != AF_INET) {
381  bb_perror_msg("the interface is not IP-based");
382  } else {
383  ((struct sockaddr_in*)(&ifrequest.ifr_addr))->sin_addr.s_addr = INADDR_ANY;
384  network_ioctl(SIOCSIFADDR, &ifrequest, "can't set interface address");
385  }
386  network_ioctl(SIOCGIFFLAGS, &ifrequest, "can't get interface flags");
387 #endif
388 }

426 static smallint detect_link(void)
427 {
428  smallint status;
429 
430  if (!G.iface_exists)
431  return (option_mask32 & FLAG_MONITOR) ? IFSTATUS_DOWN : IFSTATUS_ERR;
432 
433  /* Some drivers can't detect link status when the interface is down.
434  * I imagine detect_link_iff() is the most vulnerable.
435  * That's why -a "noauto" in an option, not a hardwired behavior.
436  */
437  if (!(option_mask32 & FLAG_NO_AUTO))
438  up_iface();
439 
440  if (G.api_method_num == API_AUTO) {
441  int i;
442  smallint sv_logmode;
443 
444  sv_logmode = logmode;
445  for (i = 0; i < ARRAY_SIZE(method_table); i++) {
446  logmode = LOGMODE_NONE;
447  status = method_table[i].func();
448  logmode = sv_logmode;
449  if (status != IFSTATUS_ERR) {
450  G.api_method_num = i;
451  bb_info_msg("using %s detection mode", method_table[i].name);
452  break;
453  }
454  }
455  } else {
456  status = method_table[G.api_method_num].func();
457  }
458 
459  if (status == IFSTATUS_ERR) {
460  if (option_mask32 & FLAG_IGNORE_FAIL)
461  status = IFSTATUS_DOWN;
462  else if (option_mask32 & FLAG_IGNORE_FAIL_POSITIVE)
463  status = IFSTATUS_UP;
464  else if (G.api_mode[0] == 'a')
465  bb_simple_error_msg("can't detect link status");
466  }
467 
468  if (status != G.iface_last_status) {
469  G.iface_prev_status = G.iface_last_status;
470  G.iface_last_status = status;
471  }
472 
473  return status;
474 }

476 static NOINLINE int check_existence_through_netlink(void)
477 {
478  int iface_len;
479  /* Buffer was 1K, but on linux-3.9.9 it was reported to be too small.
480  * netlink.h: "limit to 8K to avoid MSG_TRUNC when PAGE_SIZE is very large".
481  * Note: on error returns (-1) we exit, no need to free replybuf.
482  */
483  enum { BUF_SIZE = 8 * 1024 };
484  char *replybuf = xmalloc(BUF_SIZE);
485 
486  iface_len = strlen(G.iface);
487  while (1) {
488  struct nlmsghdr *mhdr;
489  ssize_t bytes;
490 
491  bytes = recv(netlink_fd, replybuf, BUF_SIZE, MSG_DONTWAIT);
492  if (bytes < 0) {
493  if (errno == EAGAIN)
494  goto ret;
495  if (errno == EINTR)
496  continue;
497  bb_simple_perror_msg("netlink: recv");
498  return -1;
499  }
500 
501  mhdr = (struct nlmsghdr*)replybuf;
502  while (bytes > 0) {
503  if (!NLMSG_OK(mhdr, bytes)) {
504  bb_simple_error_msg("netlink packet too small or truncated");
505  return -1;
506  }
507 
508  if (mhdr->nlmsg_type == RTM_NEWLINK || mhdr->nlmsg_type == RTM_DELLINK) {
509  struct rtattr *attr;
510  int attr_len;
511 
512  if (mhdr->nlmsg_len < NLMSG_LENGTH(sizeof(struct ifinfomsg))) {
513  bb_simple_error_msg("netlink packet too small or truncated");
514  return -1;
515  }
516 
517  attr = IFLA_RTA(NLMSG_DATA(mhdr));
518  attr_len = IFLA_PAYLOAD(mhdr);
519 
520  while (RTA_OK(attr, attr_len)) {
521  if (attr->rta_type == IFLA_IFNAME) {
522  int len = RTA_PAYLOAD(attr);
523  if (len > IFNAMSIZ)
524  len = IFNAMSIZ;
525  if (iface_len <= len
526  && strncmp(G.iface, RTA_DATA(attr), len) == 0
527  ) {
528  G.iface_exists = (mhdr->nlmsg_type == RTM_NEWLINK);
529  }
530  }
531  attr = RTA_NEXT(attr, attr_len);
532  }
533  }
534 
535  mhdr = NLMSG_NEXT(mhdr, bytes);
536  }
537  }
538 
539  ret:
540  free(replybuf);
541  return G.iface_exists;
542 }

561 int ifplugd_main(int argc UNUSED_PARAM, char **argv)
562 {
563  int iface_status;
564  int delay_time;
565  const char *iface_status_str;
566  struct pollfd netlink_pollfd[1];
567  unsigned opts;
568  const char *api_mode_found;

570  char *pidfile_name;
571  pid_t pid_from_pidfile;
572 #endif
573

575 
576  opts = getopt32(argv, OPTION_STR,
577  &G.iface, &G.script_name, &G.poll_time, &G.delay_up,
578  &G.delay_down, &G.api_mode, &G.extra_arg);
579  G.poll_time *= 1000;
580 
581  applet_name = xasprintf("ifplugd(%s)", G.iface);
582 
583 #if ENABLE_FEATURE_PIDFILE
584  pidfile_name = xasprintf(CONFIG_PID_FILE_PATH "/ifplugd.%s.pid", G.iface);
585  pid_from_pidfile = read_pid(pidfile_name);
586 
587  if (opts & FLAG_KILL) {
588  if (pid_from_pidfile > 0)
589  /* Upstream tool use SIGINT for -k */
590  kill(pid_from_pidfile, SIGINT);
591  return EXIT_SUCCESS;
592  }
593 
594  if (pid_from_pidfile > 0 && kill(pid_from_pidfile, 0) == 0)
595  bb_simple_error_msg_and_die("daemon already running");
596 #endif
597 
598  api_mode_found = strchr(api_modes, G.api_mode[0]);
599  if (!api_mode_found)
600  bb_error_msg_and_die("unknown API mode '%s'", G.api_mode);
601  G.api_method_num = api_mode_found - api_modes;
602 
603  if (!(opts & FLAG_NO_DAEMON))
604  bb_daemonize_or_rexec(DAEMON_CHDIR_ROOT, argv);
605 
606  xmove_fd(xsocket(AF_INET, SOCK_DGRAM, 0), ioctl_fd);
607  if (opts & FLAG_MONITOR) {
608  int fd = create_and_bind_to_netlink(NETLINK_ROUTE, RTMGRP_LINK, 0);
609  xmove_fd(fd, netlink_fd);
610  }
611 
612  write_pidfile(pidfile_name);
613 
614  /* this can't be moved before socket creation */
