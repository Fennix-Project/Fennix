70 unsigned int W_LumpNameHash(const char *s)
71 {
72  // This is the djb2 string hash function, modded to work on strings
73  // that have a maximum length of 8.
74 
75  unsigned int result = 5381;
76  unsigned int i;
77 
78  for (i=0; i < 8 && s[i] != '\0'; ++i)
79  {
80  result = ((result << 5) ^ result ) ^ toupper((int)s[i]);
81  }
82 
83  return result;
84 }

86 // Increase the size of the lumpinfo[] array to the specified size.
87 static void ExtendLumpInfo(int newnumlumps)
88 {
89  lumpinfo_t *newlumpinfo;
90  unsigned int i;
91 
92  newlumpinfo = calloc(newnumlumps, sizeof(lumpinfo_t));
93 
94  if (newlumpinfo == NULL)
95  {
96  I_Error ("Couldn't realloc lumpinfo");
97  }
98 
99  // Copy over lumpinfo_t structures from the old array. If any of
100  // these lumps have been cached, we need to update the user
101  // pointers to the new location.
102  for (i = 0; i < numlumps && i < newnumlumps; ++i)
103  {
104  memcpy(&newlumpinfo[i], &lumpinfo[i], sizeof(lumpinfo_t));
105 
106  if (newlumpinfo[i].cache != NULL)
107  {
108  Z_ChangeUser(newlumpinfo[i].cache, &newlumpinfo[i].cache);
109  }
110 
111  // We shouldn't be generating a hash table until after all WADs have
112  // been loaded, but just in case...
113  if (lumpinfo[i].next != NULL)
114  {
115  int nextlumpnum = lumpinfo[i].next - lumpinfo;
116  newlumpinfo[i].next = &newlumpinfo[nextlumpnum];
117  }
118  }
119 
120  // All done.
121  free(lumpinfo);
122  lumpinfo = newlumpinfo;
123  numlumps = newnumlumps;
124 }

139 wad_file_t *W_AddFile (char *filename)
140 {
141  wadinfo_t header;
142  lumpinfo_t *lump_p;
143  unsigned int i;
144  wad_file_t *wad_file;
145  int length;
146  int startlump;
147  filelump_t *fileinfo;
148  filelump_t *filerover;
149  int newnumlumps;
150 
151  // open the file and add to directory
152 
153  wad_file = W_OpenFile(filename);
154 
155  if (wad_file == NULL)
156  {
157  printf (" couldn't open %s\n", filename);
158  return NULL;
159  }
160 
161  newnumlumps = numlumps;
162 
163  if (strcasecmp(filename+strlen(filename)-3 , "wad" ) )
164  {
165  // single lump file
166 
167  // fraggle: Swap the filepos and size here. The WAD directory
168  // parsing code expects a little-endian directory, so will swap
169  // them back. Effectively we're constructing a "fake WAD directory"
170  // here, as it would appear on disk.
171 
172  fileinfo = Z_Malloc(sizeof(filelump_t), PU_STATIC, 0);
173  fileinfo->filepos = LONG(0);
174  fileinfo->size = LONG(wad_file->length);
175 
176  // Name the lump after the base of the filename (without the
177  // extension).
178 
179  M_ExtractFileBase (filename, fileinfo->name);
180  newnumlumps++;
181  }
182  else
183  {
184  // WAD file
185  W_Read(wad_file, 0, &header, sizeof(header));
186 
187  if (strncmp(header.identification,"IWAD",4))
188  {
189  // Homebrew levels?
190  if (strncmp(header.identification,"PWAD",4))
191  {
192  I_Error ("Wad file %s doesn't have IWAD "
193  "or PWAD id\n", filename);
194  }
195 
196  // ???modifiedgame = true;
197  }
198 
199  header.numlumps = LONG(header.numlumps);
200  header.infotableofs = LONG(header.infotableofs);
201  length = header.numlumps*sizeof(filelump_t);
202  fileinfo = Z_Malloc(length, PU_STATIC, 0);
203 
204  W_Read(wad_file, header.infotableofs, fileinfo, length);
205  newnumlumps += header.numlumps;
206  }
207 
208  // Increase size of numlumps array to accomodate the new file.
209  startlump = numlumps;
210  ExtendLumpInfo(newnumlumps);
211 
212  lump_p = &lumpinfo[startlump];
213 
214  filerover = fileinfo;
215 
216  for (i=startlump; i<numlumps; ++i)
217  {
218  lump_p->wad_file = wad_file;
219  lump_p->position = LONG(filerover->filepos);
220  lump_p->size = LONG(filerover->size);
221  lump_p->cache = NULL;
222  strncpy(lump_p->name, filerover->name, 8);
223 
224  ++lump_p;
225  ++filerover;
226  }
227 
228  Z_Free(fileinfo);
229 
230  if (lumphash != NULL)
231  {
232  Z_Free(lumphash);
233  lumphash = NULL;
234  }
235 
236  return wad_file;
237 }

252 // W_CheckNumForName
253 // Returns -1 if name not found.
254 //
255 
256 int W_CheckNumForName (char* name)
257 {
258  lumpinfo_t *lump_p;
259  int i;
260 
261  // Do we have a hash table yet?
262 
263  if (lumphash != NULL)
264  {
265  int hash;
266 
267  // We do! Excellent.
268 
269  hash = W_LumpNameHash(name) % numlumps;
270 
271  for (lump_p = lumphash[hash]; lump_p != NULL; lump_p = lump_p->next)
272  {
273  if (!strncasecmp(lump_p->name, name, 8))
274  {
275  return lump_p - lumpinfo;
276  }
277  }
278  }
279  else
280  {
281  // We don't have a hash table generate yet. Linear search :-(
282  //
283  // scan backwards so patch lump files take precedence
284 
285  for (i=numlumps-1; i >= 0; --i)
286  {
287  if (!strncasecmp(lumpinfo[i].name, name, 8))
288  {
289  return i;
290  }
291  }
292  }
293 
294  // TFB. Not found.
295 
296  return -1;
297 }

370 //
371 // W_CacheLumpNum
372 //
373 // Load a lump into memory and return a pointer to a buffer containing
374 // the lump data.
375 //
376 // 'tag' is the type of zone memory buffer to allocate for the lump
377 // (usually PU_STATIC or PU_CACHE). If the lump is loaded as
378 // PU_STATIC, it should be released back using W_ReleaseLumpNum
379 // when no longer needed (do not use Z_ChangeTag).
380 //
381 
382 void *W_CacheLumpNum(int lumpnum, int tag)
383 {
384  byte *result;
385  lumpinfo_t *lump;
386 
387  if ((unsigned)lumpnum >= numlumps)
388  {
389  I_Error ("W_CacheLumpNum: %i >= numlumps", lumpnum);
390  }
391 
392  lump = &lumpinfo[lumpnum];
393 
394  // Get the pointer to return. If the lump is in a memory-mapped
395  // file, we can just return a pointer to within the memory-mapped
396  // region. If the lump is in an ordinary file, we may already
397  // have it cached; otherwise, load it into memory.
398 
399  if (lump->wad_file->mapped != NULL)
400  {
401  // Memory mapped file, return from the mmapped region.
402 
403  result = lump->wad_file->mapped + lump->position;
404  }
405  else if (lump->cache != NULL)
406  {
407  // Already cached, so just switch the zone tag.
408 
409  result = lump->cache;
410  Z_ChangeTag(lump->cache, tag);
411  }
412  else
413  {
414  // Not yet loaded, so load it now
415 
416  lump->cache = Z_Malloc(W_LumpLength(lumpnum), tag, &lump->cache);
417  W_ReadLump (lumpnum, lump->cache);
418  result = lump->cache;
419  }
420 
421  return result;
422 }

537 // Generate a hash table for fast lookups
538 
539 void W_GenerateHashTable(void)
540 {
541  unsigned int i;
542 
543  // Free the old hash table, if there is one
544 
545  if (lumphash != NULL)
546  {
547  Z_Free(lumphash);
548  }
549 
550  // Generate hash table
551  if (numlumps > 0)
552  {
553  lumphash = Z_Malloc(sizeof(lumpinfo_t *) * numlumps, PU_STATIC, NULL);
554  memset(lumphash, 0, sizeof(lumpinfo_t *) * numlumps);
555 
556  for (i=0; i<numlumps; ++i)
557  {
558  unsigned int hash;
559 
560  hash = W_LumpNameHash(lumpinfo[i].name) % numlumps;
561 
562  // Hook into the hash table
563 
564  lumpinfo[i].next = lumphash[hash];
565  lumphash[hash] = &lumpinfo[i];
566  }
567  }
568 
569  // All done!
570 }
