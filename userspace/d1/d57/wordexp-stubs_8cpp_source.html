44 static size_t we_read_fully(int fd, char *buffer, size_t len) {
45  size_t done = 0;
46 
47  do {
48  ssize_t nread = read(fd, buffer + done, len - done);
49  if(nread == -1 && errno == EINTR)
50  continue;
51  if(nread <= 0)
52  break;
53  done += nread;
54  } while (done != len);
55 
56  return done;
57 }

59 static int we_askshell(const char *words, wordexp_t *we, int flags) {
60  int pdes[2]; /* pipe to child */
61  char bbuf[9]; /* buffer for byte count */
62  char wbuf[9]; /* buffer for word count */
63  size_t nwords = 0; /* number of words from child */
64  size_t nbytes = 0; /* number of bytes from child */
65  size_t sofs = 0; /* offset into we->we_strings */
66  size_t vofs = 0; /* offset into we->we_wordv */
67  pid_t pid; /* PID of child */
68  pid_t wpid; /* waitpid return value */
69  int status; /* child exit status */
70  int error; /* our return value */
71  int serrno; /* errno to return */
72  char *np, *p; /* handy pointers */
73  char *nstrings; /* temporary for realloc() */
74  char **new_wordv; /* temporary for realloc() */
75  sigset_t newsigblock;
76  sigset_t oldsigblock;
77  const char *ifs = getenv("IFS");
78 
79  serrno = errno;
80 
81  if(pipe2(pdes, O_CLOEXEC) < 0)
82  return WRDE_NOSPACE;
83 
84  (void)sigemptyset(&newsigblock);
85  (void)sigaddset(&newsigblock, SIGCHLD);
86  (void)sigprocmask(SIG_BLOCK, &newsigblock, &oldsigblock);
87 
88  if((pid = fork()) < 0) {
89  serrno = errno;
90  close(pdes[0]);
91  close(pdes[1]);
92  (void)sigprocmask(SIG_SETMASK, &oldsigblock, NULL);
93  errno = serrno;
94  return WRDE_NOSPACE;
95  } else if(pid == 0) {
96  /*
97  * We are the child; make /bin/sh expand `words'.
98  */
99  (void)sigprocmask(SIG_SETMASK, &oldsigblock, NULL);
100  if((pdes[1] != STDOUT_FILENO ? dup2(pdes[1], STDOUT_FILENO) : fcntl(pdes[1], F_SETFD, 0)) < 0)
101  _exit(1);
102 
103  execl(SHELL_PATH, SHELL_NAME, flags & WRDE_UNDEF ? "-u" : "+u",
104  "-c", "IFS=$1;eval \"$2\";eval \"set -- $3\";IFS=;a=\"$*\";"
105  "printf '%08x' \"$#\" \"${#a}\";printf '%s\\0' \"$@\"", "",
106  ifs != NULL ? ifs : " \t\n",
107  flags & WRDE_SHOWERR ? "" : "exec 2>/dev/null",
108  words,
109  (char *)NULL);
110  _exit(1);
111  }
112 
113  /*
114  * We are the parent; read the output of the shell wordexp function,
115  * which is a 32-bit hexadecimal word count, a 32-bit hexadecimal
116  * byte count (not including terminating null bytes), followed by
117  * the expanded words separated by nulls.
118  */
119  close(pdes[1]);
120  if(we_read_fully(pdes[0], wbuf, 8) != 8 || we_read_fully(pdes[0], bbuf, 8) != 8) {
121  error = flags & WRDE_UNDEF ? WRDE_BADVAL : WRDE_SYNTAX;
122  serrno = errno;
123  goto cleanup;
124  }
125  wbuf[8] = bbuf[8] = '\0';
126  nwords = strtol(wbuf, NULL, 16);
127  nbytes = strtol(bbuf, NULL, 16) + nwords;
128 
129  /*
130  * Allocate or reallocate (when flags & WRDE_APPEND) the word vector
131  * and string storage buffers for the expanded words we're about to
132  * read from the child.
133  */
134  sofs = we->we_nbytes;
135  vofs = we->we_wordc;
136  if((flags & (WRDE_DOOFFS|WRDE_APPEND)) == (WRDE_DOOFFS | WRDE_APPEND))
137  vofs += we->we_offs;
138  we->we_wordc += nwords;
139  we->we_nbytes += nbytes;
140 
141  if((new_wordv = (char **) realloc(we->we_wordv, (we->we_wordc + 1 + (flags & WRDE_DOOFFS ? we->we_offs : 0)) * sizeof(char *))) == NULL) {
142  error = WRDE_NOSPACE;
143  goto cleanup;
144  }
145 
146  we->we_wordv = new_wordv;
147 
148  if((nstrings = (char *) realloc(we->we_strings, we->we_nbytes)) == NULL) {
149  error = WRDE_NOSPACE;
150  goto cleanup;
151  }
152 
153  for(size_t i = 0; i < vofs; i++) {
154  if(we->we_wordv[i] != NULL)
155  we->we_wordv[i] += nstrings - we->we_strings;
156  }
157  we->we_strings = nstrings;
158 
159  if(we_read_fully(pdes[0], we->we_strings + sofs, nbytes) != nbytes) {
160  error = flags & WRDE_UNDEF ? WRDE_BADVAL : WRDE_SYNTAX;
161  serrno = errno;
162  goto cleanup;
163  }
164 
165  error = 0;
166 cleanup:
167  close(pdes[0]);
168 
169  do {
170  wpid = waitpid(pid, &status, 0);
171  } while(wpid < 0 && errno == EINTR);
172 
173  (void)sigprocmask(SIG_SETMASK, &oldsigblock, NULL);
174 
175  if(error != 0) {
176  errno = serrno;
177  return error;
178  }
179 
180  if(wpid < 0 || !WIFEXITED(status) || WEXITSTATUS(status) != 0)
181  return flags & WRDE_UNDEF ? WRDE_BADVAL : WRDE_SYNTAX;
182 
183  /*
184  * Break the null-terminated expanded word strings out into
185  * the vector.
186  */
187  if(vofs == 0 && flags & WRDE_DOOFFS) {
188  while (vofs < we->we_offs)
189  we->we_wordv[vofs++] = NULL;
190  }
191 
192  p = we->we_strings + sofs;
193  while (nwords-- != 0) {
194  we->we_wordv[vofs++] = p;
195  if((np = (char *) memchr(p, '\0', nbytes)) == NULL)
196  return WRDE_NOSPACE;
197 
198  nbytes -= np - p + 1;
199  p = np + 1;
200  }
201 
202  we->we_wordv[vofs] = NULL;
203  return 0;
204 }

206 /*
207  * we_check --
208  * Check that the string contains none of the following unquoted
209  * special characters: <newline> |&;<>(){}
210  * or command substitutions when WRDE_NOCMD is set in flags.
211  */
212 static int we_check(const char *words, int flags)
213 {
214  char c;
215  int dquote, level, quote, squote;
216 
217  quote = squote = dquote = 0;
218  while ((c = *words++) != '\0') {
219  switch (c) {
220  case '\\': {
221  if(squote == 0)
222  quote ^= 1;
223  continue;
224  }
225  case '\'': {
226  if(quote + dquote == 0)
227  squote ^= 1;
228  break;
229  }
230  case '"': {
231  if(quote + squote == 0)
232  dquote ^= 1;
233  break;
234  }
235  case '`': {
236  if(quote + squote == 0 && flags & WRDE_NOCMD)
237  return WRDE_CMDSUB;
238  while ((c = *words++) != '\0' && c != '`')
239  if(c == '\\' && (c = *words++) == '\0')
240  break;
241  if(c == '\0')
242  return WRDE_SYNTAX;
243  break;
244  }
245  case '|':
246  case '&':
247  case ';':
248  case '<':
249  case '>':
250  case '{':
251  case '}':
252  case '(':
253  case ')':
254  case '\n': {
255  if(quote + squote + dquote == 0)
256  return WRDE_BADCHAR;
257  break;
258  }
259  case '$': {
260  if((c = *words++) == '\0')
261  break;
262  else if(quote + squote == 0 && c == '(') {
263  if(flags & WRDE_NOCMD && *words != '(')
264  return WRDE_CMDSUB;
265  level = 1;
266  while ((c = *words++) != '\0') {
267  if(c == '\\') {
268  if((c = *words++) == '\0')
269  break;
270  } else if(c == '(')
271  level++;
272  else if(c == ')' && --level == 0)
273  break;
274  }
275  if(c == '\0' || level != 0)
276  return WRDE_SYNTAX;
277  } else if(quote + squote == 0 && c == '{') {
278  level = 1;
279  while ((c = *words++) != '\0') {
280  if(c == '\\') {
281  if((c = *words++) == '\0')
282  break;
283  } else if(c == '{')
284  level++;
285  else if(c == '}' && --level == 0)
286  break;
287  }
288  if(c == '\0' || level != 0)
289  return WRDE_SYNTAX;
290  } else
291  --words;
292  break;
293  }
294  default: {
295  break;
296  }
297  }
298  quote = 0;
299  }
300 
301  if(quote + squote + dquote != 0)
302  return WRDE_SYNTAX;
303 
304  return 0;
305 }

307 int wordexp(const char * __restrict words, wordexp_t * __restrict we, int flags) {
308  int error;
309 
310  if(flags & WRDE_REUSE)
311  wordfree(we);
312 
313  if((flags & WRDE_APPEND) == 0) {
314  we->we_wordc = 0;
315  we->we_wordv = NULL;
316  we->we_strings = NULL;
317  we->we_nbytes = 0;
318  }
319 
320  if((error = we_check(words, flags)) != 0) {
321  wordfree(we);
322  return error;
323  }
324 
325  if((error = we_askshell(words, we, flags)) != 0) {
326  wordfree(we);
327  return error;
328  }
329 
330  return 0;
331 }
