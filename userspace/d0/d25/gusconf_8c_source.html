61 static int SplitLine(char *line, char **fields, unsigned int max_fields)
62 {
63  unsigned int num_fields;
64  char *p;
65 
66  fields[0] = line;
67  num_fields = 1;
68 
69  for (p = line; *p != '\0'; ++p)
70  {
71  if (*p == ',')
72  {
73  *p = '\0';
74 
75  // Skip spaces following the comma.
76  do
77  {
78  ++p;
79  } while (*p != '\0' && isspace(*p));
80 
81  fields[num_fields] = p;
82  ++num_fields;
83  --p;
84 
85  if (num_fields >= max_fields)
86  {
87  break;
88  }
89  }
90  else if (*p == '#')
91  {
92  *p = '\0';
93  break;
94  }
95  }
96 
97  // Strip off trailing whitespace from the end of the line.
98  p = fields[num_fields - 1] + strlen(fields[num_fields - 1]);
99  while (p > fields[num_fields - 1] && isspace(*(p - 1)))
100  {
101  --p;
102  *p = '\0';
103  }
104 
105  return num_fields;
106 }

108 static void ParseLine(gus_config_t *config, char *line)
109 {
110  char *fields[6];
111  unsigned int num_fields;
112  unsigned int instr_id, mapped_id;
113 
114  num_fields = SplitLine(line, fields, 6);
115 
116  if (num_fields < 6)
117  {
118  return;
119  }
120 
121  instr_id = atoi(fields[0]);
122  mapped_id = atoi(fields[MappingIndex()]);
123 
124  free(config->patch_names[instr_id]);
125  config->patch_names[instr_id] = strdup(fields[5]);
126  config->mapping[instr_id] = mapped_id;
127 }

129 static void ParseDMXConfig(char *dmxconf, gus_config_t *config)
130 {
131  char *p, *newline;
132  unsigned int i;
133 
134  memset(config, 0, sizeof(gus_config_t));
135 
136  for (i = 0; i < MAX_INSTRUMENTS; ++i)
137  {
138  config->mapping[i] = -1;
139  }
140 
141  p = dmxconf;
142 
143  for (;;)
144  {
145  newline = strchr(p, '\n');
146 
147  if (newline != NULL)
148  {
149  *newline = '\0';
150  }
151 
152  ParseLine(config, p);
153 
154  if (newline == NULL)
155  {
156  break;
157  }
158  else
159  {
160  p = newline + 1;
161  }
162  }
163 }

197 static boolean WriteTimidityConfig(char *path, gus_config_t *config)
198 {
199  FILE *fstream;
200  unsigned int i;
201 
202  fstream = fopen(path, "w");
203 
204  if (fstream == NULL)
205  {
206  return false;
207  }
208 
209  fprintf(fstream, "# Autogenerated Timidity config.\n\n");
210 
211  fprintf(fstream, "dir %s\n", gus_patch_path);
212 
213  fprintf(fstream, "\nbank 0\n\n");
214 
215  for (i = 0; i < 128; ++i)
216  {
217  if (config->mapping[i] >= 0 && config->mapping[i] < MAX_INSTRUMENTS
218  && config->patch_names[config->mapping[i]] != NULL)
219  {
220  fprintf(fstream, "%i %s\n",
221  i, config->patch_names[config->mapping[i]]);
222  }
223  }
224 
225  fprintf(fstream, "\ndrumset 0\n\n");
226 
227  for (i = 128 + 25; i < MAX_INSTRUMENTS; ++i)
228  {
229  if (config->mapping[i] >= 0 && config->mapping[i] < MAX_INSTRUMENTS
230  && config->patch_names[config->mapping[i]] != NULL)
231  {
232  fprintf(fstream, "%i %s\n",
233  i - 128, config->patch_names[config->mapping[i]]);
234  }
235  }
236 
237  fprintf(fstream, "\n");
238 
239  fclose(fstream);
240 
241  return true;
242 }

244 boolean GUS_WriteConfig(char *path)
245 {
246  boolean result;
247  char *dmxconf;
248  gus_config_t config;
249 
250  if (!strcmp(gus_patch_path, ""))
251  {
252  printf("You haven't configured gus_patch_path.\n");
253  printf("gus_patch_path needs to point to the location of "
254  "your GUS patch set.\n"
255  "To get a copy of the \"standard\" GUS patches, "
256  "download a copy of dgguspat.zip.\n");
257 
258  return false;
259  }
260 
261  dmxconf = ReadDMXConfig();
262  ParseDMXConfig(dmxconf, &config);
263 
264  result = WriteTimidityConfig(path, &config);
265 
266  FreeDMXConfig(&config);
267  Z_Free(dmxconf);
268 
269  return result;
270 }
