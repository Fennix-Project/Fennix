51 int
52 mkdir_builtin (list)
53  WORD_LIST *list;
54 {
55  int opt, pflag, mflag, omode, rval, nmode, parent_mode;
56  char *mode;
57  WORD_LIST *l;
58 
59  reset_internal_getopt ();
60  pflag = mflag = 0;
61  mode = (char *)NULL;
62  while ((opt = internal_getopt(list, "m:p")) != -1)
63  switch (opt)
64  {
65  case 'p':
66  pflag = 1;
67  break;
68  case 'm':
69  mflag = 1;
70  mode = list_optarg;
71  break;
72  CASE_HELPOPT;
73  default:
74  builtin_usage();
75  return (EX_USAGE);
76  }
77  list = loptend;
78 
79  if (list == 0)
80  {
81  builtin_usage ();
82  return (EX_USAGE);
83  }
84 
85  if (mode == NULL)
86  omode = S_IRWXU | S_IRWXG | S_IRWXO; /* a=rwx */
87  else if (ISOCTAL (*mode)) /* octal number */
88  {
89  omode = read_octal (mode);
90  if (omode < 0)
91  {
92  builtin_error ("invalid file mode: %s", mode);
93  return (EXECUTION_FAILURE);
94  }
95  }
96  else /* symbolic mode */
97  {
98  /* initial bits are a=rwx; the mode argument modifies them */
99  omode = parse_symbolic_mode (mode, S_IRWXU | S_IRWXG | S_IRWXO);
100  if (omode < 0)
101  {
102  builtin_error ("invalid file mode: %s", mode);
103  return (EXECUTION_FAILURE);
104  }
105  }
106 
107  /* Make the new mode */
108  original_umask = umask (0);
109  umask (original_umask);
110 
111  nmode = (S_IRWXU | S_IRWXG | S_IRWXO) & ~original_umask;
112  parent_mode = nmode | (S_IWUSR|S_IXUSR); /* u+wx */
113 
114  /* Adjust new mode based on mode argument */
115  nmode &= omode;
116 
117  for (rval = EXECUTION_SUCCESS, l = list; l; l = l->next)
118  {
119  if (pflag && make_path (l->word->word, mflag, nmode, parent_mode))
120  {
121  rval = EXECUTION_FAILURE;
122  continue;
123  }
124  else if (pflag == 0 && mkdir (l->word->word, nmode) < 0)
125  {
126  builtin_error ("cannot create directory `%s': %s", l->word->word, strerror (errno));
127  rval = EXECUTION_FAILURE;
128  }
129  }
130  return rval;
131 }

133 /* Make all the directories leading up to PATH, then create PATH. Note that
134  this changes the process's umask; make sure that all paths leading to a
135  return reset it to ORIGINAL_UMASK */
136 static int
137 make_path (path, user_mode, nmode, parent_mode)
138  char *path;
139  int user_mode;
140  int nmode, parent_mode;
141 {
142  int oumask;
143  struct stat sb;
144  char *p, *npath;
145 
146  if (stat (path, &sb) == 0)
147  {
148  if (S_ISDIR (sb.st_mode) == 0)
149  {
150  builtin_error ("`%s': file exists but is not a directory", path);
151  return 1;
152  }
153 
154  if (user_mode && chmod (path, nmode))
155  {
156  builtin_error ("%s: %s", path, strerror (errno));
157  return 1;
158  }
159 
160  return 0;
161  }
162 
163  oumask = umask (0);
164  npath = savestring (path); /* So we can write to it. */
165 
166  /* Check whether or not we need to do anything with intermediate dirs. */
167 
168  /* Skip leading slashes. */
169  p = npath;
170  while (*p == '/')
171  p++;
172 
173  while (p = strchr (p, '/'))
174  {
175  *p = '\0';
176  if (stat (npath, &sb) != 0)
177  {
178  if (mkdir (npath, 0))
179  {
180  builtin_error ("cannot create directory `%s': %s", npath, strerror (errno));
181  umask (original_umask);
182  free (npath);
183  return 1;
184  }
185  if (chmod (npath, parent_mode) != 0)
186  {
187  builtin_error ("cannot chmod directory `%s': %s", npath, strerror (errno));
188  umask (original_umask);
189  free (npath);
190  return 1;
191  }
192  }
193  else if (S_ISDIR (sb.st_mode) == 0)
194  {
195  builtin_error ("`%s': file exists but is not a directory", npath);
196  umask (original_umask);
197  free (npath);
198  return 1;
199  }
200 
201  *p++ = '/'; /* restore slash */
202  while (*p == '/')
203  p++;
204  }
205 
206  /* Create the final directory component. */
207  if (stat (npath, &sb) && mkdir (npath, nmode))
208  {
209  builtin_error ("cannot create directory `%s': %s", npath, strerror (errno));
210  umask (original_umask);
211  free (npath);
212  return 1;
213  }
214 
215  umask (original_umask);
216  free (npath);
217  return 0;
218 }
