98 /* Modify the case of characters in STRING matching PAT based on the value of
99  FLAGS. If PAT is null, modify the case of each character */
100 char *
101 sh_modcase (string, pat, flags)
102  const char *string;
103  char *pat;
104  int flags;
105 {
106  int start, next, end, retind;
107  int inword, c, nc, nop, match, usewords;
108  char *ret, *s;
109  wchar_t wc;
110  int mb_cur_max;
111 #if defined (HANDLE_MULTIBYTE)
112  wchar_t nwc;
113  char mb[MB_LEN_MAX+1];
114  int mlen;
115  size_t m;
116  mbstate_t state;
117 #endif
118 
119  if (string == 0 || *string == 0)
120  {
121  ret = (char *)xmalloc (1);
122  ret[0] = '\0';
123  return ret;
124  }
125 
126 #if defined (HANDLE_MULTIBYTE)
127  memset (&state, 0, sizeof (mbstate_t));
128 #endif
129 
130  start = 0;
131  end = strlen (string);
132  mb_cur_max = MB_CUR_MAX;
133 
134  ret = (char *)xmalloc (2*end + 1);
135  retind = 0;
136 
137  /* See if we are supposed to split on alphanumerics and operate on each word */
138  usewords = (flags & CASE_USEWORDS);
139  flags &= ~CASE_USEWORDS;
140 
141  inword = 0;
142  while (start < end)
143  {
144  wc = cval ((char *)string, start, end);
145 
146  if (iswalnum (wc) == 0)
147  inword = 0;
148 
149  if (pat)
150  {
151  next = start;
152  ADVANCE_CHAR (string, end, next);
153  s = substring ((char *)string, start, next);
154  match = strmatch (pat, s, FNM_EXTMATCH) != FNM_NOMATCH;
155  free (s);
156  if (match == 0)
157  {
158  /* copy unmatched portion */
159  memcpy (ret + retind, string + start, next - start);
160  retind += next - start;
161  start = next;
162  inword = 1;
163  continue;
164  }
165  }
166 
167  /* XXX - for now, the toggling operators work on the individual
168  words in the string, breaking on alphanumerics. Should I
169  leave the capitalization operators to do that also? */
170  if (flags == CASE_CAPITALIZE)
171  {
172  if (usewords)
173  nop = inword ? CASE_LOWER : CASE_UPPER;
174  else
175  nop = (start > 0) ? CASE_LOWER : CASE_UPPER;
176  inword = 1;
177  }
178  else if (flags == CASE_UNCAP)
179  {
180  if (usewords)
181  nop = inword ? CASE_UPPER : CASE_LOWER;
182  else
183  nop = (start > 0) ? CASE_UPPER : CASE_LOWER;
184  inword = 1;
185  }
186  else if (flags == CASE_UPFIRST)
187  {
188  if (usewords)
189  nop = inword ? CASE_NOOP : CASE_UPPER;
190  else
191  nop = (start > 0) ? CASE_NOOP : CASE_UPPER;
192  inword = 1;
193  }
194  else if (flags == CASE_LOWFIRST)
195  {
196  if (usewords)
197  nop = inword ? CASE_NOOP : CASE_LOWER;
198  else
199  nop = (start > 0) ? CASE_NOOP : CASE_LOWER;
200  inword = 1;
201  }
202  else if (flags == CASE_TOGGLE)
203  {
204  nop = inword ? CASE_NOOP : CASE_TOGGLE;
205  inword = 1;
206  }
207  else
208  nop = flags;
209 
210  /* Can't short-circuit, some locales have multibyte upper and lower
211  case equivalents of single-byte ascii characters (e.g., Turkish) */
212  if (mb_cur_max == 1)
213  {
214 singlebyte:
215  switch (nop)
216  {
217  default:
218  case CASE_NOOP: nc = wc; break;
219  case CASE_UPPER: nc = TOUPPER (wc); break;
220  case CASE_LOWER: nc = TOLOWER (wc); break;
221  case CASE_TOGGLEALL:
222  case CASE_TOGGLE: nc = TOGGLE (wc); break;
223  }
224  ret[retind++] = nc;
225  }
226 #if defined (HANDLE_MULTIBYTE)
227  else
228  {
229  m = mbrtowc (&wc, string + start, end - start, &state);
230  /* Have to go through wide case conversion even for single-byte
231  chars, to accommodate single-byte characters where the
232  corresponding upper or lower case equivalent is multibyte. */
233  if (MB_INVALIDCH (m))
234  {
235  wc = (unsigned char)string[start];
236  goto singlebyte;
237  }
238  else if (MB_NULLWCH (m))
239  wc = L'\0';
240  switch (nop)
241  {
242  default:
243  case CASE_NOOP: nwc = wc; break;
244  case CASE_UPPER: nwc = _to_wupper (wc); break;
245  case CASE_LOWER: nwc = _to_wlower (wc); break;
246  case CASE_TOGGLEALL:
247  case CASE_TOGGLE: nwc = TOGGLE (wc); break;
248  }
249 
250  /* We don't have to convert `wide' characters that are in the
251  unsigned char range back to single-byte `multibyte' characters. */
252  if ((int)nwc <= UCHAR_MAX && is_basic ((int)nwc))
253  ret[retind++] = nwc;
254  else
255  {
256  mlen = wcrtomb (mb, nwc, &state);
257  if (mlen > 0)
258  mb[mlen] = '\0';
259  /* Don't assume the same width */
260  strncpy (ret + retind, mb, mlen);
261  retind += mlen;
262  }
263  }
264 #endif
265 
266  ADVANCE_CHAR (string, end, start);
267  }
268 
269  ret[retind] = '\0';
270  return ret;
271 }
