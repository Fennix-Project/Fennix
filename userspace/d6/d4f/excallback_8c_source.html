65 /* This little examples demonstrates the alternate interface to using readline.
66  * In the alternate interface, the user maintains control over program flow and
67  * only calls readline when STDIN is readable. Using the alternate interface,
68  * you can do anything else while still using readline (like talking to a
69  * network or another program) without blocking.
70  *
71  * Specifically, this program highlights two importants features of the
72  * alternate interface. The first is the ability to interactively change the
73  * prompt, which can't be done using the regular interface since rl_prompt is
74  * read-only.
75  *
76  * The second feature really highlights a subtle point when using the alternate
77  * interface. That is, readline will not alter the terminal when inside your
78  * callback handler. So let's so, your callback executes a user command that
79  * takes a non-trivial amount of time to complete (seconds). While your
80  * executing the command, the user continues to type keystrokes and expects them
81  * to be re-echoed on the new prompt when it returns. Unfortunately, the default
82  * terminal configuration doesn't do this. After the prompt returns, the user
83  * must hit one additional keystroke and then will see all of his previous
84  * keystrokes. To illustrate this, compile and run this program. Type "sleep" at
85  * the prompt and then type "bar" before the prompt returns (you have 3
86  * seconds). Notice how "bar" is re-echoed on the prompt after the prompt
87  * returns? This is what you expect to happen. Now comment out the 4 lines below
88  * the line that says COMMENT LINE BELOW. Recompile and rerun the program and do
89  * the same thing. When the prompt returns, you should not see "bar". Now type
90  * "f", see how "barf" magically appears? This behavior is un-expected and not
91  * desired.
92  */

104 int
105 main()
106 {
107  fd_set fds;
108 
109  setlocale (LC_ALL, "");
110 
111  /* Adjust the terminal slightly before the handler is installed. Disable
112  * canonical mode processing and set the input character time flag to be
113  * non-blocking.
114  */
115  if( tcgetattr(STDIN_FILENO, &term) < 0 ) {
116  perror("tcgetattr");
117  exit(1);
118  }
119  old_lflag = term.c_lflag;
120  old_vtime = term.c_cc[VTIME];
121  term.c_lflag &= ~ICANON;
122  term.c_cc[VTIME] = 1;
123  /* COMMENT LINE BELOW - see above */
124  if( tcsetattr(STDIN_FILENO, TCSANOW, &term) < 0 ) {
125  perror("tcsetattr");
126  exit(1);
127  }
128 
129  rl_add_defun("change-prompt", change_prompt, CTRL('t'));
130  rl_callback_handler_install(get_prompt(), process_line);
131 
132  while(1) {
133  FD_ZERO(&fds);
134  FD_SET(fileno(stdin), &fds);
135 
136  if( select(FD_SETSIZE, &fds, NULL, NULL, NULL) < 0) {
137  perror("select");
138  exit(1);
139  }
140 
141  if( FD_ISSET(fileno(stdin), &fds) ) {
142  rl_callback_read_char();
143  }
144  }
145 }

147 void
148 process_line(char *line)
149 {
150  if( line == NULL ) {
151  fprintf(stderr, "\n", line);
152 
153  /* reset the old terminal setting before exiting */
154  term.c_lflag = old_lflag;
155  term.c_cc[VTIME] = old_vtime;
156  if( tcsetattr(STDIN_FILENO, TCSANOW, &term) < 0 ) {
157  perror("tcsetattr");
158  exit(1);
159  }
160  exit(0);
161  }
162 
163  if( strcmp(line, "sleep") == 0 ) {
164  sleep(3);
165  } else {
166  fprintf(stderr, "|%s|\n", line);
167  }
168 
169  free (line);
170 }

183

190  rl_refresh_line(0, 0);
