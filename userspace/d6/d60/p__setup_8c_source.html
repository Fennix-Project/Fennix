115 //
116 // P_LoadVertexes
117 //
118 void P_LoadVertexes (int lump)
119 {
120  byte* data;
121  int i;
122  mapvertex_t* ml;
123  vertex_t* li;
124 
125  // Determine number of lumps:
126  // total lump length / vertex record length.
127  numvertexes = W_LumpLength (lump) / sizeof(mapvertex_t);
128 
129  // Allocate zone memory for buffer.
130  vertexes = Z_Malloc (numvertexes*sizeof(vertex_t),PU_LEVEL,0);
131 
132  // Load data into cache.
133  data = W_CacheLumpNum (lump, PU_STATIC);
134 
135  ml = (mapvertex_t *)data;
136  li = vertexes;
137 
138  // Copy and convert vertex coordinates,
139  // internal representation as fixed.
140  for (i=0 ; i<numvertexes ; i++, li++, ml++)
141  {
142  li->x = SHORT(ml->x)<<FRACBITS;
143  li->y = SHORT(ml->y)<<FRACBITS;
144  }
145 
146  // Free buffer memory.
147  W_ReleaseLumpNum(lump);
148 }

172 void P_LoadSegs (int lump)
173 {
174  byte* data;
175  int i;
176  mapseg_t* ml;
177  seg_t* li;
178  line_t* ldef;
179  int linedef;
180  int side;
181  int sidenum;
182 
183  numsegs = W_LumpLength (lump) / sizeof(mapseg_t);
184  segs = Z_Malloc (numsegs*sizeof(seg_t),PU_LEVEL,0);
185  memset (segs, 0, numsegs*sizeof(seg_t));
186  data = W_CacheLumpNum (lump,PU_STATIC);
187 
188  ml = (mapseg_t *)data;
189  li = segs;
190  for (i=0 ; i<numsegs ; i++, li++, ml++)
191  {
192  li->v1 = &vertexes[SHORT(ml->v1)];
193  li->v2 = &vertexes[SHORT(ml->v2)];
194 
195  li->angle = (SHORT(ml->angle))<<16;
196  li->offset = (SHORT(ml->offset))<<16;
197  linedef = SHORT(ml->linedef);
198  ldef = &lines[linedef];
199  li->linedef = ldef;
200  side = SHORT(ml->side);
201  li->sidedef = &sides[ldef->sidenum[side]];
202  li->frontsector = sides[ldef->sidenum[side]].sector;
203 
204  if (ldef-> flags & ML_TWOSIDED)
205  {
206  sidenum = ldef->sidenum[side ^ 1];
207 
208  // If the sidenum is out of range, this may be a "glass hack"
209  // impassible window. Point at side #0 (this may not be
210  // the correct Vanilla behavior; however, it seems to work for
211  // OTTAWAU.WAD, which is the one place I've seen this trick
212  // used).
213 
214  if (sidenum < 0 || sidenum >= numsides)
215  {
216  li->backsector = GetSectorAtNullAddress();
217  }
218  else
219  {
220  li->backsector = sides[sidenum].sector;
221  }
222  }
223  else
224  {
225  li->backsector = 0;
226  }
227  }
228 
229  W_ReleaseLumpNum(lump);
230 }

389 // P_LoadLineDefs
390 // Also counts secret lines for intermissions.
391 //
392 void P_LoadLineDefs (int lump)
393 {
394  byte* data;
395  int i;
396  maplinedef_t* mld;
397  line_t* ld;
398  vertex_t* v1;
399  vertex_t* v2;
400 
401  numlines = W_LumpLength (lump) / sizeof(maplinedef_t);
402  lines = Z_Malloc (numlines*sizeof(line_t),PU_LEVEL,0);
403  memset (lines, 0, numlines*sizeof(line_t));
404  data = W_CacheLumpNum (lump,PU_STATIC);
405 
406  mld = (maplinedef_t *)data;
407  ld = lines;
408  for (i=0 ; i<numlines ; i++, mld++, ld++)
409  {
410  ld->flags = SHORT(mld->flags);
411  ld->special = SHORT(mld->special);
412  ld->tag = SHORT(mld->tag);
413  v1 = ld->v1 = &vertexes[SHORT(mld->v1)];
414  v2 = ld->v2 = &vertexes[SHORT(mld->v2)];
415  ld->dx = v2->x - v1->x;
416  ld->dy = v2->y - v1->y;
417 
418  if (!ld->dx)
419  ld->slopetype = ST_VERTICAL;
420  else if (!ld->dy)
421  ld->slopetype = ST_HORIZONTAL;
422  else
423  {
424  if (FixedDiv (ld->dy , ld->dx) > 0)
425  ld->slopetype = ST_POSITIVE;
426  else
427  ld->slopetype = ST_NEGATIVE;
428  }
429 
430  if (v1->x < v2->x)
431  {
432  ld->bbox[BOXLEFT] = v1->x;
433  ld->bbox[BOXRIGHT] = v2->x;
434  }
435  else
436  {
437  ld->bbox[BOXLEFT] = v2->x;
438  ld->bbox[BOXRIGHT] = v1->x;
439  }
440 
441  if (v1->y < v2->y)
442  {
443  ld->bbox[BOXBOTTOM] = v1->y;
444  ld->bbox[BOXTOP] = v2->y;
445  }
446  else
447  {
448  ld->bbox[BOXBOTTOM] = v2->y;
449  ld->bbox[BOXTOP] = v1->y;
450  }
451 
452  ld->sidenum[0] = SHORT(mld->sidenum[0]);
453  ld->sidenum[1] = SHORT(mld->sidenum[1]);
454 
455  if (ld->sidenum[0] != -1)
456  ld->frontsector = sides[ld->sidenum[0]].sector;
457  else
458  ld->frontsector = 0;
459 
460  if (ld->sidenum[1] != -1)
461  ld->backsector = sides[ld->sidenum[1]].sector;
462  else
463  ld->backsector = 0;
464  }
465 
466  W_ReleaseLumpNum(lump);
467 }

540 //
541 // P_GroupLines
542 // Builds sector line lists and subsector sector numbers.
543 // Finds block bounding boxes for sectors.
544 //
545 void P_GroupLines (void)
546 {
547  line_t** linebuffer;
548  int i;
549  int j;
550  line_t* li;
551  sector_t* sector;
552  subsector_t* ss;
553  seg_t* seg;
554  fixed_t bbox[4];
555  int block;
556 
557  // look up sector number for each subsector
558  ss = subsectors;
559  for (i=0 ; i<numsubsectors ; i++, ss++)
560  {
561  seg = &segs[ss->firstline];
562  ss->sector = seg->sidedef->sector;
563  }
564 
565  // count number of lines in each sector
566  li = lines;
567  totallines = 0;
568  for (i=0 ; i<numlines ; i++, li++)
569  {
570  totallines++;
571  li->frontsector->linecount++;
572 
573  if (li->backsector && li->backsector != li->frontsector)
574  {
575  li->backsector->linecount++;
576  totallines++;
577  }
578  }
579 
580  // build line tables for each sector
581  linebuffer = Z_Malloc (totallines*sizeof(line_t *), PU_LEVEL, 0);
582 
583  for (i=0; i<numsectors; ++i)
584  {
585  // Assign the line buffer for this sector
586 
587  sectors[i].lines = linebuffer;
588  linebuffer += sectors[i].linecount;
589 
590  // Reset linecount to zero so in the next stage we can count
591  // lines into the list.
592 
593  sectors[i].linecount = 0;
594  }
595 
596  // Assign lines to sectors
597 
598  for (i=0; i<numlines; ++i)
599  {
600  li = &lines[i];
601 
602  if (li->frontsector != NULL)
603  {
604  sector = li->frontsector;
605 
606  sector->lines[sector->linecount] = li;
607  ++sector->linecount;
608  }
609 
610  if (li->backsector != NULL && li->frontsector != li->backsector)
611  {
612  sector = li->backsector;
613 
614  sector->lines[sector->linecount] = li;
615  ++sector->linecount;
616  }
617  }
618 
619  // Generate bounding boxes for sectors
620 
621  sector = sectors;
622  for (i=0 ; i<numsectors ; i++, sector++)
623  {
624  M_ClearBox (bbox);
625 
626  for (j=0 ; j<sector->linecount; j++)
627  {
628  li = sector->lines[j];
629 
630  M_AddToBox (bbox, li->v1->x, li->v1->y);
631  M_AddToBox (bbox, li->v2->x, li->v2->y);
632  }
633 
634  // set the degenmobj_t to the middle of the bounding box
635  sector->soundorg.x = (bbox[BOXRIGHT]+bbox[BOXLEFT])/2;
636  sector->soundorg.y = (bbox[BOXTOP]+bbox[BOXBOTTOM])/2;
637 
638  // adjust bounding box to map blocks
639  block = (bbox[BOXTOP]-bmaporgy+MAXRADIUS)>>MAPBLOCKSHIFT;
640  block = block >= bmapheight ? bmapheight-1 : block;
641  sector->blockbox[BOXTOP]=block;
642 
643  block = (bbox[BOXBOTTOM]-bmaporgy-MAXRADIUS)>>MAPBLOCKSHIFT;
644  block = block < 0 ? 0 : block;
645  sector->blockbox[BOXBOTTOM]=block;
646 
647  block = (bbox[BOXRIGHT]-bmaporgx+MAXRADIUS)>>MAPBLOCKSHIFT;
648  block = block >= bmapwidth ? bmapwidth-1 : block;
649  sector->blockbox[BOXRIGHT]=block;
650 
651  block = (bbox[BOXLEFT]-bmaporgx-MAXRADIUS)>>MAPBLOCKSHIFT;
652  block = block < 0 ? 0 : block;
653  sector->blockbox[BOXLEFT]=block;
654  }
655 
656 }

658 // Pad the REJECT lump with extra data when the lump is too small,
659 // to simulate a REJECT buffer overflow in Vanilla Doom.
660 
661 static void PadRejectArray(byte *array, unsigned int len)
662 {
663  unsigned int i;
664  unsigned int byte_num;
665  byte *dest;
666  unsigned int padvalue;
667 
668  // Values to pad the REJECT array with:
669 
670  unsigned int rejectpad[4] =
671  {
672  ((totallines * 4 + 3) & ~3) + 24, // Size
673  0, // Part of z_zone block header
674  50, // PU_LEVEL
675  0x1d4a11 // DOOM_CONST_ZONEID
676  };
677 
678  // Copy values from rejectpad into the destination array.
679 
680  dest = array;
681 
682  for (i=0; i<len && i<sizeof(rejectpad); ++i)
683  {
684  byte_num = i % 4;
685  *dest = (rejectpad[i / 4] >> (byte_num * 8)) & 0xff;
686  ++dest;
687  }
688 
689  // We only have a limited pad size. Print a warning if the
690  // REJECT lump is too small.
691 
692  if (len > sizeof(rejectpad))
693  {
694  fprintf(stderr, "PadRejectArray: REJECT lump too short to pad! (%i > %i)\n",
695  len, (int) sizeof(rejectpad));
696 
697  // Pad remaining space with 0 (or 0xff, if specified on command line).
698 
699  if (M_CheckParm("-reject_pad_with_ff"))
700  {
701  padvalue = 0xff;
702  }
703  else
704  {
705  padvalue = 0xf00;
706  }
707 
708  memset(array + sizeof(rejectpad), padvalue, len - sizeof(rejectpad));
709  }
710 }

712 static void P_LoadReject(int lumpnum)
713 {
714  int minlength;
715  int lumplen;
716 
717  // Calculate the size that the REJECT lump *should* be.
718 
719  minlength = (numsectors * numsectors + 7) / 8;
720 
721  // If the lump meets the minimum length, it can be loaded directly.
722  // Otherwise, we need to allocate a buffer of the correct size
723  // and pad it with appropriate data.
724 
725  lumplen = W_LumpLength(lumpnum);
726 
727  if (lumplen >= minlength)
728  {
729  rejectmatrix = W_CacheLumpNum(lumpnum, PU_LEVEL);
730  }
731  else
732  {
733  rejectmatrix = Z_Malloc(minlength, PU_LEVEL, &rejectmatrix);
734  W_ReadLump(lumpnum, rejectmatrix);
735 
736  PadRejectArray(rejectmatrix + lumplen, minlength - lumplen);
737  }
738 }
