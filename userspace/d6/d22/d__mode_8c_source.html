65 boolean D_ValidEpisodeMap(GameMission_t mission, GameMode_t mode,
66  int episode, int map)
67 {
68  int i;
69 
70  // Hacks for Heretic secret episodes
71 
72  if (mission == heretic)
73  {
74  if (mode == retail && episode == 6)
75  {
76  return map >= 1 && map <= 3;
77  }
78  else if (mode == registered && episode == 4)
79  {
80  return map == 1;
81  }
82  }
83 
84  // Find the table entry for this mission/mode combination.
85 
86  for (i=0; i<arrlen(valid_modes); ++i)
87  {
88  if (mission == valid_modes[i].mission
89  && mode == valid_modes[i].mode)
90  {
91  return episode >= 1 && episode <= valid_modes[i].episode
92  && map >= 1 && map <= valid_modes[i].map;
93  }
94  }
95 
96  // Unknown mode/mission combination
97 
98  return false;
