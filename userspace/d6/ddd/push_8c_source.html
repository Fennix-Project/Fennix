41 int
42 push_builtin (list)
43  WORD_LIST *list;
44 {
45  pid_t pid;
46  int xstatus, opt;
47 
48  xstatus = EXECUTION_SUCCESS;
49  reset_internal_getopt ();
50  while ((opt = internal_getopt (list, "")) != -1)
51  {
52  switch (opt)
53  {
54  CASE_HELPOPT;
55  default:
56  builtin_usage ();
57  return (EX_USAGE);
58  }
59  }
60  list = loptend;
61 
62  pid = make_child (savestring ("push"), 0);
63  if (pid == -1)
64  {
65  builtin_error ("cannot fork: %s", strerror (errno));
66  return (EXECUTION_FAILURE);
67  }
68  else if (pid == 0)
69  {
70  /* Shell variable adjustments: $SHLVL, $$, $PPID, $! */
71  adjust_shell_level (1);
72  dollar_dollar_pid = getpid ();
73  set_ppid ();
74 
75  /* Clean up job control stuff. */
76  stop_making_children ();
77  cleanup_the_pipeline ();
78  delete_all_jobs (0);
79 
80  last_asynchronous_pid = NO_PID;
81 
82  /* Make sure the job control code has the right values for
83  the shell's process group and tty process group, and that
84  the signals are set correctly for job control. */
85  initialize_job_control (0);
86  initialize_job_signals ();
87 
88  /* And read commands until exit. */
89  reader_loop ();
90  exit_shell (last_command_exit_value);
91  }
92  else
93  {
94  stop_pipeline (0, (COMMAND *)NULL);
95  xstatus = wait_for (pid, 0);
96  return (xstatus);
97  }
98 }
