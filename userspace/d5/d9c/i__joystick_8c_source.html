90 static boolean IsValidAxis(int axis)
91 {
92  int num_axes;
93 
94  if (axis < 0)
95  {
96  return true;
97  }

101  return true;

161 
162  // Initialized okay!
163 
164  printf("I_InitJoystick: %s\n", SDL_JoystickName(joystick_index));

166  I_AtExit(I_ShutdownJoystick, true);

203 static int ReadButtonState(int vbutton)
204 {
205  int physbutton;
206 
207  // Map from virtual button to physical (SDL) button.
208  if (vbutton < NUM_VIRTUAL_BUTTONS)
209  {
210  physbutton = joystick_physical_buttons[vbutton];
211  }
212  else
213  {
214  physbutton = vbutton;
215  }
216 
217  // Never read axis buttons as buttons.
218  if (IsAxisButton(physbutton))
219  {
220  return 0;
221  }
222 
223  return SDL_JoystickGetButton(joystick, physbutton);
224 }

226 // Get a bitmask of all currently-pressed buttons
227 
228 static int GetButtonsState(void)
229 {
230  int i;
231  int result;
232 
233  result = 0;
234 
235  for (i = 0; i < 20; ++i)
236  {
237  if (ReadButtonState(i))
238  {
239  result |= 1 << i;
240  }
241  }
242 
243  return result;
244 }

246 // Read the state of an axis, inverting if necessary.
247 
248 static int GetAxisState(int axis, int invert)
249 {
250  int result;
251 
252  // Axis -1 means disabled.
253 
254  if (axis < 0)
255  {
256  return 0;
257  }
258 
259  // Is this a button axis, or a hat axis?
260  // If so, we need to handle it specially.
261 
262  result = 0;
263 
264  if (IS_BUTTON_AXIS(axis))
265  {
266  if (SDL_JoystickGetButton(joystick, BUTTON_AXIS_NEG(axis)))
267  {
268  result -= 32767;
269  }
270  if (SDL_JoystickGetButton(joystick, BUTTON_AXIS_POS(axis)))
271  {
272  result += 32767;
273  }
274  }
275  else if (IS_HAT_AXIS(axis))
276  {
277  int direction = HAT_AXIS_DIRECTION(axis);
278  int hatval = SDL_JoystickGetHat(joystick, HAT_AXIS_HAT(axis));
279 
280  if (direction == HAT_AXIS_HORIZONTAL)
281  {
282  if ((hatval & SDL_HAT_LEFT) != 0)
283  {
284  result -= 32767;
285  }
286  else if ((hatval & SDL_HAT_RIGHT) != 0)
287  {
288  result += 32767;
289  }
290  }
291  else if (direction == HAT_AXIS_VERTICAL)
292  {
293  if ((hatval & SDL_HAT_UP) != 0)
294  {
295  result -= 32767;
296  }
297  else if ((hatval & SDL_HAT_DOWN) != 0)
298  {
299  result += 32767;
300  }
301  }
302  }
303  else
304  {
305  result = SDL_JoystickGetAxis(joystick, axis);
306 
307  if (result < DEAD_ZONE && result > -DEAD_ZONE)
308  {
309  result = 0;
310  }
311  }
312 
313  if (invert)
314  {
315  result = -result;
316  }
317 
318  return result;
319 }
