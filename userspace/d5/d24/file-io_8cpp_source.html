97 int abstract_file::read(char *buffer, size_t max_size, size_t *actual_size) {
98  __ensure(max_size);
99 
100  if(_init_bufmode())
101  return -1;
102 
103  size_t unget_length = 0;
104  if (__unget_ptr != __buffer_ptr) {
105  unget_length = frg::min(max_size, (size_t)(__buffer_ptr - __unget_ptr));
106  memcpy(buffer, __unget_ptr, unget_length);
107 
108  __unget_ptr += unget_length;
109  buffer += unget_length;
110  max_size -= unget_length;
111 
112  if (max_size == 0) {
113  *actual_size = unget_length;
114  return 0;
115  }
116  }
117 
118  if(globallyDisableBuffering || _bufmode == buffer_mode::no_buffer) {
119  size_t io_size;
120  if(int e = io_read(buffer, max_size, &io_size); e) {
121  __status_bits |= __MLIBC_ERROR_BIT;
122  return e;
123  }
124  if(!io_size)
125  __status_bits |= __MLIBC_EOF_BIT;
126  *actual_size = io_size + unget_length;
127  return 0;
128  }
129 
130  // Ensure correct buffer type for pipe-like streams.
131  // TODO: In order to support pipe-like streams we need to write-back the buffer.
132  if(__io_mode && __valid_limit)
133  mlibc::panicLogger() << "mlibc: Cannot read-write to same pipe-like stream"
134  << frg::endlog;
135  __io_mode = 0;
136 
137  // Clear the buffer, then buffer new data.
138  if(__offset == __valid_limit) {
139  // TODO: We only have to write-back/reset if __valid_limit reaches the buffer end.
140  if(int e = _write_back(); e)
141  return e;
142  if(int e = _reset(); e)
143  return e;
144 
145  // Perform a read-ahead.
146  _ensure_allocation();
147  size_t io_size;
148  if(int e = io_read(__buffer_ptr, __buffer_size, &io_size); e) {
149  __status_bits |= __MLIBC_ERROR_BIT;
150  return e;
151  }
152  if(!io_size) {
153  __status_bits |= __MLIBC_EOF_BIT;
154  *actual_size = 0;
155  return 0;
156  }
157 
158  __io_offset = io_size;
159  __valid_limit = io_size;
160  }
161 
162  // Return data from the buffer.
163  __ensure(__offset < __valid_limit);
164 
165  auto chunk = frg::min(size_t(__valid_limit - __offset), max_size);
166  memcpy(buffer, __buffer_ptr + __offset, chunk);
167  __offset += chunk;
168 
169  *actual_size = chunk + unget_length;
170  return 0;
171 }

173 int abstract_file::write(const char *buffer, size_t max_size, size_t *actual_size) {
174  __ensure(max_size);
175 
176  if(_init_bufmode())
177  return -1;
178  if(globallyDisableBuffering || _bufmode == buffer_mode::no_buffer) {
179  // As we do not buffer, nothing can be dirty.
180  __ensure(__dirty_begin == __dirty_end);
181  size_t io_size;
182  if(int e = io_write(buffer, max_size, &io_size); e) {
183  __status_bits |= __MLIBC_ERROR_BIT;
184  return e;
185  }
186  __ensure(io_size > 0 && "io_write() is expected to always write at least one byte");
187  *actual_size = io_size;
188  return 0;
189  }
190 
191  // Flush the buffer if necessary.
192  if(__offset == __buffer_size) {
193  if(int e = _write_back(); e)
194  return e;
195  if(int e = _reset(); e)
196  return e;
197  }
198 
199  // Ensure correct buffer type for pipe-like streams.
200  // TODO: We could full support pipe-like files
201  // by ungetc()ing all data before a write happens,
202  // however, for now we just report an error.
203  if(!__io_mode && __valid_limit) // TODO: Only check this for pipe-like streams.
204  mlibc::panicLogger() << "mlibc: Cannot read-write to same pipe-like stream"
205  << frg::endlog;
206  __io_mode = 1;
207 
208  __ensure(__offset < __buffer_size);
209  auto chunk = frg::min(__buffer_size - __offset, max_size);
210 
211  // Line-buffered streams perform I/O on full lines.
212  bool flush_line = false;
213  if(_bufmode == buffer_mode::line_buffer) {
214  auto nl = reinterpret_cast<char *>(memchr(buffer, '\n', chunk));
215  if(nl) {
216  chunk = nl + 1 - buffer;
217  flush_line = true;
218  }
219  }
220  __ensure(chunk);
221 
222  // Buffer data (without necessarily performing I/O).
223  _ensure_allocation();
224  memcpy(__buffer_ptr + __offset, buffer, chunk);
225 
226  if(__dirty_begin != __dirty_end) {
227  __dirty_begin = frg::min(__dirty_begin, __offset);
228  __dirty_end = frg::max(__dirty_end, __offset + chunk);
229  }else{
230  __dirty_begin = __offset;
231  __dirty_end = __offset + chunk;
232  }
233  __valid_limit = frg::max(__offset + chunk, __valid_limit);
234  __offset += chunk;
235 
236  // Flush line-buffered streams.
237  if(flush_line) {
238  if(_write_back())
239  return -1;
240  }
241 
242  *actual_size = chunk;
243  return 0;
244 }

346 int abstract_file::_write_back() {
347  if(int e = _init_type(); e)
348  return e;
349 
350  if(__dirty_begin == __dirty_end)
351  return 0;
352 
353  // For non-pipe streams, first do a seek to reset the
354  // I/O position to zero, then do a write().
355  if(_type == stream_type::file_like) {
356  if(__io_offset != __dirty_begin) {
357  __ensure(__dirty_begin - __io_offset > 0);
358  off_t new_offset;
359  if(int e = io_seek(off_t(__dirty_begin) - off_t(__io_offset), SEEK_CUR, &new_offset); e)
360  return e;
361  __io_offset = __dirty_begin;
362  }
363  }else{
364  __ensure(_type == stream_type::pipe_like);
365  __ensure(__io_offset == __dirty_begin);
366  }
367 
368  // Now, we are in the correct position to write-back everything.
369  while(__io_offset < __dirty_end) {
370  size_t io_size;
371  if(int e = io_write(__buffer_ptr + __io_offset, __dirty_end - __io_offset, &io_size); e) {
372  __status_bits |= __MLIBC_ERROR_BIT;
373  return e;
374  }
375  __ensure(io_size > 0 && "io_write() is expected to always write at least one byte");
376  __io_offset += io_size;
377  __dirty_begin += io_size;
378  }
379 
380  return 0;
381 }

508 int fd_file::parse_modestring(const char *mode) {
509  // Consume the first char; this must be 'r', 'w' or 'a'.
510  int flags = 0;
511  bool has_plus = strchr(mode, '+');
512  if(*mode == 'r') {
513  if(has_plus) {
514  flags = O_RDWR;
515  }else{
516  flags = O_RDONLY;
517  }
518  }else if(*mode == 'w') {
519  if(has_plus) {
520  flags = O_RDWR;
521  }else{
522  flags = O_WRONLY;
523  }
524  flags |= O_CREAT | O_TRUNC;
525  }else if(*mode == 'a') {
526  if(has_plus) {
527  flags = O_APPEND | O_RDWR;
528  }else{
529  flags = O_APPEND | O_WRONLY;
530  }
531  flags |= O_CREAT;
532  }else{
533  mlibc::infoLogger() << "Illegal fopen() mode '" << *mode << "'" << frg::endlog;
534  }
535  mode += 1;
536 
537  // Consume additional flags.
538  while(*mode) {
539  if(*mode == '+') {
540  mode++; // This is already handled above.
541  }else if(*mode == 'b') {
542  mode++; // mlibc assumes that there is no distinction between text and binary.
543  }else if(*mode == 'e') {
544  flags |= O_CLOEXEC;
545  mode++;
546  }else{
547  mlibc::infoLogger() << "Illegal fopen() flag '" << mode << "'" << frg::endlog;
548  mode++;
549  }
550  }
551 
552  return flags;
553 }
