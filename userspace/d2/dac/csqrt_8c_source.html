85  } else {

95  }
96  if (x == 0.0) {

104  }
105  /* Rescale to avoid internal overflow or underflow. */
106  if ((fabs(x) > 4.0) || (fabs(y) > 4.0)) {
107  x *= 0.25;
108  y *= 0.25;
109  scale = 2.0;
110  } else {

113  y *= 1.8014398509481984e16;

115 #else

119 #endif
120  }
