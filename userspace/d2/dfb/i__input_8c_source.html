242 static unsigned char GetTypedChar(unsigned char key)
243 {
244  key = TranslateKey(key);
245 
246  // Is shift held down? If so, perform a translation.
247 
248  if (shiftdown > 0)
249  {
250  if (key >= 0 && key < arrlen(shiftxform))
251  {
252  key = shiftxform[key];
253  }
254  else
255  {
256  key = 0;
257  }
258  }
259 
260  return key;
261 }

279 void I_GetEvent(void)
280 {
281  event_t event;
282  int pressed;
283  unsigned char key;
284 
285 
286  while (DG_GetKey(&pressed, &key))
287  {
288  UpdateShiftStatus(pressed, key);
289 
290  // process event
291 
292  if (pressed)
293  {
294  // data1 has the key pressed, data2 has the character
295  // (shift-translated, etc)
296  event.type = ev_keydown;
297  event.data1 = TranslateKey(key);
298  event.data2 = GetTypedChar(key);
299 
300  if (event.data1 != 0)
301  {
302  D_PostEvent(&event);
303  }
304  }
305  else
306  {
307  event.type = ev_keyup;
308  event.data1 = TranslateKey(key);
309 
310  // data2 is just initialized to zero for ev_keyup.
311  // For ev_keydown it's the shifted Unicode character
312  // that was typed, but if something wants to detect
313  // key releases it should do so based on data1
314  // (key ID), not the printable char.
315 
316  event.data2 = 0;
317 
318  if (event.data1 != 0)
319  {
320  D_PostEvent(&event);
321  }
322  break;
323  }
324  }
325 
326 
327  /*
328  case SDL_MOUSEMOTION:
329  event.type = ev_mouse;
330  event.data1 = mouse_button_state;
331  event.data2 = AccelerateMouse(sdlevent.motion.xrel);
332  event.data3 = -AccelerateMouse(sdlevent.motion.yrel);
333  D_PostEvent(&event);
334  break;
335  */
