140 /* Yield the difference between (YEAR-YDAY HOUR:MIN:SEC) and (*TP),
141  measured in seconds, ignoring leap seconds.
142  YEAR uses the same numbering as TM->tm_year.
143  All values are in range, except possibly YEAR.
144  If overflow occurs, yield the low order bits of the correct answer. */
145 static time_t
146 ydhms_tm_diff (year, yday, hour, min, sec, tp)
147  int year, yday, hour, min, sec;
148  const struct tm *tp;
149 {
150  /* Compute intervening leap days correctly even if year is negative.
151  Take care to avoid int overflow. time_t overflow is OK, since
152  only the low order bits of the correct time_t answer are needed.
153  Don't convert to time_t until after all divisions are done, since
154  time_t might be unsigned. */
155  int a4 = (year >> 2) + (TM_YEAR_BASE >> 2) - ! (year & 3);
156  int b4 = (tp->tm_year >> 2) + (TM_YEAR_BASE >> 2) - ! (tp->tm_year & 3);
157  int a100 = a4 / 25 - (a4 % 25 < 0);
158  int b100 = b4 / 25 - (b4 % 25 < 0);
159  int a400 = a100 >> 2;
160  int b400 = b100 >> 2;
161  int intervening_leap_days = (a4 - b4) - (a100 - b100) + (a400 - b400);
162  time_t years = year - (time_t) tp->tm_year;
163  time_t days = (365 * years + intervening_leap_days
164  + (yday - tp->tm_yday));
165  return (60 * (60 * (24 * days + (hour - tp->tm_hour))
166  + (min - tp->tm_min))
167  + (sec - tp->tm_sec));
168 }

188 /* Convert *TP to a time_t value, inverting
189  the monotonic and mostly-unit-linear conversion function CONVERT.
190  Use *OFFSET to keep track of a guess at the offset of the result,
191  compared to what the result would be for UTC without leap seconds.
192  If *OFFSET's guess is correct, only one CONVERT call is needed. */
193 time_t
194 __mktime_internal (tp, convert, offset)
195  struct tm *tp;
196  struct tm *(*convert) PARAMS ((const time_t *, struct tm *));
197  time_t *offset;
198 {
199  time_t t, dt, t0;
200  struct tm tm;
201 
202  /* The maximum number of probes (calls to CONVERT) should be enough
203  to handle any combinations of time zone rule changes, solar time,
204  and leap seconds. Posix.1 prohibits leap seconds, but some hosts
205  have them anyway. */
206  int remaining_probes = 4;
207 
208  /* Time requested. Copy it in case CONVERT modifies *TP; this can
209  occur if TP is localtime's returned value and CONVERT is localtime. */
210  int sec = tp->tm_sec;
211  int min = tp->tm_min;
212  int hour = tp->tm_hour;
213  int mday = tp->tm_mday;
214  int mon = tp->tm_mon;
215  int year_requested = tp->tm_year;
216  int isdst = tp->tm_isdst;
217 
218  /* Ensure that mon is in range, and set year accordingly. */
219  int mon_remainder = mon % 12;
220  int negative_mon_remainder = mon_remainder < 0;
221  int mon_years = mon / 12 - negative_mon_remainder;
222  int year = year_requested + mon_years;
223 
224  /* The other values need not be in range:
225  the remaining code handles minor overflows correctly,
226  assuming int and time_t arithmetic wraps around.
227  Major overflows are caught at the end. */
228 
229  /* Calculate day of year from year, month, and day of month.
230  The result need not be in range. */
231  int yday = ((__mon_yday[__isleap (year + TM_YEAR_BASE)]
232  [mon_remainder + 12 * negative_mon_remainder])
233  + mday - 1);
234 
235 #if LEAP_SECONDS_POSSIBLE
236  /* Handle out-of-range seconds specially,
237  since ydhms_tm_diff assumes every minute has 60 seconds. */
238  int sec_requested = sec;
239  if (sec < 0)
240  sec = 0;
241  if (59 < sec)
242  sec = 59;
243 #endif
244 
245  /* Invert CONVERT by probing. First assume the same offset as last time.
246  Then repeatedly use the error to improve the guess. */
247 
248  tm.tm_year = EPOCH_YEAR - TM_YEAR_BASE;
249  tm.tm_yday = tm.tm_hour = tm.tm_min = tm.tm_sec = 0;
250  t0 = ydhms_tm_diff (year, yday, hour, min, sec, &tm);
251 
252  for (t = t0 + *offset;
253  (dt = ydhms_tm_diff (year, yday, hour, min, sec, (*convert) (&t, &tm)));
254  t += dt)
255  if (--remaining_probes == 0)
256  return -1;
257 
258  /* Check whether tm.tm_isdst has the requested value, if any. */
259  if (0 <= isdst && 0 <= tm.tm_isdst)
260  {
261  int dst_diff = (isdst != 0) - (tm.tm_isdst != 0);
262  if (dst_diff)
263  {
264  /* Move two hours in the direction indicated by the disagreement,
265  probe some more, and switch to a new time if found.
266  The largest known fallback due to daylight savings is two hours:
267  once, in Newfoundland, 1988-10-30 02:00 -> 00:00. */
268  time_t ot = t - 2 * 60 * 60 * dst_diff;
269  while (--remaining_probes != 0)
270  {
271  struct tm otm;
272  if (! (dt = ydhms_tm_diff (year, yday, hour, min, sec,
273  (*convert) (&ot, &otm))))
274  {
275  t = ot;
276  tm = otm;
277  break;
278  }
279  if ((ot += dt) == t)
280  break; /* Avoid a redundant probe. */
281  }
282  }
283  }
284 
285  *offset = t - t0;
286 
287 #if LEAP_SECONDS_POSSIBLE
288  if (sec_requested != tm.tm_sec)
289  {
290  /* Adjust time to reflect the tm_sec requested, not the normalized value.
291  Also, repair any damage from a false match due to a leap second. */
292  t += sec_requested - sec + (sec == 0 && tm.tm_sec == 60);
293  (*convert) (&t, &tm);
294  }
295 #endif
296 
297  if (TIME_T_MAX / INT_MAX / 366 / 24 / 60 / 60 < 3)
298  {
299  /* time_t isn't large enough to rule out overflows in ydhms_tm_diff,
300  so check for major overflows. A gross check suffices,
301  since if t has overflowed, it is off by a multiple of
302  TIME_T_MAX - TIME_T_MIN + 1. So ignore any component of
303  the difference that is bounded by a small value. */
304 
305  double dyear = (double) year_requested + mon_years - tm.tm_year;
306  double dday = 366 * dyear + mday;
307  double dsec = 60 * (60 * (24 * dday + hour) + min) + sec_requested;
308 
309  if (TIME_T_MAX / 3 - TIME_T_MIN / 3 < (dsec < 0 ? - dsec : dsec))
310  return -1;
311  }
312 
313  *tp = tm;
314  return t;
315 }
