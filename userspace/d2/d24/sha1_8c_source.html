195 /* Update the message digest with the contents
196  * of INBUF with length INLEN.
197  */
198 void SHA1_Update(sha1_context_t *hd, byte *inbuf, size_t inlen)
199 {
200  if (hd->count == 64)
201  {
202  /* flush the buffer */
203  Transform(hd, hd->buf);
204  hd->count = 0;
205  hd->nblocks++;
206  }
207  if (!inbuf)
208  return;
209  if (hd->count)
210  {
211  for (; inlen && hd->count < 64; inlen--)
212  hd->buf[hd->count++] = *inbuf++;
213  SHA1_Update(hd, NULL, 0);
214  if (!inlen)
215  return;
216  }
217 
218  while (inlen >= 64)
219  {
220  Transform(hd, inbuf);
221  hd->count = 0;
222  hd->nblocks++;
223  inlen -= 64;
224  inbuf += 64;
225  }
226  for (; inlen && hd->count < 64; inlen--)
227  hd->buf[hd->count++] = *inbuf++;
228 }

238 void SHA1_Final(sha1_digest_t digest, sha1_context_t *hd)
239 {
240  uint32_t t, msb, lsb;
241  byte *p;
242 
243  SHA1_Update(hd, NULL, 0); /* flush */;
244 
245  t = hd->nblocks;
246  /* multiply by 64 to make a byte count */
247  lsb = t << 6;
248  msb = t >> 26;
249  /* add the count */
250  t = lsb;
251  if ((lsb += hd->count) < t)
252  msb++;
253  /* multiply by 8 to make a bit count */
254  t = lsb;
255  lsb <<= 3;
256  msb <<= 3;
257  msb |= t >> 29;
258 
259  if (hd->count < 56)
260  {
261  /* enough room */
262  hd->buf[hd->count++] = 0x80; /* pad */
263  while (hd->count < 56)
264  hd->buf[hd->count++] = 0; /* pad */
265  }
266  else
267  {
268  /* need one extra block */
269  hd->buf[hd->count++] = 0x80; /* pad character */
270  while (hd->count < 64)
271  hd->buf[hd->count++] = 0;
272  SHA1_Update(hd, NULL, 0); /* flush */;
273  memset(hd->buf, 0, 56 ); /* fill next block with zeroes */
274  }
275  /* append the 64 bit count */
276  hd->buf[56] = msb >> 24;
277  hd->buf[57] = msb >> 16;
278  hd->buf[58] = msb >> 8;
279  hd->buf[59] = msb ;
280  hd->buf[60] = lsb >> 24;
281  hd->buf[61] = lsb >> 16;
282  hd->buf[62] = lsb >> 8;
283  hd->buf[63] = lsb ;
284  Transform(hd, hd->buf);
285 
286  p = hd->buf;
287 #ifdef SYS_BIG_ENDIAN
288 #define X(a) do { *(uint32_t*)p = hd->h##a ; p += 4; } while(0)
289 #else /* little endian */
290 #define X(a) do { *p++ = hd->h##a >> 24; *p++ = hd->h##a >> 16; \
291  *p++ = hd->h##a >> 8; *p++ = hd->h##a; } while(0)
292 #endif
293  X(0);
294  X(1);
295  X(2);
296  X(3);
297  X(4);
298 #undef X
299 
300  memcpy(digest, hd->buf, sizeof(sha1_digest_t));
301 }
